87 static uint32_t build_static_word(
88  uint16_t weight, uint16_t delay, uint32_t type,
89  uint16_t post_index, uint32_t synapse_type_bits,
90  uint32_t synapse_index_bits) {
91  uint32_t synapse_index_mask = (1 << synapse_index_bits) - 1;
92  uint32_t synapse_type_mask = (1 << synapse_type_bits) - 1;
93 
94  uint32_t wrd = post_index & synapse_index_mask;
95  wrd |= (type & synapse_type_mask) << synapse_index_bits;
96  wrd |= (delay & SYNAPSE_DELAY_MASK) <<
97  (synapse_index_bits + synapse_type_bits);
98  wrd |= (weight & SYNAPSE_WEIGHT_MASK) << SYNAPSE_WEIGHT_SHIFT;
99  return wrd;
100 }

102 static void matrix_generator_static_write_row(
103  void *data,
104  address_t synaptic_matrix, address_t delayed_synaptic_matrix,
105  uint32_t n_pre_neurons, uint32_t pre_neuron_index,
106  uint32_t max_row_n_words, uint32_t max_delayed_row_n_words,
107  uint32_t synapse_type_bits, uint32_t synapse_index_bits,
108  uint32_t synapse_type, uint32_t n_synapses,
109  uint16_t *indices, uint16_t *delays, uint16_t *weights,
110  uint32_t max_stage) {
111  use(data);
112 
113  log_debug("Max stage = %u", max_stage);
114 
115  // Row address and position for each possible delay stage (including no
116  // delay stage)
117  address_t row_address[max_stage];
118 
119  // The space available on each row
120  uint16_t space[max_stage];
121 
122  // The normal row position and space available - might be 0 if all delayed
123  row_address[0] = NULL;
124  space[0] = max_row_n_words;
125  if (synaptic_matrix != NULL) {
126  row_address[0] =
127  &synaptic_matrix[pre_neuron_index * (max_row_n_words + 3)];
128  }
129  log_debug("row[0] = 0x%08x", row_address[0]);
130 
131  // The delayed row positions and space available
132  if (delayed_synaptic_matrix != NULL) {
133  address_t delayed_address = &delayed_synaptic_matrix[
134  pre_neuron_index * (max_delayed_row_n_words + 3)];
135  uint32_t single_matrix_size =
136  n_pre_neurons * (max_delayed_row_n_words + 3);
137  for (uint32_t i = 1; i < max_stage; i++) {
138  row_address[i] = &delayed_address[single_matrix_size * (i - 1)];
139  space[i] = max_delayed_row_n_words;
140  log_debug("row[%u] = 0x%08x", i, row_address[i]);
141  }
142  } else {
143  for (uint32_t i = 1; i < max_stage; i++) {
144  row_address[i] = NULL;
145  space[i] = 0;
146  log_debug("row[%u] = 0x%08x", i, row_address[i]);
147  }
148  }
149 
150  // The address to write synapses to on each stage
151  address_t write_address[max_stage];
152  for (uint32_t i = 0; i < max_stage; i++) {
153  if (row_address[i] != NULL) {
154  log_debug("Row size at 0x%08x for stage %u",
155  &row_address[i][STATIC_FIXED_FIXED_SIZE], i);
156  row_address[i][STATIC_FIXED_FIXED_SIZE] = 0;
157  row_address[i][STATIC_PLASTIC_PLASTIC_SIZE] = 0;
158  row_address[i][STATIC_FIXED_PLASTIC_SIZE] = 0;
159  write_address[i] = &row_address[i][STATIC_FIXED_FIXED_OFFSET];
160  } else {
161  write_address[i] = NULL;
162  }
163  log_debug("write[%u] = 0x%08x", i, write_address[i]);
164  }
165 
166 
167  // Go through the synapses
168  for (uint32_t synapse = 0; synapse < n_synapses; synapse++) {
169  // Post-neuron index
170  uint32_t post_index = indices[synapse];
171 
172  // Weight
173  uint16_t weight = weights[synapse];
174 
175  // Work out the delay stage and final value
176  struct delay_value delay = get_delay(delays[synapse], max_stage);
177  if (write_address[delay.stage] == NULL) {
178  log_error("Delay stage %u has not been initialised; raw delay = %u,"
179  " delay = %u, max stage = %u", delay.stage, delays[synapse],
180  delay.delay, max_stage);
181  rt_error(RTE_SWERR);
182  }
183 
184  // Avoid errors when rows are full
185  if (space[delay.stage] == 0) {
186  log_warning("Row for delay stage %u is full - word not added!",
187  delay.stage);
188  continue;
189  }
190 
191  // Build synaptic word
192  uint32_t word = build_static_word(
193  weight, delay.delay, synapse_type, post_index, synapse_type_bits,
194  synapse_index_bits);
195 
196  // Write the word
197  log_debug("Writing word to 0x%08x", &write_address[delay.stage][0]);
198  write_address[delay.stage][0] = word;
199  write_address[delay.stage] = &write_address[delay.stage][1];
200 
201  // Increment the size of the current row
202  row_address[delay.stage][STATIC_FIXED_FIXED_SIZE]++;
203  space[delay.stage]--;
204  }
205 }
