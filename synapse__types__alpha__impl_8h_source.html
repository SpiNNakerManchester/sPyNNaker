106 static inline void add_input_alpha(alpha_params_t* a_params, input_t input) {
107  a_params->q_buff = input;
108 
109  a_params->exp_buff =
110  decay_s1615(a_params->exp_buff, a_params->decay) + ONE;
111 
112  a_params->lin_buff =
113  (a_params->lin_buff + (input * a_params->dt_divided_by_tau_sqr))
114  * (ONE - ONE/a_params->exp_buff);
115 }
