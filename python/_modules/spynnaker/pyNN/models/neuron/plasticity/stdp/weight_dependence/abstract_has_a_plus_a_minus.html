from spinn_utilities.abstract_base import AbstractBase


class AbstractHasAPlusAMinus(object, metaclass=AbstractBase):
    """
    An object that has :math:`A^+` and :math:`A^-` properties.
    """
    __slots__ = [
        '__a_plus',
        '__a_minus'
    ]

    def __init__(self):
        self.__a_plus = None
        self.__a_minus = None
