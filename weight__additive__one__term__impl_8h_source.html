85 static inline weight_t weight_get_final(weight_state_t new_state) {
86  // Scale potentiation and depression
87  // **NOTE** A2+ and A2- are pre-scaled into weight format
88  int32_t scaled_a2_plus = STDP_FIXED_MUL_16X16(
89  new_state.a2_plus, new_state.weight_region->a2_plus);
90  int32_t scaled_a2_minus = STDP_FIXED_MUL_16X16(
91  new_state.a2_minus, new_state.weight_region->a2_minus);
92 
93  // Apply all terms to initial weight
94  int32_t new_weight =
95  new_state.initial_weight + scaled_a2_plus - scaled_a2_minus;
96 
97  // Clamp new weight
98  new_weight = MIN(new_state.weight_region->max_weight,
99  MAX(new_weight, new_state.weight_region->min_weight));
100 
101  log_debug("\told_weight:%u, a2+:%d, a2-:%d, scaled a2+:%d, scaled a2-:%d,"
102  " new_weight:%d",
103  new_state.initial_weight, new_state.a2_plus, new_state.a2_minus,
104  scaled_a2_plus, scaled_a2_minus, new_weight);
105 
106  return (weight_t) new_weight;
107 }
