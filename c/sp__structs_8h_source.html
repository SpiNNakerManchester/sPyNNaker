111 static inline bool sp_structs_find_by_spike(
112  pre_pop_info_table_t *pre_pop_info_table, spike_t spike,
113  uint32_t *neuron_id, uint32_t *population_id,
114  uint32_t *sub_population_id, uint32_t *m_pop_index) {
115  // Amazing linear search inc.
116  // Loop over all populations
117  for (uint32_t i = 0; i < pre_pop_info_table->no_pre_pops; i++) {
118  pre_info_t *pre_pop_info = pre_pop_info_table->prepop_info[i];
119 
120  // Loop over all sub-populations and check if the KEY matches
121  // (with neuron ID masked out)
122  for (int j = 0; j < pre_pop_info->no_pre_vertices; j++) {
123  key_atom_info_t *kai = &pre_pop_info->key_atom_info[j];
124  if ((spike & kai->mask) == kai->key) {
125  *population_id = i;
126  *sub_population_id = j;
127  *neuron_id = spike & ~kai->mask;
128  *m_pop_index = kai->m_pop_index;
129  return true;
130  }
131  }
132  }
133  return false;
134 }

138 static inline bool sp_structs_get_sub_pop_info(
139  const pre_pop_info_table_t *pre_pop_table_info, uint32_t population_id,
140  uint32_t pop_neuron_id, uint32_t *restrict sub_population_id,
141  uint32_t *restrict sub_pop_neuron_id, uint32_t *restrict spike) {
142  const pre_info_t *app_pop_info =
143  pre_pop_table_info->prepop_info[population_id];
144  uint32_t neuron_id = pop_neuron_id;
145  for (uint32_t i = 0; i < app_pop_info->no_pre_vertices; i++) {
146  uint32_t n_atoms = app_pop_info->key_atom_info[i].n_atoms;
147  if (neuron_id < n_atoms) {
148  *sub_population_id = i;
149  *sub_pop_neuron_id = neuron_id;
150  *spike = app_pop_info->key_atom_info[i].key | neuron_id;
151  return true;
152  }
153  neuron_id -= n_atoms;
154  }
155  return false;
156 }

167 static inline bool sp_structs_add_synapse(
168  current_state_t *restrict current_state, address_t row) {
169  uint32_t appr_scaled_weight = current_state->pre_population_info->weight;
170 
171  uint32_t actual_delay;
172  uint32_t offset = current_state->pre_population_info->delay_hi -
173  current_state->pre_population_info->delay_lo;
174  actual_delay = ulrbits(mars_kiss64_seed(*(current_state->local_seed))) *
175  offset + current_state->pre_population_info->delay_lo;
176 
177  if (!synapse_dynamics_add_neuron(
178  current_state->post_syn_id, row, appr_scaled_weight, actual_delay,
179  current_state->pre_population_info->connection_type)) {
180  return false;
181  }
182 
183  *(current_state->post_to_pre_table_entry) = current_state->post_to_pre;
184  return true;
