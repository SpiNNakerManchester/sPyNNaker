129 static inline bool sp_structs_find_by_spike(
130  const pre_pop_info_table_t *pre_pop_info_table, spike_t spike,
131  uint32_t *restrict neuron_id, uint32_t *restrict population_id,
132  uint32_t *restrict sub_population_id, uint32_t *restrict m_pop_index) {
133  // Amazing linear search inc.
134  // Loop over all populations
135  for (uint32_t i = 0; i < pre_pop_info_table->no_pre_pops; i++) {
136  const pre_info_t *pre_pop_info = pre_pop_info_table->prepop_info[i];
137 
138  // Loop over all sub-populations and check if the KEY matches
139  // (with neuron ID masked out)
140  for (int j = 0; j < pre_pop_info->no_pre_vertices; j++) {
141  const key_atom_info_t *kai = &pre_pop_info->key_atom_info[j];
142  if ((spike & kai->mask) == kai->key) {
143  *population_id = i;
144  *sub_population_id = j;
145  *neuron_id = spike & ~kai->mask;
146  *m_pop_index = kai->m_pop_index;
147  return true;
148  }
149  }
150  }
151  return false;
152 }

164 static inline bool sp_structs_get_sub_pop_info(
165  const pre_pop_info_table_t *pre_pop_info_table, uint32_t population_id,
166  uint32_t pop_neuron_id, uint32_t *restrict sub_population_id,
167  uint32_t *restrict sub_pop_neuron_id, uint32_t *restrict spike) {
168  const pre_info_t *app_pop_info =
169  pre_pop_info_table->prepop_info[population_id];
170  uint32_t neuron_id = pop_neuron_id;
171  for (uint32_t i = 0; i < app_pop_info->no_pre_vertices; i++) {
172  uint32_t n_atoms = app_pop_info->key_atom_info[i].n_atoms;
173  if (neuron_id < n_atoms) {
174  *sub_population_id = i;
175  *sub_pop_neuron_id = neuron_id;
176  *spike = app_pop_info->key_atom_info[i].key | neuron_id;
177  return true;
178  }
179  neuron_id -= n_atoms;
180  }
181  return false;
182 }

201 static inline bool sp_structs_add_synapse(
202  current_state_t *restrict current_state, address_t restrict row) {
203  uint32_t appr_scaled_weight = current_state->pre_population_info->weight;
204 
205  uint32_t actual_delay;
206  uint32_t offset = current_state->pre_population_info->delay_hi -
207  current_state->pre_population_info->delay_lo;
208  actual_delay = rand_int(offset, *(current_state->local_seed)) +
209  current_state->pre_population_info->delay_lo;
210 
211  if (!synapse_dynamics_add_neuron(
212  current_state->post_syn_id, row, appr_scaled_weight, actual_delay,
213  current_state->pre_population_info->connection_type)) {
214  return false;
215  }
216 
217  // Critical: tell the compiler that this pointer is aligned so it doesn't
218  // internally convert the assignment to a memcpy(), which is a saving of
219  // hundreds of bytes...
220  post_to_pre_entry *ppentry = __builtin_assume_aligned(
221  current_state->post_to_pre_table_entry, 4);
222  *ppentry = current_state->post_to_pre;
223  return true;
224 }

237 static inline uint8_t *sp_structs_read_in_common(
238  address_t sdram_sp_address, rewiring_data_t *rewiring_data,
239  pre_pop_info_table_t *pre_info,
240  post_to_pre_entry **post_to_pre_table) {
241  uint8_t *data = (uint8_t *) sdram_sp_address;
242  spin1_memcpy(rewiring_data, data, sizeof(rewiring_data_t));
243  data += sizeof(rewiring_data_t);
244  log_debug("Topographic Map Impl, s_max=%u", rewiring_data->s_max);
245 
246  pre_info->no_pre_pops = rewiring_data->no_pre_pops;
247  log_debug(" no pre pops = %d", pre_info->no_pre_pops);
248  pre_info->prepop_info = spin1_malloc(
249  rewiring_data->no_pre_pops * sizeof(pre_info_t *));
250  if (pre_info->prepop_info == NULL) {
251  log_error("Could not initialise pre population info");
252  rt_error(RTE_SWERR);
253  }
254  for (uint32_t i = 0; i < rewiring_data->no_pre_pops; i++) {
255  pre_info->prepop_info[i] = (pre_info_t *) data;
256  uint32_t pre_size = (pre_info->prepop_info[i]->no_pre_vertices
257  * sizeof(key_atom_info_t)) + sizeof(pre_info_t);
258  pre_info->prepop_info[i] = spin1_malloc(pre_size);
259  if (pre_info->prepop_info[i] == NULL) {
260  log_error("Could not initialise pre population info %d", i);
261  rt_error(RTE_SWERR);
262  }
263  spin1_memcpy(pre_info->prepop_info[i], data, pre_size);
264 
265  log_debug("no_pre = %u, sp_control %u, "
266  "delay lo %u, delay hi %u, weight %d",
267  pre_info->prepop_info[i]->no_pre_vertices,
268  pre_info->prepop_info[i]->sp_control,
269  pre_info->prepop_info[i]->delay_lo,
270  pre_info->prepop_info[i]->delay_hi,
271  pre_info->prepop_info[i]->weight);
272  log_debug("connection_type = %d, total_no_atoms=%d",
273  pre_info->prepop_info[i]->connection_type,
274  pre_info->prepop_info[i]->total_no_atoms);
275  data += pre_size;
276  }
277 
278  *post_to_pre_table = (post_to_pre_entry *) data;
279  uint32_t n_elements =
280  rewiring_data->s_max * rewiring_data->machine_no_atoms;
281 
282  for (uint32_t i=0; i < n_elements; i++){
283  log_debug("index %d, pop index %d, sub pop index %d, neuron_index %d",
284  i, post_to_pre_table[i]->pop_index,
285  post_to_pre_table[i]->sub_pop_index,
286  post_to_pre_table[i]->neuron_index);
287  }
288  data += n_elements * sizeof(post_to_pre_entry);
289  return (uint8_t *) data;
290 }
