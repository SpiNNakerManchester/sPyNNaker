136 static inline bool sp_structs_find_by_spike(
137  const pre_pop_info_table_t *pre_pop_info_table, spike_t spike,
138  uint32_t *restrict neuron_id, uint32_t *restrict population_id,
139  uint32_t *restrict sub_population_id, uint32_t *restrict m_pop_index) {
140  // Amazing linear search inc.
141  // Loop over all populations
142  for (uint32_t i = 0; i < pre_pop_info_table->no_pre_pops; i++) {
143  const pre_info_t *pre_pop_info = pre_pop_info_table->prepop_info[i];
144 
145  // Loop over all sub-populations and check if the KEY matches
146  // (with neuron ID masked out)
147  for (int j = 0; j < pre_pop_info->no_pre_vertices; j++) {
148  const key_atom_info_t *kai = &pre_pop_info->key_atom_info[j];
149  if ((spike & kai->mask) == kai->key) {
150  *population_id = i;
151  *sub_population_id = j;
152  *neuron_id = spike & ~kai->mask;
153  *m_pop_index = kai->m_pop_index;
154  return true;
155  }
156  }
157  }
158  return false;
159 }

171 static inline bool sp_structs_get_sub_pop_info(
172  const pre_pop_info_table_t *pre_pop_info_table, uint32_t population_id,
173  uint32_t pop_neuron_id, uint32_t *restrict sub_population_id,
174  uint32_t *restrict sub_pop_neuron_id, uint32_t *restrict spike) {
175  const pre_info_t *app_pop_info =
176  pre_pop_info_table->prepop_info[population_id];
177  uint32_t neuron_id = pop_neuron_id;
178  for (uint32_t i = 0; i < app_pop_info->no_pre_vertices; i++) {
179  uint32_t n_atoms = app_pop_info->key_atom_info[i].n_atoms;
180  if (neuron_id < n_atoms) {
181  *sub_population_id = i;
182  *sub_pop_neuron_id = neuron_id;
183  *spike = app_pop_info->key_atom_info[i].key | neuron_id;
184  return true;
185  }
186  neuron_id -= n_atoms;
187  }
188  return false;
189 }

209 static inline bool sp_structs_add_synapse(
210  current_state_t *restrict current_state, synaptic_row_t restrict row) {
211  uint32_t appr_scaled_weight = current_state->pre_population_info->weight;
212 
213  uint32_t actual_delay;
214  uint32_t offset = current_state->pre_population_info->delay_hi -
215  current_state->pre_population_info->delay_lo;
216  actual_delay = rand_int(offset, *(current_state->local_seed)) +
217  current_state->pre_population_info->delay_lo;
218 
219  if (!synapse_dynamics_add_neuron(
220  current_state->post_syn_id, row, appr_scaled_weight, actual_delay,
221  current_state->pre_population_info->connection_type)) {
222  return false;
223  }
224 
225  // Critical: tell the compiler that this pointer is aligned so it doesn't
226  // internally convert the assignment to a memcpy(), which is a saving of
227  // hundreds of bytes...
228  post_to_pre_entry *ppentry = __builtin_assume_aligned(
229  current_state->post_to_pre_table_entry, 4);
230  *ppentry = current_state->post_to_pre;
231  return true;
232 }

245 static inline uint8_t *sp_structs_read_in_common(
246  address_t sdram_sp_address, rewiring_data_t *rewiring_data,
247  pre_pop_info_table_t *pre_info, post_to_pre_entry **post_to_pre_table) {
248  uint8_t *data = (uint8_t *) sdram_sp_address;
249  spin1_memcpy(rewiring_data, data, sizeof(rewiring_data_t));
250  data += sizeof(rewiring_data_t);
251  log_debug("Topographic Map Impl, s_max=%u", rewiring_data->s_max);
252 
253  pre_info->no_pre_pops = rewiring_data->no_pre_pops;
254  log_debug(" no pre pops = %d", pre_info->no_pre_pops);
255  pre_info->prepop_info = spin1_malloc(
256  rewiring_data->no_pre_pops * sizeof(pre_info_t *));
257  if (pre_info->prepop_info == NULL) {
258  log_error("Could not initialise pre population info");
259  rt_error(RTE_SWERR);
260  }
261  for (uint32_t i = 0; i < rewiring_data->no_pre_pops; i++) {
262  pre_info->prepop_info[i] = (pre_info_t *) data;
263  uint32_t pre_size = (pre_info->prepop_info[i]->no_pre_vertices
264  * sizeof(key_atom_info_t)) + sizeof(pre_info_t);
265  pre_info->prepop_info[i] = spin1_malloc(pre_size);
266  if (pre_info->prepop_info[i] == NULL) {
267  log_error("Could not initialise pre population info %d", i);
268  rt_error(RTE_SWERR);
269  }
270  spin1_memcpy(pre_info->prepop_info[i], data, pre_size);
271 
272  log_debug("no_pre = %u, sp_control %u, "
273  "delay lo %u, delay hi %u, weight %d",
274  pre_info->prepop_info[i]->no_pre_vertices,
275  pre_info->prepop_info[i]->sp_control,
276  pre_info->prepop_info[i]->delay_lo,
277  pre_info->prepop_info[i]->delay_hi,
278  pre_info->prepop_info[i]->weight);
279  log_debug("connection_type = %d, total_no_atoms=%d",
280  pre_info->prepop_info[i]->connection_type,
281  pre_info->prepop_info[i]->total_no_atoms);
282  data += pre_size;
283  }
284 
285  *post_to_pre_table = (post_to_pre_entry *) data;
286  uint32_t n_elements =
287  rewiring_data->s_max * rewiring_data->machine_no_atoms;
288 
289  for (uint32_t i=0; i < n_elements; i++){
290  log_debug("index %d, pop index %d, sub pop index %d, neuron_index %d",
291  i, (*post_to_pre_table)[i].pop_index,
292  (*post_to_pre_table)[i].sub_pop_index,
293  (*post_to_pre_table)[i].neuron_index);
294  }
295  data += n_elements * sizeof(post_to_pre_entry);
296  return (uint8_t *) data;
297 }
