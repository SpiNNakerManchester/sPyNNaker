42 address_t weight_initialise(
43  address_t address, uint32_t n_synapse_types,
44  uint32_t *ring_buffer_to_input_buffer_left_shifts) {
45  log_debug("weight_initialise: starting");
46  log_debug("\tSTDP multiplicative weight dependence");
47 
48  // Copy plasticity region data from address
49  // **NOTE** this seems somewhat safer than relying on sizeof
50  plasticity_weight_region_data_t *dtcm_copy = plasticity_weight_region_data =
51  spin1_malloc(sizeof(plasticity_weight_region_data_t) * n_synapse_types);
52  if (dtcm_copy == NULL) {
53  log_error("Could not initialise weight region data");
54  return NULL;
55  }
56  weight_multiply_right_shift =
57  spin1_malloc(sizeof(uint32_t) * n_synapse_types);
58  if (weight_multiply_right_shift == NULL) {
59  log_error("Could not initialise weight region data");
60  return NULL;
61  }
62 
63  multiplicative_config_t *config = (multiplicative_config_t *) address;
64  for (uint32_t s = 0; s < n_synapse_types; s++, config++) {
65  // Copy parameters
66  dtcm_copy[s].min_weight = config->min_weight;
67  dtcm_copy[s].max_weight = config->max_weight;
68  dtcm_copy[s].a2_plus = config->a2_plus;
69  dtcm_copy[s].a2_minus = config->a2_minus;
70 
71  // Calculate the right shift required to fixed-point multiply weights
72  uint32_t shift = weight_multiply_right_shift[s] =
73  16 - (ring_buffer_to_input_buffer_left_shifts[s] + 1);
74 
75  log_debug("\tSynapse type %u: Min weight:%d, Max weight:%d, A2+:%d, A2-:%d,"
76  " Weight multiply right shift:%u",
77  s, dtcm_copy[s].min_weight, dtcm_copy[s].max_weight,
78  dtcm_copy[s].a2_plus, dtcm_copy[s].a2_minus, shift);
79  }
80 
81  log_debug("weight_initialise: completed successfully");
82 
83  // Return end address of region
84  return (address_t) config;
85 }
