43 address_t weight_initialise(
44  address_t address, uint32_t n_synapse_types,
45  uint32_t *ring_buffer_to_input_buffer_left_shifts) {
46  log_debug("weight_initialise: starting");
47  log_debug("\tSTDP multiplicative weight dependence");
48 
49  // Copy plasticity region data from address
50  // **NOTE** this seems somewhat safer than relying on sizeof
51  plasticity_weight_region_data_t *dtcm_copy = plasticity_weight_region_data =
52  spin1_malloc(sizeof(plasticity_weight_region_data_t) * n_synapse_types);
53  if (dtcm_copy == NULL) {
54  log_error("Could not initialise weight region data");
55  return NULL;
56  }
57  weight_shift = spin1_malloc(sizeof(uint32_t) * n_synapse_types);
58  if (weight_shift == NULL) {
59  log_error("Could not initialise weight region data");
60  return NULL;
61  }
62 
63  multiplicative_config_t *config = (multiplicative_config_t *) address;
64  for (uint32_t s = 0; s < n_synapse_types; s++, config++) {
65  // Copy parameters
66  dtcm_copy[s].min_weight = config->min_weight;
67  dtcm_copy[s].max_weight = config->max_weight;
68  dtcm_copy[s].a2_plus = config->a2_plus;
69  dtcm_copy[s].a2_minus = config->a2_minus;
70 
71  // Copy weight shift
72  weight_shift[s] = ring_buffer_to_input_buffer_left_shifts[s];
73 
74  log_debug("\tSynapse type %u: Min weight:%d, Max weight:%d, A2+:%d, A2-:%d,"
75  " Weight multiply right shift:%u",
76  s, dtcm_copy[s].min_weight, dtcm_copy[s].max_weight,
77  dtcm_copy[s].a2_plus, dtcm_copy[s].a2_minus, weight_shift[s]);
78  }
79 
80  log_debug("weight_initialise: completed successfully");
81 
82  // Return end address of region
83  return (address_t) config;
84 }
