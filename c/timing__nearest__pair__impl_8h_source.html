117 static inline update_state_t timing_apply_post_spike(
118  uint32_t time, UNUSED post_trace_t trace, uint32_t last_pre_time,
119  UNUSED pre_trace_t last_pre_trace, uint32_t last_post_time,
120  UNUSED post_trace_t last_post_trace, update_state_t previous_state) {
121  extern int16_lut *tau_plus_lookup;
122 
123  // Get time of event relative to last pre-synaptic event
124  uint32_t time_since_last_pre = time - last_pre_time;
125  uint32_t time_since_last_post = time - last_post_time;
126 
127  if (time_since_last_pre > 0) {
128  int32_t decayed_r1 = maths_lut_exponential_decay(
129  time_since_last_pre, tau_plus_lookup);
130 
131  log_debug("\t\t\ttime_since_last_pret=%u, decayed_r1=%d\n",
132  time_since_last_pre, decayed_r1);
133  log_debug("\t\t\ttime_since_last_post=%u, decayed_r1=%d\n",
134  time_since_last_post, decayed_r1);
135 
136  if (time_since_last_post < time_since_last_pre) {
137  log_debug("\t\t\tSetting trace to zero as not first pre-post pairing");
138  decayed_r1 = 0;
139  }
140 
141  // Apply potentiation to state (which is a weight_state)
142  return weight_one_term_apply_potentiation(previous_state, decayed_r1);
143  } else {
144  return previous_state;
145  }
