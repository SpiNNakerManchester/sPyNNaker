48 static inline bool threshold_type_is_above_threshold(
49  state_t value, threshold_type_t *threshold_type) {
50  UREAL random_number = ukbits(mars_kiss64_simp() & 0xFFFF);
51 
52  REAL exponent = (value - threshold_type->v_thresh)
53  * threshold_type->du_th_inv;
54 
55  // if exponent is large, further calculation is unnecessary
56  // (result --> prob_saturation).
57  UREAL result;
58  if (exponent < 5.0k) {
59  REAL hazard = expk(exponent) * threshold_type->tau_th_inv;
60  result = (1. - expk(hazard *
61  threshold_type->neg_machine_time_step_ms_div_10)) *
62  PROB_SATURATION;
63  } else {
64  result = PROB_SATURATION;
65  }
66 
67  return REAL_COMPARE(result, >=, random_number);
68 }
