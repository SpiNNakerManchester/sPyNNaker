83 static uint32_t connection_generator_one_to_one_generate(
84  void *generator, UNUSED uint32_t pre_slice_start,
85  UNUSED uint32_t pre_slice_count, uint32_t pre_neuron_index,
86  uint32_t post_slice_start, UNUSED uint32_t post_slice_count,
87  uint32_t max_row_length, uint16_t *indices) {
88  struct one_to_one *obj = generator;
89  log_debug("pre_neuron_index %u", pre_neuron_index);
90 
91  // If no space, generate nothing
92  if (max_row_length < 1) {
93  return 0;
94  }
95 
96  // If not in the pre-population view range, then don't generate
97  if ((pre_neuron_index < obj->pre_lo) ||
98  (pre_neuron_index > obj->pre_hi)) {
99  return 0;
100  }
101 
102  // Post-index = view-relative pre-index
103  // Note that this could be negative (but see later)
104  int post_neuron_index = pre_neuron_index - obj->pre_lo + obj->post_lo;
105 
106  // If not in the post-population view range, then don't generate.
107  // This will filter negatives
108  if ((post_neuron_index < (int) obj->post_lo) ||
109  (post_neuron_index > (int) obj->post_hi)) {
110  return 0;
111  }
112 
113  // If out of range, don't generate anything
114  if ((post_neuron_index < (int) post_slice_start) ||
115  (post_neuron_index >= (int) (post_slice_start + post_slice_count))) {
116  return 0;
117  }
118 
119  // Post-index = core-relative post-index
120  indices[0] = post_neuron_index - post_slice_start;
121  return 1;
122 }
