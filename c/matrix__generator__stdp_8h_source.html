94 static uint16_t build_fixed_plastic_half_word(
95  uint16_t delay, uint32_t type,
96  uint32_t post_index, uint32_t synapse_type_bits,
97  uint32_t synapse_index_bits) {
98  uint16_t synapse_index_mask = (1 << synapse_index_bits) - 1;
99  uint16_t synapse_type_mask = (1 << synapse_type_bits) - 1;
100 
101  uint16_t wrd = post_index & synapse_index_mask;
102  wrd |= (type & synapse_type_mask) << synapse_index_bits;
103  wrd |= (delay & SYNAPSE_DELAY_MASK) <<
104  (synapse_index_bits + synapse_type_bits);
105  // wrd |= (delay & SYNAPSE_DELAY_MASK) << synapse_type_bits;
106 
107  return wrd;
108 }

132 void matrix_generator_stdp_write_row(
133  void *generator,
134  address_t synaptic_matrix, address_t delayed_synaptic_matrix,
135  uint32_t n_pre_neurons, uint32_t pre_neuron_index,
136  uint32_t max_row_n_words, uint32_t max_delayed_row_n_words,
137  uint32_t synapse_type_bits, uint32_t synapse_index_bits,
138  uint32_t synapse_type, uint32_t n_synapses,
139  uint16_t *indices, uint16_t *delays, uint16_t *weights,
140  uint32_t max_stage, uint32_t max_delay_per_stage) {
141  struct matrix_generator_stdp *obj = generator;
142 
143  // Row address for each possible delay stage (including no delay stage)
144  row_plastic_t row[max_stage];
145 
146  // Space available in each row
147  uint16_t space_half_words[max_stage];
148 
149  // The number of words in a row including headers
150  uint32_t n_row_words = max_row_n_words + 3;
151  uint32_t n_delay_row_words = max_delayed_row_n_words + 3;
152 
153  // The normal row position and space available - might be 0 if all delayed
154  row[0] = NULL;
155  space_half_words[0] = max_row_n_words * 2;
156  if (synaptic_matrix != NULL) {
157  row[0] = (row_plastic_t)
158  &synaptic_matrix[pre_neuron_index * n_row_words];
159  }
160 
161  // The delayed row positions and space available
162  if (delayed_synaptic_matrix != NULL) {
163  address_t delayed_address =
164  &delayed_synaptic_matrix[pre_neuron_index * n_delay_row_words];
165  uint32_t single_matrix_size = n_pre_neurons * n_delay_row_words;
166  for (uint32_t i = 1; i < max_stage; i++) {
167  row[i] = (row_plastic_t)
168  &delayed_address[single_matrix_size * (i - 1)];
169  space_half_words[i] = max_delayed_row_n_words * 2;
170  }
171  } else {
172  for (uint32_t i = 1; i < max_stage; i++) {
173  row[i] = NULL;
174  space_half_words[i] = 0;
175  }
176  }
177 
178  // Add the header half words (zero initialised) to each row
179  for (uint32_t i = 0; i < max_stage; i++) {
180  if (row[i] != NULL) {
181  row[i]->plastic_plastic_size =
182  obj->n_half_words_per_pp_row_header >> 1;
183  uint16_t *header = row[i]->plastic_plastic_data;
184  for (uint32_t j = 0;
185  j < obj->n_half_words_per_pp_row_header; j++) {
186  header[j] = 0;
187  }
188  space_half_words[i] -= obj->n_half_words_per_pp_row_header;
189  }
190  }
191 
192  // Get the plastic-plastic position at the start of each row and keep track
193  // of the number of half-words per row (to allow padding later)
194  uint16_t *pp_address[max_stage];
195  uint16_t n_half_words_per_row[max_stage];
196  for (uint32_t i = 0; i < max_stage; i++) {
197  n_half_words_per_row[i] = 0;
198  if (row[i] != NULL) {
199  pp_address[i] = &row[i]->plastic_plastic_data[
200  obj->n_half_words_per_pp_row_header];
201  } else {
202  pp_address[i] = NULL;
203  }
204  }
205 
206  // Write the plastic-plastic part of the row
207  for (uint32_t synapse = 0; synapse < n_synapses; synapse++) {
208  // Weight
209  uint16_t weight = weights[synapse];
210  // Delay (mostly to get the stage)
211  struct delay_value delay = get_delay(
212  delays[synapse], max_stage, max_delay_per_stage);
213 
214  // Check that the position is valid
215  if (delay.stage >= max_stage || pp_address[delay.stage] == NULL) {
216  log_error("Delay stage %u has not been initialised", delay.stage);
217  rt_error(RTE_SWERR);
218  }
219 
220  // Check there is enough space
221  if (space_half_words[delay.stage] < obj->n_half_words_per_pp_synapse) {
222  log_warning("Row %u only has %u half words of %u free - not writing",
223  delay.stage, space_half_words[delay.stage],
224  obj->n_half_words_per_pp_synapse);
225  continue;
226  }
227 
228  // Put the weight words in place
229  uint16_t *weight_words = pp_address[delay.stage];
230  pp_address[delay.stage] =
231  &pp_address[delay.stage][obj->n_half_words_per_pp_synapse];
232  for (uint32_t i = 0; i < obj->n_half_words_per_pp_synapse; i++) {
233  weight_words[i] = 0;
234  }
235  weight_words[obj->weight_half_word] = weight;
236  n_half_words_per_row[delay.stage] += obj->n_half_words_per_pp_synapse;
237  space_half_words[delay.stage] -= obj->n_half_words_per_pp_synapse;
238  }
239 
240  // Add padding to any rows that are not word-aligned
241  // and set the size in words
242  for (uint32_t i = 0; i < max_stage; i++) {
243  if (row[i] != NULL) {
244  if (n_half_words_per_row[i] & 0x1) {
245  *pp_address[i]++ = 0;
246  n_half_words_per_row[i]++;
247  }
248  row[i]->plastic_plastic_size += n_half_words_per_row[i] >> 1;
249  }
250  }
251 
252  // PP address is now fixed region address
253  // Set the fixed-fixed size to 0 and point to the fixed-plastic region
254  row_fixed_t fixed[max_stage];
255  uint16_t *fp_address[max_stage];
256  for (uint32_t i = 0; i < max_stage; i++) {
257  fixed[i] = (row_fixed_t) pp_address[i];
258  if (fixed[i] != NULL) {
259  fp_address[i] = fixed[i]->fixed_plastic_data;
260  fixed[i]->fixed_fixed_size = 0;
261  fixed[i]->fixed_plastic_size = 0;
262  } else {
263  fp_address[i] = NULL;
264  }
265  }
266 
267  // Write the fixed-plastic part of the row
268  for (uint32_t synapse = 0; synapse < n_synapses; synapse++) {
269  // Post-neuron index
270  uint32_t post_index = indices[synapse];
271 
272  struct delay_value delay = get_delay(
273  delays[synapse], max_stage, max_delay_per_stage);
274 
275  // Build synaptic word
276  uint16_t fp_half_word = build_fixed_plastic_half_word(
277  delay.delay, synapse_type, post_index, synapse_type_bits,
278  synapse_index_bits);
279 
280  // Write the half-word
281  *fp_address[delay.stage]++ = fp_half_word;
282 
283  // Increment the size of the current row
284  fixed[delay.stage]->fixed_plastic_size++;
285  }
286 }
