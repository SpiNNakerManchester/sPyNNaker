91 static uint16_t build_fixed_plastic_half_word(
92  uint16_t delay, uint32_t type,
93  uint32_t post_index, uint32_t synapse_type_bits,
94  uint32_t synapse_index_bits, uint32_t delay_bits) {
95  uint16_t synapse_index_mask = (1 << synapse_index_bits) - 1;
96  uint16_t synapse_type_mask = (1 << synapse_type_bits) - 1;
97  uint16_t delay_mask = (1 << delay_bits) - 1;
98 
99  uint16_t wrd = post_index & synapse_index_mask;
100  wrd |= (type & synapse_type_mask) << synapse_index_bits;
101  wrd |= (delay & delay_mask) <<
102  (synapse_index_bits + synapse_type_bits);
103  // wrd |= (delay & SYNAPSE_DELAY_MASK) << synapse_type_bits;
104 
105  return wrd;
106 }

130 void matrix_generator_stdp_write_row(
131  void *generator,
132  address_t synaptic_matrix, address_t delayed_synaptic_matrix,
133  uint32_t n_pre_neurons, uint32_t pre_neuron_index,
134  uint32_t max_row_n_words, uint32_t max_delayed_row_n_words,
135  uint32_t synapse_type_bits, uint32_t synapse_index_bits,
136  uint32_t synapse_type, uint32_t n_synapses,
137  uint16_t *indices, uint16_t *delays, uint16_t *weights,
138  uint32_t max_stage, uint32_t max_delay_per_stage) {
139  struct matrix_generator_stdp *obj = generator;
140 
141  // Row address for each possible delay stage (including no delay stage)
142  row_plastic_t row[max_stage];
143 
144  // Space available in each row
145  uint16_t space_half_words[max_stage];
146 
147  // The number of words in a row including headers
148  uint32_t n_row_words = max_row_n_words + 3;
149  uint32_t n_delay_row_words = max_delayed_row_n_words + 3;
150 
151  // The normal row position and space available - might be 0 if all delayed
152  row[0] = NULL;
153  space_half_words[0] = max_row_n_words * 2;
154  if (synaptic_matrix != NULL) {
155  row[0] = (row_plastic_t)
156  &synaptic_matrix[pre_neuron_index * n_row_words];
157  }
158 
159  // The delayed row positions and space available
160  if (delayed_synaptic_matrix != NULL) {
161  address_t delayed_address =
162  &delayed_synaptic_matrix[pre_neuron_index * n_delay_row_words];
163  uint32_t single_matrix_size = n_pre_neurons * n_delay_row_words;
164  for (uint32_t i = 1; i < max_stage; i++) {
165  row[i] = (row_plastic_t)
166  &delayed_address[single_matrix_size * (i - 1)];
167  space_half_words[i] = max_delayed_row_n_words * 2;
168  }
169  } else {
170  for (uint32_t i = 1; i < max_stage; i++) {
171  row[i] = NULL;
172  space_half_words[i] = 0;
173  }
174  }
175 
176  // Add the header half words (zero initialised) to each row
177  for (uint32_t i = 0; i < max_stage; i++) {
178  if (row[i] != NULL) {
179  row[i]->plastic_plastic_size =
180  obj->n_half_words_per_pp_row_header >> 1;
181  uint16_t *header = row[i]->plastic_plastic_data;
182  for (uint32_t j = 0;
183  j < obj->n_half_words_per_pp_row_header; j++) {
184  header[j] = 0;
185  }
186  space_half_words[i] -= obj->n_half_words_per_pp_row_header;
187  }
188  }
189 
190  // Get the plastic-plastic position at the start of each row and keep track
191  // of the number of half-words per row (to allow padding later)
192  uint16_t *pp_address[max_stage];
193  uint16_t n_half_words_per_row[max_stage];
194  for (uint32_t i = 0; i < max_stage; i++) {
195  n_half_words_per_row[i] = 0;
196  if (row[i] != NULL) {
197  pp_address[i] = &row[i]->plastic_plastic_data[
198  obj->n_half_words_per_pp_row_header];
199  } else {
200  pp_address[i] = NULL;
201  }
202  }
203 
204  // Write the plastic-plastic part of the row
205  for (uint32_t synapse = 0; synapse < n_synapses; synapse++) {
206  // Weight
207  uint16_t weight = weights[synapse];
208  // Delay (mostly to get the stage)
209  struct delay_value delay = get_delay(
210  delays[synapse], max_stage, max_delay_per_stage);
211 
212  // Check that the position is valid
213  if (delay.stage >= max_stage || pp_address[delay.stage] == NULL) {
214  log_error("Delay stage %u has not been initialised", delay.stage);
215  rt_error(RTE_SWERR);
216  }
217 
218  // Check there is enough space
219  if (space_half_words[delay.stage] < obj->n_half_words_per_pp_synapse) {
220  log_warning("Row %u only has %u half words of %u free - not writing",
221  delay.stage, space_half_words[delay.stage],
222  obj->n_half_words_per_pp_synapse);
223  continue;
224  }
225 
226  // Put the weight words in place
227  uint16_t *weight_words = pp_address[delay.stage];
228  pp_address[delay.stage] =
229  &pp_address[delay.stage][obj->n_half_words_per_pp_synapse];
230  for (uint32_t i = 0; i < obj->n_half_words_per_pp_synapse; i++) {
231  weight_words[i] = 0;
232  }
233  weight_words[obj->weight_half_word] = weight;
234  n_half_words_per_row[delay.stage] += obj->n_half_words_per_pp_synapse;
235  space_half_words[delay.stage] -= obj->n_half_words_per_pp_synapse;
236  }
237 
238  // Add padding to any rows that are not word-aligned
239  // and set the size in words
240  for (uint32_t i = 0; i < max_stage; i++) {
241  if (row[i] != NULL) {
242  if (n_half_words_per_row[i] & 0x1) {
243  *pp_address[i]++ = 0;
244  n_half_words_per_row[i]++;
245  }
246  row[i]->plastic_plastic_size += n_half_words_per_row[i] >> 1;
247  }
248  }
249 
250  // PP address is now fixed region address
251  // Set the fixed-fixed size to 0 and point to the fixed-plastic region
252  row_fixed_t fixed[max_stage];
253  uint16_t *fp_address[max_stage];
254  for (uint32_t i = 0; i < max_stage; i++) {
255  fixed[i] = (row_fixed_t) pp_address[i];
256  if (fixed[i] != NULL) {
257  fp_address[i] = fixed[i]->fixed_plastic_data;
258  fixed[i]->fixed_fixed_size = 0;
259  fixed[i]->fixed_plastic_size = 0;
260  } else {
261  fp_address[i] = NULL;
262  }
263  }
264 
265  uint32_t max_delay_power_2 = max_delay_per_stage;
266  uint32_t log_max_delay = 1;
267  if (max_delay_power_2 != 1) {
268  if (!is_power_of_2(max_delay_power_2)) {
269  max_delay_power_2 = next_power_of_2(max_delay_power_2);
270  }
271  log_max_delay = ilog_2(max_delay_power_2);
272  }
273 
274  // Write the fixed-plastic part of the row
275  for (uint32_t synapse = 0; synapse < n_synapses; synapse++) {
276  // Post-neuron index
277  uint32_t post_index = indices[synapse];
278 
279  struct delay_value delay = get_delay(
280  delays[synapse], max_stage, max_delay_per_stage);
281 
282  // Build synaptic word
283  uint16_t fp_half_word = build_fixed_plastic_half_word(
284  delay.delay, synapse_type, post_index, synapse_type_bits,
285  synapse_index_bits, log_max_delay);
286 
287  // Write the half-word
288  *fp_address[delay.stage]++ = fp_half_word;
289 
290  // Increment the size of the current row
291  fixed[delay.stage]->fixed_plastic_size++;
292  }
293 }
