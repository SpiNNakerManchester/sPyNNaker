100 static uint16_t build_fixed_plastic_half_word(
101  uint16_t delay, uint32_t type,
102  uint32_t post_index, uint32_t synapse_type_bits,
103  uint32_t synapse_index_bits) {
104  uint16_t synapse_index_mask = (1 << synapse_index_bits) - 1;
105  uint16_t synapse_type_mask = (1 << synapse_type_bits) - 1;
106 
107  uint16_t wrd = post_index & synapse_index_mask;
108  wrd |= (type & synapse_type_mask) << synapse_index_bits;
109  wrd |= (delay & SYNAPSE_DELAY_MASK) <<
110  (synapse_index_bits + synapse_type_bits);
111  // wrd |= (delay & SYNAPSE_DELAY_MASK) << synapse_type_bits;
112 
113  return wrd;
114 }

116 void matrix_generator_stdp_write_row(
117  void *data,
118  address_t synaptic_matrix, address_t delayed_synaptic_matrix,
119  uint32_t n_pre_neurons, uint32_t pre_neuron_index,
120  uint32_t max_row_n_words, uint32_t max_delayed_row_n_words,
121  uint32_t synapse_type_bits, uint32_t synapse_index_bits,
122  uint32_t synapse_type, uint32_t n_synapses,
123  uint16_t *indices, uint16_t *delays, uint16_t *weights,
124  uint32_t max_stage) {
125  struct matrix_generator_stdp *obj = data;
126 
127  // Row address for each possible delay stage (including no delay stage)
128  address_t row_address[max_stage];
129 
130  // Space available in each row
131  uint16_t space_half_words[max_stage];
132 
133  // The number of words in a row including headers
134  uint32_t n_row_words = max_row_n_words + 3;
135  uint32_t n_delay_row_words = max_delayed_row_n_words + 3;
136 
137  // The normal row position and space available - might be 0 if all delayed
138  row_address[0] = NULL;
139  space_half_words[0] = max_row_n_words * 2;
140  if (synaptic_matrix != NULL) {
141  row_address[0] = &synaptic_matrix[pre_neuron_index * n_row_words];
142  }
143 
144  // The delayed row positions and space available
145  if (delayed_synaptic_matrix != NULL) {
146  address_t delayed_address =
147  &delayed_synaptic_matrix[pre_neuron_index * n_delay_row_words];
148  uint32_t single_matrix_size = n_pre_neurons * n_delay_row_words;
149  for (uint32_t i = 1; i < max_stage; i++) {
150  row_address[i] = &delayed_address[single_matrix_size * (i - 1)];
151  space_half_words[i] = max_delayed_row_n_words * 2;
152  }
153  } else {
154  for (uint32_t i = 1; i < max_stage; i++) {
155  row_address[i] = NULL;
156  space_half_words[i] = 0;
157  }
158  }
159 
160  // Add the header half words (zero initialised) to each row
161  for (uint32_t i = 0; i < max_stage; i++) {
162  if (row_address[i] != NULL) {
163  row_address[i][STDP_PLASTIC_PLASTIC_SIZE] =
164  obj->n_half_words_per_pp_row_header >> 1;
165  uint16_t *header = (uint16_t *)
166  &row_address[i][STDP_PLASTIC_PLASTIC_OFFSET];
167  for (uint32_t j = 0;
168  j < obj->n_half_words_per_pp_row_header; j++) {
169  header[j] = 0;
170  }
171  space_half_words[i] -= obj->n_half_words_per_pp_row_header;
172  }
173  }
174 
175  // Get the plastic-plastic position at the start of each row and keep track
176  // of the number of half-words per row (to allow padding later)
177  uint16_t *pp_address[max_stage];
178  uint16_t n_half_words_per_row[max_stage];
179  for (uint32_t i = 0; i < max_stage; i++) {
180  n_half_words_per_row[i] = 0;
181  if (row_address[i] != NULL) {
182  pp_address[i] = (uint16_t *) &row_address[i][
183  STDP_PLASTIC_PLASTIC_OFFSET +
184  (obj->n_half_words_per_pp_row_header >> 1)];
185  } else {
186  pp_address[i] = NULL;
187  }
188  }
189 
190  // Write the plastic-plastic part of the row
191  for (uint32_t synapse = 0; synapse < n_synapses; synapse++) {
192  // Weight
193  uint16_t weight = weights[synapse];
194  // Delay (mostly to get the stage)
195  struct delay_value delay = get_delay(delays[synapse], max_stage);
196 
197  // Check that the position is valid
198  if (pp_address[delay.stage] == NULL) {
199  log_error("Delay stage %u has not been initialised", delay.stage);
200  rt_error(RTE_SWERR);
201  }
202 
203  // Check there is enough space
204  if (space_half_words[delay.stage] < obj->n_half_words_per_pp_synapse) {
205  log_warning("Row %u only has %u half words of %u free - not writing",
206  delay.stage, space_half_words[delay.stage],
207  obj->n_half_words_per_pp_synapse);
208  continue;
209  }
210 
211  // Put the weight words in place
212  uint16_t *weight_words = pp_address[delay.stage];
213  pp_address[delay.stage] =
214  &pp_address[delay.stage][obj->n_half_words_per_pp_synapse];
215  for (uint32_t i = 0; i < obj->n_half_words_per_pp_synapse; i++) {
216  weight_words[i] = 0;
217  }
218  weight_words[obj->weight_half_word] = weight;
219  n_half_words_per_row[delay.stage] +=
220  obj->n_half_words_per_pp_synapse;
221  space_half_words[delay.stage] -= obj->n_half_words_per_pp_synapse;
222  }
223 
224  // Add padding to any rows that are not word-aligned
225  // and set the size in words
226  for (uint32_t i = 0; i < max_stage; i++) {
227  if (row_address[i] != NULL) {
228  if (n_half_words_per_row[i] & 0x1) {
229  pp_address[i][0] = 0;
230  pp_address[i] = &pp_address[i][1];
231  n_half_words_per_row[i]++;
232  }
233  row_address[i][STDP_PLASTIC_PLASTIC_SIZE] +=
234  n_half_words_per_row[i] >> 1;
235  }
236  }
237 
238  // PP address is now fixed region address
239  // Set the fixed-fixed size to 0 and point to the fixed-plastic region
240  uint32_t *fixed_address[max_stage];
241  uint16_t *fp_address[max_stage];
242  for (uint32_t i = 0; i < max_stage; i++) {
243  fixed_address[i] = (uint32_t *) pp_address[i];
244  if (pp_address[i] != NULL) {
245  fp_address[i] = (uint16_t *)
246  &fixed_address[i][STDP_FIXED_PLASTIC_OFFSET];
247  fixed_address[i][STDP_FIXED_FIXED_SIZE] = 0;
248  fixed_address[i][STDP_FIXED_PLASTIC_SIZE] = 0;
249  } else {
250  fp_address[i] = NULL;
251  }
252  }
253 
254  // Write the fixed-plastic part of the row
255  for (uint32_t synapse = 0; synapse < n_synapses; synapse++) {
256  // Post-neuron index
257  uint32_t post_index = indices[synapse];
258 
259  struct delay_value delay = get_delay(delays[synapse], max_stage);
260 
261  // Build synaptic word
262  uint16_t fp_half_word = build_fixed_plastic_half_word(
263  delay.delay, synapse_type, post_index, synapse_type_bits,
264  synapse_index_bits);
265 
266  // Write the half-word
267  fp_address[delay.stage][0] = fp_half_word;
268  fp_address[delay.stage] = &fp_address[delay.stage][1];
269 
270  // Increment the size of the current row
271  fixed_address[delay.stage][STDP_FIXED_PLASTIC_SIZE]++;
272  }
273 }
