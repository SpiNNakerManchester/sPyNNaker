106 static uint16_t build_fixed_plastic_half_word(
107  uint16_t delay, uint32_t type,
108  uint32_t post_index, uint32_t synapse_type_bits,
109  uint32_t synapse_index_bits) {
110  uint16_t synapse_index_mask = (1 << synapse_index_bits) - 1;
111  uint16_t synapse_type_mask = (1 << synapse_type_bits) - 1;
112 
113  uint16_t wrd = post_index & synapse_index_mask;
114  wrd |= (type & synapse_type_mask) << synapse_index_bits;
115  wrd |= (delay & SYNAPSE_DELAY_MASK) <<
116  (synapse_index_bits + synapse_type_bits);
117  // wrd |= (delay & SYNAPSE_DELAY_MASK) << synapse_type_bits;
118 
119  return wrd;
120 }

143 void matrix_generator_stdp_write_row(
144  void *generator,
145  address_t synaptic_matrix, address_t delayed_synaptic_matrix,
146  uint32_t n_pre_neurons, uint32_t pre_neuron_index,
147  uint32_t max_row_n_words, uint32_t max_delayed_row_n_words,
148  uint32_t synapse_type_bits, uint32_t synapse_index_bits,
149  uint32_t synapse_type, uint32_t n_synapses,
150  uint16_t *indices, uint16_t *delays, uint16_t *weights,
151  uint32_t max_stage) {
152  struct matrix_generator_stdp *obj = generator;
153 
154  // Row address for each possible delay stage (including no delay stage)
155  address_t row_address[max_stage];
156 
157  // Space available in each row
158  uint16_t space_half_words[max_stage];
159 
160  // The number of words in a row including headers
161  uint32_t n_row_words = max_row_n_words + 3;
162  uint32_t n_delay_row_words = max_delayed_row_n_words + 3;
163 
164  // The normal row position and space available - might be 0 if all delayed
165  row_address[0] = NULL;
166  space_half_words[0] = max_row_n_words * 2;
167  if (synaptic_matrix != NULL) {
168  row_address[0] = &synaptic_matrix[pre_neuron_index * n_row_words];
169  }
170 
171  // The delayed row positions and space available
172  if (delayed_synaptic_matrix != NULL) {
173  address_t delayed_address =
174  &delayed_synaptic_matrix[pre_neuron_index * n_delay_row_words];
175  uint32_t single_matrix_size = n_pre_neurons * n_delay_row_words;
176  for (uint32_t i = 1; i < max_stage; i++) {
177  row_address[i] = &delayed_address[single_matrix_size * (i - 1)];
178  space_half_words[i] = max_delayed_row_n_words * 2;
179  }
180  } else {
181  for (uint32_t i = 1; i < max_stage; i++) {
182  row_address[i] = NULL;
183  space_half_words[i] = 0;
184  }
185  }
186 
187  // Add the header half words (zero initialised) to each row
188  for (uint32_t i = 0; i < max_stage; i++) {
189  if (row_address[i] != NULL) {
190  row_address[i][STDP_PLASTIC_PLASTIC_SIZE] =
191  obj->n_half_words_per_pp_row_header >> 1;
192  uint16_t *header = (uint16_t *)
193  &row_address[i][STDP_PLASTIC_PLASTIC_OFFSET];
194  for (uint32_t j = 0;
195  j < obj->n_half_words_per_pp_row_header; j++) {
196  header[j] = 0;
197  }
198  space_half_words[i] -= obj->n_half_words_per_pp_row_header;
199  }
200  }
201 
202  // Get the plastic-plastic position at the start of each row and keep track
203  // of the number of half-words per row (to allow padding later)
204  uint16_t *pp_address[max_stage];
205  uint16_t n_half_words_per_row[max_stage];
206  for (uint32_t i = 0; i < max_stage; i++) {
207  n_half_words_per_row[i] = 0;
208  if (row_address[i] != NULL) {
209  pp_address[i] = (uint16_t *) &row_address[i][
210  STDP_PLASTIC_PLASTIC_OFFSET +
211  (obj->n_half_words_per_pp_row_header >> 1)];
212  } else {
213  pp_address[i] = NULL;
214  }
215  }
216 
217  // Write the plastic-plastic part of the row
218  for (uint32_t synapse = 0; synapse < n_synapses; synapse++) {
219  // Weight
220  uint16_t weight = weights[synapse];
221  // Delay (mostly to get the stage)
222  struct delay_value delay = get_delay(delays[synapse], max_stage);
223 
224  // Check that the position is valid
225  if (pp_address[delay.stage] == NULL) {
226  log_error("Delay stage %u has not been initialised", delay.stage);
227  rt_error(RTE_SWERR);
228  }
229 
230  // Check there is enough space
231  if (space_half_words[delay.stage] < obj->n_half_words_per_pp_synapse) {
232  log_warning("Row %u only has %u half words of %u free - not writing",
233  delay.stage, space_half_words[delay.stage],
234  obj->n_half_words_per_pp_synapse);
235  continue;
236  }
237 
238  // Put the weight words in place
239  uint16_t *weight_words = pp_address[delay.stage];
240  pp_address[delay.stage] =
241  &pp_address[delay.stage][obj->n_half_words_per_pp_synapse];
242  for (uint32_t i = 0; i < obj->n_half_words_per_pp_synapse; i++) {
243  weight_words[i] = 0;
244  }
245  weight_words[obj->weight_half_word] = weight;
246  n_half_words_per_row[delay.stage] +=
247  obj->n_half_words_per_pp_synapse;
248  space_half_words[delay.stage] -= obj->n_half_words_per_pp_synapse;
249  }
250 
251  // Add padding to any rows that are not word-aligned
252  // and set the size in words
253  for (uint32_t i = 0; i < max_stage; i++) {
254  if (row_address[i] != NULL) {
255  if (n_half_words_per_row[i] & 0x1) {
256  pp_address[i][0] = 0;
257  pp_address[i] = &pp_address[i][1];
258  n_half_words_per_row[i]++;
259  }
260  row_address[i][STDP_PLASTIC_PLASTIC_SIZE] +=
261  n_half_words_per_row[i] >> 1;
262  }
263  }
264 
265  // PP address is now fixed region address
266  // Set the fixed-fixed size to 0 and point to the fixed-plastic region
267  uint32_t *fixed_address[max_stage];
268  uint16_t *fp_address[max_stage];
269  for (uint32_t i = 0; i < max_stage; i++) {
270  fixed_address[i] = (uint32_t *) pp_address[i];
271  if (pp_address[i] != NULL) {
272  fp_address[i] = (uint16_t *)
273  &fixed_address[i][STDP_FIXED_PLASTIC_OFFSET];
274  fixed_address[i][STDP_FIXED_FIXED_SIZE] = 0;
275  fixed_address[i][STDP_FIXED_PLASTIC_SIZE] = 0;
276  } else {
277  fp_address[i] = NULL;
278  }
279  }
280 
281  // Write the fixed-plastic part of the row
282  for (uint32_t synapse = 0; synapse < n_synapses; synapse++) {
283  // Post-neuron index
284  uint32_t post_index = indices[synapse];
285 
286  struct delay_value delay = get_delay(delays[synapse], max_stage);
287 
288  // Build synaptic word
289  uint16_t fp_half_word = build_fixed_plastic_half_word(
290  delay.delay, synapse_type, post_index, synapse_type_bits,
291  synapse_index_bits);
292 
293  // Write the half-word
294  fp_address[delay.stage][0] = fp_half_word;
295  fp_address[delay.stage] = &fp_address[delay.stage][1];
296 
297  // Increment the size of the current row
298  fixed_address[delay.stage][STDP_FIXED_PLASTIC_SIZE]++;
299  }
300 }
