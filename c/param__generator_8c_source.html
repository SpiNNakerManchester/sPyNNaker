107param_generator_t param_generator_init(uint32_t hash, void **in_region) {
108 // Look through the known generators
109 for (uint32_t i = 0; i < N_PARAM_GENERATORS; i++) {
110 const param_generator_info *type = &param_generators[i];
111
112 // If the hash requested matches the hash of the generator, use it
113 if (hash == type->hash) {
114 // Prepare a space for the data
115 struct param_generator *generator =
116 spin1_malloc(sizeof(struct param_generator));
117 if (generator == NULL) {
118 log_error("Could not create generator");
119 return NULL;
120 }
121
122 // Store the index
123 generator->type = type;
124
125 // Initialise the generator and store the data
126 generator->data = type->initialize(in_region);
127 return generator;
128 }
129 }
130 log_error("Param generator with hash %u not found", hash);
131 return NULL;
132}
