115 param_generator_t param_generator_init(uint32_t hash, address_t *in_region) {
116  // Look through the known generators
117  for (uint32_t i = 0; i < N_PARAM_GENERATORS; i++) {
118  const param_generator_info *type = &param_generators[i];
119 
120  // If the hash requested matches the hash of the generator, use it
121  if (hash == type->hash) {
122  // Prepare a space for the data
123  struct param_generator *generator =
124  spin1_malloc(sizeof(struct param_generator));
125  if (generator == NULL) {
126  log_error("Could not create generator");
127  return NULL;
128  }
129 
130  // Store the index
131  generator->type = type;
132 
133  // Initialise the generator and store the data
134  generator->data = type->initialize(in_region);
135  return generator;
136  }
137  }
138  log_error("Param generator with hash %u not found", hash);
139  return NULL;
140 }
