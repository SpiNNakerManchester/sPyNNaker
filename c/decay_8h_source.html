99 // The following permits us to do a type-generic macro for decay manipulation
100 /*----------------------------------
101  * This method is currently assumed to be faulty. Please do not use it yet.
102  * Plan is to fix method and not need to use the private methods directly.
103  * issue seems to be in __builtin_types_compatible_p always returning False
104  * or at least, that there's the impression that on some types it results in
105  * the abort statement and therefore kills scripts dead on SpiNNaker.
106  * ---------------------------------
107  */
108 #define decay(x, d) ({ \
109  __typeof__(x) tmp = (x); \
110  if (__builtin_types_compatible_p(__typeof__(x), s1615)) {\
111  tmp = decay_s1615(x, d); \
112  } else if (__builtin_types_compatible_p(__typeof__(x), u1616)) {\
113  tmp = decay_u1616(x, d); \
114  } else if (__builtin_types_compatible_p(__typeof__(x), s015)) {\
115  tmp = decay_s015(x, d); \
116  } else if (__builtin_types_compatible_p(__typeof__(x), u016)) {\
117  tmp = decay_u016(x, d); \
118  } else {\
119  abort(1); \
120  }\
121  tmp; \
122 })
