45 static struct delay_value get_delay(
46  uint16_t delay_value, uint32_t max_stage,
47  uint32_t max_delay_per_stage) {
48  uint16_t delay = delay_value;
49 
50  // Ensure delay is at least 1
51  if (delay < 1) {
52  log_debug("Delay of %u is too small", delay);
53  delay = 1;
54  }
55 
56  // Ensure that the delay is less than the maximum
57  uint16_t stage = (delay - 1) / max_delay_per_stage;
58  if (stage >= max_stage) {
59  log_debug("Delay of %u is too big", delay);
60  stage = max_stage - 1;
61  delay = (stage * max_delay_per_stage);
62  }
63 
64  // Get the remainder of the delay
65  delay = ((delay - 1) % max_delay_per_stage) + 1;
66  return (struct delay_value) {.delay = delay, .stage = stage};
67 }
