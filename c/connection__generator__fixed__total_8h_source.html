100 static uint32_t binomial(uint32_t n, uint32_t N, uint32_t K, rng_t rng) {
101  uint32_t count = 0;
102  uint32_t not_K = N - K;
103  for (uint32_t i = 0; i < n; i++) {
104  if (_pick(rng, K, not_K) < K) {
105  count++;
106  }
107  }
108  return count;
109 }

119 static uint32_t hypergeom(uint32_t n, uint32_t N, uint32_t K, rng_t rng) {
120  uint32_t count = 0;
121  uint32_t K_remaining = K;
122  uint32_t not_K_remaining = N - K;
123  for (uint32_t i = 0; i < n; i++) {
124  if (_pick(rng, K_remaining, not_K_remaining) < K_remaining) {
125  count++;
126  K_remaining--;
127  } else {
128  not_K_remaining--;
129  }
130  }
131  return count;
132 }

153 static uint32_t connection_generator_fixed_total_generate(
154  void *generator, UNUSED uint32_t pre_slice_start,
155  UNUSED uint32_t pre_slice_count,
156  uint32_t pre_neuron_index, uint32_t post_slice_start,
157  uint32_t post_slice_count, uint32_t max_row_length, uint16_t *indices) {
158  // If there are no connections left or none to be made, return 0
159  struct fixed_total *obj = generator;
160  if (max_row_length == 0 || obj->params.n_connections == 0) {
161  return 0;
162  }
163 
164  // If not in the pre-population view range, then don't generate
165  if ((pre_neuron_index < obj->params.pre_lo) ||
166  (pre_neuron_index > obj->params.pre_hi)) {
167  return 0;
168  }
169 
170  // Work out how many values can be sampled from (on this slice)
171  uint32_t slice_lo = post_slice_start;
172  uint32_t slice_hi = post_slice_start + post_slice_count - 1;
173 
174  // If everything is off the current slice then don't generate
175  if ((obj->params.post_hi < slice_lo) ||
176  (obj->params.post_lo > slice_hi)) {
177  return 0;
178  }
179 
180  // Otherwise work out how many values can be sampled from
181  if ((obj->params.post_lo >= post_slice_start) &&
182  (obj->params.post_lo < post_slice_start + post_slice_count)) {
183  slice_lo = obj->params.post_lo;
184  if (obj->params.post_hi >= post_slice_start + post_slice_count) {
185  slice_hi = post_slice_start + post_slice_count - 1;
186  } else {
187  slice_hi = obj->params.post_hi;
188  }
189  } else { // post_lo is less than slice_start
190  slice_lo = post_slice_start;
191  if (obj->params.post_hi >= post_slice_start + post_slice_count) {
192  slice_hi = post_slice_start + post_slice_count - 1;
193  }
194  else {
195  slice_hi = obj->params.post_hi;
196  }
197  }
198 
199  uint32_t n_values = slice_hi - slice_lo + 1;
200  if (!obj->params.allow_self_connections
201  && pre_neuron_index >= obj->params.post_lo
202  && pre_neuron_index <= obj->params.post_hi) {
203  n_values--;
204  }
205  uint32_t n_conns = 0;
206 
207  // If we're on the last row of the sub-matrix, then all of the remaining
208  // sub-matrix connections get allocated to this row
209  if (pre_neuron_index == obj->params.pre_hi) {
210  n_conns = obj->params.n_connections;
211  } else {
212  // If with replacement, generate a binomial for this row
213  if (obj->params.with_replacement) {
214  n_conns = binomial(
215  obj->params.n_connections,
216  obj->params.n_potential_synapses, n_values, obj->rng);
217  // If without replacement, generate a hyper-geometric for this row
218  } else {
219  n_conns = hypergeom(
220  obj->params.n_connections,
221  obj->params.n_potential_synapses, n_values, obj->rng);
222  }
223  }
224 
225  // If too many connections, limit
226  if (n_conns > max_row_length) {
227  if (pre_neuron_index == obj->params.pre_hi) {
228  log_warning("Could not create %u connections",
229  n_conns - max_row_length);
230  }
231  n_conns = max_row_length;
232  }
233  log_debug("Generating %u of %u synapses",
234  n_conns, obj->params.n_connections);
235 
236  // Sample from the possible connections in this row n_conns times
237  if (obj->params.with_replacement) {
238  // Sample them with replacement
239  for (unsigned int i = 0; i < n_conns; i++) {
240  uint32_t u01 = rng_generator(obj->rng) & 0x00007fff;
241  uint32_t j = (u01 * n_values) >> 15;
242  indices[i] = j + slice_lo - post_slice_start;
243  }
244  } else {
245  // Sample them without replacement using reservoir sampling
246  for (unsigned int i = 0; i < n_conns; i++) {
247  indices[i] = i + slice_lo - post_slice_start;
248  }
249  for (unsigned int i = n_conns; i < n_values; i++) {
250  // j = random(0, i) (inclusive)
251  const unsigned int u01 = rng_generator(obj->rng) & 0x00007fff;
252  const unsigned int j = (u01 * (i + 1)) >> 15;
253  if (j < n_conns) {
254  indices[j] = i + slice_lo - post_slice_start;
255  }
256  }
257  }
258 
259  obj->params.n_connections -= n_conns;
260  obj->params.n_potential_synapses -= n_values;
261 
262  return n_conns;
263 }
