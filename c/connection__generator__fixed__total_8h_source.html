89 static uint32_t binomial(uint32_t n, uint32_t N, uint32_t K, rng_t rng) {
90  uint32_t count = 0;
91  uint32_t not_K = N - K;
92  for (uint32_t i = 0; i < n; i++) {
93  unsigned long fract value = ulrbits(rng_generator(rng));
94  uint32_t pos = (uint32_t) (value * (K + not_K));
95  if (pos < K) {
96  count++;
97  }
98  }
99  return count;
100 }

110 static uint32_t hypergeom(uint32_t n, uint32_t N, uint32_t K, rng_t rng) {
111  uint32_t count = 0;
112  uint32_t K_remaining = K;
113  uint32_t not_K_remaining = N - K;
114  for (uint32_t i = 0; i < n; i++) {
115  unsigned long fract value = ulrbits(rng_generator(rng));
116  uint32_t pos = (uint32_t) (value * (K_remaining + not_K_remaining));
117  if (pos < K_remaining) {
118  count++;
119  K_remaining--;
120  } else {
121  not_K_remaining--;
122  }
123  }
124  return count;
125 }

127 static uint32_t connection_generator_fixed_total_generate(
128  void *data, uint32_t pre_slice_start, uint32_t pre_slice_count,
129  uint32_t pre_neuron_index, uint32_t post_slice_start,
130  uint32_t post_slice_count, uint32_t max_row_length, uint16_t *indices) {
131  use(pre_slice_start);
132  use(pre_slice_count);
133 
134  // If there are no connections left or none to be made, return 0
135  struct fixed_total *obj = data;
136  if (max_row_length == 0 || obj->params.n_connections == 0) {
137  return 0;
138  }
139 
140  // If not in the pre-population view range, then don't generate
141  if ((pre_neuron_index < obj->params.pre_lo) ||
142  (pre_neuron_index > obj->params.pre_hi)) {
143  return 0;
144  }
145 
146  // Work out how many values can be sampled from (on this slice)
147  uint32_t slice_lo = post_slice_start;
148  uint32_t slice_hi = post_slice_start + post_slice_count - 1;
149 
150  // If everything is off the current slice then don't generate
151  if ((obj->params.post_hi < slice_lo) ||
152  (obj->params.post_lo > slice_hi)) {
153  return 0;
154  }
155 
156  // Otherwise work out how many values can be sampled from
157  if ((obj->params.post_lo >= post_slice_start) &&
158  (obj->params.post_lo < post_slice_start + post_slice_count)) {
159  slice_lo = obj->params.post_lo;
160  if (obj->params.post_hi >= post_slice_start + post_slice_count) {
161  slice_hi = post_slice_start + post_slice_count - 1;
162  } else {
163  slice_hi = obj->params.post_hi;
164  }
165  } else { // post_lo is less than slice_start
166  slice_lo = post_slice_start;
167  if (obj->params.post_hi >= post_slice_start + post_slice_count) {
168  slice_hi = post_slice_start + post_slice_count - 1;
169  }
170  else {
171  slice_hi = obj->params.post_hi;
172  }
173  }
174 
175  uint32_t n_values = slice_hi - slice_lo + 1;
176  if (!obj->params.allow_self_connections
177  && pre_neuron_index >= obj->params.post_lo
178  && pre_neuron_index <= obj->params.post_hi) {
179  n_values--;
180  }
181  uint32_t n_conns = 0;
182 
183  // If we're on the last row of the sub-matrix, then all of the remaining
184  // sub-matrix connections get allocated to this row
185  if (pre_neuron_index == obj->params.pre_hi) {
186  n_conns = obj->params.n_connections;
187  } else {
188  // If with replacement, generate a binomial for this row
189  if (obj->params.with_replacement) {
190  n_conns = binomial(
191  obj->params.n_connections,
192  obj->params.n_potential_synapses, n_values, obj->rng);
193  // If without replacement, generate a hyper-geometric for this row
194  } else {
195  n_conns = hypergeom(
196  obj->params.n_connections,
197  obj->params.n_potential_synapses, n_values, obj->rng);
198  }
199  }
200 
201  // If too many connections, limit
202  if (n_conns > max_row_length) {
203  if (pre_neuron_index == obj->params.pre_hi) {
204  log_warning("Could not create %u connections",
205  n_conns - max_row_length);
206  }
207  n_conns = max_row_length;
208  }
209  log_debug("Generating %u of %u synapses",
210  n_conns, obj->params.n_connections);
211 
212  // Sample from the possible connections in this row n_conns times
213  if (obj->params.with_replacement) {
214  // Sample them with replacement
215  for (unsigned int i = 0; i < n_conns; i++) {
216  uint32_t u01 = rng_generator(obj->rng) & 0x00007fff;
217  uint32_t j = (u01 * n_values) >> 15;
218  indices[i] = j + slice_lo - post_slice_start;
219  }
220  } else {
221  // Sample them without replacement using reservoir sampling
222  for (unsigned int i = 0; i < n_conns; i++) {
223  indices[i] = i + slice_lo - post_slice_start;
224  }
225  for (unsigned int i = n_conns; i < n_values; i++) {
226  // j = random(0, i) (inclusive)
227  const unsigned int u01 = rng_generator(obj->rng) & 0x00007fff;
228  const unsigned int j = (u01 * (i + 1)) >> 15;
229  if (j < n_conns) {
230  indices[j] = i + slice_lo - post_slice_start;
231  }
232  }
233  }
234 
235  obj->params.n_connections -= n_conns;
236  obj->params.n_potential_synapses -= n_values;
237 
238  return n_conns;
239 }
