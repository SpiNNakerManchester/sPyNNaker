178 static bool connection_generator_fixed_total_generate(
179  void *generator, uint32_t pre_lo, uint32_t pre_hi,
180  uint32_t post_lo, uint32_t post_hi, UNUSED uint32_t post_index,
181  uint32_t post_slice_start, uint32_t post_slice_count,
182  unsigned long accum weight_scale, accum timestep_per_delay,
183  param_generator_t weight_generator, param_generator_t delay_generator,
184  matrix_generator_t matrix_generator) {
185  struct fixed_total *obj = generator;
186 
187  uint32_t n_pre = pre_hi - pre_lo + 1;
188  uint32_t n_post = post_hi - post_lo + 1;
189  uint32_t post_slice_end = post_slice_start + post_slice_count;
190  uint32_t n_conns = obj->params.n_synapses_total;
191 
192  // Generate the connections for all cores then filter for this one
193  if (obj->params.with_replacement) {
194  for (uint32_t i = 0; i < n_conns; i++) {
195  uint32_t post = random_in_range(population_rng, n_post) + post_lo;
196  if (post >= post_slice_start && post < post_slice_end) {
197  uint32_t local_post = post - post_slice_start;
198  accum weight = param_generator_generate(weight_generator);
199  uint16_t delay = rescale_delay(
200  param_generator_generate(delay_generator), timestep_per_delay);
201 
202  uint32_t pre;
203  bool written = false;
204  uint32_t n_retries = 0;
205  do {
206  pre = random_in_range(core_rng, n_pre) + pre_lo;
207  if (obj->params.allow_self_connections || pre != post) {
208  written = matrix_generator_write_synapse(matrix_generator,
209  pre, local_post, weight, delay, weight_scale);
210  n_retries++;
211  }
212  } while (!written && n_retries < 10);
213  if (!written) {
214  log_error("Couldn't find a row to write to!");
215  return false;
216  }
217  }
218  }
219  } else {
220  uint32_t pre = pre_lo;
221  uint32_t post = post_lo;
222  uint32_t i = 0;
223  struct {
224  uint32_t pre;
225  uint32_t post;
226  } conns[n_conns];
227  for (i = 0; i < n_conns; i++) {
228  conns[i].pre = pre;
229  conns[i].post = post;
230  fixed_total_next(&pre, &post, pre_lo, pre_hi);
231  if (!obj->params.allow_self_connections && (pre == post)) {
232  fixed_total_next(&pre, &post, pre_lo, pre_hi);
233  }
234  }
235  while (pre <= pre_hi && post <= post_hi) {
236  uint32_t r = random_in_range(population_rng, i + 1);
237  if (r < n_conns) {
238  conns[r].pre = pre;
239  conns[r].post = post;
240  }
241  fixed_total_next(&pre, &post, pre_lo, pre_hi);
242  if (!obj->params.allow_self_connections && (pre == post)) {
243  fixed_total_next(&pre, &post, pre_lo, pre_hi);
244  }
245  i++;
246  }
247  for (uint32_t i = 0; i < n_conns; i++) {
248  uint32_t local_post = conns[i].post;
249  if (local_post >= post_slice_start && local_post < post_slice_end) {
250  local_post -= post_slice_start;
251  uint32_t local_pre = conns[i].pre;
252  accum weight = param_generator_generate(weight_generator);
253  uint16_t delay = rescale_delay(
254  param_generator_generate(delay_generator), timestep_per_delay);
255  if (!matrix_generator_write_synapse(matrix_generator, local_pre, local_post,
256  weight, delay, weight_scale)) {
257  // Not a lot we can do here...
258  log_error("Couldn't write matrix!");
259  return false;
260  }
261  }
262  }
263  }
264  return true;
265 }
