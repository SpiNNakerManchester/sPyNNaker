50 void pre_in_post_world(uint16_t in_row, uint16_t in_col, uint16_t start_row,

52  int16_t *out_row, int16_t *out_col) {
53  int16_t d = (int16_t) (in_row - start_row - 1);
54  uint16_t r;
55  if (d == 0) {
56  *out_row = 1;
57  } else if (d < 0) {
