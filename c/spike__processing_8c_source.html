122 static inline void do_dma_read(
123  synaptic_row_t row, size_t n_bytes_to_transfer, spike_t spike) {
124  // Write the SDRAM address of the plastic region and the
125  // Key of the originating spike to the beginning of DMA buffer
126  dma_buffer *next_buffer = &dma_buffers[next_buffer_to_fill];
127  next_buffer->sdram_writeback_address = row;
128  next_buffer->originating_spike = spike;
129  next_buffer->n_bytes_transferred = n_bytes_to_transfer;
130 
131  // Start a DMA transfer to fetch this synaptic row into current
132  // buffer
133  buffer_being_read = next_buffer_to_fill;
134  while (!spin1_dma_transfer(
135  DMA_TAG_READ_SYNAPTIC_ROW, row, next_buffer->row, DMA_READ,
136  n_bytes_to_transfer)) {
137  // Do Nothing
138  }
139  next_buffer_to_fill = (next_buffer_to_fill + 1) % N_DMA_BUFFERS;
140 }

150 static inline bool is_something_to_do(
151  synaptic_row_t *row, size_t *n_bytes_to_transfer,
152  spike_t *spike, uint32_t *n_rewire, uint32_t *n_process_spike) {
153  // Disable interrupts here as dma_busy modification is a critical section
154  uint cpsr = spin1_int_disable();
155 
156  // Check for synaptic rewiring
157  while (rewires_to_do) {
158  rewires_to_do--;
159  spin1_mode_restore(cpsr);
160  if (synaptogenesis_dynamics_rewire(time, spike, row,
161  n_bytes_to_transfer)) {
162  *n_rewire += 1;
163  return true;
164  }
165  cpsr = spin1_int_disable();
166  }
167 
168  // Is there another address in the population table?
169  spin1_mode_restore(cpsr);
170  if (population_table_get_next_address(spike, row, n_bytes_to_transfer)) {
171  *n_process_spike += 1;
172  return true;
173  }
174  cpsr = spin1_int_disable();
175 
176  // track for provenance
177  uint32_t input_buffer_filled_size = in_spikes_size();
178  if (biggest_fill_size_of_input_buffer < input_buffer_filled_size) {
179  biggest_fill_size_of_input_buffer = input_buffer_filled_size;
180  }
181 
182  // Are there any more spikes to process?
183  while (in_spikes_get_next_spike(spike)) {
184  // Enable interrupts while looking up in the master pop table,
185  // as this can be slow
186  spin1_mode_restore(cpsr);
187  if (population_table_get_first_address(
188  *spike, row, n_bytes_to_transfer)) {
189  synaptogenesis_spike_received(time, *spike);
190  *n_process_spike += 1;
191  return true;
192  }
193 
194  // Disable interrupts before checking if there is another spike
195  cpsr = spin1_int_disable();
196  }
197 
198  // If nothing to do, the DMA is not busy
199  dma_busy = false;
200 
201  // Restore interrupts
202  spin1_mode_restore(cpsr);
203  return false;
204 }

219 static bool setup_synaptic_dma_read(dma_buffer *current_buffer,
220  uint32_t *n_rewires, uint32_t *n_synapse_processes) {
221  // Set up to store the DMA location and size to read
222  synaptic_row_t row;
223  size_t n_bytes_to_transfer;
224  spike_t spike;
225  dma_n_spikes = 0;
226  dma_n_rewires = 0;
227 
228  // Keep looking if there is something to do until a DMA can be done
229  bool setup_done = false;
230  while (!setup_done && is_something_to_do(&row, &n_bytes_to_transfer,
231  &spike, &dma_n_rewires, &dma_n_spikes)) {
232  if (current_buffer != NULL &&
233  current_buffer->sdram_writeback_address == row) {
234  // If we can reuse the row, add on what we can use it for
235  // Note that only one of these will have a value of 1 with the
236  // other being set to 0, but we add both as it is simple
237  *n_rewires += dma_n_rewires;
238  *n_synapse_processes += dma_n_spikes;
239  dma_n_rewires = 0;
240  dma_n_spikes = 0;
241  } else if (n_bytes_to_transfer == 0) {
242  // If the row is in DTCM, process the row now
243  synaptic_row_t single_fixed_synapse =
244  direct_synapses_get_direct_synapse(row);
245  bool write_back;
246  synapses_process_synaptic_row(
247  time, single_fixed_synapse, &write_back);
248  dma_n_rewires = 0;
249  dma_n_spikes = 0;
250  } else {
251  // If the row is in SDRAM, set up the transfer and we are done
252  do_dma_read(row, n_bytes_to_transfer, spike);
253  setup_done = true;
254  }
255 
256  // needs to be here to ensure that its only recording actual spike
257  // processing and not the surplus DMA requests.
258  spike_processing_count++;
259  }
260  return setup_done;
261 }

267 static inline void setup_synaptic_dma_write(
268  uint32_t dma_buffer_index, bool plastic_only) {
269  // Get pointer to current buffer
270  dma_buffer *buffer = &dma_buffers[dma_buffer_index];
271 
272  // Get the number of plastic bytes and the write back address from the
273  // synaptic row
274  size_t write_size = buffer->n_bytes_transferred;
275  void *sdram_start_address = buffer->sdram_writeback_address;
276  void *dtcm_start_address = buffer->row;
277  if (plastic_only) {
278  write_size = synapse_row_plastic_size(buffer->row) * sizeof(uint32_t);
279  sdram_start_address = synapse_row_plastic_region(
280  buffer->sdram_writeback_address);
281  dtcm_start_address = synapse_row_plastic_region(buffer->row);
282  }
283 
284  log_debug("Writing back %u bytes of plastic region to %08x for spike %u",
285  write_size, sdram_start_address, buffer->originating_spike);
286 
287  // Start transfer
288  while (!spin1_dma_transfer(DMA_TAG_WRITE_PLASTIC_REGION, sdram_start_address,
289  dtcm_start_address, DMA_WRITE, write_size)) {
290  // Do Nothing
291  }
292 }

295 static inline void start_dma_loop(void) {
296  // If we're not already processing synaptic DMAs,
297  // flag pipeline as busy and trigger a feed event
298  // NOTE: locking is not used here because this is assumed to be FIQ
299  if (!dma_busy) {
300  log_debug("Sending user event for new spike");
301  // Only set busy if successful.
302  // NOTE: Counts when unsuccessful are handled by the API
303  if (spin1_trigger_user_event(0, 0)) {
304  dma_busy = true;
305  }
306  }

341 static void dma_complete_callback(UNUSED uint unused, uint tag) {
342 
343  // increment the dma complete count for provenance generation
344  dma_complete_count++;
345 
346  log_debug("DMA transfer complete at time %u with tag %u", time, tag);
347 
348  // Get pointer to current buffer
349  uint32_t current_buffer_index = buffer_being_read;
350  dma_buffer *current_buffer = &dma_buffers[current_buffer_index];
351 
352  // Start the next DMA transfer and get a count of the rewires and spikes
353  // that can be done on this row now (there might be more while the DMA
354  // was in progress). Note that either dma_n_rewires or dma_n_spikes is set
355  // to 1 here, with the other being 0. We take a copy of the count and this
356  // is the value added to for this processing, as setup_synaptic_dma will
357  // count repeats of the current spike
358  uint32_t n_rewires = dma_n_rewires;
359  uint32_t n_spikes = dma_n_spikes;
360  setup_synaptic_dma_read(current_buffer, &n_rewires, &n_spikes);
361 
362  // Assume no write back but assume any write back is plastic only
363  bool write_back = false;
364  bool plastic_only = true;
365 
366  // If rewiring, do rewiring first
367  for (uint32_t i = n_rewires; i > 0; i--) {
368  if (synaptogenesis_row_restructure(time, current_buffer->row)) {
369  write_back = true;
370  plastic_only = false;
371  n_successful_rewires++;
372  }
373  }
374 
375  // Process synaptic row repeatedly for any upcoming spikes
376  while (n_spikes > 0) {
377 
378  // Process synaptic row, writing it back if it's the last time
379  // it's going to be processed
380  bool write_back_now = false;
381  if (!synapses_process_synaptic_row(
382  time, current_buffer->row, &write_back_now)) {
383  log_error(
384  "Error processing spike 0x%.8x for address 0x%.8x"
385  " (local=0x%.8x)",
386  current_buffer->originating_spike,
387  current_buffer->sdram_writeback_address,
388  current_buffer->row);
389 
390  // Print out the row for debugging
391  address_t row = (address_t) current_buffer->row;
392  for (uint32_t i = 0;
393  i < (current_buffer->n_bytes_transferred >> 2); i++) {
394  log_error("%u: 0x%08x", i, row[i]);
395  }
396  rt_error(RTE_SWERR);
397  }
398 
399  write_back |= write_back_now;
400  n_spikes--;
401  }
402 
403  if (write_back) {
404  setup_synaptic_dma_write(current_buffer_index, plastic_only);
405  }
406 }

411 void user_event_callback(UNUSED uint unused0, UNUSED uint unused1) {
412  // Reset the counters as this is a new process
413  dma_n_rewires = 0;
414  dma_n_spikes = 0;
415 
416  if (buffer_being_read < N_DMA_BUFFERS) {
417  // If the DMA buffer is full of valid data, attempt to reuse it on the
418  // next data to be used, as this might be able to make use of the buffer
419  // without transferring data
420  dma_complete_callback(0, DMA_TAG_READ_SYNAPTIC_ROW);
421  } else {
422  // If the DMA buffer is invalid, just do the first transfer possible
423  setup_synaptic_dma_read(NULL, NULL, NULL);
424  }
425 }

428 void spike_processing_clear_input_buffer(timer_t time) {
429  uint32_t n_spikes = in_spikes_size();
430  if (clear_input_buffers_of_late_packets) {
431  spin1_dma_flush();

436  // Record the number of packets received last timer tick

438  recording_record(p_per_ts_region, &p_per_ts_struct, sizeof(p_per_ts_struct));
439  p_per_ts_struct.packets_this_time_step = 0;
440 
441  // Record the count whether clearing or not for provenance
442  count_input_buffer_packets_late += n_spikes;
443 }

445 bool spike_processing_initialise( // EXPORTED
446  size_t row_max_n_words, uint mc_packet_callback_priority,
447  uint user_event_priority, uint incoming_spike_buffer_size,
448  bool clear_input_buffers_of_late_packets_init,
449  uint32_t packets_per_timestep_region) {
450  // Allocate the DMA buffers
451  for (uint32_t i = 0; i < N_DMA_BUFFERS; i++) {
452  dma_buffers[i].row = spin1_malloc(row_max_n_words * sizeof(uint32_t));
453  if (dma_buffers[i].row == NULL) {
454  log_error("Could not initialise DMA buffers");
455  return false;
456  }
457  log_debug("DMA buffer %u allocated at 0x%08x",
458  i, dma_buffers[i].row);
459  }
460  dma_busy = false;
461  clear_input_buffers_of_late_packets =
462  clear_input_buffers_of_late_packets_init;
463  next_buffer_to_fill = 0;
464  buffer_being_read = N_DMA_BUFFERS;
465  p_per_ts_region = packets_per_timestep_region;
466 
467  // Allocate incoming spike buffer
468  if (!in_spikes_initialize_spike_buffer(incoming_spike_buffer_size)) {
469  return false;
470  }
471 
472  // Set up the callbacks
473  spin1_callback_on(MC_PACKET_RECEIVED,
474  multicast_packet_received_callback, mc_packet_callback_priority);
475  spin1_callback_on(MCPL_PACKET_RECEIVED,
476  multicast_packet_pl_received_callback, mc_packet_callback_priority);
477  simulation_dma_transfer_done_callback_on(
478  DMA_TAG_READ_SYNAPTIC_ROW, dma_complete_callback);
479  spin1_callback_on(USER_EVENT, user_event_callback, user_event_priority);
480 
481  return true;
482 }
