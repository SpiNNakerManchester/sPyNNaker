122 static inline void do_dma_read(
123  synaptic_row_t row, size_t n_bytes_to_transfer, spike_t spike) {
124  // Write the SDRAM address of the plastic region and the
125  // Key of the originating spike to the beginning of DMA buffer
126  dma_buffer *next_buffer = &dma_buffers[next_buffer_to_fill];
127  next_buffer->sdram_writeback_address = row;
128  next_buffer->originating_spike = spike;
129  next_buffer->n_bytes_transferred = n_bytes_to_transfer;
130 
131  // Start a DMA transfer to fetch this synaptic row into current
132  // buffer
133  buffer_being_read = next_buffer_to_fill;
134  while (!spin1_dma_transfer(
135  DMA_TAG_READ_SYNAPTIC_ROW, row, next_buffer->row, DMA_READ,
136  n_bytes_to_transfer)) {
137  // Do Nothing
138  }
139  next_buffer_to_fill = (next_buffer_to_fill + 1) % N_DMA_BUFFERS;
140 }

150 static inline bool is_something_to_do(
151  synaptic_row_t *row, size_t *n_bytes_to_transfer,
152  spike_t *spike, uint32_t *n_rewire, uint32_t *n_process_spike) {
153  // Disable interrupts here as dma_busy modification is a critical section
154  uint cpsr = spin1_int_disable();
155 
156  // Check for synaptic rewiring
157  while (rewires_to_do) {
158  rewires_to_do--;
159  spin1_mode_restore(cpsr);
160  if (synaptogenesis_dynamics_rewire(time, spike, row,
161  n_bytes_to_transfer)) {
162  *n_rewire += 1;
163  return true;
164  }
165  cpsr = spin1_int_disable();
166  }
167 
168  // Is there another address in the population table?
169  spin1_mode_restore(cpsr);
170  if (population_table_get_next_address(spike, row, n_bytes_to_transfer)) {
171  *n_process_spike += 1;
172  return true;
173  }
174  cpsr = spin1_int_disable();
175 
176  // track for provenance
177  uint32_t input_buffer_filled_size = in_spikes_size();
178  if (biggest_fill_size_of_input_buffer < input_buffer_filled_size) {
179  biggest_fill_size_of_input_buffer = input_buffer_filled_size;
180  }
181 
182  // Are there any more spikes to process?
183  while (in_spikes_get_next_spike(spike)) {
184  // Enable interrupts while looking up in the master pop table,
185  // as this can be slow
186  spin1_mode_restore(cpsr);
187  if (population_table_get_first_address(
188  *spike, row, n_bytes_to_transfer)) {
189  synaptogenesis_spike_received(time, *spike);
190  *n_process_spike += 1;
191  return true;
192  }
193 
194  // Disable interrupts before checking if there is another spike
195  cpsr = spin1_int_disable();
196  }
197 
198  // If nothing to do, the DMA is not busy
199  dma_busy = false;
200 
201  // Restore interrupts
202  spin1_mode_restore(cpsr);
203  return false;
204 }

218 static void setup_synaptic_dma_read(dma_buffer *current_buffer,
219  uint32_t *n_rewires, uint32_t *n_synapse_processes) {
220  // Set up to store the DMA location and size to read
221  synaptic_row_t row;
222  size_t n_bytes_to_transfer;
223  spike_t spike;
224  dma_n_spikes = 0;
225  dma_n_rewires = 0;
226 
227  // Keep looking if there is something to do until a DMA can be done
228  bool setup_done = false;
229  while (!setup_done && is_something_to_do(&row, &n_bytes_to_transfer,
230  &spike, &dma_n_rewires, &dma_n_spikes)) {
231  if (current_buffer != NULL &&
232  current_buffer->sdram_writeback_address == row) {
233  // If we can reuse the row, add on what we can use it for
234  // Note that only one of these will have a value of 1 with the
235  // other being set to 0, but we add both as it is simple
236  *n_rewires += dma_n_rewires;
237  *n_synapse_processes += dma_n_spikes;
238  dma_n_rewires = 0;
239  dma_n_spikes = 0;
240  } else if (n_bytes_to_transfer == 0) {
241  // If the row is in DTCM, process the row now
242  synaptic_row_t single_fixed_synapse =
243  direct_synapses_get_direct_synapse(row);
244  bool write_back;
245  synapses_process_synaptic_row(
246  time, single_fixed_synapse, &write_back);
247  dma_n_rewires = 0;
248  dma_n_spikes = 0;
249  } else {
250  // If the row is in SDRAM, set up the transfer and we are done
251  do_dma_read(row, n_bytes_to_transfer, spike);
252  setup_done = true;
253  }
254 
255  // needs to be here to ensure that its only recording actual spike
256  // processing and not the surplus DMA requests.
257  spike_processing_count++;
258  }
259 }

265 static inline void setup_synaptic_dma_write(
266  uint32_t dma_buffer_index, bool plastic_only) {
267  // Get pointer to current buffer
268  dma_buffer *buffer = &dma_buffers[dma_buffer_index];
269 
270  // Get the number of plastic bytes and the write back address from the
271  // synaptic row
272  size_t write_size = buffer->n_bytes_transferred;
273  void *sdram_start_address = buffer->sdram_writeback_address;
274  void *dtcm_start_address = buffer->row;
275  if (plastic_only) {
276  write_size = synapse_row_plastic_size(buffer->row) * sizeof(uint32_t);
277  sdram_start_address = synapse_row_plastic_region(
278  buffer->sdram_writeback_address);
279  dtcm_start_address = synapse_row_plastic_region(buffer->row);
280  }
281 
282  log_debug("Writing back %u bytes of plastic region to %08x for spike %u",
283  write_size, sdram_start_address, buffer->originating_spike);
284 
285  // Start transfer
286  while (!spin1_dma_transfer(DMA_TAG_WRITE_PLASTIC_REGION, sdram_start_address,
287  dtcm_start_address, DMA_WRITE, write_size)) {
288  // Do Nothing
289  }
290 }

295 static void multicast_packet_received_callback(uint key, uint payload) {
296  p_per_ts_struct.packets_this_time_step += 1;
297 
298  // handle the 2 cases separately
299  if (payload == 0) {
300  log_debug(
301  "Received spike %x at %d, DMA Busy = %d", key, time, dma_busy);
302  // set to 1 to work with the loop.
303  payload = 1;
304  } else {
305  log_debug(
306  "Received spike %x with payload %d at %d, DMA Busy = %d",
307  key, payload, time, dma_busy);
308  }
309 
310  // cycle through the packet insertion
311  for (uint count = payload; count > 0; count--) {
312  in_spikes_add_spike(key);
313  }
314 
315  // If we're not already processing synaptic DMAs,
316  // flag pipeline as busy and trigger a feed event
317  // NOTE: locking is not used here because this is assumed to be FIQ
318  if (!dma_busy) {
319  log_debug("Sending user event for new spike");
320  if (spin1_trigger_user_event(0, 0)) {
321  dma_busy = true;
322  } else {
323  log_warning("Could not trigger user event\n");
324  }
325  }

331 static void dma_complete_callback(UNUSED uint unused, uint tag) {
332 
333  // increment the dma complete count for provenance generation
334  dma_complete_count++;
335 
336  log_debug("DMA transfer complete at time %u with tag %u", time, tag);
337 
338  // Get pointer to current buffer
339  uint32_t current_buffer_index = buffer_being_read;
340  dma_buffer *current_buffer = &dma_buffers[current_buffer_index];
341 
342  // Start the next DMA transfer and get a count of the rewires and spikes
343  // that can be done on this row now (there might be more while the DMA
344  // was in progress). Note that either dma_n_rewires or dma_n_spikes is set
345  // to 1 here, with the other being 0. We take a copy of the count and this
346  // is the value added to for this processing, as setup_synaptic_dma will
347  // count repeats of the current spike
348  uint32_t n_rewires = dma_n_rewires;
349  uint32_t n_spikes = dma_n_spikes;
350  setup_synaptic_dma_read(current_buffer, &n_rewires, &n_spikes);
351 
352  // Assume no write back but assume any write back is plastic only
353  bool write_back = false;
354  bool plastic_only = true;
355 
356  // If rewiring, do rewiring first
357  for (uint32_t i = 0; i < n_rewires; i++) {
358  if (synaptogenesis_row_restructure(time, current_buffer->row)) {
359  write_back = true;
360  plastic_only = false;
361  n_successful_rewires++;
362  }
363  }
364 
365  // Process synaptic row repeatedly for any upcoming spikes
366  while (n_spikes > 0) {
367 
368  // Process synaptic row, writing it back if it's the last time
369  // it's going to be processed
370  bool write_back_now = false;
371  if (!synapses_process_synaptic_row(
372  time, current_buffer->row, &write_back_now)) {
373  log_error(
374  "Error processing spike 0x%.8x for address 0x%.8x"
375  " (local=0x%.8x)",
376  current_buffer->originating_spike,
377  current_buffer->sdram_writeback_address,
378  current_buffer->row);
379 
380  // Print out the row for debugging
381  address_t row = (address_t) current_buffer->row;
382  for (uint32_t i = 0;
383  i < (current_buffer->n_bytes_transferred >> 2); i++) {
384  log_error("%u: 0x%.8x", i, row[i]);
385  }
386  rt_error(RTE_SWERR);
387  }
388 
389  write_back |= write_back_now;
390  n_spikes--;
391  }
392 
393  if (write_back) {
394  setup_synaptic_dma_write(current_buffer_index, plastic_only);
395  }
396 }

401 void user_event_callback(UNUSED uint unused0, UNUSED uint unused1) {
402  // Reset the counters as this is a new process
403  dma_n_rewires = 0;
404  dma_n_spikes = 0;
405 
406  if (buffer_being_read < N_DMA_BUFFERS) {
407  // If the DMA buffer is full of valid data, attempt to reuse it on the
408  // next data to be used, as this might be able to make use of the buffer
409  // without transferring data
410  dma_complete_callback(0, DMA_TAG_READ_SYNAPTIC_ROW);
411  } else {
412  // If the DMA buffer is invalid, just do the first transfer possible
413  setup_synaptic_dma_read(NULL, NULL, NULL);
414  }
415 }

435 bool spike_processing_initialise( // EXPORTED
436  size_t row_max_n_words, uint mc_packet_callback_priority,
437  uint user_event_priority, uint incoming_spike_buffer_size,
438  bool clear_input_buffers_of_late_packets_init,
439  uint32_t packets_per_timestep_region) {
440  // Allocate the DMA buffers
441  for (uint32_t i = 0; i < N_DMA_BUFFERS; i++) {
442  dma_buffers[i].row = spin1_malloc(row_max_n_words * sizeof(uint32_t));
443  if (dma_buffers[i].row == NULL) {
444  log_error("Could not initialise DMA buffers");
445  return false;
446  }
447  log_debug("DMA buffer %u allocated at 0x%08x",
448  i, dma_buffers[i].row);
449  }
450  dma_busy = false;
451  clear_input_buffers_of_late_packets =
452  clear_input_buffers_of_late_packets_init;
453  next_buffer_to_fill = 0;
454  buffer_being_read = N_DMA_BUFFERS;
455  p_per_ts_region = packets_per_timestep_region;
456 
457  // Allocate incoming spike buffer
458  if (!in_spikes_initialize_spike_buffer(incoming_spike_buffer_size)) {
459  return false;
460  }
461 
462  // Set up the callbacks
463  spin1_callback_on(MC_PACKET_RECEIVED,
464  multicast_packet_received_callback, mc_packet_callback_priority);
465  spin1_callback_on(MCPL_PACKET_RECEIVED,
466  multicast_packet_received_callback, mc_packet_callback_priority);
467  simulation_dma_transfer_done_callback_on(
468  DMA_TAG_READ_SYNAPTIC_ROW, dma_complete_callback);
469  spin1_callback_on(USER_EVENT, user_event_callback, user_event_priority);
470 
471  return true;
472 }

501 bool spike_processing_do_rewiring(int number_of_rewires) {
502  // disable interrupts
503  uint cpsr = spin1_int_disable();
504  rewires_to_do += number_of_rewires;
505 
506  // If we're not already processing synaptic DMAs,
507  // flag pipeline as busy and trigger a feed event
