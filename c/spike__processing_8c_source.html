98 static inline void do_dma_read(
99  address_t row_address, size_t n_bytes_to_transfer, spike_t spike) {
100  // Write the SDRAM address of the plastic region and the
101  // Key of the originating spike to the beginning of DMA buffer
102  dma_buffer *next_buffer = &dma_buffers[next_buffer_to_fill];
103  next_buffer->sdram_writeback_address = row_address;
104  next_buffer->originating_spike = spike;
105  next_buffer->n_bytes_transferred = n_bytes_to_transfer;
106 
107  // Start a DMA transfer to fetch this synaptic row into current
108  // buffer
109  buffer_being_read = next_buffer_to_fill;
110  while (!spin1_dma_transfer(
111  DMA_TAG_READ_SYNAPTIC_ROW, row_address, next_buffer->row, DMA_READ,
112  n_bytes_to_transfer)) {
113  // Do Nothing
114  }
115  next_buffer_to_fill = (next_buffer_to_fill + 1) % N_DMA_BUFFERS;
116 }

126 static inline bool is_something_to_do(
127  address_t *row_address, size_t *n_bytes_to_transfer,
128  spike_t *spike, uint32_t *n_rewire, uint32_t *n_process_spike) {
129  // Disable interrupts here as dma_busy modification is a critical section
130  uint cpsr = spin1_int_disable();
131 
132  // Check for synaptic rewiring
133  while (rewires_to_do) {
134  rewires_to_do--;
135  spin1_mode_restore(cpsr);
136  if (synaptogenesis_dynamics_rewire(time, spike, row_address,
137  n_bytes_to_transfer)) {
138  *n_rewire += 1;
139  return true;
140  }
141  cpsr = spin1_int_disable();
142  }
143 
144  // Is there another address in the population table?
145  spin1_mode_restore(cpsr);
146  if (population_table_get_next_address(
147  spike, row_address, n_bytes_to_transfer)) {
148  *n_process_spike += 1;
149  return true;
150  }
151  cpsr = spin1_int_disable();
152  // Are there any more spikes to process?
153  while (in_spikes_get_next_spike(spike)) {
154  // Enable interrupts while looking up in the master pop table,
155  // as this can be slow
156  spin1_mode_restore(cpsr);
157  if (population_table_get_first_address(
158  *spike, row_address, n_bytes_to_transfer)) {
159  synaptogenesis_spike_received(time, *spike);
160  *n_process_spike += 1;
161  return true;
162  }
163 
164  // Disable interrupts before checking if there is another spike
165  cpsr = spin1_int_disable();
166  }
167 
168  // If nothing to do, the DMA is not busy
169  dma_busy = false;
170 
171  // Restore interrupts
172  spin1_mode_restore(cpsr);
173  return false;
174 }

188 static void setup_synaptic_dma_read(dma_buffer *current_buffer,
189  uint32_t *n_rewires, uint32_t *n_synapse_processes) {
190  // Set up to store the DMA location and size to read
191  address_t row_address;
192  size_t n_bytes_to_transfer;
193  spike_t spike;
194  dma_n_spikes = 0;
195  dma_n_rewires = 0;
196 
197  // Keep looking if there is something to do until a DMA can be done
198  bool setup_done = false;
199  while (!setup_done && is_something_to_do(&row_address,
200  &n_bytes_to_transfer, &spike, &dma_n_rewires, &dma_n_spikes)) {
201  if (current_buffer != NULL &&
202  current_buffer->sdram_writeback_address == row_address) {
203  // If we can reuse the row, add on what we can use it for
204  // Note that only one of these will have a value of 1 with the
205  // other being set to 0, but we add both as it is simple
206  *n_rewires += dma_n_rewires;
207  *n_synapse_processes += dma_n_spikes;
208  dma_n_rewires = 0;
209  dma_n_spikes = 0;
210  } else if (n_bytes_to_transfer == 0) {
211  // If the row is in DTCM, process the row now
212  synaptic_row_t single_fixed_synapse =
213  direct_synapses_get_direct_synapse(row_address);
214  bool write_back;
215  synapses_process_synaptic_row(
216  time, single_fixed_synapse, &write_back);
217  dma_n_rewires = 0;
218  dma_n_spikes = 0;
219  } else {
220  // If the row is in SDRAM, set up the transfer and we are done
221  do_dma_read(row_address, n_bytes_to_transfer, spike);
222  setup_done = true;
223  }
224  }
225  spike_processing_count++;
226 }

232 static inline void setup_synaptic_dma_write(
233  uint32_t dma_buffer_index, bool plastic_only) {
234  // Get pointer to current buffer
235  dma_buffer *buffer = &dma_buffers[dma_buffer_index];
236 
237  // Get the number of plastic bytes and the write back address from the
238  // synaptic row
239  size_t write_size = buffer->n_bytes_transferred;
240  address_t sdram_start_address = buffer->sdram_writeback_address;
241  address_t dtcm_start_address = buffer->row;
242  if (plastic_only) {
243  write_size = synapse_row_plastic_size(buffer->row) * sizeof(uint32_t);
244  sdram_start_address = synapse_row_plastic_region(sdram_start_address);
245  dtcm_start_address = synapse_row_plastic_region(dtcm_start_address);
246  }
247 
248  log_debug("Writing back %u bytes of plastic region to %08x for spike %u",
249  write_size, sdram_start_address, buffer->originating_spike);
250 
251  // Start transfer
252  while (!spin1_dma_transfer(DMA_TAG_WRITE_PLASTIC_REGION, sdram_start_address,
253  dtcm_start_address, DMA_WRITE, write_size)) {
254  // Do Nothing
255  }
256 }

261 static void multicast_packet_received_callback(uint key, UNUSED uint payload) {
262  log_debug("Received spike %x at %d, DMA Busy = %d", key, time, dma_busy);
263 
264  // If there was space to add spike to incoming spike queue
265  if (in_spikes_add_spike(key)) {
266  // If we're not already processing synaptic DMAs,
267  // flag pipeline as busy and trigger a feed event
268  // NOTE: locking is not used here because this is assumed to be FIQ
269  if (!dma_busy) {
270  log_debug("Sending user event for new spike");
271  if (spin1_trigger_user_event(0, 0)) {
272  dma_busy = true;
273  } else {
274  log_debug("Could not trigger user event\n");
275  }
276  }
277  } else {
278  log_debug("Could not add spike");
279  }
280 }

285 static void dma_complete_callback(UNUSED uint unused, uint tag) {
286  // increment the dma complete count for provenance generation
287  dma_complete_count++;
288 
289  log_debug("DMA transfer complete at time %u with tag %u", time, tag);
290 
291  // Get pointer to current buffer
292  uint32_t current_buffer_index = buffer_being_read;
293  dma_buffer *current_buffer = &dma_buffers[current_buffer_index];
294 
295  // Start the next DMA transfer and get a count of the rewires and spikes
296  // that can be done on this row now (there might be more while the DMA
297  // was in progress). Note that either dma_n_rewires or dma_n_spikes is set
298  // to 1 here, with the other being 0. We take a copy of the count and this
299  // is the value added to for this processing, as setup_synaptic_dma will
300  // count repeats of the current spike
301  uint32_t n_rewires = dma_n_rewires;
302  uint32_t n_spikes = dma_n_spikes;
303  setup_synaptic_dma_read(current_buffer, &n_rewires, &n_spikes);
304 
305  // Assume no write back but assume any write back is plastic only
306  bool write_back = false;
307  bool plastic_only = true;
308 
309  // If rewiring, do rewiring first
310  for (uint32_t i = 0; i < n_rewires; i++) {
311  if (synaptogenesis_row_restructure(time, current_buffer->row)) {
312  write_back = true;
313  plastic_only = false;
314  n_successful_rewires++;
315  }
316  }
317 
318  // Process synaptic row repeatedly for any upcoming spikes
319  while (n_spikes > 0) {
320 
321  // Process synaptic row, writing it back if it's the last time
322  // it's going to be processed
323  bool write_back_now = false;
324  if (!synapses_process_synaptic_row(
325  time, current_buffer->row, &write_back_now)) {
326  log_error(
327  "Error processing spike 0x%.8x for address 0x%.8x"
328  " (local=0x%.8x)",
329  current_buffer->originating_spike,
330  current_buffer->sdram_writeback_address,
331  current_buffer->row);
332 
333  // Print out the row for debugging
334  for (uint32_t i = 0;
335  i < (current_buffer->n_bytes_transferred >> 2); i++) {
336  log_error("%u: 0x%.8x", i, current_buffer->row[i]);
337  }
338  rt_error(RTE_SWERR);
339  }
340 
341  write_back |= write_back_now;
342  n_spikes--;
343  }
344 
345  if (write_back) {
346  setup_synaptic_dma_write(current_buffer_index, plastic_only);
347  }
348 }

353 void user_event_callback(UNUSED uint unused0, UNUSED uint unused1) {
354  // Reset the counters as this is a new process
355  dma_n_rewires = 0;
356  dma_n_spikes = 0;
357 
358  if (buffer_being_read < N_DMA_BUFFERS) {
359  // If the DMA buffer is full of valid data, attempt to reuse it on the
360  // next data to be used, as this might be able to make use of the buffer
361  // without transferring data
362  dma_complete_callback(0, DMA_TAG_READ_SYNAPTIC_ROW);
363  } else {
364  // If the DMA buffer is invalid, just do the first transfer possible
365  setup_synaptic_dma_read(NULL, NULL, NULL);
366  }
367 }

371 bool spike_processing_initialise( // EXPORTED
372  size_t row_max_n_words, uint mc_packet_callback_priority,
373  uint user_event_priority, uint incoming_spike_buffer_size) {
374  // Allocate the DMA buffers
375  for (uint32_t i = 0; i < N_DMA_BUFFERS; i++) {
376  dma_buffers[i].row = spin1_malloc(row_max_n_words * sizeof(uint32_t));
377  if (dma_buffers[i].row == NULL) {
378  log_error("Could not initialise DMA buffers");
379  return false;
380  }
381  log_debug("DMA buffer %u allocated at 0x%08x",
382  i, dma_buffers[i].row);
383  }
384  dma_busy = false;
385  next_buffer_to_fill = 0;
386  buffer_being_read = N_DMA_BUFFERS;
387 
388  // Allocate incoming spike buffer
389  if (!in_spikes_initialize_spike_buffer(incoming_spike_buffer_size)) {
390  return false;
391  }
392 
393  // Set up the callbacks
394  spin1_callback_on(MC_PACKET_RECEIVED,
395  multicast_packet_received_callback, mc_packet_callback_priority);
396  simulation_dma_transfer_done_callback_on(
397  DMA_TAG_READ_SYNAPTIC_ROW, dma_complete_callback);
398  spin1_callback_on(USER_EVENT, user_event_callback, user_event_priority);
399 
400  return true;
401 }

435 bool spike_processing_do_rewiring(int number_of_rewires) {
436  // disable interrupts
437  uint cpsr = spin1_int_disable();
438  rewires_to_do += number_of_rewires;
439 
440  // If we're not already processing synaptic DMAs,
441  // flag pipeline as busy and trigger a feed event

443  log_debug("Sending user event for rewiring");

445  dma_busy = true;
446  } else {
447  log_debug("Could not trigger user event\n");
448  }
449  }
450  // enable interrupts
451  spin1_mode_restore(cpsr);
452  return true;
