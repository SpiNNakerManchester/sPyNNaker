114 static inline weight_t weight_get_final(weight_state_t new_state) {
115  // Scale potentiation and depression
116  // **NOTE** A2+, A2-, A3+ and A3- are pre-scaled into weight format
117  int32_t scaled_a2_plus = STDP_FIXED_MUL_16X16(
118  new_state.a2_plus, new_state.weight_region->a2_plus);
119  int32_t scaled_a2_minus = STDP_FIXED_MUL_16X16(
120  new_state.a2_minus, new_state.weight_region->a2_minus);
121  int32_t scaled_a3_plus = STDP_FIXED_MUL_16X16(
122  new_state.a3_plus, new_state.weight_region->a3_plus);
123  int32_t scaled_a3_minus = STDP_FIXED_MUL_16X16(
124  new_state.a3_minus, new_state.weight_region->a3_minus);
125 
126  // Apply all terms to initial weight
127  int32_t new_weight = new_state.initial_weight + scaled_a2_plus
128  + scaled_a3_plus - scaled_a2_minus - scaled_a3_minus;
129 
130  // Clamp new weight
131  new_weight = MIN(new_state.weight_region->max_weight,
132  MAX(new_weight, new_state.weight_region->min_weight));
133 
134  log_debug("\told_weight:%u, a2+:%d, a2-:%d, a3+:%d, a3-:%d",
135  new_state.initial_weight, new_state.a2_plus,
136  new_state.a2_minus, new_state.a3_plus, new_state.a3_minus);
137  log_debug("\tscaled a2+:%d, scaled a2-:%d, scaled a3+:%d, scaled a3-:%d,"
138  " new_weight:%d",
139  scaled_a2_plus, scaled_a2_minus, scaled_a3_plus,
140  scaled_a3_minus, new_weight);
141 
142  return (weight_t) new_weight;
143 }
