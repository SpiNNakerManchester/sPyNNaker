261 static inline uint32_t rng(void) {
262  ssp_params.spike_source_seed.x = 314527869 * ssp_params.spike_source_seed.x + 1234567;
263  ssp_params.spike_source_seed.y ^= ssp_params.spike_source_seed.y << 5;
264  ssp_params.spike_source_seed.y ^= ssp_params.spike_source_seed.y >> 7;
265  ssp_params.spike_source_seed.y ^= ssp_params.spike_source_seed.y << 22;
266  uint64_t t = 4294584393ULL * ssp_params.spike_source_seed.z + ssp_params.spike_source_seed.c;
267  ssp_params.spike_source_seed.c = t >> 32;
268  ssp_params.spike_source_seed.z = t;
269 
270  return (uint32_t) ssp_params.spike_source_seed.x
271  + ssp_params.spike_source_seed.y + ssp_params.spike_source_seed.z;
272 }

277 static inline uint32_t n_spikes_poisson_fast(UFRACT exp_minus_lambda) {
278  UFRACT p = UFRACT_CONST(1.0);
279  uint32_t k = 0;
280 
281  do {
282  k++;
283  // p = p * ulrbits(uni_rng(seed_arg));
284  // Possibly faster multiplication by using DRL's routines
285  p = ulrbits(__stdfix_smul_ulr(bitsulr(p), rng()));
286  } while (bitsulr(p) > bitsulr(exp_minus_lambda));
287  return k - 1;
288 }

292 static inline REAL n_steps_until_next(void) {
293  REAL A = REAL_CONST(0.0);
294  uint32_t U, U0, USTAR;
295 
296  while (true) {
297  U = rng();
298  U0 = U;
299 
300  do {
301  USTAR = rng();
302  if (U < USTAR) {
303  return A + (REAL) ulrbits(U0);
304  }
305 
306  U = rng();
307  } while (U < USTAR);
308 
309  A += 1.0k;
310  }
311 }

319 static inline uint32_t slow_spike_source_get_time_to_spike(
320  uint32_t mean_inter_spike_interval_in_ticks) {
321  // Round (dist variate * ISI_SCALE_FACTOR), convert to uint32
322  uint32_t value = (uint32_t) roundk(
323  n_steps_until_next() * ISI_SCALE_FACTOR, (15));
324  // Now multiply by the mean ISI
325  uint32_t exp_variate = value * mean_inter_spike_interval_in_ticks;
326  // Note that this will be compared to ISI_SCALE_FACTOR in the main loop!
327  return exp_variate;
328 }

335 static inline uint32_t fast_spike_source_get_num_spikes(
336  UFRACT exp_minus_lambda) {
337  // If the value of exp_minus_lambda is very small then it's not worth
338  // using the algorithm, so just return 0
339  if (bitsulr(exp_minus_lambda) == bitsulr(UFRACT_CONST(0.0))) {

351 static inline uint32_t faster_spike_source_get_num_spikes(
352  REAL sqrt_lambda) {
353  // First we do x = (inv_gauss_cdf(U(0, 1)) * 0.5) + sqrt(lambda)
354  uint32_t U = rng();
355  REAL x = (norminv_urt(U) * HALF) + sqrt_lambda;
356  // Then we return int(roundk(x * x))
357  return (uint32_t) roundk(x * x, 15);
358 }

364 void set_spike_source_rate(uint32_t sub_id, unsigned long accum rate) {
365 
366  REAL rate_per_tick = kbits(
367  (__U64(bitsk(rate)) * __U64(bitsulr(ssp_params.seconds_per_tick))) >> 32);
368  log_debug("Setting rate of %u to %kHz (%k per tick)",
369  sub_id, rate, rate_per_tick);
370  spike_source_t *spike_source = &source[sub_id];
371 
372  if (rate_per_tick >= ssp_params.slow_rate_per_tick_cutoff) {
373  spike_source->is_fast_source = 1;
374  spike_source->mean_isi_ticks = 0;
375  spike_source->time_to_spike_ticks = 0;
376  if (rate_per_tick >= ssp_params.fast_rate_per_tick_cutoff) {
377  spike_source->sqrt_lambda = SQRT(rate_per_tick);
378  spike_source->exp_minus_lambda = UFRACT_CONST(0);
379  } else {
380  spike_source->exp_minus_lambda = (UFRACT) EXP(-rate_per_tick);
381  spike_source->sqrt_lambda = ZERO;
382  }
383  } else {
384  if (rate > 0) {
385  spike_source->mean_isi_ticks =
386  (uint32_t) ((bitsulk((unsigned long accum) ts_per_second)) / bitsulk(rate));
387  } else {
388  spike_source->mean_isi_ticks = 0;
389  }
390 
391  spike_source->exp_minus_lambda = UFRACT_CONST(0);
392  spike_source->sqrt_lambda = ZERO;
393  spike_source->is_fast_source = 0;
394  spike_source->time_to_spike_ticks =
395  slow_spike_source_get_time_to_spike(spike_source->mean_isi_ticks);
396  }
397 }

416 static inline void set_spike_source_details(uint32_t id, bool rate_changed) {
417  uint32_t index = source_data[id]->index;
418  log_debug("Source %u is at index %u", id, index);
419  source_details details = source_data[id]->details[index];
420  if (rate_changed) {
421  log_debug("Setting rate of %u to %k at %u", id, (s1615) details.rate, time);
422  set_spike_source_rate(id, details.rate);
423  }
424  spike_source_t *p = &(source[id]);
425  p->start_ticks = ms_to_ticks(details.start);
426  log_debug("Start of %u is %u", id, p->start_ticks);
427  if (details.duration == END_OF_TIME) {
428  log_debug("Duration of %u is forever", id);
429  p->end_ticks = END_OF_TIME;
430  } else {
431  uint32_t duration_ticks = ms_to_ticks(details.duration);
432  p->end_ticks = p->start_ticks + duration_ticks;
433  log_debug("Duration of %u is %u, end = %u", id, duration_ticks, p->end_ticks);
434  }
435  if ((index + 1) >= source_data[id]->n_rates) {
436  log_debug("Next of %u never happens", id);
437  p->next_ticks = END_OF_TIME;
438  } else {
439  accum next_start = source_data[id]->details[index + 1].start;
440  p->next_ticks = ms_to_ticks(next_start);
441  log_debug("Next of %u at %u", id, p->next_ticks);
442  }
443 }

544 static bool read_rates(source_info *sdram_sources, bool rate_changed, uint32_t next_time) {
545  // Allocate DTCM for array of spike sources and copy block of data
546  if (ssp_params.n_spike_sources > 0) {
547  // the first time around, the array is set to NULL, afterwards,
548  // assuming all goes well, there's an address here.
549  if (source == NULL) {
550  source = spin1_malloc(
551  ssp_params.n_spike_sources * sizeof(spike_source_t));
552  // if failed to alloc memory, report and fail.
553  if (source == NULL) {
554  log_error("Failed to allocate local sources");
555  return false;
556  }
557  source_data = spin1_malloc(
558  ssp_params.n_spike_sources * sizeof(source_info *));
559  if (source_data == NULL) {
560  log_error("Failed to allocate SDRAM source links");
561  return false;
562  }
563 
564  // Copy the address of each source
565  source_info *sdram_source = sdram_sources;
566  for (uint32_t i = 0; i < ssp_params.n_spike_sources; i++) {
567  source_data[i] = sdram_source;
568  sdram_source = (source_info *)
569  &sdram_source->details[sdram_source->n_rates];
570  }
571  }
572 
573  // Put the correct values into the current source information
574  for (uint32_t i = 0; i < ssp_params.n_spike_sources; i++) {
575 
576  // Find the index for the current time
577  uint32_t index = 0;
578  uint32_t n_rates = source_data[i]->n_rates;
579  while ((index + 1) < n_rates
580  && next_time >= ms_to_ticks(source_data[i]->details[index + 1].start)) {
581  index++;
582  }
583  bool new_index = source_data[i]->index != index;
584  source_data[i]->index = index;
585  set_spike_source_details(i, rate_changed || new_index);
586  }
587  }
588  log_info("read_poisson_parameters: completed successfully");
589  return true;
590 }

592 static bool expand_rates(source_expand_region *items, source_info *sdram_sources) {
593 
594  if (!items->rate_changed) {
595  return false;
596  }
597 
598  // We need a pointer here as each item is dynamically sized. This pointer
599  // will be updated each time with the start of the next item to be read
600  source_expand_details *item = &(items->items[0]);
601 
602  // Similarly, we need a pointer for the current SDRAM source
603  source_info *source = &(sdram_sources[0]);
604 
605  // Go though each expander item, which says how many times to repeat
606  for (uint32_t i = 0; i < items->n_items; i++) {
607 
608  // Copy SDRAM data to local
609  uint32_t n_rates = item->info.n_rates;
610  log_debug("Reading %u rates", n_rates);
611  source_details details[n_rates];
612  for (uint32_t k = 0; k < n_rates; k++) {
613  details[k] = item->info.details[k];
614  log_debug("Repeating rate %k %u times",
615  (accum) details[k].rate, item->count);
616  }
617 
618  // Repeat the same thing this many times
619  for (uint32_t j = 0; j < item->count; j++) {
620  source->n_rates = n_rates;
621  source->index = 0;
622  for (uint32_t k = 0; k < n_rates; k++) {
623  source->details[k] = details[k];
624  }
625 
626  // Update the source point to just after the last item written
627  source = (source_info *) &(source->details[n_rates]);
628  }
629 
630  // Update the item pointer to just after the last item read
631  item = (source_expand_details *) &(item->info.details[n_rates]);
632  }
633 
634  items->rate_changed = false;
635  return true;
636 }

654 static inline void expand_spike_recording_buffer(uint32_t n_spikes) {
655  uint32_t new_size = 8 + (n_spikes * spike_buffer_size);
656  timed_out_spikes *new_spikes = spin1_malloc(new_size);
657  if (new_spikes == NULL) {
658  log_error("Cannot reallocate spike buffer");
659  rt_error(RTE_SWERR);
660  }
661 
662  // bzero the new buffer
663  uint32_t *data = (uint32_t *) new_spikes;
664  for (uint32_t n = new_size >> 2; n > 0; n--) {
665  data[n - 1] = 0;
666  }
667 
668  // Copy over old buffer if we have it
669  if (spikes != NULL) {
670  spin1_memcpy(new_spikes, spikes,
671  8 + n_spike_buffers_allocated * spike_buffer_size);
672  sark_free(spikes);
673  }
674 
675  spikes = new_spikes;
676  n_spike_buffers_allocated = n_spikes;
677 }

850 static void process_fast_source(index_t s_id, spike_source_t *source) {
851  if ((time >= source->start_ticks) && (time < source->end_ticks)) {
852  // Get number of spikes to send this tick
853  uint32_t num_spikes = 0;
854 
855  // If sqrt_lambda has been set then use the Gaussian algorithm for
856  // faster sources
857  if (REAL_COMPARE(source->sqrt_lambda, >, ZERO)) {
858  profiler_write_entry_disable_irq_fiq(
859  PROFILER_ENTER | PROFILER_PROB_FUNC);
860  num_spikes = faster_spike_source_get_num_spikes(
861  source->sqrt_lambda);
862  profiler_write_entry_disable_irq_fiq(
863  PROFILER_EXIT | PROFILER_PROB_FUNC);
864  } else {
865  // Call the fast source Poisson algorithm
866  profiler_write_entry_disable_irq_fiq(
867  PROFILER_ENTER | PROFILER_PROB_FUNC);
868  num_spikes = fast_spike_source_get_num_spikes(
869  source->exp_minus_lambda);
870  profiler_write_entry_disable_irq_fiq(
871  PROFILER_EXIT | PROFILER_PROB_FUNC);
872  }
873 
874  log_debug("Generating %d spikes", num_spikes);
875 
876  // If there are any
877  if (num_spikes > 0) {
878  // Write spike to out spikes
879  mark_spike(s_id, num_spikes);
880 
881  // If no key has been given, do not send spikes to fabric
882  if (ssp_params.has_key) {
883  // Send spikes
884  const uint32_t spike_key = keys[s_id] | colour;
885  send_spike_mc_payload(spike_key, num_spikes);
886  } else if (sdram_inputs->address != 0) {
887  input_this_timestep[sdram_inputs->offset + s_id] +=
888  sdram_inputs->weights[s_id] * num_spikes;
889  }
890  }
891  }
892 }

897 static void process_slow_source(index_t s_id, spike_source_t *source) {
898  if ((time >= source->start_ticks) && (time < source->end_ticks)
899  && (source->mean_isi_ticks != 0)) {
900  uint32_t count = 0;
901  // Mark a spike while the "timer" is below the scale factor value
902  while (source->time_to_spike_ticks < ISI_SCALE_FACTOR) {
903  count++;
904 
905  // Update time to spike (note, this might not get us back above
906  // the scale factor, particularly if the mean_isi is smaller)
907  profiler_write_entry_disable_irq_fiq(
908  PROFILER_ENTER | PROFILER_PROB_FUNC);
909  source->time_to_spike_ticks +=
910  slow_spike_source_get_time_to_spike(source->mean_isi_ticks);
911  profiler_write_entry_disable_irq_fiq(
912  PROFILER_EXIT | PROFILER_PROB_FUNC);
913  }
914  if (count) {
915  // Write spike to out_spikes
916  mark_spike(s_id, count);
917 
918  // if no key has been given, do not send spike to fabric.
919  if (ssp_params.has_key) {
920  // Send package
921  const uint32_t spike_key = keys[s_id] | colour;
922  send_spike_mc_payload(spike_key, count);
923  } else if (sdram_inputs->address != 0) {
924  input_this_timestep[sdram_inputs->offset + s_id] +=
925  sdram_inputs->weights[s_id] * count;
926  }
927  }
928 
929  // Now we have finished for this tick, subtract the scale factor
930  source->time_to_spike_ticks -= ISI_SCALE_FACTOR;
931  }
932 }

940 static void timer_callback(UNUSED uint timer_count, UNUSED uint unused) {
941  profiler_write_entry_disable_irq_fiq(PROFILER_ENTER | PROFILER_TIMER);
942 
943  time++;
944 
945  log_debug("Timer tick %u", time);
946 
947  // If a fixed number of simulation ticks are specified and these have passed
948  if (simulation_is_finished()) {
949  // go into pause and resume state to avoid another tick
950  simulation_handle_pause_resume(resume_callback);
951 
952  profiler_write_entry_disable_irq_fiq(PROFILER_EXIT | PROFILER_TIMER);
953 
954  // Finalise any recordings that are in progress, writing back the final
955  // amounts of samples recorded to SDRAM
956  if (recording_flags > 0) {
957  recording_finalise();
958  }
959 
960  profiler_finalise();
961 
962  // Subtract 1 from the time so this tick gets done again on the next
963  // run
964  time--;
965  simulation_ready_to_read();
966  return;
967  }
968 
969  // Set the colour for the time step
970  colour = time & colour_mask;
971 
972  // Do any rate changes
973  while (circular_buffer_size(rate_change_buffer) >= 2) {
974  uint32_t id = 0;
975  REAL rate = 0.0k;
976  circular_buffer_get_next(rate_change_buffer, &id);
977  circular_buffer_get_next(rate_change_buffer, (uint32_t *) &rate);
978  set_spike_source_rate(id, rate);
979  }
980 
981  // Reset the inputs this timestep if using them
982  if (sdram_inputs->address != 0) {
983  sark_word_set(input_this_timestep, 0, sdram_inputs->size_in_bytes);
984  }
985 
986  // Loop through spike sources and see if they need updating
987  // NOTE: This full loop needs to happen first with processing in a second
988  // separate loop. This is to ensure that the random generator use matches
989  // between a single run and a split run (as slow sources can produce
990  // multiple spikes in a single time step).
991  for (index_t s_id = 0; s_id < ssp_params.n_spike_sources; s_id++) {
992  spike_source_t *spike_source = &source[s_id];
993 
994  // Move to the next tick now if needed
995  if (time >= spike_source->next_ticks) {
996  log_debug("Moving to next rate at time %d", time);
997  read_next_rates(s_id);
998 #if LOG_LEVEL >= LOG_DEBUG
999  print_spike_source(s_id);
1000 #endif
1001  }
1002  }
1003 
1004  // Loop through the sources and process them
1005  for (index_t s_id = 0; s_id < ssp_params.n_spike_sources; s_id++) {
1006  spike_source_t *spike_source = &source[s_id];
1007  if (spike_source->is_fast_source) {
1008  process_fast_source(s_id, spike_source);
1009  } else {
1010  process_slow_source(s_id, spike_source);
1011  }
1012  }
1013 
1014  profiler_write_entry_disable_irq_fiq(PROFILER_EXIT | PROFILER_TIMER);
1015 
1016  // If transferring over SDRAM, transfer now
1017  if (sdram_inputs->address != 0) {
1018  spin1_dma_transfer(0, sdram_inputs->address, input_this_timestep,
1019  DMA_WRITE, sdram_inputs->size_in_bytes);
1020  }
1021 
1022  // Record output spikes if required
1023  if (recording_flags > 0) {
1024  record_spikes(time);
1025  }
1026 }
