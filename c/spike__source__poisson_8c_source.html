230 static inline uint32_t slow_spike_source_get_time_to_spike(
231  uint32_t mean_inter_spike_interval_in_ticks) {
232  // Round (dist variate * ISI_SCALE_FACTOR), convert to uint32
233  int nbits = 15;
234  uint32_t value = (uint32_t) roundk(
235  exponential_dist_variate(
236  mars_kiss64_seed, ssp_params.spike_source_seed)
237  * ISI_SCALE_FACTOR, nbits);
238  // Now multiply by the mean ISI
239  uint32_t exp_variate = value * mean_inter_spike_interval_in_ticks;
240  // Note that this will be compared to ISI_SCALE_FACTOR in the main loop!
241  return exp_variate;
242 }

249 static inline uint32_t fast_spike_source_get_num_spikes(
250  UFRACT exp_minus_lambda) {
251  // If the value of exp_minus_lambda is very small then it's not worth
252  // using the algorithm, so just return 0
253  if (bitsulr(exp_minus_lambda) == bitsulr(UFRACT_CONST(0.0))) {
254  return 0;
255  }
256  return poisson_dist_variate_exp_minus_lambda(
257  mars_kiss64_seed, ssp_params.spike_source_seed, exp_minus_lambda);
258 }

266 static inline uint32_t faster_spike_source_get_num_spikes(
267  REAL sqrt_lambda) {
268  // First we do x = (inv_gauss_cdf(U(0, 1)) * 0.5) + sqrt(lambda)
269  REAL x = (gaussian_dist_variate(mars_kiss64_seed, ssp_params.spike_source_seed)
270  * HALF) + sqrt_lambda;
271  // Then we return int(roundk(x * x))
272  int nbits = 15;
273  return (uint32_t) roundk(x * x, nbits);
274 }

344 static bool read_rates(source_info *sdram_sources) {
345  // Allocate DTCM for array of spike sources and copy block of data
346  if (ssp_params.n_spike_sources > 0) {
347  // the first time around, the array is set to NULL, afterwards,
348  // assuming all goes well, there's an address here.
349  if (source == NULL) {
350  source = spin1_malloc(
351  ssp_params.n_spike_sources * sizeof(spike_source_t));
352  // if failed to alloc memory, report and fail.
353  if (source == NULL) {
354  log_error("Failed to allocate local sources");
355  return false;
356  }
357  source_data = spin1_malloc(
358  ssp_params.n_spike_sources * sizeof(source_info *));
359  if (source_data == NULL) {
360  log_error("Failed to allocate SDRAM source links");
361  return false;
362  }
363 
364  // Copy the address of each source
365  source_info *sdram_source = sdram_sources;
366  for (uint32_t i = 0; i < ssp_params.n_spike_sources; i++) {
367  source_data[i] = sdram_source;
368  sdram_source = (source_info *)
369  &sdram_source->poissons[sdram_source->n_rates];
370  }
371  }
372 
373  // Put the correct values into the current source information
374  for (uint32_t i = 0; i < ssp_params.n_spike_sources; i++) {
375  spike_source_t *p = &source[i];
376  spin1_memcpy(p, get_source_data(i), sizeof(spike_source_t));
377  if (!p->is_fast_source && p->time_to_spike_ticks == 0) {
378  p->time_to_spike_ticks =
379  slow_spike_source_get_time_to_spike(p->mean_isi_ticks);
380  }
381  }
382  }
383  log_info("read_poisson_parameters: completed successfully");
384  return true;
385 }

532 static inline void expand_spike_recording_buffer(uint32_t n_spikes) {
533  uint32_t new_size = 8 + (n_spikes * spike_buffer_size);
534  timed_out_spikes *new_spikes = spin1_malloc(new_size);
535  if (new_spikes == NULL) {
536  log_error("Cannot reallocate spike buffer");
537  rt_error(RTE_SWERR);
538  }
539 
540  // bzero the new buffer
541  uint32_t *data = (uint32_t *) new_spikes;
542  for (uint32_t n = new_size >> 2; n > 0; n--) {
543  data[n - 1] = 0;
544  }
545 
546  // Copy over old buffer if we have it
547  if (spikes != NULL) {
548  spin1_memcpy(new_spikes, spikes,
549  8 + n_spike_buffers_allocated * spike_buffer_size);
550  sark_free(spikes);
551  }
552 
553  spikes = new_spikes;
554  n_spike_buffers_allocated = n_spikes;
555 }

588 static void process_fast_source(
589  index_t s_id, spike_source_t *source, uint timer_count) {
590  if ((time >= source->start_ticks) && (time < source->end_ticks)) {
591  // Get number of spikes to send this tick
592  uint32_t num_spikes = 0;
593 
594  // If sqrt_lambda has been set then use the Gaussian algorithm for
595  // faster sources
596  if (REAL_COMPARE(source->sqrt_lambda, >, ZERO)) {
597  profiler_write_entry_disable_irq_fiq(
598  PROFILER_ENTER | PROFILER_PROB_FUNC);
599  num_spikes = faster_spike_source_get_num_spikes(
600  source->sqrt_lambda);
601  profiler_write_entry_disable_irq_fiq(
602  PROFILER_EXIT | PROFILER_PROB_FUNC);
603  } else {
604  // Call the fast source Poisson algorithm
605  profiler_write_entry_disable_irq_fiq(
606  PROFILER_ENTER | PROFILER_PROB_FUNC);
607  num_spikes = fast_spike_source_get_num_spikes(
608  source->exp_minus_lambda);
609  profiler_write_entry_disable_irq_fiq(
610  PROFILER_EXIT | PROFILER_PROB_FUNC);
611  }
612 
613  log_debug("Generating %d spikes", num_spikes);
614 
615  // If there are any
616  if (num_spikes > 0) {
617  // Write spike to out spikes
618  mark_spike(s_id, num_spikes);
619 
620  // If no key has been given, do not send spikes to fabric
621  if (ssp_params.has_key) {
622  // Send spikes
623  const uint32_t spike_key = ssp_params.key | s_id;
624  tdma_processing_send_packet(
625  spike_key, num_spikes, WITH_PAYLOAD, timer_count);
626  }
627  }
628  }
629 }

635 static void process_slow_source(
636  index_t s_id, spike_source_t *source, uint timer_count) {
637  if ((time >= source->start_ticks) && (time < source->end_ticks)
638  && (source->mean_isi_ticks != 0)) {
639  uint32_t count = 0;
640  // Mark a spike while the "timer" is below the scale factor value
641  while (source->time_to_spike_ticks < ISI_SCALE_FACTOR) {
642  count++;
643 
644  // Update time to spike (note, this might not get us back above
645  // the scale factor, particularly if the mean_isi is smaller)
646  profiler_write_entry_disable_irq_fiq(
647  PROFILER_ENTER | PROFILER_PROB_FUNC);
648  source->time_to_spike_ticks +=
649  slow_spike_source_get_time_to_spike(source->mean_isi_ticks);
650  profiler_write_entry_disable_irq_fiq(
651  PROFILER_EXIT | PROFILER_PROB_FUNC);
652  }
653  if (count) {
654  // Write spike to out_spikes
655  mark_spike(s_id, count);
656 
657  // if no key has been given, do not send spike to fabric.
658  if (ssp_params.has_key) {
659  // Send package
660  tdma_processing_send_packet(
661  ssp_params.key | s_id, count, WITH_PAYLOAD, timer_count);
662  }
663  }
664 
665  // Now we have finished for this tick, subtract the scale factor
666  source->time_to_spike_ticks -= ISI_SCALE_FACTOR;
667  }
668 }

676 static void timer_callback(uint timer_count, UNUSED uint unused) {
677  profiler_write_entry_disable_irq_fiq(PROFILER_ENTER | PROFILER_TIMER);
678 
679  time++;
680 
681  log_debug("Timer tick %u", time);
682 
683  // If a fixed number of simulation ticks are specified and these have passed
684  if (simulation_is_finished()) {
685  // go into pause and resume state to avoid another tick
686  simulation_handle_pause_resume(resume_callback);
687 
688  // rewrite poisson params to SDRAM for reading out if needed
689  if (!store_poisson_parameters()) {
690  log_error("Failed to write poisson parameters to SDRAM");
691  rt_error(RTE_SWERR);
692  }
693 
694  profiler_write_entry_disable_irq_fiq(PROFILER_EXIT | PROFILER_TIMER);
695 
696  // Finalise any recordings that are in progress, writing back the final
697  // amounts of samples recorded to SDRAM
698  if (recording_flags > 0) {
699  recording_finalise();
700  }
701 
702  profiler_finalise();
703 
704  // Subtract 1 from the time so this tick gets done again on the next
705  // run
706  time--;
707  simulation_ready_to_read();
708  return;
709  }
710 
711  // Loop through spike sources
712  tdma_processing_reset_phase();
713  for (index_t s_id = 0; s_id < ssp_params.n_spike_sources; s_id++) {
714  // If this spike source is active this tick
715  spike_source_t *spike_source = &source[s_id];
716  if (spike_source->is_fast_source) {
717  process_fast_source(s_id, spike_source, timer_count);
718  } else {
719  process_slow_source(s_id, spike_source, timer_count);
720  }
721 
722  if ((time + 1) >= spike_source->next_ticks) {
723  log_debug("Moving to next rate at time %d", time);
724  read_next_rates(s_id);
725 #if LOG_LEVEL >= LOG_DEBUG
726  // print_spike_source(s_id);
727 #endif
728  }
729  }
730 
731  profiler_write_entry_disable_irq_fiq(PROFILER_EXIT | PROFILER_TIMER);
732 
733  // Record output spikes if required
734  if (recording_flags > 0) {
735  record_spikes(time);
736  }
737 }

743 void set_spike_source_rate(uint32_t id, REAL rate) {
744  if ((id < ssp_params.first_source_id) ||
745  (id - ssp_params.first_source_id >= ssp_params.n_spike_sources)) {
746  return;
747  }
748 
749  uint32_t sub_id = id - ssp_params.first_source_id;
750  REAL rate_per_tick = rate * ssp_params.seconds_per_tick;
751  log_debug("Setting rate of %u (%u) to %kHz (%k per tick)",
752  id, sub_id, rate, rate_per_tick);
753  spike_source_t *spike_source = &source[sub_id];
754 
755  if (rate_per_tick >= ssp_params.slow_rate_per_tick_cutoff) {
756  spike_source->is_fast_source = true;
757  if (rate_per_tick >= ssp_params.fast_rate_per_tick_cutoff) {
758  spike_source->sqrt_lambda = SQRT(rate_per_tick);
759  // warning: sqrtk is untested...
760  } else {
761  spike_source->exp_minus_lambda = (UFRACT) EXP(-rate_per_tick);
762  spike_source->sqrt_lambda = ZERO;
763  }
764  } else if (rate_per_tick == 0) {
765  spike_source->is_fast_source = false;
766  spike_source->mean_isi_ticks = 0;
767  spike_source->time_to_spike_ticks = 0;
768  } else {
769  spike_source->is_fast_source = false;
770  spike_source->mean_isi_ticks = (uint32_t) (ONE / rate_per_tick);
771  spike_source->time_to_spike_ticks =
772  slow_spike_source_get_time_to_spike(spike_source->mean_isi_ticks);
773  }
774 }
775 
779 static void multicast_packet_callback(uint key, uint payload) {
780  uint32_t id = key & ssp_params.set_rate_neuron_id_mask;
781  REAL rate = kbits(payload);
782  set_spike_source_rate(id, rate);
783 }

787  // Load DTCM data
788  if (!initialize()) {
789  log_error("Error in initialisation - exiting!");
790  rt_error(RTE_SWERR);
791  }
792 
793  // Start the time at "-1" so that the first tick will be 0
794  time = UINT32_MAX;
795 
796  // Set timer tick (in microseconds)
797  spin1_set_timer_tick(timer_period);
798 
799  // Register callback
800  spin1_callback_on(TIMER_TICK, timer_callback, TIMER);
801  spin1_callback_on(
802  MCPL_PACKET_RECEIVED, multicast_packet_callback, MULTICAST);
803 
804  simulation_run();
805 }
