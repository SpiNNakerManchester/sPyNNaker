42 static inline final_state_t plasticity_update_synapse(
43  const uint32_t time,
44  const uint32_t last_pre_time, const pre_trace_t last_pre_trace,
45  const pre_trace_t new_pre_trace, const uint32_t delay_dendritic,
46  const uint32_t delay_axonal, update_state_t current_state,
47  const post_event_history_t *post_event_history) {
48  // Apply axonal delay to time of last presynaptic spike
49  const uint32_t delayed_last_pre_time = last_pre_time + delay_axonal;
50 
51  // Get the post-synaptic window of events to be processed
52  const uint32_t window_begin_time =
53  (delayed_last_pre_time >= delay_dendritic)
54  ? (delayed_last_pre_time - delay_dendritic) : 0;
55  const uint32_t delayed_pre_time = time + delay_axonal;
56  const uint32_t window_end_time =
57  (delayed_pre_time >= delay_dendritic)
58  ? (delayed_pre_time - delay_dendritic) : 0;
59  post_event_window_t post_window = post_events_get_window_delayed(
60  post_event_history, window_begin_time, window_end_time);
61 
62  log_debug("\tPerforming deferred synapse update at time:%u", time);
63  log_debug("\t\tbegin_time:%u, end_time:%u - prev_time:%u (valid %u), num_events:%u",
64  window_begin_time, window_end_time, post_window.prev_time,
65  post_window.prev_time_valid, post_window.num_events);
66 
67 #if LOG_LEVEL >= LOG_DEBUG
68  print_event_history(post_event_history);
69  print_delayed_window_events(post_event_history, window_begin_time,
70  window_end_time, delay_dendritic);
71 #endif
72 
73  // Process events in post-synaptic window
74  while (post_window.num_events > 0) {
75  const uint32_t delayed_post_time = *post_window.next_time + delay_dendritic;
76 
77  log_debug("\t\tApplying post-synaptic event at delayed time:%u, pre:%u\n",
78  delayed_post_time, delayed_last_pre_time);
79 
80  // Apply spike to state
81  current_state = timing_apply_post_spike(
82  delayed_post_time, *post_window.next_trace, delayed_last_pre_time,
83  last_pre_trace, post_window.prev_time, post_window.prev_trace,
84  current_state);
85 
86  // Go onto next event
87  post_window = post_events_next(post_window);
88  }
89 
90  // Apply spike to state only if there has been a post spike ever
91  if (post_window.prev_time_valid) {
92  const uint32_t delayed_last_post = post_window.prev_time + delay_dendritic;
93  log_debug("\t\tApplying pre-synaptic event at time:%u last post time:%u\n",
94  delayed_pre_time, delayed_last_post);
95  current_state = timing_apply_pre_spike(
96  delayed_pre_time, new_pre_trace, delayed_last_pre_time, last_pre_trace,
97  delayed_last_post, post_window.prev_trace, current_state);
98  }
99 
100  // Return final synaptic word and weight
101  return synapse_structure_get_final_state(current_state);
102 }

152  update_state);

158  log_debug("nA) d: %2u, n = %3u)] - {%08x %08x}\n",
159  synapse_row_sparse_delay(control_word, synapse_type_index_bits, synapse_delay_mask),

195 static inline plastic_synapse_t process_plastic_synapse(
196  uint32_t control_word, uint32_t last_pre_time, pre_trace_t last_pre_trace,
197  pre_trace_t new_pre_trace, weight_t *ring_buffers, uint32_t time,
198  plastic_synapse_t synapse) {
199  fixed_stdp_synapse s = synapse_dynamics_stdp_get_fixed(control_word, time);
200 
201  // Create update state from the plastic synaptic word
202  update_state_t current_state = synapse_structure_get_update_state(
203  synapse, s.type);
204 
205  // Update the synapse state
206  uint32_t post_delay = s.delay_dendritic;
207  if (!params.backprop_delay) {
208  post_delay = 0;
209  }
210  final_state_t final_state = plasticity_update_synapse(
211  time, last_pre_time, last_pre_trace, new_pre_trace,
212  post_delay, s.delay_axonal, current_state,
213  &post_event_history[s.index]);
214 
215  // Add weight to ring-buffer entry
216  int32_t weight = synapse_structure_get_final_weight(final_state);
217  synapse_dynamics_stdp_update_ring_buffers(ring_buffers, s, weight);
218 
219  return synapse_structure_get_final_synaptic_word(final_state);
220 }

222 bool synapse_dynamics_process_plastic_synapses(
223  synapse_row_plastic_data_t *plastic_region_address,
224  synapse_row_fixed_part_t *fixed_region,
225  weight_t *ring_buffers, uint32_t time, bool *write_back) {
226  // Extract separate arrays of plastic synapses (from plastic region),
227  // Control words (from fixed region) and number of plastic synapses
228  plastic_synapse_t *plastic_words = plastic_region_address->synapses;
229  const control_t *control_words = synapse_row_plastic_controls(fixed_region);
230  size_t n_plastic_synapses = synapse_row_num_plastic_controls(fixed_region);
231 
232  num_plastic_pre_synaptic_events += n_plastic_synapses;
233 
234  // Get last pre-synaptic event from event history
235  const uint32_t last_pre_time = plastic_region_address->history.prev_time;
236  const pre_trace_t last_pre_trace = plastic_region_address->history.prev_trace;
237 
238  // Update pre-synaptic trace
239  log_debug("Adding pre-synaptic event to trace at time:%u", time);
240  plastic_region_address->history.prev_time = time;
241  plastic_region_address->history.prev_trace =
242  timing_add_pre_spike(time, last_pre_time, last_pre_trace);
243 
244  // Loop through plastic synapses
245  for (; n_plastic_synapses > 0; n_plastic_synapses--) {
246  // Get next control word (auto incrementing)
247  uint32_t control_word = *control_words++;
248 
249  plastic_words[0] = process_plastic_synapse(
250  control_word, last_pre_time, last_pre_trace,
251  plastic_region_address->history.prev_trace, ring_buffers, time,
252  plastic_words[0]);
253  plastic_words++;
254  }
255  *write_back = true;
256  return true;
257 }

259 bool synapse_dynamics_find_neuron(
260  uint32_t id, synaptic_row_t row, weight_t *weight, uint16_t *delay,
261  uint32_t *offset, uint32_t *synapse_type) {
262  synapse_row_fixed_part_t *fixed_region = synapse_row_fixed_region(row);
263  const synapse_row_plastic_data_t *plastic_data = (void *)
264  synapse_row_plastic_region(row);
265  const plastic_synapse_t *plastic_words = plastic_data->synapses;
266  const control_t *control_words = synapse_row_plastic_controls(fixed_region);
267  const size_t n_plastic_synapses = synapse_row_num_plastic_controls(fixed_region);
268 
269  // Loop through plastic synapses
270  for (size_t plastic_synapse = n_plastic_synapses; plastic_synapse > 0;
271  plastic_synapse--) {
272  // Take the weight anyway as this updates the plastic words
273  *weight = synapse_structure_get_weight(*plastic_words++);
274 
275  // Check if index is the one I'm looking for
276  uint32_t control_word = *control_words++;
277  if (synapse_row_sparse_index(control_word, synapse_index_mask) == id) {
278  *offset = n_plastic_synapses - plastic_synapse;
279  *delay = synapse_row_sparse_delay(control_word,
280  synapse_type_index_bits, synapse_delay_mask);
281  *synapse_type = synapse_row_sparse_type(
282  control_word, synapse_index_bits, synapse_type_mask);
283  return true;
284  }
285  }
286 
287  return false;
288 }

311 bool synapse_dynamics_add_neuron(uint32_t id, synaptic_row_t row,
312  weight_t weight, uint32_t delay, uint32_t type) {
313  synapse_row_fixed_part_t *fixed_region = synapse_row_fixed_region(row);
314  synapse_row_plastic_data_t *plastic_data = synapse_row_plastic_region(row);
315  plastic_synapse_t *plastic_words = plastic_data->synapses;
316  plastic_synapse_t new_weight = synapse_structure_create_synapse(weight);
317  control_t new_control = control_conversion(id, delay, type);
318 
319  control_t *control_words = synapse_row_plastic_controls(fixed_region);
320  int32_t plastic_synapse = synapse_row_num_plastic_controls(fixed_region);
321 
322  // Add weight at offset
323  plastic_words[plastic_synapse] = new_weight;
324 
325  // Add control word at offset
326  control_words[plastic_synapse] = new_control;
327 
328  // Increment FP
329  fixed_region->num_plastic++;
330  return true;
