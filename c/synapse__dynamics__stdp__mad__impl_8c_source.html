141 static inline final_state_t plasticity_update_synapse(
142  const uint32_t time,
143  const uint32_t last_pre_time, const pre_trace_t last_pre_trace,
144  const pre_trace_t new_pre_trace, const uint32_t delay_dendritic,
145  const uint32_t delay_axonal, update_state_t current_state,
146  const post_event_history_t *post_event_history) {
147  // Apply axonal delay to time of last presynaptic spike
148  const uint32_t delayed_last_pre_time = last_pre_time + delay_axonal;
149 
150  // Get the post-synaptic window of events to be processed
151  const uint32_t window_begin_time =
152  (delayed_last_pre_time >= delay_dendritic)
153  ? (delayed_last_pre_time - delay_dendritic) : 0;
154  const uint32_t delayed_pre_time = time + delay_axonal;
155  const uint32_t window_end_time =
156  (delayed_pre_time >= delay_dendritic)
157  ? (delayed_pre_time - delay_dendritic) : 0;
158  post_event_window_t post_window = post_events_get_window_delayed(
159  post_event_history, window_begin_time, window_end_time);
160 
161  log_debug("\tPerforming deferred synapse update at time:%u", time);
162  log_debug("\t\tbegin_time:%u, end_time:%u - prev_time:%u (valid %u), num_events:%u",
163  window_begin_time, window_end_time, post_window.prev_time,
164  post_window.prev_time_valid, post_window.num_events);
165 
166 #if LOG_LEVEL >= LOG_DEBUG
167  print_event_history(post_event_history);
168  print_delayed_window_events(post_event_history, window_begin_time,
169  window_end_time, delay_dendritic);
170 #endif
171 
172  // Process events in post-synaptic window
173  while (post_window.num_events > 0) {
174  const uint32_t delayed_post_time = *post_window.next_time + delay_dendritic;
175 
176  log_debug("\t\tApplying post-synaptic event at delayed time:%u, pre:%u\n",
177  delayed_post_time, delayed_last_pre_time);
178 
179  // Apply spike to state
180  current_state = timing_apply_post_spike(
181  delayed_post_time, *post_window.next_trace, delayed_last_pre_time,
182  last_pre_trace, post_window.prev_time, post_window.prev_trace,
183  current_state);
184 
185  // Go onto next event
186  post_window = post_events_next(post_window);
187  }
188 
189  // Apply spike to state only if there has been a post spike ever
190  if (post_window.prev_time_valid) {
191  const uint32_t delayed_last_post = post_window.prev_time + delay_dendritic;
192  log_debug("\t\tApplying pre-synaptic event at time:%u last post time:%u\n",
193  delayed_pre_time, delayed_last_post);
194  current_state = timing_apply_pre_spike(
195  delayed_pre_time, new_pre_trace, delayed_last_pre_time, last_pre_trace,
196  delayed_last_post, post_window.prev_trace, current_state);
197  }
198 
199  // Return final synaptic word and weight
200  return synapse_structure_get_final_state(current_state);
201 }

244  log_debug("nA) d: %2u, %s, n = %3u)] - {%08x %08x}\n",
245  synapse_row_sparse_delay(control_word, synapse_type_index_bits),
246  synapse_types_get_type_char(synapse_type),

267 bool synapse_dynamics_initialise(
268  address_t address, uint32_t n_neurons, uint32_t n_synapse_types,
269  uint32_t *ring_buffer_to_input_buffer_left_shifts) {
270  stdp_params *sdram_params = (stdp_params *) address;
271  spin1_memcpy(&params, sdram_params, sizeof(stdp_params));
272  address = (address_t) &sdram_params[1];
273 
274  // Load timing dependence data
275  address_t weight_region_address = timing_initialise(address);
276  if (address == NULL) {
277  return false;
278  }
279 
280  // Load weight dependence data
281  address_t weight_result = weight_initialise(
282  weight_region_address, n_synapse_types,
283  ring_buffer_to_input_buffer_left_shifts);
284  if (weight_result == NULL) {
285  return false;
286  }
287 
288  post_event_history = post_events_init_buffers(n_neurons);
289  if (post_event_history == NULL) {
290  return false;
291  }
292 
293  uint32_t n_neurons_power_2 = n_neurons;
294  uint32_t log_n_neurons = 1;
295  if (n_neurons != 1) {
296  if (!is_power_of_2(n_neurons)) {
297  n_neurons_power_2 = next_power_of_2(n_neurons);
298  }
299  log_n_neurons = ilog_2(n_neurons_power_2);
300  }
301 
302  uint32_t n_synapse_types_power_2 = n_synapse_types;
303  uint32_t log_n_synapse_types = 1;
304  if (n_synapse_types != 1) {
305  if (!is_power_of_2(n_synapse_types)) {
306  n_synapse_types_power_2 = next_power_of_2(n_synapse_types);
307  }
308  log_n_synapse_types = ilog_2(n_synapse_types_power_2);
309  }
310 
311  synapse_type_index_bits = log_n_neurons + log_n_synapse_types;
312  synapse_type_index_mask = (1 << synapse_type_index_bits) - 1;
313  synapse_index_bits = log_n_neurons;
314  synapse_index_mask = (1 << synapse_index_bits) - 1;
315  synapse_delay_index_type_bits =
316  SYNAPSE_DELAY_BITS + synapse_type_index_bits;
317  synapse_type_mask = (1 << log_n_synapse_types) - 1;
318  return true;
319 }

321 bool synapse_dynamics_process_plastic_synapses(
322  address_t plastic_region_address, address_t fixed_region_address,
323  weight_t *ring_buffers, uint32_t time) {
324  synapse_row_plastic_data_t *plastic_data =
325  (synapse_row_plastic_data_t *) plastic_region_address;
326  // Extract separate arrays of plastic synapses (from plastic region),
327  // Control words (from fixed region) and number of plastic synapses
328  plastic_synapse_t *plastic_words = plastic_data->synapses;
329  const control_t *control_words =
330  synapse_row_plastic_controls(fixed_region_address);
331  size_t plastic_synapse =
332  synapse_row_num_plastic_controls(fixed_region_address);
333 
334  num_plastic_pre_synaptic_events += plastic_synapse;
335 
336  // Get last pre-synaptic event from event history
337  const uint32_t last_pre_time = plastic_data->history.prev_time;
338  const pre_trace_t last_pre_trace = plastic_data->history.prev_trace;
339 
340  // Update pre-synaptic trace
341  log_debug("Adding pre-synaptic event to trace at time:%u", time);
342  plastic_data->history.prev_time = time;
343  plastic_data->history.prev_trace =
344  timing_add_pre_spike(time, last_pre_time, last_pre_trace);
345 
346  // Loop through plastic synapses
347  for (; plastic_synapse > 0; plastic_synapse--) {
348  // Get next control word (auto incrementing)
349  uint32_t control_word = *control_words++;
350 
351  // Extract control-word components
352  // **NOTE** cunningly, control word is just the same as lower
353  // 16-bits of 32-bit fixed synapse so same functions can be used
354  uint32_t delay_axonal = sparse_axonal_delay(control_word);
355  uint32_t delay_dendritic = synapse_row_sparse_delay(
356  control_word, synapse_type_index_bits);
357  uint32_t type = synapse_row_sparse_type(
358  control_word, synapse_index_bits, synapse_type_mask);
359  uint32_t index =
360  synapse_row_sparse_index(control_word, synapse_index_mask);
361  uint32_t type_index = synapse_row_sparse_type_index(
362  control_word, synapse_type_index_mask);
363 
364  // Create update state from the plastic synaptic word
365  update_state_t current_state =
366  synapse_structure_get_update_state(*plastic_words, type);
367 
368  // Convert into ring buffer offset
369  uint32_t ring_buffer_index = synapses_get_ring_buffer_index_combined(
370  delay_axonal + delay_dendritic + time, type_index,
371  synapse_type_index_bits);
372 
373  // Update the synapse state
374  uint32_t post_delay = delay_dendritic;
375  if (!params.backprop_delay) {
376  post_delay = 0;
377  }
378  final_state_t final_state = plasticity_update_synapse(
379  time, last_pre_time, last_pre_trace,
380  plastic_data->history.prev_trace,
381  post_delay, delay_axonal, current_state,
382  &post_event_history[index]);
383 
384  // Add weight to ring-buffer entry
385  // **NOTE** Dave suspects that this could be a
386  // potential location for overflow
387 
388  uint32_t accumulation = ring_buffers[ring_buffer_index] +
389  synapse_structure_get_final_weight(final_state);
390 
391  uint32_t sat_test = accumulation & 0x10000;
392  if (sat_test) {
393  accumulation = sat_test - 1;
394  plastic_saturation_count++;
395  }
396 
397  ring_buffers[ring_buffer_index] = accumulation;
398 
399  // Write back updated synaptic word to plastic region
400  *plastic_words++ =
401  synapse_structure_get_final_synaptic_word(final_state);
402  }
403  return true;
404 }

432 bool synapse_dynamics_find_neuron(
433  uint32_t id, address_t row, weight_t *weight, uint16_t *delay,
434  uint32_t *offset, uint32_t *synapse_type) {
435  address_t fixed_region = synapse_row_fixed_region(row);
436  synapse_row_plastic_data_t *plastic_data = (void *)
437  synapse_row_plastic_region(row);
438  const plastic_synapse_t *plastic_words = plastic_data->synapses;
439  const control_t *control_words = synapse_row_plastic_controls(fixed_region);
440  int32_t plastic_synapse = synapse_row_num_plastic_controls(fixed_region);
441 
442  // Loop through plastic synapses
443  for (; plastic_synapse > 0; plastic_synapse--) {
444  // Take the weight anyway as this updates the plastic words
445  *weight = synapse_structure_get_weight(*plastic_words++);
446 
447  // Check if index is the one I'm looking for
448  uint32_t control_word = *control_words++;
449  if (synapse_row_sparse_index(control_word, synapse_index_mask) == id) {
450  *offset = synapse_row_num_plastic_controls(fixed_region) - plastic_synapse;
451  *delay = synapse_row_sparse_delay(control_word, synapse_type_index_bits);
452  *synapse_type = synapse_row_sparse_type(
453  control_word, synapse_index_bits, synapse_type_mask);
454  return true;
455  }
456  }
457 
458  return false;
459 }

487 static inline control_t control_conversion(
488  uint32_t id, uint32_t delay, uint32_t type) {
489  control_t new_control =
490  (delay & ((1 << SYNAPSE_DELAY_BITS) - 1)) << synapse_type_index_bits;
491  new_control |= (type & ((1 << synapse_type_index_bits) - 1)) << synapse_index_bits;
492  new_control |= id & ((1 << synapse_index_bits) - 1);
493  return new_control;
494 }
