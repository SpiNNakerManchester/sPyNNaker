89 static uint32_t connection_generator_all_to_all_generate(
90  void *generator, UNUSED uint32_t pre_slice_start,
91  UNUSED uint32_t pre_slice_count,
92  uint32_t pre_neuron_index, uint32_t post_slice_start,
93  uint32_t post_slice_count, uint32_t max_row_length, uint16_t *indices) {
94  log_debug("Generating for %u", pre_neuron_index);
95 
96  struct all_to_all *obj = generator;
97 
98  // If no space, generate nothing
99  if (max_row_length < 1) {
100  return 0;
101  }
102 
103  // If not in the pre-population view range, then don't generate
104  if ((pre_neuron_index < obj->pre_lo) ||
105  (pre_neuron_index > obj->pre_hi)) {
106  return 0;
107  }
108 
109  // Add a connection to this pre-neuron for each post-neuron...
110  uint32_t n_conns = 0;
111  for (uint32_t i = 0; i < post_slice_count; i++) {
112  // ... unless this is a self connection and these are disallowed
113  if (!obj->allow_self_connections &&
114  (pre_neuron_index == post_slice_start + i)) {
115  log_debug("Not generating for post %u", post_slice_start + i);
116  continue;
117  }
118  // ... or if the value is not in the range of the post-population view
119  if ((i + post_slice_start < obj->post_lo) || (i + post_slice_start > obj->post_hi)) {
120  continue;
121  }
122  indices[n_conns++] = i;
123  }
124 
125  return n_conns;
126 }
