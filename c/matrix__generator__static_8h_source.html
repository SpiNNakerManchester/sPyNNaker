93 static uint32_t build_static_word(
94  uint16_t weight, uint16_t delay, uint32_t type,
95  uint16_t post_index, uint32_t synapse_type_bits,
96  uint32_t synapse_index_bits) {
97  uint32_t synapse_index_mask = (1 << synapse_index_bits) - 1;
98  uint32_t synapse_type_mask = (1 << synapse_type_bits) - 1;
99 
100  uint32_t wrd = post_index & synapse_index_mask;
101  wrd |= (type & synapse_type_mask) << synapse_index_bits;
102  wrd |= (delay & SYNAPSE_DELAY_MASK) <<
103  (synapse_index_bits + synapse_type_bits);
104  wrd |= (weight & SYNAPSE_WEIGHT_MASK) << SYNAPSE_WEIGHT_SHIFT;
105  return wrd;
106 }

129 static void matrix_generator_static_write_row(
130  void *generator,
131  address_t synaptic_matrix, address_t delayed_synaptic_matrix,
132  uint32_t n_pre_neurons, uint32_t pre_neuron_index,
133  uint32_t max_row_n_words, uint32_t max_delayed_row_n_words,
134  uint32_t synapse_type_bits, uint32_t synapse_index_bits,
135  uint32_t synapse_type, uint32_t n_synapses,
136  uint16_t *indices, uint16_t *delays, uint16_t *weights,
137  uint32_t max_stage) {
138  use(generator);
139 
140  log_debug("Max stage = %u", max_stage);
141 
142  // Row address and position for each possible delay stage (including no
143  // delay stage)
144  address_t row_address[max_stage];
145 
146  // The space available on each row
147  uint16_t space[max_stage];
148 
149  // The normal row position and space available - might be 0 if all delayed
150  row_address[0] = NULL;
151  space[0] = max_row_n_words;
152  if (synaptic_matrix != NULL) {
153  row_address[0] =
154  &synaptic_matrix[pre_neuron_index * (max_row_n_words + 3)];
155  }
156  log_debug("row[0] = 0x%08x", row_address[0]);
157 
158  // The delayed row positions and space available
159  if (delayed_synaptic_matrix != NULL) {
160  address_t delayed_address = &delayed_synaptic_matrix[
161  pre_neuron_index * (max_delayed_row_n_words + 3)];
162  uint32_t single_matrix_size =
163  n_pre_neurons * (max_delayed_row_n_words + 3);
164  for (uint32_t i = 1; i < max_stage; i++) {
165  row_address[i] = &delayed_address[single_matrix_size * (i - 1)];
166  space[i] = max_delayed_row_n_words;
167  log_debug("row[%u] = 0x%08x", i, row_address[i]);
168  }
169  } else {
170  for (uint32_t i = 1; i < max_stage; i++) {
171  row_address[i] = NULL;
172  space[i] = 0;
173  log_debug("row[%u] = 0x%08x", i, row_address[i]);
174  }
175  }
176 
177  // The address to write synapses to on each stage
178  address_t write_address[max_stage];
179  for (uint32_t i = 0; i < max_stage; i++) {
180  if (row_address[i] != NULL) {
181  log_debug("Row size at 0x%08x for stage %u",
182  &row_address[i][STATIC_FIXED_FIXED_SIZE], i);
183  row_address[i][STATIC_FIXED_FIXED_SIZE] = 0;
184  row_address[i][STATIC_PLASTIC_PLASTIC_SIZE] = 0;
185  row_address[i][STATIC_FIXED_PLASTIC_SIZE] = 0;
186  write_address[i] = &row_address[i][STATIC_FIXED_FIXED_OFFSET];
187  } else {
188  write_address[i] = NULL;
189  }
190  log_debug("write[%u] = 0x%08x", i, write_address[i]);
191  }
192 
193 
194  // Go through the synapses
195  for (uint32_t synapse = 0; synapse < n_synapses; synapse++) {
196  // Post-neuron index
197  uint32_t post_index = indices[synapse];
198 
199  // Weight
200  uint16_t weight = weights[synapse];
201 
202  // Work out the delay stage and final value
203  struct delay_value delay = get_delay(delays[synapse], max_stage);
204  if (write_address[delay.stage] == NULL) {
205  log_error("Delay stage %u has not been initialised; raw delay = %u,"
206  " delay = %u, max stage = %u", delay.stage, delays[synapse],
207  delay.delay, max_stage);
208  rt_error(RTE_SWERR);
209  }
210 
211  // Avoid errors when rows are full
212  if (space[delay.stage] == 0) {
213  log_warning("Row for delay stage %u is full - word not added!",
214  delay.stage);
215  continue;
216  }
217 
218  // Build synaptic word
219  uint32_t word = build_static_word(
220  weight, delay.delay, synapse_type, post_index, synapse_type_bits,
221  synapse_index_bits);
222 
223  // Write the word
224  log_debug("Writing word to 0x%08x", &write_address[delay.stage][0]);
225  write_address[delay.stage][0] = word;
226  write_address[delay.stage] = &write_address[delay.stage][1];
227 
228  // Increment the size of the current row
229  row_address[delay.stage][STATIC_FIXED_FIXED_SIZE]++;
230  space[delay.stage]--;
231  }
232 }
