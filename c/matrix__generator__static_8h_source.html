111 static uint32_t build_static_word(
112  uint16_t weight, uint16_t delay, uint32_t type,
113  uint16_t post_index, uint32_t synapse_type_bits,
114  uint32_t synapse_index_bits, uint32_t delay_bits) {
115  uint32_t synapse_index_mask = (1 << synapse_index_bits) - 1;
116  uint32_t synapse_type_mask = (1 << synapse_type_bits) - 1;
117  uint32_t synapse_delay_mask = (1 << delay_bits) - 1;
118 
119  uint32_t wrd = post_index & synapse_index_mask;
120  wrd |= (type & synapse_type_mask) << synapse_index_bits;
121  wrd |= (delay & synapse_delay_mask) <<
122  (synapse_index_bits + synapse_type_bits);
123  wrd |= (weight & SYNAPSE_WEIGHT_MASK) << SYNAPSE_WEIGHT_SHIFT;
124  return wrd;
125 }

134 static void *matrix_generator_static_initialize(void **region,
135  void *synaptic_matrix) {
136  matrix_genetator_static_data_t *sdram_data = *region;
137  *region = &sdram_data[1];
138  matrix_genetator_static_data_t *data = spin1_malloc(
139  sizeof(matrix_genetator_static_data_t));
140  *data = *sdram_data;
141 
142  // Offsets are in words
143  uint32_t *syn_mat = synaptic_matrix;
144  if (data->synaptic_matrix_offset != 0xFFFFFFFF) {
145  data->synaptic_matrix = &(syn_mat[data->synaptic_matrix_offset]);
146  setup_rows(data->synaptic_matrix, data->n_pre_neurons,
147  data->max_row_n_words);
148  } else {
149  data->synaptic_matrix = NULL;
150  }
151  if (data->delayed_matrix_offset != 0xFFFFFFFF) {
152  data->delayed_synaptic_matrix = &(syn_mat[data->delayed_matrix_offset]);
153  setup_rows(data->delayed_synaptic_matrix,
154  data->n_pre_neurons * (data->max_stage - 1),
155  data->max_delayed_row_n_words);
156  } else {
157  data->delayed_synaptic_matrix = NULL;
158  }
159 
160  return data;
161 }

182 static bool matrix_generator_static_write_synapse(void *generator,
183  uint32_t pre_index, uint16_t post_index, accum weight, uint16_t delay,
184  unsigned long accum weight_scale) {
185  matrix_genetator_static_data_t *data = generator;
186  struct delay_value delay_and_stage = get_delay(delay, data->max_stage,
187  data->max_delay_per_stage);
188  static_row_t *row;
189  uint32_t pos;
190  if (delay_and_stage.stage == 0) {
191  row = get_row(data->synaptic_matrix, data->max_row_n_words, pre_index);
192  pos = row->fixed_fixed_size;
193  if (pos >= data->max_row_n_words) {
194  log_warning("Row %u at 0x%08x of matrix 0x%08x is already full (%u of %u)",
195  pre_index, row, data->synaptic_matrix, pos, data->max_row_n_words);
196  return false;
197  }
198  } else {
199  row = get_delay_row(data->delayed_synaptic_matrix,
200  data->max_delayed_row_n_words, pre_index, delay_and_stage.stage,
201  data->n_pre_neurons_per_core, data->max_stage, data->n_pre_neurons);
202  pos = row->fixed_fixed_size;
203  if (pos >= data->max_delayed_row_n_words) {
204  log_warning("Row %u, stage %u at 0x%08x of delayed matrix 0x%08x"
205  "is already full (%u of %u)",
206  pre_index, delay_and_stage.stage, row,
207  data->delayed_synaptic_matrix, pos, data->max_delayed_row_n_words);
208  return false;
209  }
210  }
211 
212  uint16_t scaled_weight = rescale_weight(weight, weight_scale);
213 
214  row->fixed_fixed_size = pos + 1;
215  row->fixed_fixed_data[pos] = build_static_word(scaled_weight, delay_and_stage.delay,
216  data->synapse_type, post_index, data->synapse_type_bits,
217  data->synapse_index_bits, data->delay_bits);
218  return true;
219 }
