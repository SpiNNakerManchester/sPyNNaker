76 static uint32_t build_static_word(
77  uint16_t weight, uint16_t delay, uint32_t type,
78  uint16_t post_index, uint32_t synapse_type_bits,
79  uint32_t synapse_index_bits, uint32_t delay_bits) {
80  uint32_t synapse_index_mask = (1 << synapse_index_bits) - 1;
81  uint32_t synapse_type_mask = (1 << synapse_type_bits) - 1;
82  uint32_t synapse_delay_mask = (1 << delay_bits) - 1;
83 
84  uint32_t wrd = post_index & synapse_index_mask;
85  wrd |= (type & synapse_type_mask) << synapse_index_bits;
86  wrd |= (delay & synapse_delay_mask) <<
87  (synapse_index_bits + synapse_type_bits);
88  wrd |= (weight & SYNAPSE_WEIGHT_MASK) << SYNAPSE_WEIGHT_SHIFT;
89  return wrd;
90 }

114 static void matrix_generator_static_write_row(
115  UNUSED void *generator,
116  address_t synaptic_matrix, address_t delayed_synaptic_matrix,
117  uint32_t n_pre_neurons, uint32_t pre_neuron_index,
118  uint32_t max_row_n_words, uint32_t max_delayed_row_n_words,
119  uint32_t synapse_type_bits, uint32_t synapse_index_bits,
120  uint32_t synapse_type, uint32_t n_synapses,
121  uint16_t *indices, uint16_t *delays, uint16_t *weights,
122  uint32_t max_stage, uint32_t max_delay_per_stage) {
123  log_debug("Max stage = %u", max_stage);
124 
125  // Row address and position for each possible delay stage (including no
126  // delay stage)
127  static_row_t row[max_stage];
128 
129  // The space available on each row
130  uint16_t space[max_stage];
131 
132  // The normal row position and space available - might be 0 if all delayed
133  row[0] = NULL;
134  space[0] = max_row_n_words;
135  if (synaptic_matrix != NULL) {
136  row[0] = (static_row_t)
137  &synaptic_matrix[pre_neuron_index * (max_row_n_words + 3)];
138  }
139  log_debug("row[0] = 0x%08x", row[0]);
140 
141  // The delayed row positions and space available
142  if (delayed_synaptic_matrix != NULL) {
143  address_t delayed_address = &delayed_synaptic_matrix[
144  pre_neuron_index * (max_delayed_row_n_words + 3)];
145  uint32_t single_matrix_size =
146  n_pre_neurons * (max_delayed_row_n_words + 3);
147  for (uint32_t i = 1; i < max_stage; i++) {
148  row[i] = (static_row_t)
149  &delayed_address[single_matrix_size * (i - 1)];
150  space[i] = max_delayed_row_n_words;
151  log_debug("row[%u] = 0x%08x", i, row[i]);
152  }
153  } else {
154  for (uint32_t i = 1; i < max_stage; i++) {
155  row[i] = NULL;
156  space[i] = 0;
157  log_debug("row[%u] = 0x%08x", i, row[i]);
158  }
159  }
160 
161  // The address to write synapses to on each stage
162  address_t write_address[max_stage];
163  for (uint32_t i = 0; i < max_stage; i++) {
164  if (row[i] != NULL) {
165  log_debug("Row size at 0x%08x for stage %u",
166  &row[i]->fixed_fixed_size, i);
167  row[i]->fixed_fixed_size = 0;
168  row[i]->plastic_plastic_size = 0;
169  row[i]->fixed_plastic_size = 0;
170  write_address[i] = row[i]->fixed_fixed_data;
171  } else {
172  write_address[i] = NULL;
173  }
174  log_debug("write[%u] = 0x%08x", i, write_address[i]);
175  }
176 
177  uint32_t max_delay_power_2 = max_delay_per_stage;
178  uint32_t log_max_delay = 1;
179  if (max_delay_power_2 != 1) {
180  if (!is_power_of_2(max_delay_power_2)) {
181  max_delay_power_2 = next_power_of_2(max_delay_power_2);
182  }
183  log_max_delay = ilog_2(max_delay_power_2);
184  }
185 
186 
187  // Go through the synapses
188  for (uint32_t synapse = 0; synapse < n_synapses; synapse++) {
189  // Post-neuron index
190  uint32_t post_index = indices[synapse];
191 
192  // Weight
193  uint16_t weight = weights[synapse];
194 
195  // Work out the delay stage and final value
196  struct delay_value delay = get_delay(
197  delays[synapse], max_stage, max_delay_per_stage);
198  if (write_address[delay.stage] == NULL) {
199  log_error("Delay stage %u has not been initialised; raw delay = %u,"
200  " delay = %u, max stage = %u", delay.stage, delays[synapse],
201  delay.delay, max_stage);
202  rt_error(RTE_SWERR);
203  }
204 
205  // Avoid errors when rows are full
206  if (space[delay.stage] == 0) {
207  log_warning("Row for delay stage %u is full - word not added!",
208  delay.stage);
209  continue;
210  }
211 
212  // Build synaptic word
213  uint32_t word = build_static_word(
214  weight, delay.delay, synapse_type, post_index, synapse_type_bits,
215  synapse_index_bits, log_max_delay);
216 
217  // Write the word
218  log_debug("Writing word to 0x%08x", &write_address[delay.stage][0]);
219  write_address[delay.stage][0] = word;
220  write_address[delay.stage] = &write_address[delay.stage][1];
221 
222  // Increment the size of the current row
223  row[delay.stage]->fixed_fixed_size++;
224  space[delay.stage]--;
225  }
226 }
