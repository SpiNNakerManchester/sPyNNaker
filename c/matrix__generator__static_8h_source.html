91 static uint32_t build_static_word(
92  uint16_t weight, uint16_t delay, uint32_t type,
93  uint16_t post_index, uint32_t synapse_type_bits,
94  uint32_t synapse_index_bits) {
95  uint32_t synapse_index_mask = (1 << synapse_index_bits) - 1;
96  uint32_t synapse_type_mask = (1 << synapse_type_bits) - 1;
97 
98  uint32_t wrd = post_index & synapse_index_mask;
99  wrd |= (type & synapse_type_mask) << synapse_index_bits;
100  wrd |= (delay & SYNAPSE_DELAY_MASK) <<
101  (synapse_index_bits + synapse_type_bits);
102  wrd |= (weight & SYNAPSE_WEIGHT_MASK) << SYNAPSE_WEIGHT_SHIFT;
103  return wrd;
104 }

127 static void matrix_generator_static_write_row(
128  UNUSED void *generator,
129  address_t synaptic_matrix, address_t delayed_synaptic_matrix,
130  uint32_t n_pre_neurons, uint32_t pre_neuron_index,
131  uint32_t max_row_n_words, uint32_t max_delayed_row_n_words,
132  uint32_t synapse_type_bits, uint32_t synapse_index_bits,
133  uint32_t synapse_type, uint32_t n_synapses,
134  uint16_t *indices, uint16_t *delays, uint16_t *weights,
135  uint32_t max_stage) {
136  log_debug("Max stage = %u", max_stage);
137 
138  // Row address and position for each possible delay stage (including no
139  // delay stage)
140  address_t row_address[max_stage];
141 
142  // The space available on each row
143  uint16_t space[max_stage];
144 
145  // The normal row position and space available - might be 0 if all delayed
146  row_address[0] = NULL;
147  space[0] = max_row_n_words;
148  if (synaptic_matrix != NULL) {
149  row_address[0] =
150  &synaptic_matrix[pre_neuron_index * (max_row_n_words + 3)];
151  }
152  log_debug("row[0] = 0x%08x", row_address[0]);
153 
154  // The delayed row positions and space available
155  if (delayed_synaptic_matrix != NULL) {
156  address_t delayed_address = &delayed_synaptic_matrix[
157  pre_neuron_index * (max_delayed_row_n_words + 3)];
158  uint32_t single_matrix_size =
159  n_pre_neurons * (max_delayed_row_n_words + 3);
160  for (uint32_t i = 1; i < max_stage; i++) {
161  row_address[i] = &delayed_address[single_matrix_size * (i - 1)];
162  space[i] = max_delayed_row_n_words;
163  log_debug("row[%u] = 0x%08x", i, row_address[i]);
164  }
165  } else {
166  for (uint32_t i = 1; i < max_stage; i++) {
167  row_address[i] = NULL;
168  space[i] = 0;
169  log_debug("row[%u] = 0x%08x", i, row_address[i]);
170  }
171  }
172 
173  // The address to write synapses to on each stage
174  address_t write_address[max_stage];
175  for (uint32_t i = 0; i < max_stage; i++) {
176  if (row_address[i] != NULL) {
177  log_debug("Row size at 0x%08x for stage %u",
178  &row_address[i][STATIC_FIXED_FIXED_SIZE], i);
179  row_address[i][STATIC_FIXED_FIXED_SIZE] = 0;
180  row_address[i][STATIC_PLASTIC_PLASTIC_SIZE] = 0;
181  row_address[i][STATIC_FIXED_PLASTIC_SIZE] = 0;
182  write_address[i] = &row_address[i][STATIC_FIXED_FIXED_OFFSET];
183  } else {
184  write_address[i] = NULL;
185  }
186  log_debug("write[%u] = 0x%08x", i, write_address[i]);
187  }
188 
189 
190  // Go through the synapses
191  for (uint32_t synapse = 0; synapse < n_synapses; synapse++) {
192  // Post-neuron index
193  uint32_t post_index = indices[synapse];
194 
195  // Weight
196  uint16_t weight = weights[synapse];
197 
198  // Work out the delay stage and final value
199  struct delay_value delay = get_delay(delays[synapse], max_stage);
200  if (write_address[delay.stage] == NULL) {
201  log_error("Delay stage %u has not been initialised; raw delay = %u,"
202  " delay = %u, max stage = %u", delay.stage, delays[synapse],
203  delay.delay, max_stage);
204  rt_error(RTE_SWERR);
205  }
206 
207  // Avoid errors when rows are full
208  if (space[delay.stage] == 0) {
209  log_warning("Row for delay stage %u is full - word not added!",
210  delay.stage);
211  continue;
212  }
213 
214  // Build synaptic word
215  uint32_t word = build_static_word(
216  weight, delay.delay, synapse_type, post_index, synapse_type_bits,
217  synapse_index_bits);
218 
219  // Write the word
220  log_debug("Writing word to 0x%08x", &write_address[delay.stage][0]);
221  write_address[delay.stage][0] = word;
222  write_address[delay.stage] = &write_address[delay.stage][1];
223 
224  // Increment the size of the current row
225  row_address[delay.stage][STATIC_FIXED_FIXED_SIZE]++;
226  space[delay.stage]--;
227  }
228 }
