148 static inline void print_ring_buffers(uint32_t time) {
149 #if LOG_LEVEL >= LOG_DEBUG
150  log_debug("Ring Buffer at %u\n", time);

153  for (uint32_t t = 0; t < n_synapse_types; t++) {

200 static inline void process_fixed_synapses(
201  address_t fixed_region_address, uint32_t time) {
202  register uint32_t *synaptic_words =
203  synapse_row_fixed_weight_controls(fixed_region_address);
204  register uint32_t fixed_synapse =
205  synapse_row_num_fixed_synapses(fixed_region_address);
206 
207  num_fixed_pre_synaptic_events += fixed_synapse;
208 
209  for (; fixed_synapse > 0; fixed_synapse--) {
210  // Get the next 32 bit word from the synaptic_row
211  // (should auto increment pointer in single instruction)
212  uint32_t synaptic_word = *synaptic_words++;
213 
214  // Extract components from this word
215  uint32_t delay =
216  synapse_row_sparse_delay(synaptic_word, synapse_type_index_bits);
217  uint32_t combined_synapse_neuron_index = synapse_row_sparse_type_index(
218  synaptic_word, synapse_type_index_mask);
219  uint32_t weight = synapse_row_sparse_weight(synaptic_word);
220 
221  // Convert into ring buffer offset
222  uint32_t ring_buffer_index = synapses_get_ring_buffer_index_combined(
223  delay + time, combined_synapse_neuron_index,
224  synapse_type_index_bits);
225 
226  // Add weight to current ring buffer value
227  uint32_t accumulation = ring_buffers[ring_buffer_index] + weight;
228 
229  // If 17th bit is set, saturate accumulator at UINT16_MAX (0xFFFF)
230  // **NOTE** 0x10000 can be expressed as an ARM literal,
231  // but 0xFFFF cannot. Therefore, we use (0x10000 - 1)
232  // to obtain this value
233  uint32_t sat_test = accumulation & 0x10000;
234  if (sat_test) {
235  accumulation = sat_test - 1;
236  synapses_saturation_count++;
237  }
238 
239  // Store saturated value back in ring-buffer
240  ring_buffers[ring_buffer_index] = accumulation;
241  }
242 }

255 bool synapses_initialise(
256  address_t synapse_params_address, uint32_t n_neurons_value,
257  uint32_t n_synapse_types_value,
258  uint32_t **ring_buffer_to_input_buffer_left_shifts,
259  bool* clear_input_buffers_of_late_packets_init) {
260  log_debug("synapses_initialise: starting");
261  n_neurons = n_neurons_value;
262  n_synapse_types = n_synapse_types_value;
263 
264  // Set up ring buffer left shifts
265  ring_buffer_to_input_left_shifts =
266  spin1_malloc(n_synapse_types * sizeof(uint32_t));
267  if (ring_buffer_to_input_left_shifts == NULL) {
268  log_error("Not enough memory to allocate ring buffer");
269  return false;
270  }
271 
272  // read bool flag about dropping packets that arrive too late
273  *clear_input_buffers_of_late_packets_init = synapse_params_address[0];
274 
275  // shift read by 1 word.
276  synapse_params_address += 1;
277 
278  // read in ring buffer to input left shifts
279  spin1_memcpy(
280  ring_buffer_to_input_left_shifts, synapse_params_address,
281  n_synapse_types * sizeof(uint32_t));
282  *ring_buffer_to_input_buffer_left_shifts =
283  ring_buffer_to_input_left_shifts;
284 
285  log_debug("synapses_initialise: completed successfully");
286  print_synapse_parameters();
287 
288  uint32_t n_neurons_power_2 = n_neurons;
289  uint32_t log_n_neurons = 1;
290  if (n_neurons != 1) {
291  if (!is_power_of_2(n_neurons)) {
292  n_neurons_power_2 = next_power_of_2(n_neurons);
293  }
294  log_n_neurons = ilog_2(n_neurons_power_2);
295  }
296 
297  uint32_t n_synapse_types_power_2 = n_synapse_types;
298  if (!is_power_of_2(n_synapse_types)) {
299  n_synapse_types_power_2 = next_power_of_2(n_synapse_types);
300  }
301  uint32_t log_n_synapse_types = ilog_2(n_synapse_types_power_2);
302 
303  uint32_t n_ring_buffer_bits =
304  log_n_neurons + log_n_synapse_types + SYNAPSE_DELAY_BITS;
305  ring_buffer_size = 1 << (n_ring_buffer_bits);
306 
307  ring_buffers = spin1_malloc(ring_buffer_size * sizeof(weight_t));
308  if (ring_buffers == NULL) {
309  log_error("Could not allocate %u entries for ring buffers",
310  ring_buffer_size);
311  }
312  for (uint32_t i = 0; i < ring_buffer_size; i++) {
313  ring_buffers[i] = 0;
314  }
315 
316  synapse_type_index_bits = log_n_neurons + log_n_synapse_types;
317  synapse_type_index_mask = (1 << synapse_type_index_bits) - 1;
318  synapse_index_bits = log_n_neurons;
319  synapse_index_mask = (1 << synapse_index_bits) - 1;
320  synapse_type_bits = log_n_synapse_types;
321  synapse_type_mask = (1 << log_n_synapse_types) - 1;
322  return true;
323 }

325 void synapses_do_timestep_update(timer_t time) {
326  print_ring_buffers(time);
327 
328  // Disable interrupts to stop DMAs interfering with the ring buffers
329  uint32_t state = spin1_irq_disable();
330 
331  // Clear any outstanding spikes
332  spike_processing_clear_input_buffer(time);
333 
334  // Transfer the input from the ring buffers into the input buffers
335  for (uint32_t neuron_index = 0; neuron_index < n_neurons;
336  neuron_index++) {
337  // Loop through all synapse types
338  for (uint32_t synapse_type_index = 0;
339  synapse_type_index < n_synapse_types; synapse_type_index++) {
340  // Get index in the ring buffers for the current time slot for
341  // this synapse type and neuron
342  uint32_t ring_buffer_index = synapses_get_ring_buffer_index(
343  time, synapse_type_index, neuron_index,
344  synapse_type_index_bits, synapse_index_bits);
345 
346  // Convert ring-buffer entry to input and add on to correct
347  // input for this synapse type and neuron
348  neuron_add_inputs(
349  synapse_type_index, neuron_index,
350  synapses_convert_weight_to_input(
351  ring_buffers[ring_buffer_index],
352  ring_buffer_to_input_left_shifts[synapse_type_index]));
353 
354  // Clear ring buffer
355  ring_buffers[ring_buffer_index] = 0;
356  }
357  }
358 
359  print_inputs();
360 
361  // Re-enable the interrupts
362  spin1_mode_restore(state);
363 }
