41static inline bool synaptogenesis_elimination_rule(
42 current_state_t *restrict current_state,
43 const elimination_params_t *params,
44 UNUSED uint32_t time, synaptic_row_t restrict row) {
45 uint32_t random_number = mars_kiss64_seed(*(current_state->local_seed));
46
47 // Is weight depressed?
48 if (current_state->weight < params->threshold &&
49 random_number > params->prob_elim_depression) {
50 return false;
51 }
52
53 // Is weight potentiated or unchanged?
54 if (current_state->weight >= params->threshold &&
55 random_number > params->prob_elim_potentiation) {
56 return false;
57 }
58
59 return sp_structs_remove_synapse(current_state, row);
60}
