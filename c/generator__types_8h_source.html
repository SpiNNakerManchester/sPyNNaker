126 static inline uint16_t rescale_delay(accum delay, accum timestep_per_delay) {
127  accum ts_delay = delay * timestep_per_delay;
128  if (ts_delay < 0) {
129  ts_delay = 1;
130  }
131  uint16_t delay_int = (uint16_t) ts_delay;
132  if (ts_delay != delay_int) {
133  log_debug("Rounded delay %k to %u", delay, delay_int);
134  }
135  return delay_int;
136 }

143 static inline uint16_t rescale_weight(accum weight, unsigned long accum weight_scale) {
144  unsigned long accum uweight = 0;
145  if (weight < 0) {
146  uweight = -weight;
147  } else {
148  uweight = weight;
149  }
150  unsigned long accum weight_scaled = uweight * weight_scale;
151  unsigned long accum weight_rounded = roundulk(weight_scaled, 32);
152  uint16_t weight_int = (uint16_t) (bitsulk(weight_rounded) >> 32);
153  if (weight_scaled != weight_int) {
154  log_debug("Rounded weight %k to %u (scale is %k)",
155  weight_scaled, weight_int, weight_scale);
156  }
157  return weight_int;
158 }
