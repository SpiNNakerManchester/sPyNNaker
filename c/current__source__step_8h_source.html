45 static bool current_source_step_init(
46  address_t cs_address, uint32_t n_step_current_sources, uint32_t *next) {
47  if (n_step_current_sources > 0) {
48  step_cs_times = spin1_malloc(n_step_current_sources * sizeof(uint32_t*));
49  step_cs_amps = spin1_malloc(n_step_current_sources * sizeof(uint32_t*));
50  step_cs_amp_last = spin1_malloc(n_step_current_sources * sizeof(REAL));
51  step_cs_index = spin1_malloc(n_step_current_sources * sizeof(uint32_t));
52  if (step_cs_amp_last == NULL) {
53  log_error("Unable to allocate step current source amp last - out of DTCM");
54  return false;
55  }
56  if (step_cs_index == NULL) {
57  log_error("Unable to allocate step current source index - out of DTCM");
58  return false;
59  }
60  }
61  for (uint32_t n_step=0; n_step < n_step_current_sources; n_step++) {
62  uint32_t arr_len = (uint32_t) cs_address[*next];
63  uint32_t struct_size = (arr_len + 1) * sizeof(uint32_t);
64  step_cs_times[n_step] = spin1_malloc(struct_size);
65  if (step_cs_times[n_step] == NULL) {
66  log_error("Unable to allocate step current source times - out of DTCM",
67  "struct_size is %u next %u n_step %u)", struct_size, *next, n_step);
68  return false;
69  }
70 
71  step_cs_amps[n_step] = spin1_malloc(struct_size);
72  if (step_cs_amps[n_step] == NULL) {
73  log_error("Unable to allocate step current source amplitudes - out of DTCM",
74  "(struct_size is %u next %u n_step %u)", struct_size, *next, n_step);
75  return false;
76  }
77 
78  *next += 2 * (arr_len + 1);
79  // Initialise last value and current index along the array for this source
80  step_cs_amp_last[n_step] = ZERO;
81  step_cs_index[n_step] = 0;
82  }
83  return true;
84 }

86 static bool current_source_step_load_parameters(
87  address_t cs_address, uint32_t n_step_current_sources, uint32_t *next) {
88  for (uint32_t n_step=0; n_step < n_step_current_sources; n_step++) {
89  uint32_t arr_len = (uint32_t) cs_address[*next];
90  uint32_t struct_size = (arr_len + 1) * sizeof(uint32_t);
91 
92  spin1_memcpy(step_cs_times[n_step], &cs_address[*next], struct_size);
93  spin1_memcpy(step_cs_amps[n_step], &cs_address[*next+arr_len+1], struct_size);
94 
95  *next += 2 * (arr_len + 1);
96 
97  // Does this need to happen here too? (What happens on reload??)
98  step_cs_amp_last[n_step] = ZERO;
99  step_cs_index[n_step] = 0;
100  }

104 static REAL current_source_step_get_offset(uint32_t cs_index, uint32_t time) {
105  if (time >= step_cs_times[cs_index]->times[step_cs_index[cs_index]]) {
106  step_cs_amp_last[cs_index] =
107  step_cs_amps[cs_index]->amplitudes[step_cs_index[cs_index]];
108  step_cs_index[cs_index]++;
109  }
110  return step_cs_amp_last[cs_index];
