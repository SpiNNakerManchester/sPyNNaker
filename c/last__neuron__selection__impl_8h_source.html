36static inline void partner_spike_received(uint32_t time, spike_t spike) {
37 extern uint32_t last_spikes_buffer_size;
38 extern uint32_t last_time;
39
40 uint32_t buffer = time & 0x1;
41 if (time != last_time) {
42 last_time = time;
43 n_spikes[buffer] = 0;
44 }
45 if (n_spikes[buffer] < last_spikes_buffer_size) {
46 last_spikes_buffer[buffer][n_spikes[buffer]++] = spike;
47 }
48}
