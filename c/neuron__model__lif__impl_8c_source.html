29  neuron_t *neuron, REAL V_prev, input_t input_this_timestep) {

41 state_t neuron_model_state_update(
42  uint16_t num_excitatory_inputs, const input_t *exc_input,
43  uint16_t num_inhibitory_inputs, const input_t *inh_input,
44  input_t external_bias, neuron_t *restrict neuron) {
45  log_debug("Exc 1: %12.6k, Exc 2: %12.6k", exc_input[0], exc_input[1]);
46  log_debug("Inh 1: %12.6k, Inh 2: %12.6k", inh_input[0], inh_input[1]);
47 
48  // If outside of the refractory period
49  if (neuron->refract_timer <= 0) {
50  REAL total_exc = 0;
51  REAL total_inh = 0;
52 
53  for (int i=0; i < num_excitatory_inputs; i++) {
54  total_exc += exc_input[i];
55  }
56  for (int i=0; i< num_inhibitory_inputs; i++) {
57  total_inh += inh_input[i];
58  }
59  // Get the input in nA
60  input_t input_this_timestep =
61  total_exc - total_inh + external_bias + neuron->I_offset;
62 
63  lif_neuron_closed_form(
64  neuron, neuron->V_membrane, input_this_timestep);
65  } else {
66  // countdown refractory timer
67  neuron->refract_timer--;
68  }
69  return neuron->V_membrane;
70 }
