105 static uint32_t post_random_in_range(struct fixed_post *obj, uint32_t range) {
106  uint32_t u01 = rng_generator(obj->rng) & 0x00007fff;
107  return (u01 * range) >> 15;

129 static uint32_t connection_generator_fixed_post_generate(
130  void *generator, UNUSED uint32_t pre_slice_start,
131  UNUSED uint32_t pre_slice_count,
132  uint32_t pre_neuron_index, uint32_t post_slice_start,
133  uint32_t post_slice_count, uint32_t max_row_length, uint16_t *indices) {
134  // If there are no connections to be made, return 0
135  struct fixed_post *obj = generator;
136  if (max_row_length == 0 || obj->params.n_post == 0) {
137  return 0;
138  }
139 
140  // If not in the pre-population view range, then don't generate
141  if ((pre_neuron_index < obj->params.pre_lo) ||
142  (pre_neuron_index > obj->params.pre_hi)) {
143  return 0;
144  }
145 
146  // Get how many values can be sampled from
147  uint32_t n_values = obj->params.n_post_neurons;
148  // Get the number of connections on this row
149  uint32_t n_conns = obj->params.n_post;
150 
151  log_debug("Generating %u from %u possible synapses", n_conns, n_values);
152 
153  uint16_t full_indices[n_conns];
154  // Sample from the possible connections in this section n_conns times
155  if (obj->params.with_replacement) {
156  // Sample them with replacement
157  if (obj->params.allow_self_connections) {
158  // self connections are allowed so sample
159  for (uint32_t i = 0; i < n_conns; i++) {
160  full_indices[i] = post_random_in_range(obj, n_values);
161  }
162  } else {
163  // self connections are not allowed (on this slice)
164  for (uint32_t i = 0; i < n_conns; i++) {
165  // Set j to the disallowed value, then test against it
166  uint32_t j;
167 
168  do {
169  j = post_random_in_range(obj, n_values);
170  } while (j == pre_neuron_index);
171 
172  full_indices[i] = j;
173  }
174  }
175  } else {
176  // Sample them without replacement using reservoir sampling
177  if (obj->params.allow_self_connections) {
178  // Self-connections are allowed so do this normally
179  for (uint32_t i = 0; i < n_conns; i++) {
180  full_indices[i] = i;
181  }
182  // And now replace values if chosen at random to be replaced
183  for (uint32_t i = n_conns; i < n_values; i++) {
184  // j = random(0, i) (inclusive)
185  uint32_t j = post_random_in_range(obj, i + 1);
186 
187  if (j < n_conns) {
188  full_indices[j] = i;
189  }
190  }
191  } else {
192  // Self-connections are not allowed
193  uint32_t replace_start = n_conns;
194  for (uint32_t i = 0; i < n_conns; i++) {
195  if (i == pre_neuron_index) {
196  // set to a value not equal to i for now
197  full_indices[i] = n_conns;
198  replace_start = n_conns + 1;
199  } else {
200  full_indices[i] = i;
201  }
202  }
203  // And now "replace" values if chosen at random to be replaced
204  for (uint32_t i = replace_start; i < n_values; i++) {
205  if (i != pre_neuron_index) {
206  // j = random(0, i) (inclusive)
207  uint32_t j = post_random_in_range(obj, i + 1);
208 
209  if (j < n_conns) {
210  full_indices[j] = i;
211  }
212  }
213  }
214  }
215  }
216 
217  // Loop over the full indices array, and only keep indices that are on this
218  // post-slice and within the range of the specified post-population view
219  uint32_t count_indices = 0;
220  for (uint32_t i = 0; i < n_conns; i++) {
221  uint32_t j = full_indices[i] + obj->params.post_lo;
222  if ((j >= post_slice_start) && (j < post_slice_start + post_slice_count)) {
223 // (j >= obj->params.post_lo) && (j <= obj->params.post_hi)) {
224  indices[count_indices] = j - post_slice_start; // On this slice!
225  count_indices++;
226  }
227  }
228 
229  // Double-check for debug purposes
230 #if 0
231  for (unsigned int i = 0; i < count_indices; i++) {
232  log_info("Check: indices[%u] is %u", i, indices[i]);
233  }
234  log_info("pre_neuron_index is %u count_indices is %u", pre_neuron_index, count_indices);
235 #endif
236 
237  return count_indices;
238 }
