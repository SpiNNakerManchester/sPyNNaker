122 static inline update_state_t timing_apply_pre_spike(
123  uint32_t time, UNUSED pre_trace_t trace, uint32_t last_pre_time,
124  UNUSED pre_trace_t last_pre_trace, uint32_t last_post_time,
125  UNUSED post_trace_t last_post_trace, update_state_t previous_state) {
126  switch (previous_state.state) {
127  case STATE_IDLE:
128  // If we're idle, transition to pre-open state
129  log_debug("\tOpening pre-window");
130  previous_state.state = STATE_PRE_OPEN;
131  previous_state =
132  timing_recurrent_calculate_pre_window(previous_state);
133  break;
134  case STATE_PRE_OPEN:
135  // If we're in pre-open state
136  _no_op(); // <<< empty statement for C syntax reasons
137  // Get time of event relative to last pre-synaptic event
138  uint32_t time_since_last_pre = time - last_pre_time;
139 
140  log_debug("\tTime_since_last_pre_event=%u", time_since_last_pre);
141 
142  if (timing_recurrent_in_pre_window(time_since_last_pre, previous_state)) {
143  // If pre-window is still open
144  log_debug("\t\tClosing pre-window");
145  previous_state.state = STATE_IDLE;
146  } else {
147  // Otherwise, leave state alone (essentially re-opening window)
148  log_debug("\t\tRe-opening pre-window");
149  previous_state =
150  timing_recurrent_calculate_pre_window(previous_state);
151  }
152  break;
153  case STATE_POST_OPEN:
154  // Otherwise, if we're in post-open
155  _no_op(); // <<< empty statement for C syntax reasons
156  // Get time of event relative to last post-synaptic event
157  uint32_t time_since_last_post = time - last_post_time;
158 
159  log_debug("\tTime_since_last_post_event=%u", time_since_last_post);
160 
161  if (timing_recurrent_in_post_window(
162  time_since_last_post, previous_state)) {
163  extern plasticity_trace_region_data_t plasticity_trace_region_data;
164 
165  // Otherwise, if post-window is still open
166  if (previous_state.accumulator >
167  plasticity_trace_region_data.accumulator_depression_plus_one) {
168  // If accumulator's not going to hit depression limit, decrement
169  // it
170  previous_state.accumulator--;
171  log_debug("\t\tDecrementing accumulator=%d",
172  previous_state.accumulator);
173  } else {
174  // Otherwise, reset accumulator and apply depression
175  log_debug("\t\tApplying depression");
176 
177  previous_state.accumulator = 0;
178  previous_state.weight_state = weight_one_term_apply_depression(
179  previous_state.weight_state, STDP_FIXED_POINT_ONE);
180  }
181 
182  // Transition back to idle
183  previous_state.state = STATE_IDLE;
184  } else {
185  // Otherwise, if post-window has closed, skip idle state and go
186  // straight to pre-open
187  log_debug("\t\tPost-window closed - Opening pre-window");
188  previous_state.state = STATE_PRE_OPEN;
189  previous_state =
190  timing_recurrent_calculate_pre_window(previous_state);
191  }
192  break;
193  default:
194  log_debug("\tInvalid state %u", previous_state.state);
195  }
196 
197  return previous_state;
198 }
