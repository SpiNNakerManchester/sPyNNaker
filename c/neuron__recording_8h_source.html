163 static inline void neuron_recording_record(uint32_t time) {
164  // go through all recordings
165  for (uint32_t i = N_RECORDED_VARS; i > 0; i--) {
166  recording_info_t *rec_info = &recording_info[i - 1];
167  // if the rate says record, record now
168  if (rec_info->count == rec_info->rate) {
169  // Reset the count
170  rec_info->count = 1;
171  // Note we are recording
172  n_recordings_outstanding += 1;
173  // Set the time and record the data
174  rec_info->values->time = time;
175  recording_record_and_notify(
176  i - 1, rec_info->values, rec_info->size, recording_done_callback);
177  } else {
178 
179  // Not recording this time, so increment by specified amount
180  rec_info->count += rec_info->increment;
181  }
182  }
183 
184  for (uint32_t i = N_BITFIELD_VARS; i > 0; i--) {
185  bitfield_info_t *bf_info = &bitfield_info[i - 1];
186  // if the rate says record, record now
187  if (bf_info->count == bf_info->rate) {
188  // Reset the count
189  bf_info->count = 1;
190  // Skip empty bitfields
191  if (empty_bit_field(bf_info->values->bits, bf_info->n_words)) {
192  continue;
193  }
194  // Note we are recording
195  n_recordings_outstanding += 1;
196  // Set the time and record the data (note index is after recorded_vars)
197  bf_info->values->time = time;
198  recording_record_and_notify(
199  i + N_RECORDED_VARS - 1, bf_info->values, bf_info->size,
200  recording_done_callback);
201  } else {
202 
203  // Not recording this time, so increment by specified amount
204  bf_info->count += bf_info->increment;
205  }
206  }
207 }

210 static inline void neuron_recording_setup_for_next_recording(void) {
211  // Wait until recordings have completed, to ensure the recording space
212  // can be re-written
213  while (n_recordings_outstanding > 0) {
214  spin1_wfi();
215  }
216 
217  // Reset the bitfields before starting if a beginning of recording
218  for (uint32_t i = N_BITFIELD_VARS; i > 0; i--) {
219  bitfield_info_t *b_info = &bitfield_info[i - 1];
220  if (b_info->count == 1) {
221  clear_bit_field(b_info->values->bits, b_info->n_words);
222  }
223  }
224 }
