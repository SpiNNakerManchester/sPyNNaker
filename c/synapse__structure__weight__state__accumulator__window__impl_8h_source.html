47 static inline update_state_t synapse_structure_get_update_state(
48  plastic_synapse_t synaptic_word, index_t synapse_type) {
49  // Create update state, using weight dependance to initialise the weight
50  // state and copying other parameters from the synaptic word into 32-bit
51  // form
52  update_state_t update_state;
53  update_state.weight_state =
54  weight_get_initial(synaptic_word.weight, synapse_type);
55  update_state.accumulator = (int32_t) synaptic_word.accumulator;
56  update_state.state = (uint32_t) synaptic_word.state;
57  update_state.window_length = (uint32_t) synaptic_word.window_length;
58  return update_state;
59 }
