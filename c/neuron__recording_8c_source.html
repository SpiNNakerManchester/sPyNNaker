55 static void reset_record_counter(void) {
56  for (uint32_t i = 0; i < N_RECORDED_VARS; i++) {
57  if (recording_info[i].rate == 0) {
58  // Setting increment to zero means count will never equal rate
59  recording_info[i].increment = 0;
60 
61  // Count is not rate so does not record, but not 1 so it does not reset!
62  recording_info[i].count = 2;
63  } else {
64  // Increase one each call so count gets to rate
65  recording_info[i].increment = 1;
66 
67  // Using rate here so that the zero time is recorded
68  recording_info[i].count = recording_info[i].rate;
69  }
70  }
71 
72  // clear the bitfields
73  for (uint32_t i = 0; i < N_BITFIELD_VARS; i++) {
74  if (bitfield_info[i].rate == 0) {
75  // Setting increment to zero means count will never equal rate
76  bitfield_info[i].increment = 0;
77 
78  // Count is not rate so does not record, but not 1 so it does not reset!
79  bitfield_info[i].count = 2;
80  } else {
81  // Increase one each call so count gets to rate
82  bitfield_info[i].increment = 1;
83 
84  // Using rate here so that the zero time is recorded
85  bitfield_info[i].count = bitfield_info[i].rate;
86  clear_bit_field(bitfield_info[i].values->bits,
87  bitfield_info[i].n_words);
88  }
89  }
90 }

103 static bool neuron_recording_read_in_elements(
104  void *recording_address, uint32_t n_neurons) {
105  // Round up the number of bytes to align at a word boundary i.e. round to
106  // the next multiple of 4
107  uint32_t ceil_n_entries = (n_neurons + CEIL_TO_4) & FLOOR_TO_4;
108 
109  // GCC lets you define a struct like this!
110  typedef struct neuron_recording_data {
111  uint32_t rate;
112  uint32_t n_neurons_recording;
113  uint32_t element_size;
114  uint8_t indices[ceil_n_entries];
115  } neuron_recording_data_t;
116 
117  neuron_recording_data_t *data = recording_address;
118 
119  for (uint32_t i = 0; i < N_RECORDED_VARS; i++) {
120  recording_info[i].rate = data[i].rate;
121  uint32_t n_neurons_rec = data[i].n_neurons_recording;
122  recording_info[i].element_size = data[i].element_size;
123  recording_info[i].size = sizeof(recording_values_t)
124  + (n_neurons_rec * recording_info[i].element_size);
125  // There is an extra "neuron" in the data used when one of the neurons
126  // is *not* recording, to avoid a check
127  uint32_t alloc_size = recording_info[i].size +
128  recording_info[i].element_size;
129 
130  // allocate memory for the recording
131  if (recording_info[i].values == NULL) {
132  recording_info[i].values = spin1_malloc(alloc_size);
133  if (recording_info[i].values == NULL) {
134  log_error("couldn't allocate recording data space for %d", i);
135  return false;
136  }
137  recording_values[i] = recording_info[i].values->data;
138  }
139 
140  // copy over the indexes
141  spin1_memcpy(neuron_recording_indexes[i], data[i].indices,
142  n_neurons * sizeof(uint8_t));
143  }

146  uint32_t rate;
147  uint32_t n_neurons_recording;
148  uint8_t indices[ceil_n_entries];
149  } bitfield_recording_data_t;
150 
151  bitfield_recording_data_t *bitfield_data =
152  (bitfield_recording_data_t *) &data[N_RECORDED_VARS];
153 
154  for (uint32_t i = 0; i < N_BITFIELD_VARS; i++) {
155  bitfield_info[i].rate = bitfield_data[i].rate;
156  uint32_t n_neurons_rec = bitfield_data[i].n_neurons_recording;
157  bitfield_info[i].size = bitfield_data_size(n_neurons_rec);
158  // There is an extra "neuron" in the data used when one of the neurons
159  // is *not* recording, to avoid a check
160  uint32_t alloc_size = bitfield_data_size(n_neurons_rec + 1);
161 
162  // allocate memory for the recording
163  if (bitfield_info[i].values == NULL) {
164  bitfield_info[i].values = spin1_malloc(alloc_size);
165  if (bitfield_info[i].values == NULL) {
166  log_error("couldn't allocate bitfield recording data space for %d", i);
167  return false;
168  }
169  // There is an extra "neuron" in the data used when one of the
170  // neurons is *not* recording, to avoid a check
171  bitfield_info[i].n_words = get_bit_field_size(n_neurons_rec + 1);
172  bitfield_values[i] = bitfield_info[i].values->bits;
173  }
174 
175  // copy over the indexes
176  spin1_memcpy(bitfield_recording_indexes[i], bitfield_data[i].indices,
177  n_neurons * sizeof(uint8_t));
178  }
179  return true;
180 }

279 bool neuron_recording_initialise(
280  void *recording_address, uint32_t n_neurons,
281  uint32_t *n_rec_regions_used) {
282  // boot up the basic recording
283  void *data_addr = recording_address;
284 
285  // Verify the number of recording and bitfield elements
286  neuron_recording_header_t *header = data_addr;
287  if (header->n_recorded_vars != N_RECORDED_VARS) {
288  log_error("Data spec number of recording variables %d != "
289  "neuron implementation number of recorded variables %d",
290  header->n_recorded_vars, N_RECORDED_VARS);
291  return false;
292  }
293  if (header->n_bitfield_vars != N_BITFIELD_VARS) {
294  log_error("Data spec number of bitfield variables %d != "
295  "neuron implementation number of bitfield variables %d",
296  header->n_bitfield_vars, N_BITFIELD_VARS);
297  return false;
298  }
299  // Copy the number of regions used
300  *n_rec_regions_used = header->n_recorded_vars + header->n_bitfield_vars;
301  data_addr = &header[1];
302  log_debug("Recording %d variables and %d bitfield variables",
303  N_RECORDED_VARS, N_BITFIELD_VARS);
304 
305  if (!allocate_word_dtcm(n_neurons)) {
306  log_error("failed to allocate DTCM for the neuron recording structs.");
307  return false;
308  }
309  if (!allocate_bitfield_dtcm(n_neurons)) {
310  log_error("failed to allocate DTCM for the bitfield recording structs");
311  return false;
312  }
313 
314  // read in the sdram params into the allocated data objects
315  reset_address = data_addr;
316  if (!neuron_recording_read_in_elements(data_addr, n_neurons)) {
317  log_error("failed to read in the elements");
318  return false;
319  }
320 
321  // reset stuff
322  reset_record_counter();
323 
324  return true;
325 }
