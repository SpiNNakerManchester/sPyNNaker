55 static void reset_record_counter(void) {
56  for (uint32_t i = 0; i < N_RECORDED_VARS; i++) {
57  if (recording_info[i].rate == 0) {
58  // Setting increment to zero means count will never equal rate
59  recording_info[i].increment = 0;
60 
61  // Count is not rate so does not record, but not 1 so it does not reset!
62  recording_info[i].count = 2;
63  } else {
64  // Increase one each call so count gets to rate
65  recording_info[i].increment = 1;
66 
67  // Using rate here so that the zero time is recorded
68  recording_info[i].count = recording_info[i].rate;
69  }
70  }
71 
72  // clear the bitfields
73  for (uint32_t i = 0; i < N_BITFIELD_VARS; i++) {
74  if (bitfield_info[i].rate == 0) {
75  // Setting increment to zero means count will never equal rate
76  bitfield_info[i].increment = 0;
77 
78  // Count is not rate so does not record, but not 1 so it does not reset!
79  bitfield_info[i].count = 2;
80  } else {
81  // Increase one each call so count gets to rate
82  bitfield_info[i].increment = 1;
83 
84  // Using rate here so that the zero time is recorded
85  bitfield_info[i].count = bitfield_info[i].rate;
86  clear_bit_field(bitfield_info[i].values->bits,
87  bitfield_info[i].n_words);
88  }
89  }
90 }

108 static bool neuron_recording_read_in_elements(
109  void *recording_address, uint32_t n_neurons) {
110  // Round up the number of bytes to align at a word boundary i.e. round to
111  // the next multiple of 4
112  uint32_t ceil_n_entries = (n_neurons + CEIL_TO_4) & FLOOR_TO_4;
113 
114  // GCC lets you define a struct like this!
115  typedef struct neuron_recording_data {
116  uint32_t rate;
117  uint32_t n_neurons_recording;
118  uint32_t element_size;
119  uint8_t indices[ceil_n_entries];
120  } neuron_recording_data_t;
121 
122  neuron_recording_data_t *data = recording_address;
123 
124  for (uint32_t i = 0; i < N_RECORDED_VARS; i++) {
125  recording_info[i].rate = data[i].rate;
126  uint32_t n_neurons_rec = data[i].n_neurons_recording;
127  recording_info[i].element_size = data[i].element_size;
128  recording_info[i].size = sizeof(recording_values_t)
129  + (n_neurons_rec * recording_info[i].element_size);
130  // There is an extra "neuron" in the data used when one of the neurons
131  // is *not* recording, to avoid a check
132  uint32_t alloc_size = recording_info[i].size +
133  recording_info[i].element_size;
134 
135  // allocate memory for the recording
136  if (recording_info[i].values == NULL) {
137  recording_info[i].values = spin1_malloc(alloc_size);
138  if (recording_info[i].values == NULL) {
139  log_error("couldn't allocate recording data space for %d", i);
140  return false;
141  }
142  recording_values[i] = recording_info[i].values->data;
143  }
144 
145  // copy over the indexes
146  spin1_memcpy(neuron_recording_indexes[i], data[i].indices,
147  n_neurons * sizeof(uint8_t));
148  }

151  uint32_t rate;
152  uint32_t n_neurons_recording;
153  uint8_t indices[ceil_n_entries];
154  } bitfield_recording_data_t;
155 
156  bitfield_recording_data_t *bitfield_data =
157  (bitfield_recording_data_t *) &data[N_RECORDED_VARS];
158 
159  for (uint32_t i = 0; i < N_BITFIELD_VARS; i++) {
160  bitfield_info[i].rate = bitfield_data[i].rate;
161  uint32_t n_neurons_rec = bitfield_data[i].n_neurons_recording;
162  bitfield_info[i].size = bitfield_data_size(n_neurons_rec);
163  // There is an extra "neuron" in the data used when one of the neurons
164  // is *not* recording, to avoid a check
165  uint32_t alloc_size = bitfield_data_size(n_neurons_rec + 1);
166 
167  // allocate memory for the recording
168  if (bitfield_info[i].values == NULL) {
169  bitfield_info[i].values = spin1_malloc(alloc_size);
170  if (bitfield_info[i].values == NULL) {
171  log_error("couldn't allocate bitfield recording data space for %d", i);
172  return false;
173  }
174  // There is an extra "neuron" in the data used when one of the
175  // neurons is *not* recording, to avoid a check
176  bitfield_info[i].n_words = get_bit_field_size(n_neurons_rec + 1);
177  bitfield_values[i] = bitfield_info[i].values->bits;
178  }
179 
180  // copy over the indexes
181  spin1_memcpy(bitfield_recording_indexes[i], bitfield_data[i].indices,
182  n_neurons * sizeof(uint8_t));
183  }
184  return true;

285 bool neuron_recording_initialise(
286  void *recording_address, uint32_t *recording_flags,
287  uint32_t n_neurons, uint32_t *n_rec_regions_used) {
288  // boot up the basic recording
289  void *data_addr = recording_address;
290  bool success = recording_initialize(&data_addr, recording_flags);
291  if (!success) {
292  log_error("failed to init basic recording.");
293  return false;
294  }
295 
296  // Verify the number of recording and bitfield elements
297  neuron_recording_header_t *header = data_addr;
298  if (header->n_recorded_vars != N_RECORDED_VARS) {
299  log_error("Data spec number of recording variables %d != "
300  "neuron implementation number of recorded variables %d",
301  header->n_recorded_vars, N_RECORDED_VARS);
302  return false;
303  }
304  if (header->n_bitfield_vars != N_BITFIELD_VARS) {
305  log_error("Data spec number of bitfield variables %d != "
306  "neuron implementation number of bitfield variables %d",
307  header->n_bitfield_vars, N_BITFIELD_VARS);
308  return false;
309  }
310  // Copy the number of regions used
311  *n_rec_regions_used = header->n_recorded_vars + header->n_bitfield_vars;
312  data_addr = &header[1];
313  log_debug("Recording %d variables and %d bitfield variables",
314  N_RECORDED_VARS, N_BITFIELD_VARS);
315 
316  if (!allocate_word_dtcm(n_neurons)) {
317  log_error("failed to allocate DTCM for the neuron recording structs.");
318  return false;
319  }
320  if (!allocate_bitfield_dtcm(n_neurons)) {
321  log_error("failed to allocate DTCM for the bitfield recording structs");
322  return false;
323  }
324 
325  // read in the sdram params into the allocated data objects
326  reset_address = data_addr;
327  if (!neuron_recording_read_in_elements(data_addr, n_neurons)) {
328  log_error("failed to read in the elements");
329  return false;
330  }
331 
332  // reset stuff
333  reset_record_counter();
334 
335  return true;
336 }
