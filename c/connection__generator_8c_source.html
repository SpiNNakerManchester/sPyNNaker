103 connection_generator_t connection_generator_init(
104  uint32_t hash, void **in_region) {
105  // Look through the known generators
106  for (uint32_t i = 0; i < N_CONNECTION_GENERATORS; i++) {
107  const connection_generator_info *type = &connection_generators[i];
108 
109  // If the hash requested matches the hash of the generator, use it
110  if (hash == type->hash) {
111  // Prepare a space for the data
112  struct connection_generator *generator =
113  spin1_malloc(sizeof(struct connection_generator));
114  if (generator == NULL) {
115  log_error("Could not create generator");
116  return NULL;
117  }
118 
119  // Store the index
120  generator->type = type;
121 
122  // Initialise the generator and store the data
123  generator->data = type->initialize(in_region);
124  return generator;
125  }
126  }
127  log_error("Connection generator with hash %u not found", hash);
128  return NULL;
129 }
