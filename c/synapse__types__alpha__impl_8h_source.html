87 static inline void get_alpha_state(alpha_state_t *state, alpha_params_t *params,
88  REAL time_step_ms, uint32_t n_steps_per_timestep) {
89  REAL ts = kdivui(time_step_ms, n_steps_per_timestep);
90  decay_t decay = expulr(-kdivk(ts, params->tau));
91  decay_t init = kdivk(ts, (params->tau * params->tau));
92  state->lin_buff = params->lin_init;
93  state->exp_buff = params->exp_init;
94  state->q_buff = params->q_init;
95  state->dt_divided_by_tau_sqr = init;
96  state->decay = decay;
97 }

116 static inline void alpha_shaping(alpha_state_t* a_params) {
117  a_params->lin_buff = a_params->lin_buff + (
118  a_params->q_buff * a_params->dt_divided_by_tau_sqr);
119 
120  // Update exponential buffer
121  a_params->exp_buff = decay_s1615(a_params->exp_buff, a_params->decay);
122 }

142 static inline void add_input_alpha(alpha_state_t *a_params, input_t input) {
143  a_params->q_buff = input;
144 
145  a_params->exp_buff =
146  decay_s1615(a_params->exp_buff, a_params->decay) + ONE;
147 
148  a_params->lin_buff =
149  (a_params->lin_buff + (input * a_params->dt_divided_by_tau_sqr))
150  * (ONE - kdivk(ONE, a_params->exp_buff));
151 }
