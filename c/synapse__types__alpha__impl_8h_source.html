107 static inline void add_input_alpha(alpha_params_t *a_params, input_t input) {
108  a_params->q_buff = input;
109 
110  a_params->exp_buff =
111  decay_s1615(a_params->exp_buff, a_params->decay) + ONE;
112 
113  a_params->lin_buff =
114  (a_params->lin_buff + (input * a_params->dt_divided_by_tau_sqr))
115  * (ONE - ONE/a_params->exp_buff);
116 }
