80 static inline void alpha_shaping(alpha_params_t* a_params) {
81  a_params->lin_buff = a_params->lin_buff + (
82  a_params->q_buff * a_params->dt_divided_by_tau_sqr);
83 
84  // Update exponential buffer
85  a_params->exp_buff = decay_s1615(a_params->exp_buff, a_params->decay);
86 }

112 static inline void add_input_alpha(alpha_params_t *a_params, input_t input) {
113  a_params->q_buff = input;
114 
115  a_params->exp_buff =
116  decay_s1615(a_params->exp_buff, a_params->decay) + ONE;
117 
118  a_params->lin_buff =
119  (a_params->lin_buff + (input * a_params->dt_divided_by_tau_sqr))
120  * (ONE - ONE/a_params->exp_buff);
121 }
