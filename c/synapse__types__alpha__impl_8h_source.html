75 static inline void alpha_shaping(alpha_params_t* a_params) {
76  a_params->lin_buff = a_params->lin_buff + (

101 static inline void add_input_alpha(alpha_params_t *a_params, input_t input) {
102  a_params->q_buff = input;
103 
104  a_params->exp_buff =
105  decay_s1615(a_params->exp_buff, a_params->decay) + ONE;
106 
107  a_params->lin_buff =
108  (a_params->lin_buff + (input * a_params->dt_divided_by_tau_sqr))
109  * (ONE - ONE/a_params->exp_buff);
110 }
