43 bool direct_synapses_initialise(
44  void *direct_matrix_address, address_t *direct_synapses_address) {
45  direct_matrix_data_t *direct_matrix = direct_matrix_address;
46  // Work out the positions of the direct and indirect synaptic matrices
47  // and copy the direct matrix to DTCM
48  uint32_t direct_matrix_size = direct_matrix->size;
49  log_debug("Direct matrix malloc size is %d", direct_matrix_size);
50 
51  if (direct_matrix_size != 0) {
52  void *dtcm_copy = spin1_malloc(direct_matrix_size);
53  if (dtcm_copy == NULL) {
54  log_error("Not enough memory to allocate direct matrix");
55  return false;
56  }
57  log_debug("Copying %u bytes of direct synapses to 0x%08x",
58  direct_matrix_size, dtcm_copy);
59  spin1_memcpy(dtcm_copy, direct_matrix->data, direct_matrix_size);
60  *direct_synapses_address = dtcm_copy;
61  }
62 
63  return true;
64 }
