39 static bool current_source_ac_init(uint32_t n_ac_sources, uint32_t *next) {
40  ac_source = spin1_malloc(n_ac_sources * sizeof(uint32_t*));
41  for (uint32_t n_ac=0; n_ac < n_ac_sources; n_ac++) {
42  ac_source[n_ac] = spin1_malloc(sizeof(ac_source_t));
43  if (ac_source[n_ac] == NULL) {
44  log_error("Unable to allocate DC source parameters - out of DTCM");
45  return false;

61 static REAL current_source_ac_get_offset(uint32_t cs_index, uint32_t time) {
62  if ((time >= ac_source[cs_index]->start) && (time < ac_source[cs_index]->stop)) {
63  REAL time_value = kbits((time - ac_source[cs_index]->start) << 15);
64  REAL sin_value = sink((time_value * ac_source[cs_index]->frequency) +
65  ac_source[cs_index]->phase);
66  REAL ac_current_offset = ac_source[cs_index]->offset + (
67  ac_source[cs_index]->amplitude * sin_value);
68  return ac_current_offset;
69  }
70  return ZERO;
71 }
