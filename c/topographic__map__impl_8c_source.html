131 bool synaptogenesis_dynamics_initialise(address_t sdram_sp_address) {
132  log_debug("SR init.");
133 
134  uint8_t *data = sp_structs_read_in_common(
135  sdram_sp_address, &rewiring_data, &pre_info, &post_to_pre_table);
136 
137  // Allocate current states
138  uint32_t n_states = 1;
139  if (rewiring_data.fast) {
140  n_states = rewiring_data.p_rew;
141  }
142  log_debug("Rewiring period %u, fast=%u, n_states=%u",
143  rewiring_data.p_rew, rewiring_data.fast, n_states);
144  // Add one to number of states as buffer wastes an entry
145  current_state_queue = circular_buffer_initialize(n_states + 1);
146  if (current_state_queue == NULL) {
147  log_error("Could not allocate current state queue");
148  rt_error(RTE_SWERR);
149  }
150  // Add one to number of states as buffer wastes an entry
151  free_states = circular_buffer_initialize(n_states + 1);
152  if (free_states == NULL) {
153  log_error("Could not allocate free state queue");
154  }
155  current_state_t *states = spin1_malloc(n_states * sizeof(current_state_t));
156  if (states == NULL) {
157  log_error("Could not allocate states");
158  rt_error(RTE_SWERR);
159  }
160  for (uint32_t i = 0; i < n_states; i++) {
161  _free_state(&states[i]);
162  }
163 
164  partner_init(&data);
165 
166  formation_params = spin1_malloc(
167  rewiring_data.no_pre_pops * sizeof(struct formation_params *));
168  if (formation_params == NULL) {
169  log_error("Could not initialise formation parameters");
170  rt_error(RTE_SWERR);
171  }
172  for (uint32_t i = 0; i < rewiring_data.no_pre_pops; i++) {
173  formation_params[i] = synaptogenesis_formation_init(&data);
174  }
175 
176  elimination_params = spin1_malloc(
177  rewiring_data.no_pre_pops * sizeof(struct elimination_params *));
178  if (elimination_params == NULL) {
179  log_error("Could not initialise elimination parameters");
180  rt_error(RTE_SWERR);
181  }
182  for (uint32_t i = 0; i < rewiring_data.no_pre_pops; i++) {
183  elimination_params[i] = synaptogenesis_elimination_init(&data);
184  }
185 
186  return true;
187 }

189 bool synaptogenesis_dynamics_rewire(
190  uint32_t time, spike_t *spike, address_t *synaptic_row_address,
191  uint32_t *n_bytes) {
192 
193  // Randomly choose a postsynaptic (application neuron)
194  uint32_t post_id = rand_int(rewiring_data.app_no_atoms,
195  rewiring_data.shared_seed);
196 
197  // Check if neuron is in the current machine vertex
198  if (post_id < rewiring_data.low_atom ||
199  post_id > rewiring_data.high_atom) {
200  return false;
201  }
202  post_id -= rewiring_data.low_atom;
203 
204  // Select an arbitrary synaptic element for the neurons
205  uint32_t row_offset = post_id * rewiring_data.s_max;
206  uint32_t column_offset = rand_int(rewiring_data.s_max,
207  rewiring_data.local_seed);
208  uint32_t total_offset = row_offset + column_offset;
209  post_to_pre_entry entry = post_to_pre_table[total_offset];
210  uint32_t pre_app_pop = 0, pre_sub_pop = 0, m_pop_index = 0, neuron_id = 0;
211  if (entry.neuron_index == 0xFFFF) {
212  if (!potential_presynaptic_partner(time, &pre_app_pop, &pre_sub_pop,
213  &neuron_id, spike, &m_pop_index)) {
214  return false;
215  }
216  } else {
217  pre_app_pop = entry.pop_index;
218  pre_sub_pop = entry.sub_pop_index;
219  neuron_id = entry.neuron_index;
220  }
221  pre_info_t *prepop_info = pre_info.prepop_info[pre_app_pop];
222  key_atom_info_t *key_atom_info = &prepop_info->key_atom_info[pre_sub_pop];
223  if (entry.neuron_index != 0xFFFF) {
224  *spike = key_atom_info->key | neuron_id;
225  m_pop_index = key_atom_info->m_pop_index;
226  }
227 
228  if (!population_table_get_first_address(
229  *spike, synaptic_row_address, n_bytes)) {
230  log_error("FAIL@key %d", *spike);
231  rt_error(RTE_SWERR);
232  }
233  uint32_t index = 0;
234  while (index < m_pop_index) {
235  if (!population_table_get_next_address(
236  spike, synaptic_row_address, n_bytes)) {
237  log_error("FAIL@key %d, index %d (failed at %d)",
238  *spike, m_pop_index, index);
239  rt_error(RTE_SWERR);
240  }
241  index++;
242  }
243 
244  // Saving current state
245  current_state_t *current_state = _alloc_state();
246  current_state->pre_syn_id = neuron_id;
247  current_state->post_syn_id = post_id;
248  current_state->element_exists = entry.neuron_index != 0xFFFF;
249  current_state->post_to_pre_table_entry = &post_to_pre_table[total_offset];
250  current_state->pre_population_info = prepop_info;
251  current_state->key_atom_info = key_atom_info;
252  current_state->post_to_pre.neuron_index = neuron_id;
253  current_state->post_to_pre.pop_index = pre_app_pop;
254  current_state->post_to_pre.sub_pop_index = pre_sub_pop;
255  current_state->local_seed = &rewiring_data.local_seed;
256  current_state->post_low_atom = rewiring_data.low_atom;
257  _queue_state(current_state);
258  return true;
259 }

261 bool synaptogenesis_row_restructure(uint32_t time, address_t row) {
262  current_state_t *current_state = _get_state();
263 
264  // the selected pre- and postsynaptic IDs are in current_state
265  bool return_value;
266  if (current_state->element_exists) {
267  // find the offset of the neuron in the current row
268  if (synapse_dynamics_find_neuron(
269  current_state->post_syn_id, row,
270  &current_state->weight, &current_state->delay,
271  &current_state->offset, &current_state->synapse_type)) {
272  return_value = synaptogenesis_elimination_rule(current_state,
273  elimination_params[current_state->post_to_pre.pop_index],
274  time, row);
275  } else {
276  log_debug("Post neuron %d not in row",
277  current_state->post_syn_id);
278  return_value = false;
279  }
280  } else {
281  // Can't form if the row is full
282  uint32_t no_elems = synapse_dynamics_n_connections_in_row(
283  synapse_row_fixed_region(row));
284  if (no_elems >= rewiring_data.s_max) {
285  log_debug("row is full");
286  return_value = false;
287  } else {
288  return_value = synaptogenesis_formation_rule(current_state,
289  formation_params[current_state->post_to_pre.pop_index],
290  time, row);
291  }
292  }
293 
294  _free_state(current_state);
295  return return_value;
296 }
