149 bool synaptogenesis_dynamics_initialise(
150  address_t sdram_sp_address, uint32_t *recording_regions_used) {
151  log_debug("SR init.");
152 
153  uint8_t *data = sp_structs_read_in_common(
154  sdram_sp_address, &rewiring_data, &pre_info, &post_to_pre_table);
155 
156  // Allocate current states
157  uint32_t n_states = 1;
158  if (rewiring_data.fast) {
159  n_states = rewiring_data.p_rew;
160  }
161  log_info("Rewiring period %u, fast=%u, n_states=%u",
162  rewiring_data.p_rew, rewiring_data.fast, n_states);
163  // Add one to number of states as buffer wastes an entry
164  current_state_queue = circular_buffer_initialize(n_states + 1);
165  if (current_state_queue == NULL) {
166  log_error("Could not allocate current state queue");
167  rt_error(RTE_SWERR);
168  }
169  // Add one to number of states as buffer wastes an entry
170  free_states = circular_buffer_initialize(n_states + 1);
171  if (free_states == NULL) {
172  log_error("Could not allocate free state queue");
173  }
174  current_state_t *states = spin1_malloc(n_states * sizeof(current_state_t));
175  if (states == NULL) {
176  log_error("Could not allocate states");
177  rt_error(RTE_SWERR);
178  }
179  for (uint32_t i = 0; i < n_states; i++) {
180  _free_state(&states[i]);
181  }
182 
183  partner_init(&data);
184 
185  formation_params = spin1_malloc(
186  rewiring_data.no_pre_pops * sizeof(struct formation_params *));
187  if (formation_params == NULL) {
188  log_error("Could not initialise formation parameters");
189  rt_error(RTE_SWERR);
190  }
191  for (uint32_t i = 0; i < rewiring_data.no_pre_pops; i++) {
192  formation_params[i] = synaptogenesis_formation_init(&data);
193  }
194 
195  elimination_params = spin1_malloc(
196  rewiring_data.no_pre_pops * sizeof(struct elimination_params *));
197  if (elimination_params == NULL) {
198  log_error("Could not initialise elimination parameters");
199  rt_error(RTE_SWERR);
200  }
201  for (uint32_t i = 0; i < rewiring_data.no_pre_pops; i++) {
202  elimination_params[i] = synaptogenesis_elimination_init(&data);
203  }
204 
205  rewiring_recording_index = *recording_regions_used;
206  *recording_regions_used = rewiring_recording_index + 1;
207 
208  log_info("The rewiring_recording_index is %u", rewiring_recording_index);
209 
210  return true;
211 }

213 bool synaptogenesis_dynamics_rewire(
214  uint32_t time, spike_t *spike, synaptic_row_t *synaptic_row,
215  uint32_t *n_bytes) {
216 
217  // Randomly choose a postsynaptic (application neuron)
218  uint32_t post_id = rand_int(rewiring_data.app_no_atoms,
219  rewiring_data.shared_seed);
220 
221  // Check if neuron is in the current machine vertex
222  if (post_id < rewiring_data.low_atom ||
223  post_id > rewiring_data.high_atom) {
224  return false;
225  }
226  post_id -= rewiring_data.low_atom;
227 
228  // Select an arbitrary synaptic element for the neurons
229  uint32_t row_offset = post_id * rewiring_data.s_max;
230  uint32_t column_offset = rand_int(rewiring_data.s_max,
231  rewiring_data.local_seed);
232  uint32_t total_offset = row_offset + column_offset;
233  post_to_pre_entry entry = post_to_pre_table[total_offset];
234  uint32_t pre_app_pop = 0, pre_sub_pop = 0, m_pop_index = 0, neuron_id = 0;
235  if (entry.neuron_index == 0xFFFF) {
236  if (!potential_presynaptic_partner(time, &pre_app_pop, &pre_sub_pop,
237  &neuron_id, spike, &m_pop_index)) {
238  return false;
239  }
240  } else {
241  pre_app_pop = entry.pop_index;
242  pre_sub_pop = entry.sub_pop_index;
243  neuron_id = entry.neuron_index;
244  }
245  pre_info_t *prepop_info = pre_info.prepop_info[pre_app_pop];
246  key_atom_info_t *key_atom_info = &prepop_info->key_atom_info[pre_sub_pop];
247  if (entry.neuron_index != 0xFFFF) {
248  *spike = key_atom_info->key | neuron_id;
249  m_pop_index = key_atom_info->m_pop_index;
250  }
251 
252  if (!population_table_get_first_address(*spike, synaptic_row, n_bytes)) {
253  log_error("FAIL@key %d", *spike);
254  rt_error(RTE_SWERR);
255  }
256  uint32_t index = 0;
257  while (index < m_pop_index) {
258  if (!population_table_get_next_address(spike, synaptic_row, n_bytes)) {
259  log_error("FAIL@key %d, index %d (failed at %d)",
260  *spike, m_pop_index, index);
261  rt_error(RTE_SWERR);
262  }
263  index++;
264  }
265 
266  // Saving current state
267  current_state_t *current_state = _alloc_state();
268  current_state->pre_syn_id = neuron_id;
269  current_state->post_syn_id = post_id;
270  current_state->element_exists = entry.neuron_index != 0xFFFF;
271  current_state->post_to_pre_table_entry = &post_to_pre_table[total_offset];
272  current_state->pre_population_info = prepop_info;
273  current_state->key_atom_info = key_atom_info;
274  current_state->post_to_pre.neuron_index = neuron_id;
275  current_state->post_to_pre.pop_index = pre_app_pop;
276  current_state->post_to_pre.sub_pop_index = pre_sub_pop;
277  current_state->local_seed = &rewiring_data.local_seed;
278  current_state->post_low_atom = rewiring_data.low_atom;
279  current_state->with_replacement = rewiring_data.with_replacement;
280  _queue_state(current_state);
281  return true;
282 }

290 static inline bool row_restructure(
291  uint32_t time, synaptic_row_t restrict row,
292  current_state_t *restrict current_state) {
293  // the selected pre- and postsynaptic IDs are in current_state
294 
295  if (current_state->element_exists) {
296  // find the offset of the neuron in the current row
297  if (!synapse_dynamics_find_neuron(
298  current_state->post_syn_id, row,
299  &current_state->weight, &current_state->delay,
300  &current_state->offset, &current_state->synapse_type)) {
301  log_debug("Post neuron %u not in row", current_state->post_syn_id);
302  return false;
303  }
304 
305  if (synaptogenesis_elimination_rule(current_state,
306  elimination_params[current_state->post_to_pre.pop_index],
307  time, row)) {
308  // Create recorded value
309  // (bottom bit: add/remove, next 8: local id, remainder: pre global id)
310  uint32_t pre_id = current_state->key_atom_info->lo_atom + current_state->pre_syn_id;
311  uint32_t id = current_state->post_syn_id;
312  uint32_t record_value = ELIM_FLAG | (id << ID_SHIFT) | (pre_id << PRE_ID_SHIFT);
313  structural_recording_values.time = time;
314  structural_recording_values.value = record_value;
315  recording_record(rewiring_recording_index, &structural_recording_values,
316  sizeof(structural_recording_values_t));
317 
318  return true;
319  } else {
320  return false;
321  }
322 
323  } else {
324  // Can't form if the row is full
325  uint32_t no_elems = synapse_dynamics_n_connections_in_row(
326  synapse_row_fixed_region(row));
327  if (no_elems >= rewiring_data.s_max) {
328  log_debug("row is full");
329  return false;
330  } else {
331  if (current_state->with_replacement) {
332  // A synapse can be added anywhere on the current row, so just do it
333  if (synaptogenesis_formation_rule(current_state,
334  formation_params[current_state->post_to_pre.pop_index], time, row)) {
335  // Create recorded value
336  // (bottom bit: add/remove, next 8: local id, remainder: pre global id)
337  uint32_t pre_id = current_state->key_atom_info->lo_atom
338  + current_state->pre_syn_id;
339  uint32_t id = current_state->post_syn_id;
340  uint32_t record_value = FORM_FLAG | (id << ID_SHIFT) |
341  (pre_id << PRE_ID_SHIFT);
342  structural_recording_values.time = time;
343  structural_recording_values.value = record_value;
344  recording_record(rewiring_recording_index, &structural_recording_values,
345  sizeof(structural_recording_values_t));
346 
347  return true;
348  } else {
349  return false;
350  }
351  } else {
352  // A synapse cannot be added if one exists between the current pair of neurons
353  if (!synapse_dynamics_find_neuron(
354  current_state->post_syn_id, row,
355  &(current_state->weight), &(current_state->delay),
356  &(current_state->offset), &(current_state->synapse_type))) {
357  if (synaptogenesis_formation_rule(current_state,
358  formation_params[current_state->post_to_pre.pop_index], time, row)) {
359  // Create recorded value
360  // (bottom bit: add/remove, next 8: local id, remainder: pre global id)
361  uint32_t pre_id = current_state->key_atom_info->lo_atom
362  + current_state->pre_syn_id;
363  uint32_t id = current_state->post_syn_id;
364  uint32_t record_value = FORM_FLAG | (id << ID_SHIFT) |
365  (pre_id << PRE_ID_SHIFT);
366  structural_recording_values.time = time;
367  structural_recording_values.value = record_value;
368  recording_record(rewiring_recording_index, &structural_recording_values,
369  sizeof(structural_recording_values_t));
370 
371  return true;
372  } else {
373  return false;
374  }
375  } else {
376  log_debug("Post neuron %u already in row", current_state->post_syn_id);
377  return false;
378  }
379  }
380  }
381  }
382 }
