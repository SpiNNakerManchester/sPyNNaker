106 bool neuron_initialise(
107  void *core_params_address, void *neuron_params_address,
108  void *current_sources_address, void *recording_address,
109  void *initial_values_address, uint32_t *n_rec_regions_used) {
110  // Read the neuron parameters
111  struct neuron_core_parameters *params = core_params_address;
112 
113  // Check if there is a key to use
114  use_key = params->has_key;
115 
116  // Read the neuron details
117  n_neurons = params->n_neurons_to_simulate;
118  n_neurons_peak = params->n_neurons_peak;
119  n_synapse_types = params->n_synapse_types;
120 
121  // Get colour details
122  colour_mask = (1 << params->n_colour_bits) - 1;
123 
124  // Set up ring buffer left shifts
125  uint32_t ring_buffer_bytes = n_synapse_types * sizeof(uint32_t);
126  ring_buffer_to_input_left_shifts = spin1_malloc(ring_buffer_bytes);
127  if (ring_buffer_to_input_left_shifts == NULL) {
128  log_error("Not enough memory to allocate ring buffer");
129  return false;
130  }
131 
132  // read in ring buffer to input left shifts
133  spin1_memcpy(
134  ring_buffer_to_input_left_shifts, params->ring_buffer_shifts,
135  ring_buffer_bytes);
136 
137  // The key list comes after the ring buffer shifts
138  uint32_t *neuron_keys_sdram =
139  (uint32_t *) &params->ring_buffer_shifts[n_synapse_types];
140  uint32_t neuron_keys_size = n_neurons * sizeof(uint32_t);
141  neuron_keys = spin1_malloc(neuron_keys_size);
142  if (neuron_keys == NULL) {
143  log_error("Not enough memory to allocate neuron keys");
144  return false;
145  }
146  spin1_memcpy(neuron_keys, neuron_keys_sdram, neuron_keys_size);
147 
148  // Store where the actual neuron parameters start
149  saved_neuron_params_address = neuron_params_address;
150  current_source_address = current_sources_address;
151  saved_initial_values_address = initial_values_address;
152 
153  log_info("\t n_neurons = %u, peak %u, n_synapse_types %u",
154  n_neurons, n_neurons_peak, n_synapse_types);
155 
156  // Call the neuron implementation initialise function to setup DTCM etc.
157  if (!neuron_impl_initialise(n_neurons)) {
158  return false;
159  }
160 
161  // load the neuron data into the allocated DTCM spaces.
162  if (!neuron_load_neuron_parameters(0)) {
163  return false;
164  }
165 
166  // Initialise for current sources
167  if (!current_source_initialise(current_sources_address, n_neurons)) {
168  return false;
169  }
170 
171  // load the current source data into the allocated DTCM spaces
172  if (!current_source_load_parameters(current_sources_address)) {
173  return false;
174  }
175 
176  // setup recording region
177  if (!neuron_recording_initialise(
178  recording_address, n_neurons, n_rec_regions_used)) {
179  return false;
180  }
181 
182  return true;
183 }

205 void neuron_transfer(weight_t *syns) { // EXPORTED
206  uint32_t synapse_index = 0;
207  uint32_t ring_buffer_index = 0;
208  for (uint32_t s_i = n_synapse_types; s_i > 0; s_i--) {
209  uint32_t rb_shift = ring_buffer_to_input_left_shifts[synapse_index];
210  uint32_t neuron_index = 0;
211  for (uint32_t n_i = n_neurons_peak; n_i > 0; n_i--) {
212  weight_t value = syns[ring_buffer_index];
213  if (value > 0) {
214  if (neuron_index > n_neurons) {
215  log_error("Neuron index %u out of range", neuron_index);
216  rt_error(RTE_SWERR);
217  }
218  input_t val_to_add = synapse_row_convert_weight_to_input(
219  value, rb_shift);
220  neuron_impl_add_inputs(synapse_index, neuron_index, val_to_add);
221  }
222  syns[ring_buffer_index] = 0;
223  ring_buffer_index++;
224  neuron_index++;
225  }
226  synapse_index++;
227  }
228 }
