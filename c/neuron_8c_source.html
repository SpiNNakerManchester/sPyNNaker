78 bool neuron_initialise(
79  address_t address, address_t recording_address, // EXPORTED
80  uint32_t *n_neurons_value, uint32_t *n_synapse_types_value,
81  uint32_t *incoming_spike_buffer_size, uint32_t *n_rec_regions_used) {
82  log_debug("neuron_initialise: starting");
83 
84  // init the TDMA
85  void *data_addr = address;
86  tdma_processing_initialise(&data_addr);
87 
88  // cast left over SDRAM into neuron struct.
89  struct neuron_parameters *params = data_addr;
90 
91  // Check if there is a key to use
92  use_key = params->has_key;
93 
94  // Read the spike key to use
95  key = params->transmission_key;
96 
97  // output if this model is expecting to transmit
98  if (!use_key) {
99  log_debug("\tThis model is not expecting to transmit as it has no key");
100  } else {
101  log_debug("\tThis model is expected to transmit with key = %08x", key);
102  }
103 
104  // Read the neuron details
105  n_neurons = params->n_neurons_to_simulate;
106  *n_neurons_value = n_neurons;
107  *n_synapse_types_value = params->n_synapse_types;
108 
109  // Read the size of the incoming spike buffer to use
110  *incoming_spike_buffer_size = params->incoming_spike_buffer_size;
111 
112  log_debug("\t n_neurons = %u, spike buffer size = %u", n_neurons,
113  *incoming_spike_buffer_size);
114 
115  // Call the neuron implementation initialise function to setup DTCM etc.
116  if (!neuron_impl_initialise(n_neurons)) {
117  return false;
118  }
119 
120  // load the data into the allocated DTCM spaces.
121  if (!neuron_load_neuron_parameters(address)) {
122  return false;
123  }
124 
125  // setup recording region
126  if (!neuron_recording_initialise(
127  recording_address, &recording_flags, n_neurons, n_rec_regions_used)) {
128  return false;
129  }
130 
131  return true;
132 }

147 void neuron_do_timestep_update(timer_t time, uint timer_count) { // EXPORTED
148 
149  // the phase in this timer tick im in (not tied to neuron index)
150  tdma_processing_reset_phase();
151 
152  // Prepare recording for the next timestep
153  neuron_recording_setup_for_next_recording();
154 
155  // update each neuron individually
156  for (index_t neuron_index = 0; neuron_index < n_neurons; neuron_index++) {
157 
158  // Get external bias from any source of intrinsic plasticity
159  input_t external_bias =
160  synapse_dynamics_get_intrinsic_bias(time, neuron_index);
161 
162  // call the implementation function (boolean for spike)
163  bool spike = neuron_impl_do_timestep_update(
164  neuron_index, external_bias);
165 
166  // If the neuron has spiked
167  if (spike) {
168  log_debug("neuron %u spiked at time %u", neuron_index, time);
169 
170  // Do any required synapse processing
171  synapse_dynamics_process_post_synaptic_event(time, neuron_index);
172 
173  if (use_key) {
174  tdma_processing_send_packet(
175  (key | neuron_index), 0, NO_PAYLOAD, timer_count);
176  }
177  } else {
178  log_debug("the neuron %d has been determined to not spike",
179  neuron_index);
180  }
181  }
182 
183  log_debug("time left of the timer after tdma is %d", tc[T1_COUNT]);
184 
185  // Disable interrupts to avoid possible concurrent access
186  uint cpsr = spin1_int_disable();
187 
188  // Record the recorded variables
189  neuron_recording_record(time);
190 
191  // Re-enable interrupts
192  spin1_mode_restore(cpsr);
193 }
