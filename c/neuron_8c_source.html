91 bool neuron_initialise(
92  address_t address, address_t recording_address, // EXPORTED
93  uint32_t *n_rec_regions_used) {
94  log_debug("neuron_initialise: starting");
95 
96  // init the TDMA
97  void *data_addr = address;
98  tdma_processing_initialise(&data_addr);
99 
100  // cast left over SDRAM into neuron struct.
101  struct neuron_parameters *params = data_addr;
102 
103  // Check if there is a key to use
104  use_key = params->has_key;
105 
106  // Read the spike key to use
107  key = params->transmission_key;
108 
109  // output if this model is expecting to transmit
110  if (!use_key) {
111  log_debug("\tThis model is not expecting to transmit as it has no key");
112  } else {
113  log_debug("\tThis model is expected to transmit with key = %08x", key);
114  }
115 
116  // Read the neuron details
117  n_neurons = params->n_neurons_to_simulate;
118  n_neurons_peak = params->n_neurons_peak;
119  n_synapse_types = params->n_synapse_types;
120 
121  // Set up ring buffer left shifts
122  uint32_t ring_buffer_bytes = n_synapse_types * sizeof(uint32_t);
123  ring_buffer_to_input_left_shifts = spin1_malloc(ring_buffer_bytes);
124  if (ring_buffer_to_input_left_shifts == NULL) {
125  log_error("Not enough memory to allocate ring buffer");
126  return false;
127  }
128 
129  // read in ring buffer to input left shifts
130  spin1_memcpy(
131  ring_buffer_to_input_left_shifts, params->ring_buffer_shifts,
132  ring_buffer_bytes);
133 
134  // Store where the actual neuron parameters start
135  saved_params_address = &params->ring_buffer_shifts[n_synapse_types];
136 
137  log_info("\t n_neurons = %u, peak %u", n_neurons, n_neurons_peak);
138 
139  // Call the neuron implementation initialise function to setup DTCM etc.
140  if (!neuron_impl_initialise(n_neurons)) {
141  return false;
142  }
143 
144  // load the data into the allocated DTCM spaces.
145  if (!neuron_load_neuron_parameters()) {
146  return false;
147  }
148 
149  // setup recording region
150  if (!neuron_recording_initialise(
151  recording_address, n_neurons, n_rec_regions_used)) {
152  return false;
153  }
154 
155  return true;
156 }

180 void neuron_transfer(weight_t *syns) { // EXPORTED
181  uint32_t synapse_index = 0;
182  uint32_t ring_buffer_index = 0;
183  for (uint32_t s_i = n_synapse_types; s_i > 0; s_i--) {
184  uint32_t rb_shift = ring_buffer_to_input_left_shifts[synapse_index];
185  uint32_t neuron_index = 0;
186  for (uint32_t n_i = n_neurons_peak; n_i > 0; n_i--) {
187  weight_t value = syns[ring_buffer_index];
188  if (value > 0) {
189  if (neuron_index > n_neurons) {
190  log_error("Neuron index %u out of range", neuron_index);
191  rt_error(RTE_SWERR);
192  }
193  input_t val_to_add = synapse_row_convert_weight_to_input(
194  value, rb_shift);
195  neuron_impl_add_inputs(synapse_index, neuron_index, val_to_add);
196  }
197  syns[ring_buffer_index] = 0;
198  ring_buffer_index++;
199  neuron_index++;
200  }
201  synapse_index++;
202  }
203 }
