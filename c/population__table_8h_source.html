177 static inline void get_row_addr_and_size(address_list_entry item,
178  uint32_t synaptic_rows_base_address, uint32_t neuron_id,
179  pop_table_lookup_result_t *result) {
180  uint32_t row_length = get_row_length(item);
181  uint32_t block_address = get_address(item, synaptic_rows_base_address);
182  uint32_t stride = (row_length + N_SYNAPSE_ROW_HEADER_WORDS);
183  uint32_t neuron_offset = neuron_id * stride * sizeof(uint32_t);
184 
185  result->row_address = (synaptic_row_t) (block_address + neuron_offset);
186  result->n_bytes_to_transfer = stride * sizeof(uint32_t);
187 
188  log_debug("neuron_id = %u, block_address = 0x%.8x, "
189  "row_length = %u, row_address = 0x%.8x, n_bytes = %u",
190  neuron_id, block_address, row_length, result->row_address,
191  result->n_bytes_to_transfer);
192 }
