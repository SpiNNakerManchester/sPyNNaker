42address_t weight_initialise(
43 address_t address, uint32_t n_synapse_types,
44 UNUSED uint32_t *ring_buffer_to_input_buffer_left_shifts) {
45 // Copy plasticity region data from address
46 // **NOTE** this seems somewhat safer than relying on sizeof
47 additive_one_term_config_t *config = (additive_one_term_config_t *) address;
48
49 plasticity_weight_region_data_t *dtcm_copy = plasticity_weight_region_data =
50 spin1_malloc(sizeof(plasticity_weight_region_data_t) * n_synapse_types);
51 if (dtcm_copy == NULL) {
52 log_error("Could not initialise weight region data");
53 return NULL;
54 }
55
56 weight_shift = spin1_malloc(sizeof(uint32_t) * n_synapse_types);
57 if (weight_shift == NULL) {
58 log_error("Could not initialise weight region data");
59 return NULL;
60 }
61
62 for (uint32_t s = 0; s < n_synapse_types; s++, config++) {
63 dtcm_copy[s].min_weight = config->min_weight;
64 dtcm_copy[s].max_weight = config->max_weight;
65 dtcm_copy[s].a2_plus = config->a2_plus;
66 dtcm_copy[s].a2_minus = config->a2_minus;
67
68 // Copy weight shift
69 weight_shift[s] = ring_buffer_to_input_buffer_left_shifts[s];
70
71 log_debug("\tSynapse type %u: Min weight:%k, Max weight:%k, A2+:%k, A2-:%k",
72 s, dtcm_copy[s].min_weight, dtcm_copy[s].max_weight,
73 dtcm_copy[s].a2_plus, dtcm_copy[s].a2_minus);
74 }
75
76 // Return end address of region
77 return (address_t) config;
78}
