142 static inline input_t *synapse_types_get_excitatory_input(
143  synapse_param_t *parameters) {
144  if (parameters->exc2.synaptic_input_value >= 0.001
145  && parameters->multiplicator == 0
146  && parameters->exc2_old == 0) {
147  parameters->multiplicator = parameters->exc.synaptic_input_value;
148  } else if (parameters->exc2.synaptic_input_value < 0.001) {
149  parameters->multiplicator = 0;
150  }
151 
152  parameters->exc2_old = parameters->exc2.synaptic_input_value;
153 
154  excitatory_response[0] = 0; // I think?
155  excitatory_response[1] =
156  parameters->exc2.synaptic_input_value * parameters->multiplicator
157  * SCALING_FACTOR;
158  return &excitatory_response[0];
159 }
