137 static inline input_t *synapse_types_get_excitatory_input(
138  input_t *excitatory_response, synapse_param_t *parameters) {
139  if (parameters->exc2.synaptic_input_value >= 0.001
140  && parameters->multiplicator == 0
141  && parameters->exc2_old == 0) {
142  parameters->multiplicator = parameters->exc.synaptic_input_value;
143  } else if (parameters->exc2.synaptic_input_value < 0.001) {
144  parameters->multiplicator = 0;
145  }
146 
147  parameters->exc2_old = parameters->exc2.synaptic_input_value;
148 
149  excitatory_response[0] = 0;
150  excitatory_response[1] =
151  parameters->exc2.synaptic_input_value * parameters->multiplicator
152  * parameters->scaling_factor;
153  return &excitatory_response[0];
154 }
