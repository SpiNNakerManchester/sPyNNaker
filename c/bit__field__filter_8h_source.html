36 static bool bit_field_filter_initialise(address_t bitfield_region_address) {
37  filter_region_t *filter_region = (filter_region_t *)
38  bitfield_region_address;
39 
40  log_info("Found %d bitfields of which %d have redundancy "
41  "of which %d merged in", filter_region->n_filters,
42  filter_region->n_redundancy_filters,
43  filter_region->n_merged_filters);
44 
45  // try allocating DTCM for starting array for bitfields
46  connectivity_bit_field =
47  spin1_malloc(sizeof(bit_field_t) * population_table_length());
48  if (connectivity_bit_field == NULL) {
49  log_warning(
50  "couldn't initialise basic bit field holder. Will end up doing"
51  " possibly more DMA's during the execution than required");
52  return true;
53  }
54 
55  // set all to NULL for when they not filled in.
56  for (uint32_t cur_bit_field = 0; cur_bit_field < population_table_length();
57  cur_bit_field++) {
58  connectivity_bit_field[cur_bit_field] = NULL;
59  }
60 
61  // try allocating dtcm for each bit field which has some redundancy.
62  // NOTE: bitfields with no redundancy are better suited to not being read
63  // in as the master pop table will not bother checking for speed boosts.
64  for (int cur_bit_field = filter_region->n_merged_filters;
65  cur_bit_field < filter_region->n_redundancy_filters;
66  cur_bit_field++) {
67  // get the key associated with this bitfield
68  uint32_t key = filter_region->filters[cur_bit_field].key;
69 
70  // get n words needed for this bitfield
71  uint32_t n_words = get_bit_field_size(
72  filter_region->filters[cur_bit_field].n_atoms);
73 
74  // locate the position in the array to match the master pop element.
75  int position_in_array =
76  population_table_position_in_the_master_pop_array(key);
77 
78  log_debug("putting key %d in position %d", key, position_in_array);
79 
80  // alloc sdram into right region
81  connectivity_bit_field[position_in_array] = spin1_malloc(
82  sizeof(bit_field_t) * n_words);
83  if (connectivity_bit_field[position_in_array] == NULL) {
84  log_debug(
85  "could not initialise bit field for key %d, packets with "
86  "that key will use a DMA to check if the packet targets "
87  "anything within this core. Potentially slowing down the "
88  "execution of neurons on this core.", key);
89  failed_bit_field_reads++;
90  } else { // read in bit field into correct location
91  // read in the bits for the bitfield (think this avoids a for loop)
92  spin1_memcpy(
93  connectivity_bit_field[position_in_array],
94  filter_region->filters[cur_bit_field].data,
95  sizeof(uint32_t) * n_words);
96 
97  // print out the bit field for debug purposes
98  log_debug("bit field for key %d is :", key);
99  for (uint32_t bit_field_word_index = 0;
100  bit_field_word_index < n_words;
101  bit_field_word_index++){
102  log_debug("%x",
103  connectivity_bit_field[position_in_array][
104  bit_field_word_index]);
105  }
106  }
107  }
108  population_table_set_connectivity_bit_field(connectivity_bit_field);
109  return true;
110 }
