110 static inline post_event_window_t post_events_get_window_delayed(
111  const post_event_history_t *events, uint32_t begin_time,
112  uint32_t end_time) {
113  // Start at end event - beyond end of post-event history
114  const uint32_t count = events->count_minus_one + 1;
115  const uint32_t *end_event_time = events->times + count;
116  const uint32_t *event_time = end_event_time;
117  const post_trace_t *event_trace = events->traces + count;
118 
119  post_event_window_t window;
120  do {
121  // If this event is still in the future, set it as the end
122  if (*event_time > end_time) {
123  end_event_time = event_time;
124  }
125 
126  // Cache pointer to this event as potential next event and go back one
127  // event.
128  // **NOTE** next_time can be invalid
129  window.next_time = event_time--;
130  window.next_trace = event_trace--;
131 
132  // Keep looping while event occurred after start of window and we
133  // haven't hit beginning of array...
134  } while (*event_time > begin_time && event_time != events->times);
135 
136  // Deference event to use as previous
137  window.prev_time = *event_time;
138  window.prev_trace = *event_trace;
139  window.prev_time_valid = event_time != events->times;
140 
141  // Calculate number of events
142  window.num_events = (end_event_time - window.next_time);
143 
144  // Return window
145  return window;
146 }

152 static inline post_event_window_t post_events_next(
153  post_event_window_t window) {
154  // Update previous time and increment next time
155  window.prev_time = *window.next_time++;
156  window.prev_trace = *window.next_trace++;
157 
158  // Time will now be valid for sure!
159  window.prev_time_valid = 1;
160 
161  // Decrement remaining events
162  window.num_events--;
163  return window;
164 }

171 static inline void post_events_add(
172  uint32_t time, post_event_history_t *events, post_trace_t trace) {
173  if (events->count_minus_one < MAX_POST_SYNAPTIC_EVENTS - 1) {
174  // If there's still space, store time at current end
175  // and increment count minus 1
176  const uint32_t new_index = ++events->count_minus_one;
177  events->times[new_index] = time;
178  events->traces[new_index] = trace;
179  } else {
180  // Otherwise Shuffle down elements
181  // **NOTE** 1st element is always an entry at time 0
182  for (uint32_t e = 2; e < MAX_POST_SYNAPTIC_EVENTS; e++) {
183  events->times[e - 1] = events->times[e];
184  events->traces[e - 1] = events->traces[e];
185  }
186 
187  // Stick new time at end
188  events->times[MAX_POST_SYNAPTIC_EVENTS - 1] = time;
189  events->traces[MAX_POST_SYNAPTIC_EVENTS - 1] = trace;
190  }
191 }

199  const post_event_history_t *post_event_history,
200  uint32_t begin_time, uint32_t end_time, uint32_t delay_dendritic) {
201  log_info(" ## printing post window ##");

204

209  post_window.num_events, delayed_post_time,
210  *post_window.next_trace);
211 
212  post_window = post_events_next(post_window);
213  }
