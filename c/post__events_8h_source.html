89 static inline post_event_window_t post_events_get_window_delayed(
90  const post_event_history_t *events, uint32_t begin_time,
91  uint32_t end_time) {
92  // Start at end event - beyond end of post-event history
93  const uint32_t count = events->count_minus_one + 1;
94  const uint32_t *end_event_time = events->times + count;
95  const uint32_t *event_time = end_event_time;
96 
97  post_event_window_t window;
98  do {
99  // If this event is still in the future, set it as the end
100  if (*event_time > end_time) {
101  end_event_time = event_time;
102  }
103 
104  // Cache pointer to this event as potential
105  // Next event and go back one event
106  // **NOTE** next_time can be invalid
107  window.next_time = event_time--;
108  }
109 
110  // Keep looping while event occurred after start
111  // Of window and we haven't hit beginning of array
112  while (*event_time > begin_time && event_time != events->times);
113 
114  // Deference event to use as previous
115  window.prev_time = *event_time;
116  window.prev_time_valid = event_time != events->times;
117 
118  // Calculate number of events
119  window.num_events = (end_event_time - window.next_time);
120 
121  // Using num_events, find next and previous traces
122  const post_trace_t *end_event_trace = events->traces + count;
123  window.next_trace = (end_event_trace - window.num_events);
124  window.prev_trace = *(window.next_trace - 1);
125 
126  // Return window
127  return window;
128 }

131 static inline post_event_window_t post_events_next(
132  post_event_window_t window) {
133  // Update previous time and increment next time
134  window.prev_time = *window.next_time++;
135  window.prev_trace = *window.next_trace++;
136 
137  // Time will now be valid for sure!
138  window.prev_time_valid = 1;
139 
140  // Decrement remaining events
141  window.num_events--;
142  return window;
143 }

146 static inline void post_events_add(
147  uint32_t time, post_event_history_t *events, post_trace_t trace) {
148  if (events->count_minus_one < MAX_POST_SYNAPTIC_EVENTS - 1) {
149  // If there's still space, store time at current end
150  // and increment count minus 1
151  const uint32_t new_index = ++events->count_minus_one;
152  events->times[new_index] = time;
153  events->traces[new_index] = trace;
154  } else {
155  // Otherwise Shuffle down elements
156  // **NOTE** 1st element is always an entry at time 0
157  for (uint32_t e = 2; e < MAX_POST_SYNAPTIC_EVENTS; e++) {
158  events->times[e - 1] = events->times[e];
159  events->traces[e - 1] = events->traces[e];
160  }
161 
162  // Stick new time at end
163  events->times[MAX_POST_SYNAPTIC_EVENTS - 1] = time;
164  events->traces[MAX_POST_SYNAPTIC_EVENTS - 1] = trace;
165  }
166 }
