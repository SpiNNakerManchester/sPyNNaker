122 static inline update_state_t timing_apply_pre_spike(
123  uint32_t time, UNUSED pre_trace_t trace, UNUSED uint32_t last_pre_time,
124  UNUSED pre_trace_t last_pre_trace, uint32_t last_post_time,
125  post_trace_t last_post_trace, update_state_t previous_state) {
126  // Get time of event relative to last post-synaptic event
127  uint32_t time_since_last_post = time - last_post_time;
128 
129  log_debug("\t\t\ttime_since_last_post:%u, post_window_length:%u",
130  time_since_last_post, last_post_trace);
131 
132  if (time_since_last_post < last_post_trace) {
133  if (previous_state.accumulator >
134  plasticity_trace_region_data.accumulator_depression_plus_one) {
135  // If accumulator's not going to hit depression limit,
136  // decrement it
137  previous_state.accumulator--;
138  log_debug("\t\t\t\tDecrementing accumulator=%d",
139  previous_state.accumulator);
140  } else {
141  // Otherwise, reset accumulator and apply depression
142  log_debug("\t\t\t\tApplying depression");
143 
144  previous_state.accumulator = 0;
145  previous_state.weight_state = weight_one_term_apply_depression(
146  previous_state.weight_state, STDP_FIXED_POINT_ONE);
147  }
148  }
149 
150  return previous_state;
151 }

172 
173  // If spikes don't coincide
174  if (time_since_last_pre > 0) {
175  // If this post-spike has arrived within the last pre window
176  if (time_since_last_pre < last_pre_trace) {
