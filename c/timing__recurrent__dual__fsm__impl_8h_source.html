117 static inline update_state_t timing_apply_pre_spike(
118  uint32_t time, UNUSED pre_trace_t trace, UNUSED uint32_t last_pre_time,
119  UNUSED pre_trace_t last_pre_trace, uint32_t last_post_time,
120  post_trace_t last_post_trace, update_state_t previous_state) {
121  // Get time of event relative to last post-synaptic event
122  uint32_t time_since_last_post = time - last_post_time;
123 
124  log_debug("\t\t\ttime_since_last_post:%u, post_window_length:%u",
125  time_since_last_post, last_post_trace);
126 
127  if (time_since_last_post < last_post_trace) {
128  if (previous_state.accumulator >
129  plasticity_trace_region_data.accumulator_depression_plus_one) {
130  // If accumulator's not going to hit depression limit,
131  // decrement it
132  previous_state.accumulator--;
133  log_debug("\t\t\t\tDecrementing accumulator=%d",
134  previous_state.accumulator);
135  } else {
136  // Otherwise, reset accumulator and apply depression
137  log_debug("\t\t\t\tApplying depression");
138 
139  previous_state.accumulator = 0;
140  previous_state.weight_state = weight_one_term_apply_depression(
141  previous_state.weight_state, STDP_FIXED_POINT_ONE);
142  }
143  }
144 
145  return previous_state;
146 }

158 static inline update_state_t timing_apply_post_spike(
159  uint32_t time, UNUSED post_trace_t trace, uint32_t last_pre_time,
160  pre_trace_t last_pre_trace, UNUSED uint32_t last_post_time,
161  UNUSED post_trace_t last_post_trace, update_state_t previous_state) {
162  // Get time of event relative to last pre-synaptic event
163  uint32_t time_since_last_pre = time - last_pre_time;
164 
165  log_debug("\t\t\ttime_since_last_pre:%u, pre_window_length:%u",
166  time_since_last_pre, last_pre_trace);
167 
168  // If spikes don't coincide
169  if (time_since_last_pre > 0) {
170  // If this post-spike has arrived within the last pre window
171  if (time_since_last_pre < last_pre_trace) {
172  if (previous_state.accumulator <
173  plasticity_trace_region_data.accumulator_potentiation_minus_one) {
174  // If accumulator's not going to hit potentiation limit,
175  // increment it
176  previous_state.accumulator++;
177  log_debug("\t\t\t\tIncrementing accumulator=%d",
178  previous_state.accumulator);
179  } else {
180  // Otherwise, reset accumulator and apply potentiation
181  log_debug("\t\t\t\tApplying potentiation");
182 
183  previous_state.accumulator = 0;
184  previous_state.weight_state = weight_one_term_apply_potentiation(
185  previous_state.weight_state, STDP_FIXED_POINT_ONE);
186  }
187  }
188  }
189 
190  return previous_state;
