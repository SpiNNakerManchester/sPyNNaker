40 bool synapse_dynamics_initialise(
41  UNUSED address_t address, uint32_t n_neurons, uint32_t n_synapse_types,
42  UNUSED uint32_t *ring_buffer_to_input_buffer_left_shifts) {
43  uint32_t n_neurons_power_2 = n_neurons;
44  uint32_t log_n_neurons = 1;
45  if (n_neurons != 1) {
46  if (!is_power_of_2(n_neurons)) {
47  n_neurons_power_2 = next_power_of_2(n_neurons);
48  }
49  log_n_neurons = ilog_2(n_neurons_power_2);
50  }
51  uint32_t n_synapse_types_power_2 = n_synapse_types;
52  synapse_type_bits = 1;
53  if (n_synapse_types != 1) {
54  if (!is_power_of_2(n_synapse_types)) {
55  n_synapse_types_power_2 = next_power_of_2(n_synapse_types);
56  }
57  synapse_type_bits = ilog_2(n_synapse_types_power_2);
58  }
59  synapse_type_index_bits = log_n_neurons + synapse_type_bits;
60  synapse_index_bits = log_n_neurons;
61  synapse_index_mask = (1 << synapse_index_bits) - 1;
62  synapse_type_mask = (1 << synapse_type_bits) - 1;
63  return true;
64 }

100 bool synapse_dynamics_find_neuron(
101  uint32_t id, synaptic_row_t row, weight_t *weight, uint16_t *delay,
102  uint32_t *offset, uint32_t *synapse_type) {
103  synapse_row_fixed_part_t *fixed_region = synapse_row_fixed_region(row);
104  int32_t fixed_synapse = synapse_row_num_fixed_synapses(fixed_region);
105  uint32_t *synaptic_words = synapse_row_fixed_weight_controls(fixed_region);
106 
107  // Loop through plastic synapses
108  for (; fixed_synapse > 0; fixed_synapse--) {
109 
110  // Get next control word (auto incrementing)
111  // Check if index is the one I'm looking for
112  uint32_t synaptic_word = *synaptic_words++;
113  if (synapse_row_sparse_index(synaptic_word, synapse_index_mask) == id) {
114  *offset = synapse_row_num_fixed_synapses(fixed_region) -
115  fixed_synapse;
116  *weight = synapse_row_sparse_weight(synaptic_word);
117  *delay = synapse_row_sparse_delay(synaptic_word,
118  synapse_type_index_bits);
119  *synapse_type = synapse_row_sparse_type(
120  synaptic_word, synapse_index_bits, synapse_type_mask);
121  return true;
122  }
123  }
124 
125  return false;
126 }

142 static inline uint32_t _fixed_synapse_convert(
143  uint32_t id, weight_t weight, uint32_t delay, uint32_t type) {
144  uint32_t new_synapse = weight << (32 - SYNAPSE_WEIGHT_BITS);
145  new_synapse |= ((delay & ((1 << SYNAPSE_DELAY_BITS) - 1)) <<
146  synapse_type_index_bits);
147  new_synapse |= ((type & ((1 << synapse_type_bits) - 1)) <<
148  synapse_index_bits);
149  new_synapse |= (id & ((1 << synapse_type_index_bits) - 1));
150  return new_synapse;
151 }

153 bool synapse_dynamics_add_neuron(
154  uint32_t id, synaptic_row_t row, weight_t weight,
155  uint32_t delay, uint32_t type) {
156  synapse_row_fixed_part_t *fixed_region = synapse_row_fixed_region(row);
157  int32_t fixed_synapse = synapse_row_num_fixed_synapses(fixed_region);
158  uint32_t *synaptic_words = synapse_row_fixed_weight_controls(fixed_region);
159  uint32_t new_synapse = _fixed_synapse_convert(id, weight, delay, type);
160 
161  // Add control word at offset
162  synaptic_words[fixed_synapse] = new_synapse;
163 
164  // Increment FF
165  fixed_region->num_fixed++;
166  return true;
167 }
