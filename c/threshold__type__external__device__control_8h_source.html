75  uint32_t key, uint32_t payload, bool with_payload) {
76  // Wait until the expected time to send
77  while (tc[T1_COUNT] > expected_time) {
78  // Do Nothing
79  }
80  expected_time -= time_between_spikes;
81 
82  if (with_payload) {

93 static inline uint get_payload(enum send_type type, accum value) {
94  switch (type) {
95  case SEND_TYPE_INT:

97  case SEND_TYPE_UINT:

99  case SEND_TYPE_ACCUM:

101  case SEND_TYPE_UACCUM:
102  return bitsuk((unsigned accum) value);
103  case SEND_TYPE_FRACT:
104  return int_bits(bitslr((long fract) value));
105  case SEND_TYPE_UFRACT:
106  return bitsulr((long unsigned fract) value);
107  default:
108  log_error("Unknown enum value %u", value);
109  rt_error(RTE_SWERR);
110  }
111  return 0;

114 static bool threshold_type_is_above_threshold(
115  state_t value, threshold_type_t *threshold_type) {
116  if (threshold_type->time_until_next_send == 0) {
117  if (threshold_type->value_as_payload) {
118  accum value_to_send = value;
119  if (value > threshold_type->max_value) {
120  value_to_send = threshold_type->max_value;
121  }
122  if (value < threshold_type->min_value) {
123  value_to_send = threshold_type->min_value;
124  }
125 
126  uint payload = get_payload(threshold_type->type,
127  value_to_send * threshold_type->value_as_payload);
128 
129  log_debug("Sending key=0x%08x payload=0x%08x",
130  threshold_type->key, payload);
131  send_packet(threshold_type->key, payload, true);
132  } else {
133  log_debug("Sending key=0x%08x", threshold_type->key);
134  send_packet(threshold_type->key, 0, false);
135  }
136 
137  threshold_type->time_until_next_send =
138  threshold_type->timesteps_between_sending;
139  }
140  --threshold_type->time_until_next_send;
141  return false;
142 }
