79 static inline void send_packet(
80  uint32_t key, uint32_t payload, bool with_payload) {
81  // Wait until the expected time to send
82  while (tc[T1_COUNT] > expected_time) {
83  // Do Nothing
84  }
85  expected_time -= time_between_spikes;
86 
87  if (with_payload) {

89  spin1_delay_us(1);
90  }
91  } else {// Send the spike

93  spin1_delay_us(1);
94  }
95  }

102 static inline uint get_payload(enum send_type type, accum value) {
103  switch (type) {
104  case SEND_TYPE_INT:
105  return _int_bits((int) value);
106  case SEND_TYPE_UINT:
107  return (uint) value;
108  case SEND_TYPE_ACCUM:
109  return _int_bits(bitsk(value));
110  case SEND_TYPE_UACCUM:
111  return bitsuk((unsigned accum) value);
112  case SEND_TYPE_FRACT:
113  return _int_bits(bitslr((long fract) value));
114  case SEND_TYPE_UFRACT:
115  return bitsulr((long unsigned fract) value);
116  default:
117  log_error("Unknown enum value %u", value);
118  rt_error(RTE_SWERR);
119  }
120  return 0;
121 }

127 static bool threshold_type_is_above_threshold(
128  state_t value, threshold_type_t *threshold_type) {
129  if (threshold_type->time_until_next_send == 0) {
130  if (threshold_type->value_as_payload) {
131  accum value_to_send = value;
132  if (value > threshold_type->max_value) {
133  value_to_send = threshold_type->max_value;
134  }
135  if (value < threshold_type->min_value) {
136  value_to_send = threshold_type->min_value;
137  }
138 
139  uint payload = get_payload(threshold_type->type,
140  value_to_send * threshold_type->value_as_payload);
141 
142  log_debug("Sending key=0x%08x payload=0x%08x",
143  threshold_type->key, payload);
144  send_packet(threshold_type->key, payload, true);
145  } else {
146  log_debug("Sending key=0x%08x", threshold_type->key);
147  send_packet(threshold_type->key, 0, false);
148  }
149 
150  threshold_type->time_until_next_send =
151  threshold_type->timesteps_between_sending;
152  }
153  --threshold_type->time_until_next_send;
154  return false;
155 }
