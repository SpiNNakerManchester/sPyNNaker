120 static inline void send_to_motor(uint32_t direction, uint32_t the_speed) {
121  uint32_t direction_key = direction | key;
122  while (!spin1_send_mc_packet(direction_key, the_speed, WITH_PAYLOAD)) {
123  spin1_delay_us(1);
124  }
125  if (delay_time > 0) {
126  spin1_delay_us(delay_time);
127  }
128 }

135 static inline void do_motion(
136  direction_t direction_index, direction_t opposite_index,
137  const char *direction, const char *opposite) {
138  int direction_count = counters[direction_index - 1];
139  int opposite_count = counters[opposite_index - 1];
140  int delta = direction_count - opposite_count;
141  log_debug("%s = %d, %s = %d, delta = %d, threshold = %u",
142  direction, direction_count, opposite, opposite_count, delta,
143  delta_threshold);
144 
145  if (delta >= delta_threshold) {
146  log_debug("Moving %s", direction);
147  last_speed[direction_index - 1] = speed;
148  last_speed[opposite_index - 1] = 0;
149  send_to_motor(direction_index, speed);
150  } else if (delta <= -delta_threshold) {
151  log_debug("Moving %s", direction);
152  last_speed[direction_index - 1] = 0;
153  last_speed[opposite_index - 1] = speed;
154  send_to_motor(opposite_index, speed);
155  } else if (!continue_if_not_different) {
156  log_debug("Motion is indeterminate in %s-%s direction",
157  direction, opposite);
158  last_speed[direction_index - 1] = 0;
159  last_speed[opposite_index - 1] = 0;
160  send_to_motor(direction_index, 0);
161  }
162 }

170 static inline void do_update(
171  direction_t direction_index, direction_t opposite_index,
172  const char *direction, const char *opposite) {
173  int direction_speed = last_speed[direction_index - 1];
174  int opposite_speed = last_speed[opposite_index - 1];
175  int delta = direction_speed - opposite_speed;
176  if (delta > 0) {
177  log_debug("Resending %s = %d", direction, direction_speed);
178  send_to_motor(direction_index, direction_speed);
179  } else if (delta < 0) {
180  log_debug("Resending %s = %d", opposite, opposite_speed);
181  send_to_motor(opposite_index, opposite_speed);
182  } else {
183  log_debug("Resending No Motion in the %s-%s direction", direction,
184  opposite);
185  send_to_motor(direction_index, 0);
186  }
187 }

194 static void timer_callback(UNUSED uint unused0, UNUSED uint unused1) {
195  time++;
196 
197  log_debug("Timer tick %d", time);
198 
199  if ((infinite_run != TRUE) && (time == simulation_ticks)) {
200  simulation_handle_pause_resume(NULL);
201  log_info("Simulation complete.\n");
202  simulation_ready_to_read();
203  }
204 
205  // Process the incoming spikes
206  spike_t s;
207  uint32_t nid;
208  while (in_spikes_get_next_spike(&s)) {
209  nid = (s & NEURON_ID_MASK);
210 
211  if (nid < N_COUNTERS) {
212  counters[nid]++;
213  } else {
214  log_debug("Received spike from unknown neuron %d", nid);
215  }
216  }
217 
218  // Work out if there is any motion
219  if (time % sample_time == 0) {
220  // Do motion in pairs
221  do_motion(MOTION_FORWARD, MOTION_BACK, "Forwards", "Backwards");
222  do_motion(MOTION_LEFT, MOTION_RIGHT, "Left", "Right");
223  do_motion(MOTION_CLOCKWISE, MOTION_C_CLOCKWISE, "Clockwise",
224  "Anti-clockwise");
225 
226  // Reset the counters
227  for (uint32_t i = 0; i < N_COUNTERS; i++) {
228  counters[i] = 0;
229  }
230  } else if ((time % update_time) == 0) {
231  // Do updates in pairs
232  do_update(MOTION_FORWARD, MOTION_BACK, "Forwards", "Backwards");
233  do_update(MOTION_LEFT, MOTION_RIGHT, "Left", "Right");
234  do_update(MOTION_CLOCKWISE, MOTION_C_CLOCKWISE, "Clockwise",
235  "Anti-clockwise");
236  }
237 }

241 static void read_parameters(motor_control_config_t *config_region) {
242  log_info("Reading parameters from 0x%.8x", config_region);
243  key = config_region->key;
244  speed = config_region->speed;
245  sample_time = config_region->sample_time;
246  update_time = config_region->update_time;
247  delay_time = config_region->delay_time;
248  delta_threshold = config_region->delta_threshold;
249  continue_if_not_different = config_region->continue_if_not_different;
250 
251  // Allocate the space for the schedule
252  counters = spin1_malloc(N_COUNTERS * sizeof(int));
253  last_speed = spin1_malloc(N_COUNTERS * sizeof(int));
254 
255  for (uint32_t i = 0; i < N_COUNTERS; i++) {
256  counters[i] = 0;
257  last_speed[i] = 0;
258  }
259 
260  log_info("Key = %d, speed = %d, sample_time = %d, update_time = %d,"
261  " delay_time = %d, delta_threshold = %d, continue_if_not_different = %d",
262  key, speed, sample_time, update_time, delay_time, delta_threshold,
263  continue_if_not_different);
264 }

305 static bool initialize(uint32_t *timer_period) {
306  log_info("initialise: started");
307 
308  // Get the address this core's DTCM data starts at from SRAM
309  data_specification_metadata_t *ds_regions =
310  data_specification_get_data_address();
311 
312  // Read the header
313  if (!data_specification_read_header(ds_regions)) {
314  return false;
315  }
316 
317  // Get the timing details and set up the simulation interface
318  if (!simulation_initialise(
319  data_specification_get_region(SYSTEM_REGION, ds_regions),
320  APPLICATION_NAME_HASH, timer_period, &simulation_ticks,
321  &infinite_run, &time, SDP, DMA)) {
322  return false;
323  }
324 
325  simulation_set_provenance_function(
326  c_main_store_provenance_data,
327  data_specification_get_region(PROVENANCE_DATA_REGION, ds_regions));
328 
329  // Get the parameters
330  read_parameters(data_specification_get_region(PARAMS_REGION, ds_regions));
331 
332  log_info("initialise: completed successfully");
333 
334  return true;
335 }

338 void c_main(void) {
339  // Initialise
340  uint32_t timer_period = 0;
341  if (!initialize(&timer_period)) {
342  log_error("Error in initialisation - exiting!");
343  rt_error(RTE_SWERR);
344  }
345 
346  // Initialise the incoming spike buffer
347  if (!in_spikes_initialize_spike_buffer(8192)) {
348  return;
349  }
350 
351  // Set timer_callback
352  spin1_set_timer_tick(timer_period);
353 
354  // Register callbacks
355  spin1_callback_on(MC_PACKET_RECEIVED, incoming_spike_callback, MC);
356  spin1_callback_on(
357  MCPL_PACKET_RECEIVED, incoming_spike_callback_payload, MC);
358  spin1_callback_on(TIMER_TICK, timer_callback, TIMER);
359 
360  // Start the time at "-1" so that the first tick will be 0
361  time = UINT32_MAX;
362  simulation_run();
363 }
