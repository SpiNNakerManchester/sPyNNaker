228 static inline void print_master_population_table(void) {
229 #if log_level >= LOG_DEBUG
230  log_info("Master_population\n");
231  for (uint32_t i = 0; i < master_population_table_length; i++) {
232  master_population_table_entry entry = master_population_table[i];
233  log_info("key: 0x%08x, mask: 0x%08x", entry.key, entry.mask);
234  int count = entry.count;
235  int start = entry.start;
236  if (entry.extra_info_flag) {
237  extra_info extra = address_list[start].extra;
238  start += 1;
239  log_info(" core_mask: 0x%08x, core_shift: %u, n_neurons: %u, n_words: %u",
240  extra.core_mask, extra.mask_shift, extra.n_neurons, extra.n_words);
241  }
242  for (uint16_t j = start; j < (start + count); j++) {
243  address_and_row_length addr = address_list[j].addr;
244  if (addr.address == INVALID_ADDRESS) {
245  log_info(" index %d: INVALID", j);
246  } else if (!addr.is_single) {
247  log_info(" index %d: offset: %u, address: 0x%08x, row_length: %u",
248  j, get_offset(addr), get_address(addr), get_row_length(addr));
249  } else {
250  log_info(" index %d: offset: %u, address: 0x%08x, single",
251  j, addr.address, get_direct_address(addr));
252  }
253  }
254  }
255  log_info("Population table has %u entries", master_population_table_length);
256 #endif

294 bool population_table_load_bitfields(filter_region_t *filter_region) {
295 
296  if (master_population_table_length == 0) {
297  return true;
298  }
299  // try allocating DTCM for starting array for bitfields
300  connectivity_bit_field =
301  spin1_malloc(sizeof(bit_field_t) * master_population_table_length);
302  if (connectivity_bit_field == NULL) {
303  log_warning(
304  "Couldn't initialise basic bit field holder. Will end up doing"
305  " possibly more DMA's during the execution than required."
306  " We required %d bytes where %d are available",
307  sizeof(bit_field_t) * master_population_table_length,
308  sark_heap_max(sark.heap, 0));
309  return true;
310  }
311 
312  // Go through the population table, and the relevant bitfield list, both
313  // of which are ordered by key...
314  uint32_t bf_i = 0;
315  uint32_t n_filters = filter_region->n_filters;
316  filter_info_t* filters = filter_region->filters;
317  for (uint32_t mp_i = 0; mp_i < master_population_table_length; mp_i++) {
318  connectivity_bit_field[mp_i] = NULL;
319 
320  log_debug("Master pop key: 0x%08x, mask: 0x%08x",
321  master_population_table[mp_i].key, master_population_table[mp_i].mask);
322 
323 #ifdef LOG_DEBUG
324  // Sanity checking code; not needed in normal operation, and costs ITCM
325  // With both things being in key order, this should never happen...
326  if (bf_i < n_filters &&
327  filters[bf_i].key < master_population_table[mp_i].key) {
328  log_error("Skipping bitfield %d for key 0x%08x", bf_i, filters[bf_i].key);
329  rt_error(RTE_SWERR);
330  }
331 #endif
332 
333  // While there is a match, keep track of the start and end; note this
334  // may recheck the first entry, but there might not be a first entry if
335  // we have already gone off the end of the bitfield array
336  uint32_t start = bf_i;
337  uint32_t n_words_total = 0;
338  uint32_t useful = 0;
339  log_debug("Starting with bit field %d with key 0x%08x", bf_i, filters[bf_i].key);
340  while (bf_i < n_filters && matches(mp_i, filters[bf_i].key)) {
341  log_debug("Using bit field %d with key 0x%08x, merged %d, redundant %d",
342  bf_i, filters[bf_i].key, filters[bf_i].merged, filters[bf_i].all_ones);
343  n_words_total += get_bit_field_size(filters[bf_i].n_atoms);
344  useful += !(filters[bf_i].merged || filters[bf_i].all_ones);
345  bf_i++;
346  }
347 
348  // If there is something to copy, copy them in now
349  log_debug("Ended with bit field %d with key 0x%08x, n_words %d, useful %d",
350  bf_i, filters[bf_i].key, n_words_total, useful);
351  if (bf_i != start && useful) {
352  // Try to allocate all the bitfields for this entry
353  connectivity_bit_field[mp_i] = spin1_malloc(
354  sizeof(bit_field_t) * n_words_total);
355  if (connectivity_bit_field[mp_i] == NULL) {
356  // If allocation fails, we can still continue
357  log_debug(
358  "Could not initialise bit field for key %d, packets with "
359  "that key will use a DMA to check if the packet targets "
360  "anything within this core. Potentially slowing down the "
361  "execution of neurons on this core.",
362  master_population_table[mp_i].key);
363  // There might be more than one that has failed
364  failed_bit_field_reads += bf_i - start;
365  } else {
366  // If allocation succeeds, copy the bitfields in
367  bit_field_t bf_pointer = &connectivity_bit_field[mp_i][0];
368  for (uint32_t i = start; i < bf_i; i++) {
369  uint32_t n_words = get_bit_field_size(filters[i].n_atoms);
370  spin1_memcpy(bf_pointer, filters[i].data, n_words * sizeof(uint32_t));
371  bf_pointer = &bf_pointer[n_words];
372  }
373 
374  print_bitfields(mp_i, start, bf_i, filters);
375  }
376  }
377  }
378  return true;
379 }

385 static inline bool population_table_position_in_the_master_pop_array(
386  spike_t spike, uint32_t *position) {
387  uint32_t imin = 0;
388  uint32_t imax = master_population_table_length;
389 
390  while (imin < imax) {
391  uint32_t imid = (imax + imin) >> 1;
392  master_population_table_entry entry = master_population_table[imid];
393  if ((spike & entry.mask) == entry.key) {
394  *position = imid;
395  return true;
396  } else if (entry.key < spike) {
397 
398  // Entry must be in upper part of the table
399  imin = imid + 1;
400  } else {
401  // Entry must be in lower part of the table
402  imax = imid;
403  }
404  }
405  return false;
406 }

412 bool population_table_initialise(
413  address_t table_address, address_t synapse_rows_address,
414  address_t direct_rows_address, uint32_t *row_max_n_words) {
415  log_debug("Population_table_initialise: starting");
416  pop_table_config_t *config = (pop_table_config_t *) table_address;
417 
418  master_population_table_length = config->table_length;
419  log_debug("Master pop table length is %d\n", master_population_table_length);
420  log_debug("Master pop table entry size is %d\n",
421  sizeof(master_population_table_entry));
422  uint32_t n_master_pop_bytes =
423  master_population_table_length * sizeof(master_population_table_entry);
424  log_debug("Pop table size is %d\n", n_master_pop_bytes);
425 
426  // only try to malloc if there's stuff to malloc.
427  if (n_master_pop_bytes != 0) {
428  master_population_table = spin1_malloc(n_master_pop_bytes);
429  if (master_population_table == NULL) {
430  log_error("Could not allocate master population table");
431  return false;
432  }
433  }
434 
435  uint32_t address_list_length = config->addr_list_length;
436  uint32_t n_address_list_bytes =
437  address_list_length * sizeof(address_list_entry);
438 
439  // only try to malloc if there's stuff to malloc.
440  if (n_address_list_bytes != 0) {
441  address_list = spin1_malloc(n_address_list_bytes);
442  if (address_list == NULL) {
443  log_error("Could not allocate master population address list");
444  return false;
445  }
446  }
447 
448  log_debug("Pop table size: %u (%u bytes)",
449  master_population_table_length, n_master_pop_bytes);
450  log_debug("Address list size: %u (%u bytes)",
451  address_list_length, n_address_list_bytes);
452 
453  // Copy the master population table
454  spin1_memcpy(master_population_table, config->data, n_master_pop_bytes);
455  spin1_memcpy(address_list, &config->data[master_population_table_length],
456  n_address_list_bytes);
457 
458  // Store the base address
459  log_info("The stored synaptic matrix base address is located at: 0x%08x",
460  synapse_rows_address);
461  log_info("The direct synaptic matrix base address is located at: 0x%08x",
462  direct_rows_address);
463  synaptic_rows_base_address = (uint32_t) synapse_rows_address;
464  direct_rows_base_address = (uint32_t) direct_rows_address;
465 
466  *row_max_n_words = 0xFF + N_SYNAPSE_ROW_HEADER_WORDS;
467 
468  print_master_population_table();
469  return true;
470 }

472 bool population_table_get_first_address(
473  spike_t spike, synaptic_row_t *row_address,
474  size_t *n_bytes_to_transfer) {
475  // locate the position in the binary search / array
476  log_debug("Searching for key %d", spike);
477 
478  // check we don't have a complete miss
479  uint32_t position;
480  if (!population_table_position_in_the_master_pop_array(spike, &position)) {
481  invalid_master_pop_hits++;
482  log_debug("Ghost searches: %u\n", ghost_pop_table_searches);
483  log_debug("Spike %u (= %x): "
484  "Population not found in master population table",
485  spike, spike);
486  return false;
487  }
488  log_debug("position = %d", position);
489 
490  master_population_table_entry entry = master_population_table[position];
491 
492  #if LOG_LEVEL >= LOG_DEBUG
493  if (entry.count == 0) {
494  log_debug("Spike %u (= %x): Population found in master population"
495  "table but count is 0", spike, spike);
496  }
497  #endif
498 
499  last_spike = spike;
500  next_item = entry.start;
501  items_to_go = entry.count;
502  uint32_t bit_field_id = 0;
503  if (entry.extra_info_flag) {
504  extra_info extra = address_list[next_item++].extra;
505  uint32_t local_neuron_id = get_local_neuron_id(entry, extra, spike);
506  last_neuron_id = local_neuron_id + get_core_sum(extra, spike);
507  bit_field_id = local_neuron_id + get_bitfield_sum(extra, spike);
508  } else {
509  last_neuron_id = get_neuron_id(entry, spike);
510  bit_field_id = last_neuron_id;
511  }
512 
513  // check we have a entry in the bit field for this (possible not to due to
514  // DTCM limitations or router table compression). If not, go to DMA check.
515  log_debug("Checking bit field");
516  if (connectivity_bit_field != NULL &&
517  connectivity_bit_field[position] != NULL) {
518  log_debug("Can be checked, bitfield is allocated");
519  // check that the bit flagged for this neuron id does hit a
520  // neuron here. If not return false and avoid the DMA check.
521  if (!bit_field_test(
522  connectivity_bit_field[position], bit_field_id)) {
523  log_debug("Tested and was not set");
524  bit_field_filtered_packets += 1;
525  items_to_go = 0;
526  return false;
527  }
528  log_debug("Was set, carrying on");
529  } else {
530  log_debug("Bit field was not set up. "
531  "either its due to a lack of DTCM, or because the "
532  "bit field was merged into the routing table");
533  }
534 
535  log_debug("spike = %08x, entry_index = %u, start = %u, count = %u",
536  spike, position, next_item, items_to_go);
537 
538  // A local address is used here as the interface requires something
539  // to be passed in but using the address of an argument is odd!
540  uint32_t local_spike_id;
541  bool get_next = population_table_get_next_address(
542  &local_spike_id, row_address, n_bytes_to_transfer);
543 
544  // tracks surplus DMAs
545  if (!get_next) {
546  log_debug("Found a entry which has a ghost entry for key %d", spike);
547  ghost_pop_table_searches++;
548  }
549  return get_next;
550 }

552 bool population_table_get_next_address(
553  spike_t *spike, synaptic_row_t *row_address,
554  size_t *n_bytes_to_transfer) {
555  // If there are no more items in the list, return false
556  if (items_to_go == 0) {
557  return false;
558  }
559 
560  bool is_valid = false;
561  do {
562  address_and_row_length item = address_list[next_item].addr;
563  if (item.address != INVALID_ADDRESS) {
564 
565  // If the row is a direct row, indicate this by specifying the
566  // n_bytes_to_transfer is 0
567  if (item.is_single) {
568  *row_address = (synaptic_row_t) (get_direct_address(item) +
569  (last_neuron_id * sizeof(uint32_t)));
570  *n_bytes_to_transfer = 0;
571  } else {
572 
573  uint32_t row_length = get_row_length(item);
574  uint32_t block_address = get_address(item);
575  uint32_t stride = (row_length + N_SYNAPSE_ROW_HEADER_WORDS);
576  uint32_t neuron_offset = last_neuron_id * stride * sizeof(uint32_t);
577 
578  *row_address = (synaptic_row_t) (block_address + neuron_offset);
579  *n_bytes_to_transfer = stride * sizeof(uint32_t);
580  log_debug("neuron_id = %u, block_address = 0x%.8x, "
581  "row_length = %u, row_address = 0x%.8x, n_bytes = %u",
582  last_neuron_id, block_address, row_length, *row_address,
583  *n_bytes_to_transfer);
584  *spike = last_spike;
585  }
586  is_valid = true;
587  }
588 
589  next_item++;
590  items_to_go--;
591  } while (!is_valid && (items_to_go > 0));
592 
593  return is_valid;
594 }
