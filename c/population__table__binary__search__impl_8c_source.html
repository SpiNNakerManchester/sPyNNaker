76 static inline void print_master_population_table(void) {
77 #if LOG_LEVEL >= LOG_DEBUG
78  log_info("Master_population\n");
79  for (uint32_t i = 0; i < master_population_table_length; i++) {
80  master_population_table_entry entry = master_population_table[i];
81  log_info("key: 0x%08x, mask: 0x%08x", entry.key, entry.mask);
82  int count = entry.count;
83  int start = entry.start;
84  log_info(" core_mask: 0x%08x, core_shift: %u, n_neurons: %u, n_words: %u, n_colour_bits: %u",
85  entry.core_mask, entry.mask_shift, entry.n_neurons, entry.n_words, entry.n_colour_bits);
86  for (uint16_t j = start; j < (start + count); j++) {
87  address_list_entry addr = address_list[j];
88  if (addr.address == INVALID_ADDRESS) {
89  log_info(" index %d: INVALID", j);
90  } else {
91  log_info(" index %d: offset: %u, address: 0x%08x, row_length: %u",
92  j, get_offset(addr), get_address(addr, synaptic_rows_base_address),
93  get_row_length(addr));
94  }
95  }
96  }
97  log_info("Population table has %u entries", master_population_table_length);
98 #endif
99 }

119 bool population_table_load_bitfields(filter_region_t *filter_region) {
120 
121  if (master_population_table_length == 0) {
122  return true;
123  }
124  // No filters = nothing to load
125  if (filter_region->n_filters == 0) {
126  return true;
127  }
128  // try allocating DTCM for starting array for bitfields
129  connectivity_bit_field =
130  spin1_malloc(sizeof(bit_field_t) * master_population_table_length);
131  if (connectivity_bit_field == NULL) {
132  log_warning(
133  "Couldn't initialise basic bit field holder. Will end up doing"
134  " possibly more DMA's during the execution than required."
135  " We required %d bytes where %d are available",
136  sizeof(bit_field_t) * master_population_table_length,
137  sark_heap_max(sark.heap, 0));
138  failed_bit_field_reads += filter_region->n_filters;
139  return true;
140  }
141 
142  // Go through the population table, and the relevant bitfield list, both
143  // of which are ordered by key...
144  if (filter_region->n_filters != master_population_table_length) {
145  log_error("The number of filters doesn't match the population table");
146  return false;
147  }
148  filter_info_t* filters = filter_region->filters;
149  for (uint32_t mp_i = 0; mp_i < master_population_table_length; mp_i++) {
150  connectivity_bit_field[mp_i] = NULL;
151 
152  // Fail if the key doesn't match
153  if (master_population_table[mp_i].key != filters[mp_i].key) {
154  log_error("Bitfield for %u keys do not match: bf=0x%08x vs mp=0x%08x",
155  mp_i, filters[mp_i].key, master_population_table[mp_i].key);
156  return false;
157  }
158  uint32_t useful = !(filters[mp_i].merged || filters[mp_i].all_ones);
159 
160  if (useful) {
161  // Try to allocate all the bitfields for this entry
162  uint32_t n_words = get_bit_field_size(filters[mp_i].n_atoms);
163  uint32_t size = sizeof(bit_field_t) * n_words;
164  connectivity_bit_field[mp_i] = spin1_malloc(size);
165  if (connectivity_bit_field[mp_i] == NULL) {
166  // There might be more than one that has failed
167  failed_bit_field_reads += 1;
168  } else {
169  spin1_memcpy(connectivity_bit_field[mp_i], filters[mp_i].data, size);
170  print_bitfields(mp_i, filters);
171  }
172  }
173  }
174  return true;
175 }

181 static inline bool population_table_position_in_the_master_pop_array(
182  spike_t spike, uint32_t *position) {
183  uint32_t imin = 0;
184  uint32_t imax = master_population_table_length;
185 
186  while (imin < imax) {
187  uint32_t imid = (imax + imin) >> 1;
188  master_population_table_entry entry = master_population_table[imid];
189  if ((spike & entry.mask) == entry.key) {
190  *position = imid;
191  return true;
192  } else if (entry.key < spike) {
193 
194  // Entry must be in upper part of the table
195  imin = imid + 1;
196  } else {
197  // Entry must be in lower part of the table
198  imax = imid;
199  }
200  }
201  return false;
202 }

204 bool population_table_setup(address_t table_address, uint32_t *row_max_n_words,
205  uint32_t *master_pop_table_length,
206  master_population_table_entry **master_pop_table,
207  address_list_entry **address_list) {
208  pop_table_config_t *config = (pop_table_config_t *) table_address;
209 
210  *master_pop_table_length = config->table_length;
211 
212  if (*master_pop_table_length == 0) {
213  return true;
214  }
215 
216  uint32_t n_master_pop_bytes =
217  *master_pop_table_length * sizeof(master_population_table_entry);
218 
219  // only try to malloc if there's stuff to malloc.
220  *master_pop_table = spin1_malloc(n_master_pop_bytes);
221  if (*master_pop_table == NULL) {
222  log_error("Could not allocate master population table of %u bytes",
223  n_master_pop_bytes);
224  return false;
225  }
226 
227  uint32_t address_list_length = config->addr_list_length;
228  uint32_t n_address_list_bytes =
229  address_list_length * sizeof(address_list_entry);
230 
231  *address_list = spin1_malloc(n_address_list_bytes);
232  if (*address_list == NULL) {
233  log_error("Could not allocate master population address list of %u bytes",
234  n_address_list_bytes);
235  return false;
236  }
237 
238  // Copy the master population table
239  spin1_memcpy(*master_pop_table, config->data, n_master_pop_bytes);
240  spin1_memcpy(*address_list, &config->data[*master_pop_table_length],
241  n_address_list_bytes);
242 
243  *row_max_n_words = 0xFF + N_SYNAPSE_ROW_HEADER_WORDS;
244  return true;
245 }

265 bool population_table_get_first_address(spike_t spike, pop_table_lookup_result_t *result) {
266 
267  // check we don't have a complete miss
268  uint32_t position;
269  if (!population_table_position_in_the_master_pop_array(spike, &position)) {
270  invalid_master_pop_hits++;
271  return false;
272  }
273 
274  master_population_table_entry entry = master_population_table[position];
275 
276  last_spike = spike;
277  next_item = entry.start;
278  items_to_go = entry.count;
279  uint32_t local_neuron_id = get_local_neuron_id(entry, spike);
280  if (entry.n_colour_bits) {
281  last_colour_mask = (1 << entry.n_colour_bits) - 1;
282  last_colour = local_neuron_id & last_colour_mask;
283  last_neuron_id = (local_neuron_id >> entry.n_colour_bits) + get_core_sum(entry, spike);
284  } else {
285  last_colour = 0;
286  last_colour_mask = 0;
287  last_neuron_id = local_neuron_id + get_core_sum(entry, spike);
288  }
289 
290  // check we have a entry in the bit field for this (possible not to due to
291  // DTCM limitations or router table compression). If not, go to DMA check.
292  if (connectivity_bit_field != NULL &&
293  connectivity_bit_field[position] != NULL) {
294  // check that the bit flagged for this neuron id does hit a
295  // neuron here. If not return false and avoid the DMA check.
296  if (!bit_field_test(
297  connectivity_bit_field[position], last_neuron_id)) {
298  bit_field_filtered_packets += 1;
299  items_to_go = 0;
300  return false;
301  }
302  }
303 
304  // A local address is used here as the interface requires something
305  // to be passed in but using the address of an argument is odd!
306  uint32_t local_spike_id;
307  bool get_next = population_table_get_next_address(&local_spike_id, result);
308 
309  // tracks surplus DMAs
310  if (!get_next) {
311  ghost_pop_table_searches++;
312  }
313  return get_next;
314 }

316 bool population_table_get_next_address(spike_t *spike, pop_table_lookup_result_t *result) {
317  // If there are no more items in the list, return false
318  if (items_to_go == 0) {
319  return false;
320  }
321 
322  bool is_valid = false;
323  do {
324  address_list_entry item = address_list[next_item];
325  if (item.address != INVALID_ADDRESS) {
326 
327  get_row_addr_and_size(item, synaptic_rows_base_address,
328  last_neuron_id, result);
329  *spike = last_spike;
330  result->colour = last_colour;
331  result->colour_mask = last_colour_mask;
332  is_valid = true;
333  }
334 
335  next_item++;
336  items_to_go--;
337  } while (!is_valid && (items_to_go > 0));
338 
339  return is_valid;
340 }
