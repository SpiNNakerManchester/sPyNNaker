144  log_info("master_population\n");
145  for (uint32_t i = 0; i < master_population_table_length; i++) {
146  master_population_table_entry entry = master_population_table[i];
147  for (uint16_t j = entry.start; j < (entry.start + entry.count); j++) {

173 bool population_table_initialise(
174  address_t table_address, address_t synapse_rows_address,
175  address_t direct_rows_address, uint32_t *row_max_n_words) {
176  log_debug("population_table_initialise: starting");
177 
178  master_population_table_length = table_address[0];
179  log_debug("master pop table length is %d\n", master_population_table_length);
180  log_debug("master pop table entry size is %d\n",
181  sizeof(master_population_table_entry));
182  uint32_t n_master_pop_bytes =
183  master_population_table_length * sizeof(master_population_table_entry);
184  uint32_t n_master_pop_words = n_master_pop_bytes >> 2;
185  log_debug("pop table size is %d\n", n_master_pop_bytes);
186 
187  // only try to malloc if there's stuff to malloc.
188  if (n_master_pop_bytes != 0) {
189  master_population_table = spin1_malloc(n_master_pop_bytes);
190  if (master_population_table == NULL) {
191  log_error("Could not allocate master population table");
192  return false;
193  }
194  }
195 
196  uint32_t address_list_length = table_address[1];
197  uint32_t n_address_list_bytes =
198  address_list_length * sizeof(address_and_row_length);
199 
200  // only try to malloc if there's stuff to malloc.
201  if (n_address_list_bytes != 0) {
202  address_list = spin1_malloc(n_address_list_bytes);
203  if (address_list == NULL) {
204  log_error("Could not allocate master population address list");
205  return false;
206  }
207  }
208 
209  log_debug("pop table size: %u (%u bytes)",
210  master_population_table_length, n_master_pop_bytes);
211  log_debug("address list size: %u (%u bytes)",
212  address_list_length, n_address_list_bytes);
213 
214  // Copy the master population table
215  spin1_memcpy(master_population_table, &table_address[2],
216  n_master_pop_bytes);
217  spin1_memcpy(address_list, &table_address[2 + n_master_pop_words],
218  n_address_list_bytes);
219 
220  // Store the base address
221  log_info("the stored synaptic matrix base address is located at: 0x%08x",
222  synapse_rows_address);
223  log_info("the direct synaptic matrix base address is located at: 0x%08x",
224  direct_rows_address);
225  synaptic_rows_base_address = synapse_rows_address;
226  direct_rows_base_address = (uint32_t) direct_rows_address;
227 
228  *row_max_n_words = 0xFF + N_SYNAPSE_ROW_HEADER_WORDS;
229 
230  print_master_population_table();
231  return true;
232 }

234 bool population_table_get_first_address(
235  spike_t spike, address_t* row_address, size_t* n_bytes_to_transfer) {
236  // locate the position in the binary search / array
237  log_debug("searching for key %d", spike);
238  int position = population_table_position_in_the_master_pop_array(spike);
239  log_debug("position = %d", position);
240 
241  // check we don't have a complete miss
242  if (position == NOT_IN_MASTER_POP_TABLE_FLAG) {
243  invalid_master_pop_hits++;
244  log_debug("Ghost searches: %u\n", ghost_pop_table_searches);
245  log_debug("spike %u (= %x): "
246  "population not found in master population table",
247  spike, spike);
248  return false;
249  }
250 
251  master_population_table_entry entry = master_population_table[position];
252  if (entry.count == 0) {
253  log_debug("spike %u (= %x): population found in master population"
254  "table but count is 0", spike, spike);
255  }
256 
257  log_debug("about to try to find neuron id");
258  last_neuron_id = get_neuron_id(entry, spike);
259 
260  // check we have a entry in the bit field for this (possible not to due to
261  // DTCM limitations or router table compression). If not, go to DMA check.
262  log_debug("checking bit field");
263  if (connectivity_bit_field != NULL &&
264  connectivity_bit_field[position] != NULL) {
265  log_debug("can be checked, bitfield is allocated");
266  // check that the bit flagged for this neuron id does hit a
267  // neuron here. If not return false and avoid the DMA check.
268  if (!bit_field_test(
269  connectivity_bit_field[position], last_neuron_id)) {
270  log_debug("tested and was not set");
271  bit_field_filtered_packets += 1;
272  return false;
273  }
274  log_debug("was set, carrying on");
275  } else {
276  log_debug("bit_field was not set up. "
277  "either its due to a lack of dtcm, or because the "
278  "bitfield was merged into the routing table");
279  }
280 
281  // going to do a DMA to read the matrix and see if there's a hit.
282  log_debug("about to set items");
283  next_item = entry.start;
284  items_to_go = entry.count;
285  last_spike = spike;
286 
287  log_debug("spike = %08x, entry_index = %u, start = %u, count = %u",
288  spike, position, next_item, items_to_go);
289 
290  // A local address is used here as the interface requires something
291  // to be passed in but using the address of an argument is odd!
292  uint32_t local_spike_id;
293  bool get_next = population_table_get_next_address(
294  &local_spike_id, row_address, n_bytes_to_transfer);
295 
296  // tracks surplus dmas
297  if (!get_next) {
298  log_debug("found a entry which has a ghost entry for key %d", spike);
299  ghost_pop_table_searches++;
300  }
301  return get_next;
302 }

304 int population_table_position_in_the_master_pop_array(spike_t spike) {
305  uint32_t imin = 0;
306  uint32_t imax = master_population_table_length;
307 
308  while (imin < imax) {
309  int imid = (imax + imin) >> 1;
310  master_population_table_entry entry = master_population_table[imid];
311  if ((spike & entry.mask) == entry.key) {
312  return imid;
313  } else if (entry.key < spike) {
314 
315  // Entry must be in upper part of the table
316  imin = imid + 1;
317  } else {
318  // Entry must be in lower part of the table
319  imax = imid;
320  }
321  }
322  return NOT_IN_MASTER_POP_TABLE_FLAG;
323 }

325 bool population_table_get_next_address(
326  spike_t *spike, address_t *row_address, size_t *n_bytes_to_transfer) {
327  // If there are no more items in the list, return false
328  if (items_to_go <= 0) {
329  return false;
330  }
331 
332  bool is_valid = false;
333  do {
334  address_and_row_length item = address_list[next_item];
335 
336  // If the row is a direct row, indicate this by specifying the
337  // n_bytes_to_transfer is 0
338  if (is_single(item)) {
339  *row_address = (address_t) (
340  get_direct_address(item) + direct_rows_base_address +
341  (last_neuron_id * sizeof(uint32_t)));
342  *n_bytes_to_transfer = 0;
343  *spike = last_spike;
344  is_valid = true;
345  } else {
346  uint32_t row_length = get_row_length(item);
347  if (row_length > 0) {
348  uint32_t block_address = get_address(item) +
349  (uint32_t) synaptic_rows_base_address;
350  uint32_t stride = (row_length + N_SYNAPSE_ROW_HEADER_WORDS);
351  uint32_t neuron_offset =
352  last_neuron_id * stride * sizeof(uint32_t);
353 
354  *row_address = (address_t) (block_address + neuron_offset);
355  *n_bytes_to_transfer = stride * sizeof(uint32_t);
356  log_debug(
357  "neuron_id = %u, block_address = 0x%.8x,"
358  "row_length = %u, row_address = 0x%.8x, n_bytes = %u",
359  last_neuron_id, block_address, row_length, *row_address,
360  *n_bytes_to_transfer);
361  *spike = last_spike;
362  is_valid = true;
363  }
364  }
365 
366  next_item++;
367  items_to_go--;
368  } while (!is_valid && (items_to_go > 0));
369 
370  return is_valid;
371 }
