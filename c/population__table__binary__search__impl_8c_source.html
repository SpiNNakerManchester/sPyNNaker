238 static inline void print_master_population_table(void) {
239 #if log_level >= LOG_DEBUG
240  log_info("Master_population\n");
241  for (uint32_t i = 0; i < master_population_table_length; i++) {
242  master_population_table_entry entry = master_population_table[i];
243  log_info("key: 0x%08x, mask: 0x%08x", entry.key, entry.mask);
244  int count = entry.count;
245  int start = entry.start;
246  if (entry.extra_info_flag) {
247  extra_info extra = address_list[start].extra;
248  start += 1;
249  log_info(" core_mask: 0x%08x, core_shift: %u, n_neurons: %u",
250  extra.core_mask, extra.mask_shift, extra.n_neurons);
251  }
252  for (uint16_t j = start; j < (start + count); j++) {
253  address_and_row_length addr = address_list[j].addr;
254  if (addr.address == INVALID_ADDRESS) {
255  log_info(" index %d: INVALID", j);
256  } else if (!addr.is_single) {
257  log_info(" index %d: offset: %u, address: 0x%08x, row_length: %u",
258  j, get_offset(addr), get_address(addr), get_row_length(addr));
259  } else {
260  log_info(" index %d: offset: %u, address: 0x%08x, single",
261  j, addr.address, get_direct_address(addr));
262  }
263  }
264  }
265  log_info("Population table has %u entries", master_population_table_length);
266 #endif

304 bool population_table_load_bitfields(filter_region_t *filter_region) {
305 
306  if (master_population_table_length == 0) {
307  return true;
308  }
309  // try allocating DTCM for starting array for bitfields
310  connectivity_bit_field =
311  spin1_malloc(sizeof(bit_field_t) * master_population_table_length);
312  if (connectivity_bit_field == NULL) {
313  log_warning(
314  "Couldn't initialise basic bit field holder. Will end up doing"
315  " possibly more DMA's during the execution than required."
316  " We required %d bytes where %d are available",
317  sizeof(bit_field_t) * master_population_table_length,
318  sark_heap_max(sark.heap, 0));
319  return true;
320  }
321 
322  // Go through the population table, and the relevant bitfield list, both
323  // of which are ordered by key...
324  uint32_t bf_i = 0;
325  uint32_t n_filters = filter_region->n_filters;
326  filter_info_t* filters = filter_region->filters;
327  for (uint32_t mp_i = 0; mp_i < master_population_table_length; mp_i++) {
328  connectivity_bit_field[mp_i] = NULL;
329 
330  log_debug("Master pop key: 0x%08x, mask: 0x%08x",
331  master_population_table[mp_i].key, master_population_table[mp_i].mask);
332 
333 #ifdef LOG_DEBUG
334  // Sanity checking code; not needed in normal operation, and costs ITCM
335  // With both things being in key order, this should never happen...
336  if (bf_i < n_filters &&
337  filters[bf_i].key < master_population_table[mp_i].key) {
338  log_error("Skipping bitfield %d for key 0x%08x", bf_i, filters[bf_i].key);
339  rt_error(RTE_SWERR);
340  }
341 #endif
342 
343  // While there is a match, keep track of the start and end; note this
344  // may recheck the first entry, but there might not be a first entry if
345  // we have already gone off the end of the bitfield array
346  uint32_t start = bf_i;
347  uint32_t n_words_total = 0;
348  uint32_t useful = 0;
349  log_debug("Starting with bit field %d with key 0x%08x", bf_i, filters[bf_i].key);
350  while (bf_i < n_filters && matches(mp_i, filters[bf_i].key)) {
351  log_debug("Using bit field %d with key 0x%08x, merged %d, redundant %d",
352  bf_i, filters[bf_i].key, filters[bf_i].merged, filters[bf_i].all_ones);
353  n_words_total += get_bit_field_size(filters[bf_i].n_atoms);
354  useful += !(filters[bf_i].merged || filters[bf_i].all_ones);
355  bf_i++;
356  }
357 
358  // If there is something to copy, copy them in now
359  log_debug("Ended with bit field %d with key 0x%08x, n_words %d, useful %d",
360  bf_i, filters[bf_i].key, n_words_total, useful);
361  if (bf_i != start && useful) {
362  // Try to allocate all the bitfields for this entry
363  connectivity_bit_field[mp_i] = spin1_malloc(
364  sizeof(bit_field_t) * n_words_total);
365  if (connectivity_bit_field[mp_i] == NULL) {
366  // If allocation fails, we can still continue
367  log_debug(
368  "Could not initialise bit field for key %d, packets with "
369  "that key will use a DMA to check if the packet targets "
370  "anything within this core. Potentially slowing down the "
371  "execution of neurons on this core.",
372  master_population_table[mp_i].key);
373  // There might be more than one that has failed
374  failed_bit_field_reads += bf_i - start;
375  } else {
376  // If allocation succeeds, copy the bitfields in
377  bit_field_t bf_pointer = &connectivity_bit_field[mp_i][0];
378  for (uint32_t i = start; i < bf_i; i++) {
379  uint32_t n_words = get_bit_field_size(filters[i].n_atoms);
380  spin1_memcpy(bf_pointer, filters[i].data, n_words * sizeof(uint32_t));
381  bf_pointer = &bf_pointer[n_words];
382  }
383 
384  print_bitfields(mp_i, start, bf_i, filters);
385  }
386  }
387  }
388  return true;
389 }

395 static inline bool population_table_position_in_the_master_pop_array(
396  spike_t spike, uint32_t *position) {
397  uint32_t imin = 0;
398  uint32_t imax = master_population_table_length;
399 
400  while (imin < imax) {
401  uint32_t imid = (imax + imin) >> 1;
402  master_population_table_entry entry = master_population_table[imid];
403  if ((spike & entry.mask) == entry.key) {
404  *position = imid;
405  return true;
406  } else if (entry.key < spike) {
407 
408  // Entry must be in upper part of the table
409  imin = imid + 1;
410  } else {
411  // Entry must be in lower part of the table
412  imax = imid;
413  }
414  }
415  return false;
416 }

422 bool population_table_initialise(
423  address_t table_address, address_t synapse_rows_address,
424  address_t direct_rows_address, uint32_t *row_max_n_words) {
425  log_debug("Population_table_initialise: starting");
426  pop_table_config_t *config = (pop_table_config_t *) table_address;
427 
428  master_population_table_length = config->table_length;
429  log_debug("Master pop table length is %d\n", master_population_table_length);
430  log_debug("Master pop table entry size is %d\n",
431  sizeof(master_population_table_entry));
432  uint32_t n_master_pop_bytes =
433  master_population_table_length * sizeof(master_population_table_entry);
434  log_debug("Pop table size is %d\n", n_master_pop_bytes);
435 
436  // only try to malloc if there's stuff to malloc.
437  if (n_master_pop_bytes != 0) {
438  master_population_table = spin1_malloc(n_master_pop_bytes);
439  if (master_population_table == NULL) {
440  log_error("Could not allocate master population table");
441  return false;
442  }
443  }
444 
445  uint32_t address_list_length = config->addr_list_length;
446  uint32_t n_address_list_bytes =
447  address_list_length * sizeof(address_list_entry);
448 
449  // only try to malloc if there's stuff to malloc.
450  if (n_address_list_bytes != 0) {
451  address_list = spin1_malloc(n_address_list_bytes);
452  if (address_list == NULL) {
453  log_error("Could not allocate master population address list");
454  return false;
455  }
456  }
457 
458  log_debug("Pop table size: %u (%u bytes)",
459  master_population_table_length, n_master_pop_bytes);
460  log_debug("Address list size: %u (%u bytes)",
461  address_list_length, n_address_list_bytes);
462 
463  // Copy the master population table
464  spin1_memcpy(master_population_table, config->data, n_master_pop_bytes);
465  spin1_memcpy(address_list, &config->data[master_population_table_length],
466  n_address_list_bytes);
467 
468  // Store the base address
469  log_info("The stored synaptic matrix base address is located at: 0x%08x",
470  synapse_rows_address);
471  log_info("The direct synaptic matrix base address is located at: 0x%08x",
472  direct_rows_address);
473  synaptic_rows_base_address = (uint32_t) synapse_rows_address;
474  direct_rows_base_address = (uint32_t) direct_rows_address;
475 
476  *row_max_n_words = 0xFF + N_SYNAPSE_ROW_HEADER_WORDS;
477 
478  print_master_population_table();
479  return true;
480 }

482 bool population_table_get_first_address(
483  spike_t spike, synaptic_row_t *row_address,
484  size_t *n_bytes_to_transfer) {
485  // locate the position in the binary search / array
486  log_debug("Searching for key %d", spike);
487 
488  // check we don't have a complete miss
489  uint32_t position;
490  if (!population_table_position_in_the_master_pop_array(spike, &position)) {
491  invalid_master_pop_hits++;
492  log_debug("Ghost searches: %u\n", ghost_pop_table_searches);
493  log_debug("Spike %u (= %x): "
494  "Population not found in master population table",
495  spike, spike);
496  return false;
497  }
498  log_debug("position = %d", position);
499 
500  master_population_table_entry entry = master_population_table[position];
501  if (entry.count == 0) {
502  log_debug("Spike %u (= %x): Population found in master population"
503  "table but count is 0", spike, spike);
504  }
505 
506  last_spike = spike;
507  next_item = entry.start;
508  items_to_go = entry.count;
509  uint32_t bits_offset = 0;
510  if (entry.extra_info_flag) {
511  extra_info extra = address_list[next_item++].extra;
512  bits_offset = get_core_index(extra, spike) * extra.n_words;
513  last_neuron_id = get_extended_neuron_id(entry, extra, spike);
514  } else {
515  last_neuron_id = get_neuron_id(entry, spike);
516  }
517 
518  // check we have a entry in the bit field for this (possible not to due to
519  // DTCM limitations or router table compression). If not, go to DMA check.
520  log_debug("Checking bit field");
521  if (connectivity_bit_field != NULL &&
522  connectivity_bit_field[position] != NULL) {
523  log_debug("Can be checked, bitfield is allocated");
524  // check that the bit flagged for this neuron id does hit a
525  // neuron here. If not return false and avoid the DMA check.
526  if (!bit_field_test(
527  &connectivity_bit_field[position][bits_offset], last_neuron_id)) {
528  log_debug("Tested and was not set");
529  bit_field_filtered_packets += 1;
530  return false;
531  }
532  log_debug("Was set, carrying on");
533  } else {
534  log_debug("Bit field was not set up. "
535  "either its due to a lack of DTCM, or because the "
536  "bit field was merged into the routing table");
537  }
538 
539  log_debug("spike = %08x, entry_index = %u, start = %u, count = %u",
540  spike, position, next_item, items_to_go);
541 
542  // A local address is used here as the interface requires something
543  // to be passed in but using the address of an argument is odd!
544  uint32_t local_spike_id;
545  bool get_next = population_table_get_next_address(
546  &local_spike_id, row_address, n_bytes_to_transfer);
547 
548  // tracks surplus DMAs
549  if (!get_next) {
550  log_debug("Found a entry which has a ghost entry for key %d", spike);
551  ghost_pop_table_searches++;
552  }
553  return get_next;
554 }

556 bool population_table_get_next_address(
557  spike_t *spike, synaptic_row_t *row_address,
558  size_t *n_bytes_to_transfer) {
559  // If there are no more items in the list, return false
560  if (items_to_go == 0) {
561  return false;
562  }
563 
564  bool is_valid = false;
565  do {
566  address_and_row_length item = address_list[next_item].addr;
567  if (item.address != INVALID_ADDRESS) {
568 
569  // If the row is a direct row, indicate this by specifying the
570  // n_bytes_to_transfer is 0
571  if (item.is_single) {
572  *row_address = (synaptic_row_t) (get_direct_address(item) +
573  (last_neuron_id * sizeof(uint32_t)));
574  *n_bytes_to_transfer = 0;
575  } else {
576 
577  uint32_t row_length = get_row_length(item);
578  uint32_t block_address = get_address(item);
579  uint32_t stride = (row_length + N_SYNAPSE_ROW_HEADER_WORDS);
580  uint32_t neuron_offset = last_neuron_id * stride * sizeof(uint32_t);
581 
582  *row_address = (synaptic_row_t) (block_address + neuron_offset);
583  *n_bytes_to_transfer = stride * sizeof(uint32_t);
584  log_debug("neuron_id = %u, block_address = 0x%.8x, "
585  "row_length = %u, row_address = 0x%.8x, n_bytes = %u",
586  last_neuron_id, block_address, row_length, *row_address,
587  *n_bytes_to_transfer);
588  *spike = last_spike;
589  }
590  is_valid = true;
591  }
592 
593  next_item++;
594  items_to_go--;
595  } while (!is_valid && (items_to_go > 0));
596 
597  return is_valid;
598 }
