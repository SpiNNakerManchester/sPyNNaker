74 static inline post_trace_t timing_decay_post(
75  uint32_t time, uint32_t last_time, post_trace_t last_trace) {
76  // Get time since last spike
77  uint32_t delta_time = time - last_time;
78 
79  // Decay previous o1 trace
80  int32_t decayed_o1 = STDP_FIXED_MUL_16X16(last_trace.o1,
81  maths_lut_exponential_decay(delta_time, tau_minus_lookup));
82 
83  // If we have already added on the last spike effect, just decay
84  // (as it's sampled BEFORE the spike),
85  // otherwise, add on energy caused by last spike and decay that
86  int32_t new_o2 = 0;
87  int32_t next_spike_time = last_trace.last_spike_time;
88  if (last_trace.last_spike_time == 0) {
89  int32_t decay = maths_lut_exponential_decay(delta_time, tau_y_lookup);
90  new_o2 = STDP_FIXED_MUL_16X16(last_trace.o2, decay);
91  } else {
92  uint32_t o2_delta = time - last_trace.last_spike_time;
93  int32_t decay = maths_lut_exponential_decay(o2_delta, tau_y_lookup);
94  new_o2 = STDP_FIXED_MUL_16X16(last_trace.o2 + STDP_FIXED_POINT_ONE, decay);
95  next_spike_time = 0;
96  }
97  return (post_trace_t) {.o1 = decayed_o1, .o2 = new_o2,
98  .last_spike_time = next_spike_time};
99 }

108  uint32_t time, uint32_t last_time, post_trace_t last_trace) {
109 
110  post_trace_t next_trace = timing_decay_post(time, last_time, last_trace);

112  next_trace.last_spike_time = time;

115  // for new spike added
116  return next_trace;

125 static inline pre_trace_t timing_add_pre_spike(
126  uint32_t time, uint32_t last_time, pre_trace_t last_trace) {
127  // Get time since last spike
128  uint32_t delta_time = time - last_time;
129 
130  // Decay previous r1 trace and add energy caused by new spike
131  int32_t decayed_r1 = STDP_FIXED_MUL_16X16(last_trace.r1,
132  maths_lut_exponential_decay(delta_time, tau_plus_lookup));
133  int32_t new_r1 = decayed_r1 + STDP_FIXED_POINT_ONE;
134 
135  // If this is the 1st pre-synaptic event, r2 trace is zero
136  // (as it's sampled BEFORE the spike),
137  // otherwise, add on energy caused by last spike and decay that
138  int32_t new_r2 = (last_time == 0) ? 0 :
139  STDP_FIXED_MUL_16X16(
140  last_trace.r2 + STDP_FIXED_POINT_ONE,
141  maths_lut_exponential_decay(delta_time, tau_x_lookup));
142 
143  log_debug("\tdelta_time=%u, r1=%d, r2=%d\n", delta_time, new_r1, new_r2);
144 
145  // Return new pre-synaptic event with decayed trace values with energy
146  // for new spike added
147  return (pre_trace_t) {.r1 = new_r1, .r2 = new_r2};
148 }

160 static inline update_state_t timing_apply_pre_spike(
161  uint32_t time, pre_trace_t trace, UNUSED uint32_t last_pre_time,
162  UNUSED pre_trace_t last_pre_trace, uint32_t last_post_time,
163  post_trace_t last_post_trace, update_state_t previous_state) {
164  // Get time of event relative to last post-synaptic event
165  uint32_t time_since_last_post = time - last_post_time;
166  int32_t decayed_o1 = STDP_FIXED_MUL_16X16(last_post_trace.o1,
167  maths_lut_exponential_decay(time_since_last_post, tau_minus_lookup));
168 
169  // Calculate triplet term
170  int32_t decayed_o1_r2 = STDP_FIXED_MUL_16X16(decayed_o1, trace.r2);
171 
172  log_debug("\t\t\ttime_since_last_post_event=%u, decayed_o1=%d, r2=%d,"
173  "decayed_o1_r2=%d\n",
174  time_since_last_post, decayed_o1, trace.r2, decayed_o1_r2);
175 
176  // Apply depression to state (which is a weight_state)
177  return weight_two_term_apply_depression(
178  previous_state, decayed_o1, decayed_o1_r2);
179 }

191 static inline update_state_t timing_apply_post_spike(
192  uint32_t time, post_trace_t trace, uint32_t last_pre_time,
193  pre_trace_t last_pre_trace, UNUSED uint32_t last_post_time,
194  UNUSED post_trace_t last_post_trace, update_state_t previous_state) {
195  // Get time of event relative to last pre-synaptic event

199  maths_lut_exponential_decay(time_since_last_pre, tau_plus_lookup));
200 
201  // Calculate triplet term
202  int32_t decayed_r1_o2 = STDP_FIXED_MUL_16X16(decayed_r1, trace.o2);
203 
204  log_debug("\t\t\ttime_since_last_pre_event=%u, decayed_r1=%d, o2=%d,"
205  "decayed_r1_o2=%d\n",
206  time_since_last_pre, decayed_r1, trace.o2, decayed_r1_o2);
207 
208  // Apply potentiation to state (which is a weight_state)
209  return weight_two_term_apply_potentiation(
210  previous_state, decayed_r1, decayed_r1_o2);
211  } else {
212  return previous_state;
