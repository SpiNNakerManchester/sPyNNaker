79 static inline post_trace_t timing_add_post_spike(
80  uint32_t time, uint32_t last_time, post_trace_t last_trace) {
81  // Get time since last spike
82  uint32_t delta_time = time - last_time;
83 
84  // Decay previous o1 trace and add energy caused by new spike
85  int32_t decayed_o1 = STDP_FIXED_MUL_16X16(last_trace.o1,
86  maths_lut_exponential_decay(delta_time, tau_minus_lookup));
87  int32_t new_o1 = decayed_o1 + STDP_FIXED_POINT_ONE;
88 
89  // If this is the 1st post-synaptic event, o2 trace is zero
90  // (as it's sampled BEFORE the spike),
91  // otherwise, add on energy caused by last spike and decay that
92  int32_t new_o2 = (last_time == 0) ? 0 :
93  STDP_FIXED_MUL_16X16(
94  last_trace.o2 + STDP_FIXED_POINT_ONE,
95  maths_lut_exponential_decay(delta_time, tau_y_lookup));
96 
97  log_debug("\tdelta_time=%d, o1=%d, o2=%d\n", delta_time, new_o1, new_o2);
98 
99  // Return new pre- synaptic event with decayed trace values with energy
100  // for new spike added
101  return (post_trace_t) {.o1 = new_o1, .o2 = new_o2};
102 }

145 static inline update_state_t timing_apply_pre_spike(
146  uint32_t time, pre_trace_t trace, UNUSED uint32_t last_pre_time,
147  UNUSED pre_trace_t last_pre_trace, uint32_t last_post_time,
148  post_trace_t last_post_trace, update_state_t previous_state) {
149  // Get time of event relative to last post-synaptic event
150  uint32_t time_since_last_post = time - last_post_time;
151  int32_t decayed_o1 = STDP_FIXED_MUL_16X16(last_post_trace.o1,
152  maths_lut_exponential_decay(time_since_last_post, tau_minus_lookup));
153 
154  // Calculate triplet term
155  int32_t decayed_o1_r2 = STDP_FIXED_MUL_16X16(decayed_o1, trace.r2);
156 
157  log_debug("\t\t\ttime_since_last_post_event=%u, decayed_o1=%d, r2=%d,"
158  "decayed_o1_r2=%d\n",
159  time_since_last_post, decayed_o1, trace.r2, decayed_o1_r2);
160 
161  // Apply depression to state (which is a weight_state)
162  return weight_two_term_apply_depression(
163  previous_state, decayed_o1, decayed_o1_r2);
164 }

176 static inline update_state_t timing_apply_post_spike(
177  uint32_t time, post_trace_t trace, uint32_t last_pre_time,
178  pre_trace_t last_pre_trace, UNUSED uint32_t last_post_time,
179  UNUSED post_trace_t last_post_trace, update_state_t previous_state) {
180  // Get time of event relative to last pre-synaptic event
181  uint32_t time_since_last_pre = time - last_pre_time;
182  if (time_since_last_pre > 0) {
183  int32_t decayed_r1 = STDP_FIXED_MUL_16X16(last_pre_trace.r1,
184  maths_lut_exponential_decay(time_since_last_pre, tau_plus_lookup));
185 
186  // Calculate triplet term
187  int32_t decayed_r1_o2 = STDP_FIXED_MUL_16X16(decayed_r1, trace.o2);
188 
189  log_debug("\t\t\ttime_since_last_pre_event=%u, decayed_r1=%d, o2=%d,"
190  "decayed_r1_o2=%d\n",
191  time_since_last_pre, decayed_r1, trace.o2, decayed_r1_o2);
192 
193  // Apply potentiation to state (which is a weight_state)
194  return weight_two_term_apply_potentiation(
195  previous_state, decayed_r1, decayed_r1_o2);
