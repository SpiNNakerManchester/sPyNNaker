46 static inline void decay_and_init(exp_state_t *state, exp_params_t *params,
47  REAL time_step_ms, uint32_t n_steps_per_timestep) {
48  REAL ts = kdivui(time_step_ms, n_steps_per_timestep);
49  REAL ts_over_tau = kdivk(ts, params->tau);
50  decay_t decay = expulr(-ts_over_tau);
51  decay_t inv_decay = 1.0ulr - decay;
52  REAL tau_over_ts = kdivk(params->tau, ts);
53  decay_t init = decay_s1615_to_u032(tau_over_ts, inv_decay);
54  state->decay = decay;
55  state->init = init;
56  state->synaptic_input_value = params->init_input;
57 }
