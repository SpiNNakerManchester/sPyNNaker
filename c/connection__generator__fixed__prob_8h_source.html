54  // Allocate memory for the data

79 static uint32_t connection_generator_fixed_prob_generate(
80  void *data, uint32_t pre_slice_start, uint32_t pre_slice_count,
81  uint32_t pre_neuron_index, uint32_t post_slice_start,
82  uint32_t post_slice_count, uint32_t max_row_length, uint16_t *indices) {
83  use(pre_slice_start);
84  use(pre_slice_count);
85 
86  struct fixed_prob *obj = data;
87 
88  // If no space, generate nothing
89  if (max_row_length < 1) {
90  return 0;
91  }
92 
93  // If not in the pre-population view range, then don't generate
94  if ((pre_neuron_index < obj->params.pre_lo) ||
95  (pre_neuron_index > obj->params.pre_hi)) {
96  return 0;
97  }
98 
99  // Randomly select connections to each post-neuron
100  uint32_t n_conns = 0;
101  for (uint32_t i = 0; i < post_slice_count; i++) {
102  // Disallow self connections if configured
103  if (!obj->params.allow_self_connections &&
104  (pre_neuron_index == post_slice_start + i)) {
105  continue;
106  }
107 
108  // Don't generate if the value is not in the range of the post-population view
109  if ((i + post_slice_start < obj->params.post_lo) ||
110  (i + post_slice_start > obj->params.post_hi)) {
111  continue;
112  }
113 
114  // Generate a random number
115  _probability_t value = ulrbits(rng_generator(obj->rng));
116 
117  // If less than our probability, generate a connection if possible
118  if ((value <= obj->params.probability) &&
119  (n_conns < max_row_length)) {
120  indices[n_conns++] = i;
121  } else if (n_conns >= max_row_length) {
122  log_warning("Row overflow");
123  }
124  }
125 
126  return n_conns;
127 }
