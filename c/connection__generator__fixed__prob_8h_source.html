55 static void *connection_generator_fixed_prob_initialise(address_t *region) {
56  // Allocate memory for the data
57  struct fixed_prob *obj = spin1_malloc(sizeof(struct fixed_prob));
58 
59  // Copy the parameters in
60  struct fixed_prob_params *params_sdram = (void *) *region;
61  obj->params = *params_sdram++;
62  *region = (void *) params_sdram;
63 
64  // Initialise the RNG for the connector

69  obj->params.pre_lo, obj->params.pre_hi, obj->params.post_lo, obj->params.post_hi,
70  obj->params.allow_self_connections,
71  (accum) obj->params.probability);

104 static uint32_t connection_generator_fixed_prob_generate(
105  void *generator, UNUSED uint32_t pre_slice_start,
106  UNUSED uint32_t pre_slice_count,
107  uint32_t pre_neuron_index, uint32_t post_slice_start,
108  uint32_t post_slice_count, uint32_t max_row_length, uint16_t *indices) {
109  struct fixed_prob *obj = generator;
110 
111  // If no space, generate nothing
112  if (max_row_length < 1) {
113  return 0;
114  }
115 
116  // If not in the pre-population view range, then don't generate
117  if ((pre_neuron_index < obj->params.pre_lo) ||
118  (pre_neuron_index > obj->params.pre_hi)) {
119  return 0;
120  }
121 
122  // Randomly select connections to each post-neuron
123  uint32_t n_conns = 0;
124  for (uint32_t i = 0; i < post_slice_count; i++) {
125  // Disallow self connections if configured
126  if (!obj->params.allow_self_connections &&
127  (pre_neuron_index == post_slice_start + i)) {
128  continue;
129  }
130 
131  // Don't generate if the value is not in the range of the post-population view
132  if ((i + post_slice_start < obj->params.post_lo) ||
133  (i + post_slice_start > obj->params.post_hi)) {
134  continue;
135  }
136 
137  // Generate a random number
138  _probability_t value = ulrbits(rng_generator(obj->rng));
139 
140  // If less than our probability, generate a connection if possible
141  if ((value <= obj->params.probability) &&
142  (n_conns < max_row_length)) {
143  indices[n_conns++] = i;
144  } else if (n_conns >= max_row_length) {
145  log_warning("Row overflow");
146  }
147  }
148 
149  return n_conns;
150 }
