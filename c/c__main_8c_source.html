161 static bool initialise(void) {
162  log_debug("Initialise: started");
163 
164  // Get the address this core's DTCM data starts at from SRAM
165  data_specification_metadata_t *ds_regions =
166  data_specification_get_data_address();
167 
168  // Read the header
169  if (!data_specification_read_header(ds_regions)) {
170  return false;
171  }
172 
173  // Get the timing details and set up the simulation interface
174  if (!simulation_initialise(
175  data_specification_get_region(SYSTEM_REGION, ds_regions),
176  APPLICATION_NAME_HASH, &timer_period, &simulation_ticks,
177  &infinite_run, &time, SDP, DMA)) {
178  return false;
179  }
180  simulation_set_provenance_function(
181  c_main_store_provenance_data,
182  data_specification_get_region(PROVENANCE_DATA_REGION, ds_regions));
183 
184  // Set up the neurons
185  uint32_t n_synapse_types;
186  uint32_t incoming_spike_buffer_size;
187  uint32_t n_regions_used;
188  if (!neuron_initialise(
189  data_specification_get_region(NEURON_PARAMS_REGION, ds_regions),
190  data_specification_get_region(NEURON_RECORDING_REGION, ds_regions),
191  &n_neurons, &n_synapse_types, &incoming_spike_buffer_size,
192  &n_regions_used)) {
193  return false;
194  }
195 
196  // Set up the synapses
197  uint32_t *ring_buffer_to_input_buffer_left_shifts;
198  bool clear_input_buffers_of_late_packets_init;
199  if (!synapses_initialise(
200  data_specification_get_region(SYNAPSE_PARAMS_REGION, ds_regions),
201  n_neurons, n_synapse_types,
202  &ring_buffer_to_input_buffer_left_shifts,
203  &clear_input_buffers_of_late_packets_init)) {
204  return false;
205  }
206 
207  // set up direct synapses
208  address_t direct_synapses_address;
209  if (!direct_synapses_initialise(
210  data_specification_get_region(DIRECT_MATRIX_REGION, ds_regions),
211  &direct_synapses_address)) {
212  return false;
213  }
214 
215  // Set up the population table
216  uint32_t row_max_n_words;
217  if (!population_table_initialise(
218  data_specification_get_region(POPULATION_TABLE_REGION, ds_regions),
219  data_specification_get_region(SYNAPTIC_MATRIX_REGION, ds_regions),
220  direct_synapses_address, &row_max_n_words)) {
221  return false;
222  }
223  // Set up the synapse dynamics
224  if (!synapse_dynamics_initialise(
225  data_specification_get_region(SYNAPSE_DYNAMICS_REGION, ds_regions),
226  n_neurons, n_synapse_types,
227  ring_buffer_to_input_buffer_left_shifts)) {
228  return false;
229  }
230 
231  // Set up structural plasticity dynamics
232  if (!synaptogenesis_dynamics_initialise(data_specification_get_region(
233  STRUCTURAL_DYNAMICS_REGION, ds_regions))) {
234  return false;
235  }
236 
237  rewiring_period = synaptogenesis_rewiring_period();
238  rewiring = rewiring_period != -1;
239 
240  if (!spike_processing_initialise(
241  row_max_n_words, MC, USER, incoming_spike_buffer_size,
242  clear_input_buffers_of_late_packets_init, n_regions_used)) {
243  return false;
244  }
245 
246  // Setup profiler
247  profiler_init(data_specification_get_region(PROFILER_REGION, ds_regions));
248 
249  // Do bitfield configuration last to only use any unused memory
250  if (!population_table_load_bitfields(
251  data_specification_get_region(BIT_FIELD_FILTER_REGION, ds_regions))) {
252  return false;
253  }
254 
255  print_post_to_pre_entry();
256 
257  log_debug("Initialise: finished");
258  return true;
259 }

286 void timer_callback(uint timer_count, UNUSED uint unused) {
287 
288  global_timer_count = timer_count;
289  profiler_write_entry_disable_irq_fiq(PROFILER_ENTER | PROFILER_TIMER);
290 
291  time++;
292  last_rewiring_time++;
293 
294  // This is the part where I save the input and output indices
295  // from the circular buffer
296  // If time == 0 as well as output == input == 0 then no rewire is
297  // supposed to happen. No spikes yet
298  log_debug("Timer tick %u \n", time);
299 
300  /* if a fixed number of simulation ticks that were specified at startup
301  * then do reporting for finishing */
302  if (simulation_is_finished()) {
303 
304  // Enter pause and resume state to avoid another tick
305  simulation_handle_pause_resume(resume_callback);
306 
307  log_debug("Completed a run");
308 
309  // rewrite neuron params to SDRAM for reading out if needed
310  data_specification_metadata_t *ds_regions =
311  data_specification_get_data_address();
312  neuron_pause(data_specification_get_region(NEURON_PARAMS_REGION, ds_regions));
313 
314  profiler_write_entry_disable_irq_fiq(PROFILER_EXIT | PROFILER_TIMER);
315 
316  profiler_finalise();
317 
318  // Subtract 1 from the time so this tick gets done again on the next
319  // run
320  time--;
321 
322  log_debug("Rewire tries = %d", count_rewire_attempts);
323  simulation_ready_to_read();
324  return;
325  }
326 
327  // First do synapses timestep update, as this is time-critical
328  synapses_do_timestep_update(time);
329 
330  // Then do rewiring
331  if (rewiring &&
332  ((last_rewiring_time >= rewiring_period && !synaptogenesis_is_fast())
333  || synaptogenesis_is_fast())) {
334  last_rewiring_time = 0;
335  // put flag in spike processing to do synaptic rewiring
336  if (synaptogenesis_is_fast()) {
337  spike_processing_do_rewiring(rewiring_period);
338  } else {
339  spike_processing_do_rewiring(1);
340  }
341  count_rewire_attempts++;
342  }
343 
344  // Now do neuron time step update
345  neuron_do_timestep_update(time, timer_count);
346 
347  profiler_write_entry_disable_irq_fiq(PROFILER_EXIT | PROFILER_TIMER);
348 }

351 void c_main(void) {
352 
353  // initialise the model
354  if (!initialise()) {
355  rt_error(RTE_API);
356  }
357 
358  // Start the time at "-1" so that the first tick will be 0
359  time = UINT32_MAX;
360 
361  // Set timer tick (in microseconds)
362  log_debug("setting timer tick callback for %d microseconds", timer_period);
363  spin1_set_timer_tick(timer_period);
364 
365  // Set up the timer tick callback (others are handled elsewhere)
366  spin1_callback_on(TIMER_TICK, timer_callback, TIMER);
367 
368  simulation_run();
369 }
