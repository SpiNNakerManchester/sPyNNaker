177 static bool initialise(void) {
178  log_debug("Initialise: started");
179 
180  // Get the address this core's DTCM data starts at from SRAM
181  data_specification_metadata_t *ds_regions =
182  data_specification_get_data_address();
183 
184  // Read the header
185  if (!data_specification_read_header(ds_regions)) {
186  return false;
187  }
188 
189  // Get the timing details and set up the simulation interface
190  if (!simulation_initialise(
191  data_specification_get_region(SYSTEM_REGION, ds_regions),
192  APPLICATION_NAME_HASH, &timer_period, &simulation_ticks,
193  &infinite_run, &time, SDP, DMA)) {
194  return false;
195  }
196  simulation_set_provenance_function(
197  c_main_store_provenance_data,
198  data_specification_get_region(PROVENANCE_DATA_REGION, ds_regions));
199 
200  // Set up the neurons
201  uint32_t n_synapse_types;
202  uint32_t incoming_spike_buffer_size;
203  uint32_t n_regions_used;
204  if (!neuron_initialise(
205  data_specification_get_region(NEURON_PARAMS_REGION, ds_regions),
206  data_specification_get_region(NEURON_RECORDING_REGION, ds_regions),
207  &n_neurons, &n_synapse_types, &incoming_spike_buffer_size,
208  &n_regions_used)) {
209  return false;
210  }
211 
212  // Set up the synapses
213  uint32_t *ring_buffer_to_input_buffer_left_shifts;
214  bool clear_input_buffers_of_late_packets_init;
215  if (!synapses_initialise(
216  data_specification_get_region(SYNAPSE_PARAMS_REGION, ds_regions),
217  n_neurons, n_synapse_types,
218  &ring_buffer_to_input_buffer_left_shifts,
219  &clear_input_buffers_of_late_packets_init)) {
220  return false;
221  }
222 
223  // set up direct synapses
224  address_t direct_synapses_address;
225  if (!direct_synapses_initialise(
226  data_specification_get_region(DIRECT_MATRIX_REGION, ds_regions),
227  &direct_synapses_address)) {
228  return false;
229  }
230 
231  // Set up the population table
232  uint32_t row_max_n_words;
233  if (!population_table_initialise(
234  data_specification_get_region(POPULATION_TABLE_REGION, ds_regions),
235  data_specification_get_region(SYNAPTIC_MATRIX_REGION, ds_regions),
236  direct_synapses_address, &row_max_n_words)) {
237  return false;
238  }
239  // Set up the synapse dynamics
240  if (!synapse_dynamics_initialise(
241  data_specification_get_region(SYNAPSE_DYNAMICS_REGION, ds_regions),
242  n_neurons, n_synapse_types,
243  ring_buffer_to_input_buffer_left_shifts)) {
244  return false;
245  }
246 
247  // Set up structural plasticity dynamics
248  if (!synaptogenesis_dynamics_initialise(data_specification_get_region(
249  STRUCTURAL_DYNAMICS_REGION, ds_regions))) {
250  return false;
251  }
252 
253  rewiring_period = synaptogenesis_rewiring_period();
254  rewiring = rewiring_period != -1;
255 
256  if (!spike_processing_initialise(
257  row_max_n_words, MC, USER, incoming_spike_buffer_size,
258  clear_input_buffers_of_late_packets_init, n_regions_used)) {
259  return false;
260  }
261 
262  // Setup profiler
263  profiler_init(data_specification_get_region(PROFILER_REGION, ds_regions));
264 
265  // Do bitfield configuration last to only use any unused memory
266  if (!population_table_load_bitfields(
267  data_specification_get_region(BIT_FIELD_FILTER_REGION, ds_regions))) {
268  return false;
269  }
270 
271  print_post_to_pre_entry();
272 
273  log_debug("Initialise: finished");
274  return true;
275 }

302 void background_callback(uint timer_count, uint local_time) {
303  global_timer_count = timer_count;
304  profiler_write_entry_disable_irq_fiq(PROFILER_ENTER | PROFILER_TIMER);
305 
306  last_rewiring_time++;
307 
308  // This is the part where I save the input and output indices
309  // from the circular buffer
310  // If time == 0 as well as output == input == 0 then no rewire is
311  // supposed to happen. No spikes yet
312  log_debug("Timer tick %u \n", local_time);
313 
314  // Then do rewiring
315  if (rewiring &&
316  ((last_rewiring_time >= rewiring_period && !synaptogenesis_is_fast())
317  || synaptogenesis_is_fast())) {
318  last_rewiring_time = 0;
319  // put flag in spike processing to do synaptic rewiring
320  if (synaptogenesis_is_fast()) {
321  spike_processing_do_rewiring(rewiring_period);
322  } else {
323  spike_processing_do_rewiring(1);
324  }
325  count_rewire_attempts++;
326  }
327 
328  // Now do neuron time step update
329  neuron_do_timestep_update(local_time, timer_count);
330 
331  profiler_write_entry_disable_irq_fiq(PROFILER_EXIT | PROFILER_TIMER);
332  n_backgrounds_queued--;
333 }

339 void timer_callback(uint timer_count, UNUSED uint unused) {
340  // Disable interrupts to stop DMAs and MC getting in the way of this bit
341  uint32_t state = spin1_int_disable();
342 
343  time++;
344 
345  // Clear any outstanding spikes
346  spike_processing_clear_input_buffer(time);
347 
348  spin1_mode_restore(state);
349  state = spin1_irq_disable();
350 
351  // Also do synapses timestep update, as this is time-critical
352  synapses_do_timestep_update(time);
353 
354  /* if a fixed number of simulation ticks that were specified at startup
355  * then do reporting for finishing */
356  if (simulation_is_finished()) {
357 
358  // Enter pause and resume state to avoid another tick
359  simulation_handle_pause_resume(resume_callback);
360 
361  log_debug("Completed a run");
362 
363  // rewrite neuron params to SDRAM for reading out if needed
364  data_specification_metadata_t *ds_regions =
365  data_specification_get_data_address();
366  neuron_pause(data_specification_get_region(NEURON_PARAMS_REGION, ds_regions));
367 
368  profiler_write_entry_disable_irq_fiq(PROFILER_EXIT | PROFILER_TIMER);
369 
370  profiler_finalise();
371 
372  // Subtract 1 from the time so this tick gets done again on the next
373  // run
374  time--;
375 
376  log_debug("Rewire tries = %d", count_rewire_attempts);
377  simulation_ready_to_read();
378  spin1_mode_restore(state);
379  return;
380  }
381 
382  // Push the rest to the background
383  if (!spin1_schedule_callback(background_callback, timer_count, time, BACKGROUND)) {
384  // We have failed to do this timer tick!
385  n_background_overloads++;
386  } else {
387  n_backgrounds_queued++;
388  if (n_backgrounds_queued > max_backgrounds_queued) {
389  max_backgrounds_queued++;
390  }
391  }
392 
393  spin1_mode_restore(state);
394 }

397 void c_main(void) {
398 
399  // initialise the model
400  if (!initialise()) {
401  rt_error(RTE_API);
402  }
403 
404  // Start the time at "-1" so that the first tick will be 0
405  time = UINT32_MAX;
406 
407  // Set timer tick (in microseconds)
408  log_debug("setting timer tick callback for %d microseconds", timer_period);
409  spin1_set_timer_tick(timer_period);
410 
411  // Set up the timer tick callback (others are handled elsewhere)
412  spin1_callback_on(TIMER_TICK, timer_callback, TIMER);
413 
414  simulation_run();
415 }
