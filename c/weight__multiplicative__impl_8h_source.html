81 static inline weight_state_t weight_one_term_apply_depression(
82  weight_state_t state, int32_t depression) {
83  // Calculate scale
84  // **NOTE** this calculation must be done at runtime-defined weight
85  // fixed-point format
86  int32_t scale = maths_fixed_mul16(
87  state.weight - state.weight_region->min_weight,
88  state.weight_region->a2_minus, state.weight_multiply_right_shift);
89 
90  // Multiply scale by depression and subtract
91  // **NOTE** using standard STDP fixed-point format handles format conversion
92  state.weight -= STDP_FIXED_MUL_16X16(scale, depression);
93  return state;
94 }

100 static inline weight_state_t weight_one_term_apply_potentiation(
101  weight_state_t state, int32_t potentiation) {
102  // Calculate scale
103  // **NOTE** this calculation must be done at runtime-defined weight
104  // fixed-point format
105  int32_t scale = maths_fixed_mul16(
106  state.weight_region->max_weight - state.weight,
107  state.weight_region->a2_plus, state.weight_multiply_right_shift);
108 
109  // Multiply scale by potentiation and add
110  // **NOTE** using standard STDP fixed-point format handles format conversion
111  state.weight += STDP_FIXED_MUL_16X16(scale, potentiation);
112  return state;
113 }
