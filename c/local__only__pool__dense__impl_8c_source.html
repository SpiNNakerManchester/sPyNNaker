95bool local_only_impl_initialise(void *address){
96 log_info("+++++++++++++++++ CONV init ++++++++++++++++++++");
97 conv_config* sdram_config = address;
98 config = *sdram_config;
99
100 log_info("num connectors = %u", config.n_connectors);
101 if (config.n_connectors == 0) {
102 return false;
103 }
104
105 log_info("num post = %u", config.n_post);
106
107 // Allocate space for connector information and pre-to-post table
108 connectors = spin1_malloc(config.n_connectors * sizeof(connectors[0]));
109 if (connectors == NULL) {
110 log_error("Can't allocate memory for connectors");
111 return false;
112 }
113
114 // The first connector comes after the configuration in SDRAM
115 connector *conn = (connector *) &sdram_config[1];
116 for (uint32_t i = 0; i < config.n_connectors; i++) {
117 log_info("Connector %u: key=0x%08x, mask=0x%08x",
118 i, conn->key_info.key, conn->key_info.mask);
119
120 uint32_t n_bytes = sizeof(*conn) + (conn->n_weights * sizeof(lc_weight_t)) +
121 (conn->n_dims * sizeof(dimension));
122
123 // Copy the data from SDRAM
124 connectors[i] = spin1_malloc(n_bytes);
125 if (connectors[i] == NULL) {
126 log_error("Can't allocate %u bytes for connectors[%u]", n_bytes, i);
127 return false;
128 }
129 spin1_memcpy(connectors[i], conn, n_bytes);
130
131 // Move to the next connector; because it is dynamically sized,
132 // this comes after the last weight in the last connector, which comes
133 // after the last dimension!
134 lc_weight_t* weights = get_weights(conn);
135 uint32_t n_weights = connectors[i]->n_weights;
136 if (n_weights & 0x1) {
137 n_weights += 1;
138 }
139 conn = (connector *) &weights[n_weights];
140 }
141
142 return true;
143}

147static inline int16_t recip_multiply(int16_t integer, int16_t recip) {
148 int32_t i = integer;
149 int32_t r = recip;
150 return (int16_t) ((i * r) >> RECIP_FRACT_BITS);
151}

153static inline bool key_to_index_lookup(uint32_t spike, uint32_t *start, uint32_t *end) {
154 for (uint32_t i = 0; i < config.n_connectors; i++) {
155 connector *c = connectors[i];
156 if ((spike & c->key_info.mask) == c->key_info.key) {
157 *start = i;
158 uint32_t e = i + 1;
159 while (e < config.n_connectors) {
160 connector *c_e = connectors[e];
161 if ((spike & c_e->key_info.mask) != c_e->key_info.mask) {
162 break;
163 }
164 e = e + 1;
165 }
166 *end = e;
167 }
168 }
169 return false;
170}

172bool get_conn_weights(uint32_t spike, uint32_t i, lc_weight_t **weights) {
173 connector *c = connectors[i];
174 uint32_t local_spike = (spike & ~c->key_info.mask) & c->key_info.n_colour_bits;
175
176 // Now work out the index into the weights from the coordinates
177 uint32_t last_extent = 1;
178 uint32_t index = 0;
179 for (uint32_t j = 0; j < c->n_dims; j++) {
180 dimension *dim = &c->dimensions[j];
181
182 // Get the coordinate for this dimension from the spike
183 uint32_t coord = (local_spike & dim->mask) >> dim->shift;
184
185 // Work out the position in the global space
186 coord += dim->pre_start;
187
188 // Work out the position after pooling
189 coord = recip_multiply(coord, dim->recip_pool_stride);
190
191 // Check that the coordinate is in range now for this core
192 if (coord < dim->pre_in_post_start || coord > dim->pre_in_post_end) {
193 return false;
194 }
195
196 // Get the local coordinate after pooling and add into the
197 // final index
198 coord -= dim->pre_in_post_start;
199 index += (coord * last_extent);
200
201 // Remember the shape from this dimension to pass to the next
202 last_extent = dim->pre_in_post_shape;
203 }
204 lc_weight_t *all_weights = get_weights(c);
205 *weights = &all_weights[index * config.n_post];
206 return true;
207}

215void local_only_impl_process_spike(
216 uint32_t time, uint32_t spike, uint16_t* ring_buffers) {
217
218 // Lookup the spike, and if found, get the appropriate parts
219 uint32_t start;
220 uint32_t end;
221 if (!key_to_index_lookup(spike, &start, &end)) {
222 return;
223 }
224
225 // Go through the weights and process them into the ring buffers
226 for (uint32_t i = start; i < end; i++) {
227 connector *connector = connectors[i];
228 lc_weight_t *weights;
229 if (!get_conn_weights(spike, i, &weights)) {
230 continue;
231 }
232
233 for (uint32_t post_index = 0; post_index < config.n_post; post_index++) {
234
235 lc_weight_t weight = weights[post_index];
236 if (weight == 0) {
237 continue;
238 }
239 uint32_t rb_index = 0;
240 if (weight > 0) {
241 rb_index = synapse_row_get_ring_buffer_index(time + connector->delay,
242 connector->positive_synapse_type, post_index,
243 synapse_type_index_bits, synapse_index_bits,
244 synapse_delay_mask);
245 } else {
246 rb_index = synapse_row_get_ring_buffer_index(time + connector->delay,
247 connector->negative_synapse_type, post_index,
248 synapse_type_index_bits, synapse_index_bits,
249 synapse_delay_mask);
250 weight = -weight;
251 }
252 log_debug("Updating ring_buffers[%u] for post neuron %u with weight %u",
253 rb_index, post_index, weight);
254
255 // Add weight to current ring buffer value, avoiding saturation
256 uint32_t accumulation = ring_buffers[rb_index] + weight;
257 uint32_t sat_test = accumulation & 0x10000;
258 if (sat_test) {
259 accumulation = sat_test - 1;
260 }
261 ring_buffers[rb_index] = accumulation;
262 }
263 }
