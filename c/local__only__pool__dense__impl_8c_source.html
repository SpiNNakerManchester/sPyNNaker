93 bool local_only_impl_initialise(void *address){
94  log_info("+++++++++++++++++ CONV init ++++++++++++++++++++");
95  conv_config* sdram_config = address;
96  config = *sdram_config;
97 
98  log_info("num connectors = %u", config.n_connectors);
99  if (config.n_connectors == 0) {
100  return false;
101  }
102 
103  log_info("num post = %u", config.n_post);
104 
105  // Allocate space for connector information and pre-to-post table
106  connectors = spin1_malloc(config.n_connectors * sizeof(connectors[0]));
107  if (connectors == NULL) {
108  log_error("Can't allocate memory for connectors");
109  return false;
110  }
111 
112  // The first connector comes after the configuration in SDRAM
113  connector *conn = (connector *) &sdram_config[1];
114  for (uint32_t i = 0; i < config.n_connectors; i++) {
115  log_info("Connector %u: key=0x%08x, mask=0x%08x",
116  i, conn->key_info.key, conn->key_info.mask);
117 
118  uint32_t n_bytes = sizeof(*conn) + (conn->n_weights * sizeof(lc_weight_t)) +
119  (conn->n_dims * sizeof(dimension));
120 
121  // Copy the data from SDRAM
122  connectors[i] = spin1_malloc(n_bytes);
123  if (connectors[i] == NULL) {
124  log_error("Can't allocate %u bytes for connectors[%u]", n_bytes, i);
125  return false;
126  }
127  spin1_memcpy(connectors[i], conn, n_bytes);
128 
129  // Move to the next connector; because it is dynamically sized,
130  // this comes after the last weight in the last connector, which comes
131  // after the last dimension!
132  lc_weight_t* weights = get_weights(conn);
133  conn = (connector *) &weights[connectors[i]->n_weights];
134  }
135 
136  return true;
137 }

147 static inline bool key_to_index_lookup(uint32_t spike, connector **conn,
148  lc_weight_t **weights) {
149  for (uint32_t i = 0; i < config.n_connectors; i++) {
150  connector *c = connectors[i];
151  if ((spike & c->key_info.mask) == c->key_info.key) {
152  *conn = c;
153 
154  // Now work out the index into the weights from the coordinates
155  uint32_t last_extent = 1;
156  uint32_t index = 0;
157  for (uint32_t j = 0; j < c->n_dims; j++) {
158  dimension *dim = &c->dimensions[j];
159 
160  // Get the coordinate for this dimension from the spike
161  uint32_t coord = (spike & dim->mask) >> dim->shift;
162 
163  // Work out the position in the global space
164  coord += dim->pre_start;
165 
166  // Work out the position after pooling
167  coord = recip_multiply(coord, dim->recip_pool_stride);
168 
169  // Check that the coordinate is in range now for this core
170  if (coord < dim->pre_in_post_start || coord > dim->pre_in_post_end) {
171  return false;
172  }
173 
174  // Get the local coordinate after pooling and add into the
175  // final index
176  coord -= dim->pre_in_post_start;
177  index += (coord * last_extent);
178 
179  // Remember the shape from this dimension to pass to the next
180  last_extent = dim->pre_in_post_shape;
181  }
182  lc_weight_t *all_weights = get_weights(c);
183  *weights = &all_weights[index * config.n_post];
184  return true;
185  }
186  }
187  return false;
188 }

196 void local_only_impl_process_spike(
197  uint32_t time, uint32_t spike, uint16_t* ring_buffers) {
198  connector *connector;
199  lc_weight_t *weights;
200 
201  // Lookup the spike, and if found, get the appropriate parts
202  if (!key_to_index_lookup(spike, &connector, &weights)) {
203  return;
204  }
205 
206  // Go through the weights and process them into the ring buffers
207  for (uint32_t post_index = 0; post_index < config.n_post; post_index++) {
208  lc_weight_t weight = weights[post_index];
209  if (weight == 0) {
210  continue;
211  }
212  uint32_t rb_index = 0;
213  if (weight > 0) {
214  rb_index = synapse_row_get_ring_buffer_index(time + 1,
215  connector->positive_synapse_type, post_index,
216  synapse_type_index_bits, synapse_index_bits,
217  synapse_delay_mask);
218  } else {
219  rb_index = synapse_row_get_ring_buffer_index(time + 1,
220  connector->negative_synapse_type, post_index,
221  synapse_type_index_bits, synapse_index_bits,
222  synapse_delay_mask);
223  weight = -weight;
224  }
225  log_debug("Updating ring_buffers[%u] for post neuron %u with weight %u",
226  rb_index, post_index, weight);
227 
228  // Add weight to current ring buffer value, avoiding saturation
229  uint32_t accumulation = ring_buffers[rb_index] + weight;
230  uint32_t sat_test = accumulation & 0x10000;
231  if (sat_test) {
232  accumulation = sat_test - 1;
233  }
234  ring_buffers[rb_index] = accumulation;
235  }
