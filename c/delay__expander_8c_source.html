61 static bool read_delay_builder_region(address_t *in_region,
62  bit_field_t *neuron_delay_stage_config, uint32_t post_slice_start,
63  uint32_t post_slice_count) {
64  // Get the parameters
65  address_t region = *in_region;
66  struct delay_builder_config config;
67  fast_memcpy(&config, region, sizeof(config));
68  region += sizeof(config) / sizeof(uint32_t);
69  // Get the connector and delay parameter generators
70  connection_generator_t connection_generator =
71  connection_generator_init(config.connector_type, &region);
72  param_generator_t delay_generator =
73  param_generator_init(config.delay_type, &region);
74  *in_region = region;
75 
76  // If any components couldn't be created return false
77  if (connection_generator == NULL || delay_generator == NULL) {
78  return false;
79  }
80 
81  // For each pre-neuron, generate the connections
82  uint32_t pre_slice_start = config.pre_slice_start;
83  uint32_t pre_slice_count = config.pre_slice_count;
84  uint32_t pre_slice_end = pre_slice_start + pre_slice_count;
85  for (uint32_t pre_neuron_index = pre_slice_start;
86  pre_neuron_index < pre_slice_end; pre_neuron_index++) {
87  // Generate the connections
88  uint32_t max_n_synapses =
89  config.max_row_n_synapses + config.max_delayed_row_n_synapses;
90  uint16_t indices[max_n_synapses];
91  uint32_t n_indices = connection_generator_generate(
92  connection_generator, pre_slice_start, pre_slice_count,
93  pre_neuron_index, post_slice_start, post_slice_count,
94  max_n_synapses, indices);
95  log_debug("Generated %u synapses", n_indices);
96 
97  // Generate delays for each index
98  accum delay_params[n_indices];
99  param_generator_generate(
100  delay_generator, n_indices, pre_neuron_index, indices,
101  delay_params);
102 
103  // Go through the delays
104  for (uint32_t i = 0; i < n_indices; i++) {
105  // Get the delay in timesteps
106  accum delay = delay_params[i] * config.timestep_per_delay;
107  if (delay < 0) {
108  delay = 1;
109  }
110 
111  // Round down to an integer number of timesteps
112  uint16_t rounded_delay = (uint16_t) delay;
113  if (delay != rounded_delay) {
114  log_debug("Rounded delay %k to %u", delay, rounded_delay);
115  }
116 
117  // Get the delay stage and update the data
118  struct delay_value delay_value =
119  get_delay(rounded_delay, config.max_stage);
120  if (delay_value.stage > 0) {
121  bit_field_set(neuron_delay_stage_config[delay_value.stage - 1],
122  pre_neuron_index - pre_slice_start);
123  }
124  }
125  }
126 
127  // Finish with the generators
128  connection_generator_free(connection_generator);
129  param_generator_free(delay_generator);
130 
131  return true;
132 }

142 static bool run_delay_expander(
143  void *delay_params_address, address_t params_address) {
144  // Read the global parameters from the delay extension
145  struct delay_parameters *delay_params = delay_params_address;
146  uint32_t num_neurons = delay_params->n_atoms;
147  uint32_t neuron_bit_field_words = get_bit_field_size(num_neurons);
148  uint32_t n_stages = delay_params->n_delay_stages;
149 
150  // Set up the bit fields
151  bit_field_t *neuron_delay_stage_config =
152  spin1_malloc(n_stages * sizeof(bit_field_t));
153  for (uint32_t d = 0; d < n_stages; d++) {
154  neuron_delay_stage_config[d] =
155  &delay_params->delay_blocks[d * neuron_bit_field_words];
156  clear_bit_field(neuron_delay_stage_config[d], neuron_bit_field_words);
157  }
158 
159  // Read the global parameters from the expander region
160  uint32_t n_out_edges = *params_address++;
161  uint32_t post_slice_start = *params_address++;
162  uint32_t post_slice_count = *params_address++;
163 
164  log_info("Generating %u delay edges for %u atoms starting at %u",
165  n_out_edges, post_slice_count, post_slice_start);
166 
167  // Go through each connector and make the delay data
168  for (uint32_t edge = 0; edge < n_out_edges; edge++) {
169  if (!read_delay_builder_region(
170  &params_address, neuron_delay_stage_config,
171  post_slice_start, post_slice_count)) {
172  return false;
173  }
174  }
175 
176  return true;
177 }
