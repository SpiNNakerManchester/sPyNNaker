62 static bool read_delay_builder_region(address_t *in_region,
63  bit_field_t *neuron_delay_stage_config, uint32_t pre_slice_start,
64  uint32_t pre_slice_count) {
65  // Get the parameters
66  address_t region = *in_region;
67  struct delay_builder_config config;
68  fast_memcpy(&config, region, sizeof(config));
69  region += sizeof(config) / sizeof(uint32_t);
70  // Get the connector and delay parameter generators
71  connection_generator_t connection_generator =
72  connection_generator_init(config.connector_type, &region);
73  param_generator_t delay_generator =
74  param_generator_init(config.delay_type, &region);
75  *in_region = region;
76 
77  // If any components couldn't be created return false
78  if (connection_generator == NULL || delay_generator == NULL) {
79  return false;
80  }
81 
82  // For each pre-neuron, generate the connections
83  uint32_t pre_slice_end = pre_slice_start + pre_slice_count;
84  for (uint32_t pre_neuron_index = pre_slice_start;
85  pre_neuron_index < pre_slice_end; pre_neuron_index++) {
86  // Generate the connections
87  uint32_t max_n_synapses =
88  config.max_row_n_synapses + config.max_delayed_row_n_synapses;
89  uint16_t indices[max_n_synapses];
90  uint32_t n_indices = connection_generator_generate(
91  connection_generator, pre_slice_start, pre_slice_count,
92  pre_neuron_index, config.post_slice_start, config.post_slice_count,
93  max_n_synapses, indices);
94  log_debug("Generated %u synapses", n_indices);
95 
96  // Generate delays for each index
97  accum delay_params[n_indices];
98  param_generator_generate(
99  delay_generator, n_indices, pre_neuron_index, indices,
100  delay_params);
101 
102  // Go through the delays
103  for (uint32_t i = 0; i < n_indices; i++) {
104  // Get the delay in timesteps
105  accum delay = delay_params[i] * config.timestep_per_delay;
106  if (delay < 0) {
107  delay = 1;
108  }
109 
110  // Round down to an integer number of timesteps
111  uint16_t rounded_delay = (uint16_t) delay;
112  if (delay != rounded_delay) {
113  log_debug("Rounded delay %k to %u", delay, rounded_delay);
114  }
115 
116  // Get the delay stage and update the data
117  struct delay_value delay_value = get_delay(
118  rounded_delay, config.max_stage, config.max_delay_per_stage);
119  if (delay_value.stage > 0) {
120  bit_field_set(neuron_delay_stage_config[delay_value.stage - 1],
121  pre_neuron_index - pre_slice_start);
122  }
123  }
124  }
125 
126  // Finish with the generators
127  connection_generator_free(connection_generator);
128  param_generator_free(delay_generator);
129 
130  return true;
131 }

141 static bool run_delay_expander(
142  void *delay_params_address, address_t params_address) {
143  // Read the global parameters from the delay extension
144  struct delay_parameters *delay_params = delay_params_address;
145  uint32_t num_neurons = delay_params->n_atoms;
146  uint32_t neuron_bit_field_words = get_bit_field_size(num_neurons);
147  uint32_t n_stages = delay_params->n_delay_stages;
148 
149  // Set up the bit fields
150  bit_field_t *neuron_delay_stage_config =
151  spin1_malloc(n_stages * sizeof(bit_field_t));
152  for (uint32_t d = 0; d < n_stages; d++) {
153  neuron_delay_stage_config[d] =
154  &delay_params->delay_blocks[d * neuron_bit_field_words];
155  clear_bit_field(neuron_delay_stage_config[d], neuron_bit_field_words);
156  }
157 
158  // Read the global parameters from the expander region
159  uint32_t n_out_edges = *params_address++;
160  uint32_t pre_slice_start = *params_address++;
161  uint32_t pre_slice_count = *params_address++;
162 
163  log_info("Generating %u delay edges for %u atoms starting at %u",
164  n_out_edges, pre_slice_count, pre_slice_start);
165 
166  // Go through each connector and make the delay data
167  for (uint32_t edge = 0; edge < n_out_edges; edge++) {
168  if (!read_delay_builder_region(
169  &params_address, neuron_delay_stage_config,
170  pre_slice_start, pre_slice_count)) {
171  return false;
172  }
173  }
174 
175  return true;
176 }
