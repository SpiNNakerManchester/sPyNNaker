77 static inline bool initialise_common_regions(
78  uint32_t *timer_period, uint32_t *simulation_ticks,
79  uint32_t *infinite_run, uint32_t *time, uint32_t *recording_flags,
80  prov_callback_t store_provenance_function, callback_t timer_callback,
81  struct common_regions regions, struct common_priorities priorities,
82  data_specification_metadata_t **ds_regions) {
83 
84  // Get the address this core's DTCM data starts at from SRAM
85  *ds_regions = data_specification_get_data_address();
86 
87  // Read the header
88  if (!data_specification_read_header(*ds_regions)) {
89  return false;
90  }
91 
92  // Get the timing details and set up the simulation interface
93  if (!simulation_initialise(
94  data_specification_get_region(regions.system, *ds_regions),
95  APPLICATION_NAME_HASH, timer_period, simulation_ticks,
96  infinite_run, time, priorities.sdp, priorities.dma)) {
97  return false;
98  }
99  simulation_set_provenance_function(
100  store_provenance_function,
101  data_specification_get_region(regions.provenance, *ds_regions));
102 
103  // Setup profiler
104  profiler_init(data_specification_get_region(regions.profiler, *ds_regions));
105 
106  // Setup recording
107  void *rec_addr = data_specification_get_region(regions.recording, *ds_regions);
108  if (!recording_initialize(&rec_addr, recording_flags)) {
109  return false;
110  }
111 
112  if (timer_callback) {
113 
114  // Set up the timer tick callback (others are handled elsewhere)
115  spin1_callback_on(TIMER_TICK, timer_callback, priorities.timer);
116  }
117 
118  return true;
119 }
