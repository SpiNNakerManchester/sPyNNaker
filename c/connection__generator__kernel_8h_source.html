107 static uint32_t connection_generator_kernel_generate(
108  void *generator, uint32_t pre_slice_start, uint32_t pre_slice_count,
109  uint32_t pre_neuron_index, uint32_t post_slice_start,
110  uint32_t post_slice_count, uint32_t max_row_length, uint16_t *indices) {
111  use(pre_slice_start);
112  use(pre_slice_count);
113 
114  log_debug("Generating for %u", pre_neuron_index);
115 
116  // If no space, generate nothing
117  if (max_row_length < 1) {
118  return 0;
119  }
120 
121  struct kernel *obj = generator;
122 
123  // start n_conns at zero
124  uint32_t n_conns = 0;
125 
126  uint16_t pre_c = 0;
127  uint16_t pre_r = uidiv(pre_neuron_index, obj->preWidth, &pre_c);
128 
129  uint16_t hlf_kw = obj->kernelWidth >> 1;
130  uint16_t hlf_kh = obj->kernelHeight >> 1;
131  int16_t k_r, k_c;
132  for (uint16_t i = 0; i < post_slice_count; i++) {
133  uint16_t post_r, post_c; //post raw
134  uint16_t pac_r, pac_c; // post as common
135  int16_t pap_r, pap_c; // post as pre
136  post_r = uidiv(post_slice_start + i, obj->postWidth, &post_c);
137 
138  //move post coords into common coordinate system
139  post_in_pre_world(post_r, post_c, obj->startPostHeight,
140  obj->startPostWidth, obj->stepPostHeight,
141  obj->stepPostWidth, &pac_r, &pac_c);
142 
143  //move common to pre coords
144  pre_in_post_world(
145  pac_r, pac_c, obj->startPreHeight, obj->startPreHeight,
146  obj->stepPreHeight, obj->stepPreWidth, &pap_r, &pap_c);
147 
148  int16_t r_diff = (int16_t) pap_r - (int16_t) pre_r;
149  int16_t c_diff = (int16_t) pap_c - (int16_t) pre_c;
150 
151  k_r = hlf_kh - r_diff;
152  k_c = hlf_kw - c_diff;
153 
154  if ((0 <= k_r) && (k_r < obj->kernelHeight) && (0 <= k_c)
155  && (k_c < obj->kernelWidth)) {
156  indices[n_conns++] = i;
157  }
158  }
159 
160  return n_conns;
161 }
