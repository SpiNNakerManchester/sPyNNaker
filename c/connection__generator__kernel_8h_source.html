121 static uint32_t connection_generator_kernel_generate(
122  void *generator, uint32_t pre_slice_start, uint32_t pre_slice_count,
123  uint32_t pre_neuron_index, uint32_t post_slice_start,
124  uint32_t post_slice_count, uint32_t max_row_length, uint16_t *indices) {
125  use(pre_slice_start);
126  use(pre_slice_count);
127 
128  log_debug("Generating for %u", pre_neuron_index);
129 
130  // If no space, generate nothing
131  if (max_row_length < 1) {
132  return 0;
133  }
134 
135  struct kernel *obj = generator;
136 
137  // start n_conns at zero
138  uint32_t n_conns = 0;
139 
140  uint16_t pre_c = 0;
141  uint16_t pre_r = uidiv(pre_neuron_index, obj->preWidth, &pre_c);
142 
143  uint16_t hlf_kw = obj->kernelWidth >> 1;
144  uint16_t hlf_kh = obj->kernelHeight >> 1;
145  int16_t k_r, k_c;
146  for (uint16_t i = 0; i < post_slice_count; i++) {
147  uint16_t post_r, post_c; //post raw
148  uint16_t pac_r, pac_c; // post as common
149  int16_t pap_r, pap_c; // post as pre
150  post_r = uidiv(post_slice_start + i, obj->postWidth, &post_c);
151 
152  //move post coords into common coordinate system
153  post_in_pre_world(post_r, post_c, obj->startPostHeight,
154  obj->startPostWidth, obj->stepPostHeight,
155  obj->stepPostWidth, &pac_r, &pac_c);
156 
157  //move common to pre coords
158  pre_in_post_world(
159  pac_r, pac_c, obj->startPreHeight, obj->startPreHeight,
160  obj->stepPreHeight, obj->stepPreWidth, &pap_r, &pap_c);
161 
162  int16_t r_diff = (int16_t) pap_r - (int16_t) pre_r;
163  int16_t c_diff = (int16_t) pap_c - (int16_t) pre_c;
164 
165  k_r = hlf_kh - r_diff;
166  k_c = hlf_kw - c_diff;
167 
168  if ((0 <= k_r) && (k_r < obj->kernelHeight) && (0 <= k_c)
169  && (k_c < obj->kernelWidth)) {
170  indices[n_conns++] = i;
171  }
172  }
173 
174  return n_conns;
175 }
