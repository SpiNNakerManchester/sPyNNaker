85 static void *connection_generator_kernel_initialise(void **region) {
86  struct kernel *params_sdram = *region;
87 
88  // Allocate the data structure for parameters
89  uint32_t kernelSize = params_sdram->kernelWidth * params_sdram->kernelHeight;
90  uint32_t size = sizeof(struct kernel);
91  uint32_t extra = 0;
92  if (params_sdram->weightsPresent) {
93  size += kernelSize * sizeof(accum);
94  extra += kernelSize;
95  }
96  if (params_sdram->delaysPresent) {
97  size += kernelSize * sizeof(accum);
98  extra += kernelSize;
99  }
100  struct kernel *obj = spin1_malloc(size);
101 
102  // Copy the parameters into the data structure
103  spin1_memcpy(obj, params_sdram, size);
104  *region = &(params_sdram->kernelWeightsAndDelays[extra]);
105 
106  log_debug("Kernel connector, m_kernelWidth, m_kernelHeight = %u %u",
107  obj->kernelWidth, obj->kernelHeight);
108 
109  return obj;
110 }

139 static bool connection_generator_kernel_generate(
140  void *generator, uint32_t pre_lo, uint32_t pre_hi,
141  uint32_t post_lo, uint32_t post_hi, UNUSED uint32_t post_index,
142  uint32_t post_slice_start, uint32_t post_slice_count,
143  unsigned long accum weight_scale, accum timestep_per_delay,
144  param_generator_t weight_generator, param_generator_t delay_generator,
145  matrix_generator_t matrix_generator) {
146 
147  struct kernel *obj = generator;
148  uint16_t hlf_kw = obj->kernelWidth >> 1;
149  uint16_t hlf_kh = obj->kernelHeight >> 1;
150  int16_t k_r, k_c;
151  uint32_t k_offset = obj->kernelWidth * obj->kernelHeight;
152 
153  // Get the actual ranges to generate within
154  uint32_t post_start = max(post_slice_start, post_lo);
155  uint32_t post_end = min(post_slice_start + post_slice_count - 1, post_hi);
156 
157  // Go through and find the coordinates needed
158  for (uint32_t pre = pre_lo; pre <= pre_hi; pre++) {
159  uint16_t pre_c = 0;
160  uint16_t pre_r = uidiv(pre, obj->preWidth, &pre_c);
161  for (uint32_t post = post_start; post <= post_end; post++) {
162  uint16_t post_r, post_c; //post raw
163  uint16_t pac_r, pac_c; // post as common
164  int16_t pap_r, pap_c; // post as pre
165  post_r = uidiv(post, obj->postWidth, &post_c);
166 
167  //move post coords into common coordinate system
168  post_in_pre_world(
169  post_r, post_c, obj->startPostHeight, obj->startPostWidth,
170  obj->stepPostHeight, obj->stepPostWidth, &pac_r, &pac_c);
171 
172  //move common to pre coords
173  pre_in_post_world(
174  pac_r, pac_c, obj->startPreHeight, obj->startPreWidth,
175  obj->stepPreHeight, obj->stepPreWidth, &pap_r, &pap_c);
176 
177  int16_t r_diff = (int16_t) pap_r - (int16_t) pre_r;
178  int16_t c_diff = (int16_t) pap_c - (int16_t) pre_c;
179 
180  k_r = hlf_kh - r_diff;
181  k_c = hlf_kw - c_diff;
182 
183  if ((0 <= k_r) && (k_r < obj->kernelHeight) && (0 <= k_c)
184  && (k_c < obj->kernelWidth)) {
185  uint32_t local_post = post - post_slice_start;
186  accum weight;
187  accum delay;
188  uint32_t k = (k_r * obj->kernelWidth) + k_c;
189  if (obj->weightsPresent && obj->delaysPresent) {
190  weight = obj->kernelWeightsAndDelays[k];
191  delay = obj->kernelWeightsAndDelays[k + k_offset];
192  } else if (obj->weightsPresent) {
193  weight = obj->kernelWeightsAndDelays[k];
194  delay = param_generator_generate(delay_generator);
195  } else if (obj->delaysPresent) {
196  weight = param_generator_generate(weight_generator);
197  delay = obj->kernelWeightsAndDelays[k];
198  } else {
199  weight = param_generator_generate(weight_generator);
200  delay = param_generator_generate(delay_generator);
201  }
202  if (!matrix_generator_write_synapse(
203  matrix_generator, pre, local_post,
204  weight, rescale_delay(delay, timestep_per_delay),
205  weight_scale)) {
206  log_error("Matrix size is wrong!");
207  return false;
208  }
209  }
210  }
211  }
212  return true;
213 }
