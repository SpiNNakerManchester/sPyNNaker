121 static uint32_t connection_generator_kernel_generate(
122  void *generator, UNUSED uint32_t pre_slice_start,
123  UNUSED uint32_t pre_slice_count,
124  uint32_t pre_neuron_index, uint32_t post_slice_start,
125  uint32_t post_slice_count, uint32_t max_row_length, uint16_t *indices) {
126  log_debug("Generating for %u", pre_neuron_index);
127 
128  // If no space, generate nothing
129  if (max_row_length < 1) {
130  return 0;
131  }
132 
133  struct kernel *obj = generator;
134 
135  // start n_conns at zero
136  uint32_t n_conns = 0;
137 
138  uint16_t pre_c = 0;
139  uint16_t pre_r = uidiv(pre_neuron_index, obj->preWidth, &pre_c);
140 
141  // Check whether these coordinates should be included based on step functions
142  if (!(((pre_r - obj->startPreHeight) % obj->stepPreHeight == 0) &&
143  ((pre_c - obj->startPreWidth) % obj->stepPreWidth == 0))) {
144  return 0;
145  }
146 
147  uint16_t hlf_kw = obj->kernelWidth >> 1;
148  uint16_t hlf_kh = obj->kernelHeight >> 1;
149  int16_t k_r, k_c;
150  for (uint16_t i = 0; i < post_slice_count; i++) {
151  uint16_t post_r, post_c; //post raw
152  uint16_t pac_r, pac_c; // post as common
153  int16_t pap_r, pap_c; // post as pre
154  post_r = uidiv(post_slice_start + i, obj->postWidth, &post_c);
155 
156  //move post coords into common coordinate system
157  post_in_pre_world(
158  post_r, post_c, obj->startPostHeight, obj->startPostWidth,
159  obj->stepPostHeight, obj->stepPostWidth, &pac_r, &pac_c);
160 
161  //move common to pre coords
162  pre_in_post_world(
163  pac_r, pac_c, obj->startPreHeight, obj->startPreWidth,
164  obj->stepPreHeight, obj->stepPreWidth, &pap_r, &pap_c);
165 
166  int16_t r_diff = (int16_t) pap_r - (int16_t) pre_r;
167  int16_t c_diff = (int16_t) pap_c - (int16_t) pre_c;
168 
169  k_r = hlf_kh - r_diff;
170  k_c = hlf_kw - c_diff;
171 
172  if ((0 <= k_r) && (k_r < obj->kernelHeight) && (0 <= k_c)
173  && (k_c < obj->kernelWidth)) {
174  indices[n_conns++] = i;
175  }
176  }
177 
178  return n_conns;
179 }
