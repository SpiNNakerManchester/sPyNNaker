93 matrix_generator_t matrix_generator_init(uint32_t hash, void **in_region,
94  void *synaptic_matrix) {
95  // Look through the known generators
96  for (uint32_t i = 0; i < N_MATRIX_GENERATORS; i++) {
97  const matrix_generator_info *type = &matrix_generators[i];
98 
99  // If the hash requested matches the hash of the generator, use it
100  if (hash == type->hash) {
101  // Prepare a space for the data
102  struct matrix_generator *generator =
103  spin1_malloc(sizeof(struct matrix_generator));
104  if (generator == NULL) {
105  log_error("Could not create generator");
106  return NULL;
107  }
108 
109  // Store the index
110  generator->type = type;
111 
112  // Initialise the generator and store the data
113  generator->data = type->initialize(in_region, synaptic_matrix);
114  return generator;
115  }
116  }
117  log_error("Matrix generator with hash %u not found", hash);
118  return NULL;
119 }
