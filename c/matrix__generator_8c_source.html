93 matrix_generator_t matrix_generator_init(uint32_t hash, address_t *in_region) {
94  // Look through the known generators
95  for (uint32_t i = 0; i < N_MATRIX_GENERATORS; i++) {
96  const matrix_generator_info *type = &matrix_generators[i];
97 
98  // If the hash requested matches the hash of the generator, use it
99  if (hash == type->hash) {
100  // Prepare a space for the data
101  struct matrix_generator *generator =
102  spin1_malloc(sizeof(struct matrix_generator));
103  if (generator == NULL) {
104  log_error("Could not create generator");
105  return NULL;
106  }
107 
108  // Store the index
109  generator->type = type;
110 
111  // Initialise the generator and store the data
112  generator->data = type->initialize(in_region);
113  return generator;
114  }
115  }
116  log_error("Matrix generator with hash %u not found", hash);
117  return NULL;
118 }

169 static inline uint16_t rescale_delay(accum delay, accum timestep_per_delay) {
170  delay = delay * timestep_per_delay;
171  if (delay < 0) {
172  delay = 1;
173  }
174  uint16_t delay_int = (uint16_t) delay;
175  if (delay != delay_int) {
176  log_debug("Rounded delay %k to %u", delay, delay_int);
177  }
178  return delay_int;
179 }
180 
186 static inline uint16_t rescale_weight(accum weight, unsigned long accum weight_scale) {
187  if (weight < 0) {
188  weight = -weight;
189  }
190  accum weight_scaled = weight * weight_scale;
191  uint16_t weight_int = (uint16_t) weight_scaled;
192  if (weight_scaled != weight_int) {
193  log_debug("Rounded weight %k to %u (scale is %k)",
194  weight_scaled, weight_int, weight_scale);
195  }
196  return weight_int;
197 }

199 bool matrix_generator_generate(
200  matrix_generator_t generator,
201  address_t synaptic_matrix, address_t delayed_synaptic_matrix,
202  uint32_t max_row_n_words, uint32_t max_delayed_row_n_words,
203  uint32_t max_row_n_synapses, uint32_t max_delayed_row_n_synapses,
204  uint32_t n_synapse_type_bits, uint32_t n_synapse_index_bits,
205  uint32_t synapse_type, unsigned long accum *weight_scales,
206  uint32_t post_slice_start, uint32_t post_slice_count,
207  uint32_t pre_slice_start, uint32_t pre_slice_count,
208  connection_generator_t connection_generator,
209  param_generator_t delay_generator, param_generator_t weight_generator,
210  uint32_t max_stage, uint32_t max_delay_in_a_stage,
211  accum timestep_per_delay) {
212  // Go through and generate connections for each pre-neuron
213  uint32_t n_connections = 0;
214  for (uint32_t i = 0; i < pre_slice_count; i++) {
215  uint32_t pre_neuron_index = pre_slice_start + i;
216 
217  // Get up to a maximum number of synapses
218  uint32_t max_n_synapses =
219  max_row_n_synapses + max_delayed_row_n_synapses;
220  uint16_t indices[max_n_synapses];
221  uint32_t n_indices = connection_generator_generate(
222  connection_generator, pre_slice_start, pre_slice_count,
223  pre_neuron_index, post_slice_start, post_slice_count,
224  max_n_synapses, indices);
225  log_debug("Generated %u synapses", n_indices);
226 
227  for (uint32_t j = 0; j < n_indices; j++) {
228  if (indices[j] >= post_slice_count) {
229  log_error("Index %u out of bounds for %u neurons", indices[j], post_slice_count);
230  rt_error(RTE_SWERR);
231  }
232  }
233 
234  accum delay_params[n_indices], weight_params[n_indices];
235  uint16_t delays[n_indices], weights[n_indices];
236 
237  // Generate delays for each index
238  param_generator_generate(
239  delay_generator, n_indices, pre_neuron_index, indices, delay_params);
240  for (uint32_t j = 0; j < n_indices; j++) {
241  delays[j] = rescale_delay(delay_params[j], timestep_per_delay);
242  }
243 
244  // Generate weights for each index
245  param_generator_generate(
246  weight_generator, n_indices, pre_neuron_index, indices, weight_params);
247  for (uint32_t j = 0; j < n_indices; j++) {
248  weights[j] = rescale_weight(weight_params[j], weight_scales[synapse_type]);
249  }
250 
251  // Write row
252  matrix_generator_write_row(
253  generator, synaptic_matrix, delayed_synaptic_matrix,
254  pre_slice_count, pre_neuron_index - pre_slice_start,
255  max_row_n_words, max_delayed_row_n_words,
256  n_synapse_type_bits, n_synapse_index_bits,
257  synapse_type, n_indices, indices, delays, weights, max_stage,
258  max_delay_in_a_stage);
259 
260  n_connections += n_indices;
261  }
262  log_debug("\t\tTotal synapses generated = %u. Done!", n_connections);
263 
264  return true;
265 }
