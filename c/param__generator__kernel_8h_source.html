82 static void param_generator_kernel_generate(void *data, uint32_t n_synapses,
83  uint32_t pre_neuron_index, uint16_t *indices, accum *values) {
84  use(pre_neuron_index);
85  use(indices);
86  struct all_kernel_params *obj = data;
87  struct param_generator_kernel *params = &obj->params;
88  uint16_t pre_c = 0;
89  uint16_t pre_r = uidiv(pre_neuron_index, params->preWidth, &pre_c);
90 
91  uint16_t hlf_kw = params->kernelWidth >> 1;
92  uint16_t hlf_kh = params->kernelHeight >> 1;
93  int16_t k_r, k_c;
94  for (uint16_t i = 0; i < n_synapses; i++) {
95  uint16_t post_r, post_c; //post raw
96  uint16_t pac_r, pac_c; // post as common
97  int16_t pap_r, pap_c; // post as pre
98  post_r = uidiv(params->post_slice_start + indices[i],
99  params->postWidth, &post_c);
100 
101  //move post coords into common coordinate system
102  post_in_pre_world(post_r, post_c, params->startPostHeight,
103  params->startPostWidth, params->stepPostHeight,
104  params->stepPostWidth, &pac_r, &pac_c);
105 
106  //move common to pre coords
107  pre_in_post_world(
108  pac_r, pac_c, params->startPreHeight, params->startPreHeight,
109  params->stepPreHeight, params->stepPreWidth, &pap_r, &pap_c);
110 
111  int16_t r_diff = (int16_t) pap_r - (int16_t) pre_r;
112  int16_t c_diff = (int16_t) pap_c - (int16_t) pre_c;
113 
114  k_r = hlf_kh - r_diff;
115  k_c = hlf_kw - c_diff;
116 
117  if ((0 <= k_r) && (k_r < params->kernelHeight) && (0 <= k_c)
118  && (k_c < params->kernelWidth)) {
119  values[i] = obj->values[k_r * params->kernelWidth + k_c];
120  // LOG_PRINT(LOG_LEVEL_INFO, "val = %5.6k", output[i]);
121  } else {
122  log_error("Kernel coordinates off range (%d, %d)", k_r, k_c);
123  }
124  }
125 }
