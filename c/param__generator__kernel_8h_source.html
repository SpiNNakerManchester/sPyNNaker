84 static void *param_generator_kernel_initialize(address_t *region) {
85  struct all_kernel_params *obj = spin1_malloc(sizeof(struct all_kernel_params));
86  struct param_generator_kernel *params_sdram = (void *) *region;
87  fast_memcpy(&obj->params, params_sdram++, sizeof(*params_sdram));
88  *region = (void *) params_sdram;

117 static void param_generator_kernel_generate(
118  void *generator, uint32_t n_synapses,
119  uint32_t pre_neuron_index, uint16_t *indices, accum *values) {
120  struct all_kernel_params *obj = generator;
121  struct param_generator_kernel *params = &obj->params;
122  uint16_t pre_c = 0;
123  uint16_t pre_r = uidiv(pre_neuron_index, params->preWidth, &pre_c);
124 
125  // Check whether these coordinates should be included based on step functions
126  if (!(((pre_r - params->startPreHeight) % params->stepPreHeight == 0) &&
127  ((pre_c - params->startPreWidth) % params->stepPreWidth == 0))) {
128  return;
129  }
130 
131  uint16_t hlf_kw = params->kernelWidth >> 1;
132  uint16_t hlf_kh = params->kernelHeight >> 1;
133  int16_t k_r, k_c;
134  for (uint16_t i = 0; i < n_synapses; i++) {
135  uint16_t post_r, post_c; //post raw
136  uint16_t pac_r, pac_c; // post as common
137  int16_t pap_r, pap_c; // post as pre
138  post_r = uidiv(params->post_slice_start + indices[i],
139  params->postWidth, &post_c);
140 
141  //move post coords into common coordinate system
142  post_in_pre_world(post_r, post_c, params->startPostHeight,
143  params->startPostWidth, params->stepPostHeight,
144  params->stepPostWidth, &pac_r, &pac_c);
145 
146  //move common to pre coords
147  pre_in_post_world(
148  pac_r, pac_c, params->startPreHeight, params->startPreWidth,
149  params->stepPreHeight, params->stepPreWidth, &pap_r, &pap_c);
150 
151  int16_t r_diff = (int16_t) pap_r - (int16_t) pre_r;
152  int16_t c_diff = (int16_t) pap_c - (int16_t) pre_c;
153 
154  k_r = hlf_kh - r_diff;
155  k_c = hlf_kw - c_diff;
156 
157  if ((0 <= k_r) && (k_r < params->kernelHeight) && (0 <= k_c)
158  && (k_c < params->kernelWidth)) {
159  values[i] = obj->values[k_r * params->kernelWidth + k_c];
160  // LOG_PRINT(LOG_LEVEL_INFO, "val = %5.6k", output[i]);
161  } else {
162  log_error("Kernel coordinates off range (%d, %d)", k_r, k_c);
163  }
164  }
165 }
