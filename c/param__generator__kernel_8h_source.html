117 static void param_generator_kernel_generate(
118  void *generator, uint32_t n_synapses,
119  uint32_t pre_neuron_index, uint16_t *indices, accum *values) {
120  use(pre_neuron_index);
121  use(indices);
122  struct all_kernel_params *obj = generator;
123  struct param_generator_kernel *params = &obj->params;
124  uint16_t pre_c = 0;
125  uint16_t pre_r = uidiv(pre_neuron_index, params->preWidth, &pre_c);
126 
127  uint16_t hlf_kw = params->kernelWidth >> 1;
128  uint16_t hlf_kh = params->kernelHeight >> 1;
129  int16_t k_r, k_c;
130  for (uint16_t i = 0; i < n_synapses; i++) {
131  uint16_t post_r, post_c; //post raw
132  uint16_t pac_r, pac_c; // post as common
133  int16_t pap_r, pap_c; // post as pre
134  post_r = uidiv(params->post_slice_start + indices[i],
135  params->postWidth, &post_c);
136 
137  //move post coords into common coordinate system
138  post_in_pre_world(post_r, post_c, params->startPostHeight,
139  params->startPostWidth, params->stepPostHeight,
140  params->stepPostWidth, &pac_r, &pac_c);
141 
142  //move common to pre coords
143  pre_in_post_world(
144  pac_r, pac_c, params->startPreHeight, params->startPreHeight,
145  params->stepPreHeight, params->stepPreWidth, &pap_r, &pap_c);
146 
147  int16_t r_diff = (int16_t) pap_r - (int16_t) pre_r;
148  int16_t c_diff = (int16_t) pap_c - (int16_t) pre_c;
149 
150  k_r = hlf_kh - r_diff;
151  k_c = hlf_kw - c_diff;
152 
153  if ((0 <= k_r) && (k_r < params->kernelHeight) && (0 <= k_c)
154  && (k_c < params->kernelWidth)) {
155  values[i] = obj->values[k_r * params->kernelWidth + k_c];
156  // LOG_PRINT(LOG_LEVEL_INFO, "val = %5.6k", output[i]);
157  } else {
158  log_error("Kernel coordinates off range (%d, %d)", k_r, k_c);
159  }
160  }
161 }
