84 static bool neuron_impl_initialise(uint32_t n_neurons) {
85 
86  // Allocate DTCM for neuron array
87  if (sizeof(neuron_t)) {
88  neuron_array = spin1_malloc(n_neurons * sizeof(neuron_t));
89  if (neuron_array == NULL) {
90  log_error("Unable to allocate neuron array - Out of DTCM");
91  return false;
92  }
93  }
94 
95  // Allocate DTCM for input type array and copy block of data
96  if (sizeof(input_type_t)) {
97  input_type_array = spin1_malloc(n_neurons * sizeof(input_type_t));
98  if (input_type_array == NULL) {
99  log_error("Unable to allocate input type array - Out of DTCM");
100  return false;
101  }
102  }
103 
104  // Allocate DTCM for additional input array and copy block of data
105  if (sizeof(additional_input_t)) {
106  additional_input_array =
107  spin1_malloc(n_neurons * sizeof(additional_input_t));
108  if (additional_input_array == NULL) {
109  log_error("Unable to allocate additional input array"
110  " - Out of DTCM");
111  return false;
112  }
113  }
114 
115  // Allocate DTCM for threshold type array and copy block of data
116  if (sizeof(threshold_type_t)) {
117  threshold_type_array =
118  spin1_malloc(n_neurons * sizeof(threshold_type_t));
119  if (threshold_type_array == NULL) {
120  log_error("Unable to allocate threshold type array - Out of DTCM");
121  return false;
122  }
123  }
124 
125  // Allocate DTCM for synapse shaping parameters
126  if (sizeof(synapse_types_t)) {
127  synapse_types_array =
128  spin1_malloc(n_neurons * sizeof(synapse_types_t));
129  if (synapse_types_array == NULL) {
130  log_error("Unable to allocate synapse types array - Out of DTCM");
131  return false;
132  }
133  }
134 
135  return true;
136 }

165 static void neuron_impl_load_neuron_parameters(
166  address_t address, uint32_t next, uint32_t n_neurons,
167  address_t save_initial_state) {
168 
169  // Read the number of steps per timestep
170  n_steps_per_timestep = address[next++];
171  if (n_steps_per_timestep == 0) {
172  log_error("bad number of steps per timestep: 0");
173  rt_error(RTE_SWERR);
174  }
175 
176  if (sizeof(neuron_t)) {
177  neuron_params_t *params = (neuron_params_t *) &address[next];
178  for (uint32_t i = 0; i < n_neurons; i++) {
179  neuron_model_initialise(&neuron_array[i], &params[i],
180  n_steps_per_timestep);
181  }
182  next += n_words_needed(n_neurons * sizeof(neuron_params_t));
183  }
184 
185  if (sizeof(input_type_t)) {
186  input_type_params_t *params = (input_type_params_t *) &address[next];
187  for (uint32_t i = 0; i < n_neurons; i++) {
188  input_type_initialise(&input_type_array[i], &params[i],
189  n_steps_per_timestep);
190  }
191  next += n_words_needed(n_neurons * sizeof(input_type_params_t));
192  }
193 
194  if (sizeof(threshold_type_t)) {
195  threshold_type_params_t *params = (threshold_type_params_t *) &address[next];
196  for (uint32_t i = 0; i < n_neurons; i++) {
197  threshold_type_initialise(&threshold_type_array[i], &params[i],
198  n_steps_per_timestep);
199  }
200  next += n_words_needed(n_neurons * sizeof(threshold_type_params_t));
201  }
202 
203  if (sizeof(synapse_types_t)) {
204  synapse_types_params_t *params = (synapse_types_params_t *) &address[next];
205  for (uint32_t i = 0; i < n_neurons; i++) {
206  synapse_types_initialise(&synapse_types_array[i], &params[i],
207  n_steps_per_timestep);
208  }
209  next += n_words_needed(n_neurons * sizeof(synapse_types_params_t));
210  }
211 
212  if (sizeof(additional_input_t)) {
213  additional_input_params_t *params = (additional_input_params_t *) &address[next];
214  for (uint32_t i = 0; i < n_neurons; i++) {
215  additional_input_initialise(&additional_input_array[i], &params[i],
216  n_steps_per_timestep);
217  }
218  next += n_words_needed(n_neurons * sizeof(additional_input_params_t));
219  }
220 
221  // If we are to save the initial state, copy the whole of the parameters
222  // to the initial state
223  if (save_initial_state) {
224  spin1_memcpy(save_initial_state, address, next * sizeof(uint32_t));
225  }
226 
227 #if LOG_LEVEL >= LOG_DEBUG
228  for (index_t n = 0; n < n_neurons; n++) {
229  neuron_model_print_parameters(&neuron_array[n]);
230  neuron_model_print_state_variables(&neuron_array[n]);
231  }
232 #endif // LOG_LEVEL >= LOG_DEBUG
233 }

240 static void neuron_impl_do_timestep_update(
241  uint32_t timer_count, uint32_t time, uint32_t n_neurons) {
242 
243  for (uint32_t neuron_index = 0; neuron_index < n_neurons; neuron_index++) {
244 
245  // Get the neuron itself
246  neuron_t *this_neuron = &neuron_array[neuron_index];
247 
248  // Get the input_type parameters and voltage for this neuron
249  input_type_t *input_types = &input_type_array[neuron_index];
250 
251  // Get threshold and additional input parameters for this neuron
252  threshold_type_t *the_threshold_type = &threshold_type_array[neuron_index];
253  additional_input_t *additional_inputs = &additional_input_array[neuron_index];
254  synapse_types_t *the_synapse_type = &synapse_types_array[neuron_index];
255 
256  // Loop however many times requested; do this in reverse for efficiency,
257  // and because the index doesn't actually matter
258  for (uint32_t i_step = n_steps_per_timestep; i_step > 0; i_step--) {
259  // Get the voltage
260  state_t soma_voltage = neuron_model_get_membrane_voltage(this_neuron);
261 
262  // Get the exc and inh values from the synapses
263  input_t exc_values[NUM_EXCITATORY_RECEPTORS];
264  input_t *exc_syn_values =
265  synapse_types_get_excitatory_input(exc_values, the_synapse_type);
266  input_t inh_values[NUM_INHIBITORY_RECEPTORS];
267  input_t *inh_syn_values =
268  synapse_types_get_inhibitory_input(inh_values, the_synapse_type);
269 
270  // Call functions to obtain exc_input and inh_input
271  input_t *exc_input_values = input_type_get_input_value(
272  exc_syn_values, input_types, NUM_EXCITATORY_RECEPTORS);
273  input_t *inh_input_values = input_type_get_input_value(
274  inh_syn_values, input_types, NUM_INHIBITORY_RECEPTORS);
275 
276  // Sum g_syn contributions from all receptors for recording
277  REAL total_exc = ZERO;
278  REAL total_inh = ZERO;
279 
280  for (int i = 0; i < NUM_EXCITATORY_RECEPTORS; i++) {
281  total_exc += exc_input_values[i];
282  }
283  for (int i = 0; i < NUM_INHIBITORY_RECEPTORS; i++) {
284  total_inh += inh_input_values[i];
285  }
286 
287  // Do recording if on the first step
288  if (i_step == n_steps_per_timestep) {
289  neuron_recording_record_accum(
290  V_RECORDING_INDEX, neuron_index, soma_voltage);
291  neuron_recording_record_accum(
292  GSYN_EXC_RECORDING_INDEX, neuron_index, total_exc);
293  neuron_recording_record_accum(
294  GSYN_INH_RECORDING_INDEX, neuron_index, total_inh);
295  }
296 
297  // Call functions to convert exc_input and inh_input to current
298  input_type_convert_excitatory_input_to_current(
299  exc_input_values, input_types, soma_voltage);
300  input_type_convert_inhibitory_input_to_current(
301  inh_input_values, input_types, soma_voltage);
302 
303  // Get any input from an injected current source
304  REAL current_offset = current_source_get_offset(time, neuron_index);
305 
306  // Get any external bias input
307  input_t external_bias = additional_input_get_input_value_as_current(
308  additional_inputs, soma_voltage);
309 
310  // update neuron parameters
311  state_t result = neuron_model_state_update(
312  NUM_EXCITATORY_RECEPTORS, exc_input_values,
313  NUM_INHIBITORY_RECEPTORS, inh_input_values,
314  external_bias, current_offset, this_neuron);
315 
316  // determine if a spike should occur
317  bool spike_now =
318  threshold_type_is_above_threshold(result, the_threshold_type);
319 
320  // If spike occurs, communicate to relevant parts of model
321  if (spike_now) {
322 
323  // Call relevant model-based functions
324  // Tell the neuron model
325  neuron_model_has_spiked(this_neuron);
326 
327  // Tell the additional input
328  additional_input_has_spiked(additional_inputs);
329 
330  // Record the spike
331  neuron_recording_record_bit(SPIKE_RECORDING_BITFIELD, neuron_index);
332 
333  // Send the spike
334  send_spike(timer_count, time, neuron_index);
335  }
336 
337  // Shape the existing input according to the included rule
338  synapse_types_shape_input(the_synapse_type);
339  }
340 
341  #if LOG_LEVEL >= LOG_DEBUG
342  neuron_model_print_state_variables(this_neuron);
343  #endif // LOG_LEVEL >= LOG_DEBUG
344  }
345 }

403 void neuron_impl_print_inputs(uint32_t n_neurons) {
404  bool empty = true;
405  for (index_t i = 0; i < n_neurons; i++) {
406  synapse_types_t *params = &synapse_types_array[i];
407  input_t exc_values[NUM_EXCITATORY_RECEPTORS];
408  input_t inh_values[NUM_INHIBITORY_RECEPTORS];
409  empty = empty && (0 == bitsk(
410  synapse_types_get_excitatory_input(exc_values, params)[0]
411  - synapse_types_get_inhibitory_input(inh_values, params)[0]));
412  }
413 
414  if (!empty) {
415  for (index_t i = 0; i < n_neurons; i++) {
416  synapse_types_t *params = &synapse_types_array[i];
417  input_t exc_values[NUM_EXCITATORY_RECEPTORS];
418  input_t inh_values[NUM_INHIBITORY_RECEPTORS];
419  input_t *exc_input = synapse_types_get_excitatory_input(exc_values, params);
420  input_t *inh_input = synapse_types_get_inhibitory_input(inh_values, params);
421  input_t input = exc_input[0] - inh_input[0];
422  if (bitsk(input) != 0) {
423  log_debug("Neuron %3u: input %12.6k (= ", i, input);
424  synapse_types_print_input(params);
425  log_debug(")");
426  }
427  }
428  }
