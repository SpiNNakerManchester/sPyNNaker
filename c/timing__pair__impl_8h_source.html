59 static inline post_trace_t timing_add_post_spike(
60  uint32_t time, uint32_t last_time, post_trace_t last_trace) {
61  // Get time since last spike
62  uint32_t delta_time = time - last_time;
63 
64  // Decay previous o1 and o2 traces
65  int32_t decayed_o1_trace = STDP_FIXED_MUL_16X16(last_trace,
66  maths_lut_exponential_decay(delta_time, tau_minus_lookup));
67 
68  // Add energy caused by new spike to trace
69  // **NOTE** o2 trace is pre-multiplied by a3_plus
70  int32_t new_o1_trace = decayed_o1_trace + STDP_FIXED_POINT_ONE;
71 
72  log_debug("\tdelta_time=%d, o1=%d\n", delta_time, new_o1_trace);
73 
74  // Return new pre- synaptic event with decayed trace values with energy
75  // for new spike added
76  return (post_trace_t) new_o1_trace;
77 }
