58 static inline int16_lut *maths_copy_int16_lut(address_t *address) {
59  int16_lut *sdram_lut = (int16_lut *) *address;
60  uint32_t size = sizeof(int16_lut) + (sdram_lut->size * sizeof(int16_t));
61  int16_lut *lut = spin1_malloc(size);
62  if (lut == NULL) {
63  log_error("Not enough space to allocate LUT. Try reducing the timestep,"
64  " the number of neurons per core, or the tau value; size = %u", size);
65  rt_error(RTE_SWERR);
66  }
67  spin1_memcpy(lut, sdram_lut, size);
68 
69  // Pad to number of words (+ 1 for size / shift header)
70  const uint32_t num_words = (lut->size / 2) + (((lut->size & 1) != 0) ? 1 : 0);
71  *address += num_words + 1;
72 
73  return lut;
74 }
