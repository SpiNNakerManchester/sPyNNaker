68 static inline bool synaptogenesis_formation_rule(
69  current_state_t *current_state, const formation_params_t *params,
70  uint32_t time, address_t row) {
71  use(time);
72 
73  // Compute distances
74  // To do this I need to take the DIV and MOD of the
75  // post-synaptic neuron ID, of the pre-synaptic neuron ID
76  // Compute the distance of these 2 measures
77  int32_t pre_x, pre_y, post_x, post_y, pre_global_id, post_global_id;
78  // Pre computation requires querying the table with global information
79  pre_global_id = current_state->key_atom_info->lo_atom +
80  current_state->pre_syn_id;
81  post_global_id = current_state->post_syn_id + current_state->post_low_atom;
82 
83  if (params->grid_x > 1) {
84  pre_x = pre_global_id / params->grid_x;
85  post_x = post_global_id / params->grid_x;
86  } else {
87  pre_x = 0;
88  post_x = 0;
89  }
90 
91  if (params->grid_y > 1) {
92  pre_y = pre_global_id % params->grid_y;
93  post_y = post_global_id % params->grid_y;
94  } else {
95  pre_y = 0;
96  post_y = 0;
97  }
98 
99  // With periodic boundary conditions
100  uint32_t delta_x, delta_y;
101  delta_x = my_abs(pre_x - post_x);
102  delta_y = my_abs(pre_y - post_y);
103 
104  if (delta_x > params->grid_x >> 1 && params->grid_x > 1) {
105  delta_x -= params->grid_x;
106  }
107 
108  if (delta_y > params->grid_y >> 1 && params->grid_y > 1) {
109  delta_y -= params->grid_y;
110  }
111 
112  uint32_t distance = delta_x * delta_x + delta_y * delta_y;
113 
114  // Distance based probability extracted from the appropriate LUT
115  uint16_t probability;
116  int16_t controls = current_state->pre_population_info->sp_control;
117  if (!(controls & IS_CONNECTION_LAT)) {
118  if (distance >= params->ff_prob_size) {
119  return false;
120  }
121  probability = params->prob_tables[distance];
122  } else {
123  if (distance >= params->lat_prob_size) {
124  return false;
125  }
126  probability = params->prob_tables[params->ff_prob_size + distance];
127  }
128  uint16_t r = ulrbits(mars_kiss64_seed(*(current_state->local_seed)))
129  * MAX_SHORT;
130  if (r > probability) {
131  return false;
132  }
133 
134  return sp_structs_add_synapse(current_state, row);
135 }
