72 static inline bool synaptogenesis_formation_rule(
73  current_state_t *current_state, const formation_params_t *params,
74  UNUSED uint32_t time, synaptic_row_t row) {
75  // Compute distances
76  // To do this I need to take the DIV and MOD of the
77  // post-synaptic neuron ID, of the pre-synaptic neuron ID
78  // Compute the distance of these 2 measures
79  uint32_t pre_x, pre_y, post_x, post_y;
80  // Pre computation requires querying the table with global information
81  uint32_t pre_global_id = current_state->key_atom_info->lo_atom +
82  current_state->pre_syn_id;
83  uint32_t post_global_id = current_state->post_syn_id +
84  current_state->post_low_atom;
85 
86  if (params->grid_x > 1) {
87  pre_x = muliulr(pre_global_id, params->grid_x_recip);
88  post_x = muliulr(post_global_id, params->grid_x_recip);
89  } else {
90  pre_x = 0;
91  post_x = 0;
92  }
93 
94  if (params->grid_y > 1) {
95  uint32_t pre_y_div = muliulr(pre_global_id, params->grid_y_recip);
96  uint32_t post_y_div = muliulr(post_global_id, params->grid_y_recip);
97  pre_y = pre_global_id - (pre_y_div * params->grid_y);
98  post_y = post_global_id - (post_y_div * params->grid_y);
99  } else {
100  pre_y = 0;
101  post_y = 0;
102  }
103 
104  // With periodic boundary conditions
105  uint32_t delta_x, delta_y;
106  delta_x = my_abs(pre_x - post_x);
107  delta_y = my_abs(pre_y - post_y);
108 
109  if (delta_x > params->grid_x >> 1 && params->grid_x > 1) {
110  delta_x -= params->grid_x;
111  }
112 
113  if (delta_y > params->grid_y >> 1 && params->grid_y > 1) {
114  delta_y -= params->grid_y;
115  }
116 
117  uint32_t distance = delta_x * delta_x + delta_y * delta_y;
118 
119  // Distance based probability extracted from the appropriate LUT
120  uint16_t probability;
121  int16_t controls = current_state->pre_population_info->sp_control;
122  if (!(controls & IS_CONNECTION_LAT)) {
123  if (distance >= params->ff_prob_size) {
124  return false;
125  }
126  probability = params->prob_tables[distance];
127  } else {
128  if (distance >= params->lat_prob_size) {
129  return false;
130  }
131  probability = params->prob_tables[params->ff_prob_size + distance];
132  }
133  uint32_t r = rand_int(MAX_SHORT, *(current_state->local_seed));
134  if (r > probability) {
135  return false;
136  }
137 
138  return sp_structs_add_synapse(current_state, row);
139 }
