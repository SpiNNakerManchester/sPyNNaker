41 static inline bool synaptogenesis_formation_rule(
42  current_state_t *current_state, const formation_params_t *params,
43  uint32_t time, address_t row) {
44  use(time);
45 
46  // Compute distances
47  // To do this I need to take the DIV and MOD of the
48  // post-synaptic neuron ID, of the pre-synaptic neuron ID
49  // Compute the distance of these 2 measures
50  int32_t pre_x, pre_y, post_x, post_y, pre_global_id, post_global_id;
51  // Pre computation requires querying the table with global information
52  pre_global_id = current_state->key_atom_info->lo_atom +
53  current_state->pre_syn_id;
54  post_global_id = current_state->post_syn_id + current_state->post_low_atom;
55 
56  if (params->grid_x > 1) {
57  pre_x = pre_global_id / params->grid_x;
58  post_x = post_global_id / params->grid_x;
59  } else {
60  pre_x = 0;
61  post_x = 0;
62  }
63 
64  if (params->grid_y > 1) {
65  pre_y = pre_global_id % params->grid_y;
66  post_y = post_global_id % params->grid_y;
67  } else {
68  pre_y = 0;
69  post_y = 0;
70  }
71 
72  // With periodic boundary conditions
73  uint delta_x, delta_y;
74  delta_x = my_abs(pre_x - post_x);
75  delta_y = my_abs(pre_y - post_y);
76 
77  if (delta_x > params->grid_x >> 1 && params->grid_x > 1) {
78  delta_x -= params->grid_x;
79  }
80 
81  if (delta_y > params->grid_y >> 1 && params->grid_y > 1) {
82  delta_y -= params->grid_y;
83  }
84 
85  uint32_t distance = delta_x * delta_x + delta_y * delta_y;
86 
87  // Distance based probability extracted from the appropriate LUT
88  uint16_t probability;
89  int16_t controls = current_state->pre_population_info->sp_control;
90  if (!(controls & IS_CONNECTION_LAT)) {
91  if (distance >= params->ff_prob_size) {
92  return false;
93  }
94  probability = params->prob_tables[distance];
95  } else {
96  if (distance >= params->lat_prob_size) {
97  return false;
98  }
99  probability = params->prob_tables[params->ff_prob_size + distance];
100  }
101  uint16_t r = ulrbits(mars_kiss64_seed(*(current_state->local_seed)))
102  * MAX_SHORT;
103  if (r > probability) {
104  return false;
105  }
106 
107  return sp_structs_add_synapse(current_state, row);
108 }
