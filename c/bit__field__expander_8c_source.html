142 void read_in_addresses(void) {
143  // get the data (linked to sdram tag 2 and assume the app ids match)
144  data_specification_metadata_t *dsg_metadata =
145  data_specification_get_data_address();
146  const builder_region_struct *builder_data =
147  (builder_region_struct *) vcpu()->user1;
148 
149  master_pop_base_address = data_specification_get_region(
150  builder_data->master_pop_region_id, dsg_metadata);
151  synaptic_matrix_base_address = data_specification_get_region(
152  builder_data->synaptic_matrix_region_id, dsg_metadata);
153  bit_field_base_address = data_specification_get_region(
154  builder_data->bit_field_region_id, dsg_metadata);
155 
156  // fill in size zero in case population table never read in
157  bit_field_base_address->n_filters = 0;
158  direct_matrix_region_base_address = data_specification_get_region(
159  builder_data->direct_matrix_region_id, dsg_metadata);
160 
161  log_info("structural matrix region id = %d",
162  builder_data->structural_matrix_region_id);
163  if (builder_data->structural_matrix_region_id != FAILED_REGION_ID) {
164  structural_matrix_region_base_address = data_specification_get_region(
165  builder_data->structural_matrix_region_id, dsg_metadata);
166  }
167  key_atom_data_t *keys_to_max_atoms_sdram = data_specification_get_region(
168  builder_data->bit_field_key_map_region_id, dsg_metadata);
169  uint32_t pair_size = sizeof(key_atom_data_t) +
170  (keys_to_max_atoms_sdram->n_pairs * sizeof(key_atom_pair_t));
171  keys_to_max_atoms = spin1_malloc(pair_size);
172  if (keys_to_max_atoms == NULL) {
173  log_error("Couldn't allocate memory for key_to_max_atoms");
174  rt_error(RTE_SWERR);
175  }
176  spin1_memcpy(keys_to_max_atoms, keys_to_max_atoms_sdram, pair_size);
177 
178  // printer
179  log_debug("master_pop_table_base_address = %0x", master_pop_base_address);
180  log_debug("synaptic_matrix_base_address = %0x",
181  synaptic_matrix_base_address);
182  log_debug("bit_field_base_address = %0x", bit_field_base_address);
183  log_debug("direct_matrix_region_base_address = %0x",
184  direct_matrix_region_base_address);
185  log_debug("Structural matrix region base address = %0x",
186  structural_matrix_region_base_address);
187  log_info("Finished reading in vertex data region addresses");
188 }

213  log_info("Direct synapse init");
214  if (!direct_synapses_initialise(
215  direct_matrix_region_base_address, &direct_synapses_address)) {
216  log_error("Failed to init the synapses. failing");
217  return false;
218  }
219 
220  // init the master pop table
221  log_info("Pop table init");
222  if (!population_table_initialise(
223  master_pop_base_address, synaptic_matrix_base_address,
224  direct_synapses_address, &row_max_n_words)) {
225  log_error("Failed to init the master pop table. failing");
226  return false;
227  }
228 
229  log_info("Structural plastic if needed");
230  if (structural_matrix_region_base_address != NULL) {
231  if (! sp_structs_read_in_common(
232  structural_matrix_region_base_address, &rewiring_data,
233  &pre_info, &post_to_pre_table)) {
234  log_error("Failed to init the synaptogenesis");
235  return false;
236  }
237  }
238 
239  if (keys_to_max_atoms->n_pairs == 0) {
240  success_shut_down();
241  log_info("There were no bitfields to process.");
242  can_run = false;
243  return true;
244  }
245 
246  // read in the correct key to max atom map
247 #ifdef __PRINT_KEY_ATOM_MAP__
248  print_key_to_max_atom_map();
249 #endif
250 
251  // set up a sdram read for a row
252  log_debug("Allocating dtcm for row data");
253  row_data = spin1_malloc(row_max_n_words * sizeof(uint32_t));
254  if (row_data == NULL) {
255  log_error("Could not allocate dtcm for the row data");
256  return false;
257  }
258  log_debug("Finished pop table set connectivity lookup");
259 
260  return true;

267  // get address of plastic region from row
268  if (synapse_row_plastic_size(row) > 0) {
269  log_debug("Plastic row had entries, so cant be pruned");
270  return true;
271  }
272 
273  // Get address of non-plastic region from row
274  synapse_row_fixed_part_t *fixed_region = synapse_row_fixed_region(row);
275  uint32_t fixed_synapse = synapse_row_num_fixed_synapses(fixed_region);
276  if (fixed_synapse == 0) {
277  log_debug("Plastic and fixed do not have entries, so can be pruned");
278  return false;
279  } else {
280  log_debug("Fixed row has entries, so cant be pruned");
281  return true;
282  }

298 static inline void sort_by_key(void) {
299  filter_info_t *filters = bit_field_base_address->filters;
300  for (uint32_t i = 1; i < bit_field_base_address->n_filters; i++) {
301  const filter_info_t temp = filters[i];
302 
303  uint32_t j;
304  for (j = i; j > 0 && filters[j - 1].key > temp.key; j--) {
305  filters[j] = filters[j - 1];

307  filters[j] = temp;

312 static void determine_redundancy(void) {
313  // Semantic sugar to keep the code a little shorter
314  filter_info_t *filters = bit_field_base_address->filters;
315  for (uint32_t i = 0; i < bit_field_base_address->n_filters; i++) {
316  filters[i].merged = 0;
317  filters[i].all_ones = 0;
318  int i_atoms = filters[i].n_atoms;
319  int i_words = get_bit_field_size(i_atoms);
320  if (i_atoms == count_bit_field(filters[i].data, i_words)) {
321  filters[i].all_ones = 1;
322  }
323  }
324 
325  for (uint32_t i = 0; i < bit_field_base_address->n_filters; i++) {
326  log_info(" Key: 0x%08x, Filter:", filters[i].key);
327  uint32_t n_words = get_bit_field_size(filters[i].n_atoms);
328  for (uint32_t j = 0; j < n_words; j++) {
329  log_info(" 0x%08x", filters[i].data[j]);
330  }
331  }

336 bool generate_bit_field(void) {
337  // write how many entries (thus bitfields) are to be generated into sdram
338  log_debug("Update by pop length");
339  bit_field_base_address->n_filters = keys_to_max_atoms->n_pairs;
340 
341  // location where to dump the bitfields into (right after the filter structs
342  address_t bit_field_words_location = (address_t)
343  &bit_field_base_address->filters[bit_field_base_address->n_filters];
344  log_debug("bit_field_words_location is %x", bit_field_words_location);
345  int position = 0;
346 
347  // iterate through the master pop entries
348  log_debug("Starting master pop entry bit field generation");
349  for (uint32_t i = 0; i < keys_to_max_atoms->n_pairs; i++) {
350 
351  // determine keys masks and n_neurons
352  spike_t key = keys_to_max_atoms->pairs[i].key;
353  uint32_t n_neurons = keys_to_max_atoms->pairs[i].n_atoms;
354 
355  // generate the bitfield for this master pop entry
356  uint32_t n_words = get_bit_field_size(n_neurons);
357 
358  log_debug("Bitfield %d, key = %d, n_neurons = %d",
359  i, (uint32_t) key, n_neurons);
360  bit_field_t bit_field = bit_field_alloc(n_neurons);
361  if (bit_field == NULL) {
362  log_error("Could not allocate dtcm for bit field");
363  return false;
364  }
365 
366  // set the bitfield to 0. so assuming a miss on everything
367  clear_bit_field(bit_field, n_words);
368 
369  // iterate through neurons and ask for rows from master pop table
370  log_debug("Searching neuron ids");
371  for (uint32_t neuron_id=0; neuron_id < n_neurons; neuron_id++) {
372  // update key with neuron id
373  spike_t new_key = (spike_t) (key + neuron_id);
374  log_debug("New key for neurons %d is %0x", neuron_id, new_key);
375 
376  // check if this is governed by the structural stuff. if so,
377  // avoid filtering as it could change over time
378  bool bit_found = false;
379  if (structural_matrix_region_base_address != NULL) {
380  uint32_t dummy1 = 0, dummy2 = 0, dummy3 = 0, dummy4 = 0;
381  bit_found = sp_structs_find_by_spike(&pre_info, new_key,
382  &dummy1, &dummy2, &dummy3, &dummy4);
383  }
384 
385  // holder for the bytes to transfer if we need to read SDRAM.
386  size_t n_bytes_to_transfer;
387 
388  // used to store the row from the master pop / synaptic matrix,
389  // not going to be used in reality.
390  synaptic_row_t row;
391  if (!bit_found) {
392  if (population_table_get_first_address(
393  new_key, &row, &n_bytes_to_transfer)) {
394  log_debug("%d", neuron_id);
395 
396  // This is a direct row to process, so will have 1 target,
397  // so no need to go further
398  if (n_bytes_to_transfer == 0) {
399  log_debug("Direct synapse");
400  bit_found = true;
401  } else {
402  // sdram read (faking dma transfer)
403  log_debug("DMA read synapse");
404  bit_found = do_sdram_read_and_test(
405  row, n_bytes_to_transfer);
406  }
407 
408  while (!bit_found && population_table_get_next_address(
409  &new_key, &row, &n_bytes_to_transfer)){
410  log_debug("%d", neuron_id);
411 
412  // This is a direct row to process, so will have 1
413  // target, so no need to go further
414  if (n_bytes_to_transfer == 0) {
415  log_debug("Direct synapse");
416  bit_found = true;
417  } else {
418  // sdram read (faking dma transfer)
419  log_debug("DMA read synapse");
420  bit_found = do_sdram_read_and_test(
421  row, n_bytes_to_transfer);
422  }
423  }
424  }
425  }
426 
427  // if returned false, then the bitfield should be set to 0.
428  // Which its by default already set to. so do nothing. so no else.
429  log_debug("bit_found %d", bit_found);
430  if (bit_found) {
431  bit_field_set(bit_field, neuron_id);
432  }
433  }
434 
435  log_debug("Writing bitfield to sdram for core use");
436  bit_field_base_address->filters[i].key = key;
437  log_debug("Putting master pop key %d in entry %d", key, i);
438  bit_field_base_address->filters[i].n_atoms = n_neurons;
439  log_debug("Putting n_atom %d in entry %d", n_neurons, i);
440  // write bitfield to sdram.
441  log_debug("Writing to address %0x, %d words to write",
442  &bit_field_words_location[position], n_words);
443  spin1_memcpy(&bit_field_words_location[position], bit_field,
444  n_words * BYTE_TO_WORD_CONVERSION);
445  // update pointer to correct place
446  bit_field_base_address->filters[i].data =
447  (bit_field_t) &bit_field_words_location[position];
448 
449  // update tracker
450  position += n_words;
451 
452  // free dtcm of bitfield.
453  log_debug("Freeing the bitfield dtcm");
454  sark_free(bit_field);
455  }
456  determine_redundancy();
457  sort_by_key();
458  return true;
459 }

462 void c_main(void) {
463  // set to running state
464  sark_cpu_state(CPU_STATE_RUN);
465 
466  log_info("Starting the bit field expander");
467 
468  // read in sdram data
469  read_in_addresses();

471  // generate bit field for each vertex regions
472  if (!initialise()) {
473  log_error("Failed to init the master pop and synaptic matrix");
474  fail_shut_down();
475  }
476 
477  if (can_run) {
478  log_info("Generating bit field");
479  if (!generate_bit_field()) {
480  log_error("Failed to generate bitfield");
