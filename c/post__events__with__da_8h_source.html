124 static inline post_event_window_t post_events_get_window_delayed(
125  const post_event_history_t *events, uint32_t begin_time,
126  uint32_t end_time) {
127  // Start at end event - beyond end of post-event history
128  const uint32_t count = events->count_minus_one + 1;
129  const uint32_t *end_event_time = events->times + count;
130  const uint32_t *event_time = end_event_time;
131  const nm_post_trace_t *event_trace = events->traces + count;
132 
133  post_event_window_t window;
134  do {
135  // If this event is still in the future, set it as the end
136  if (*event_time > end_time) {
137  end_event_time = event_time;
138  }
139 
140  // Cache pointer to this event as potential next event and go back one
141  // event.
142  // **NOTE** next_time can be invalid
143  window.next_time = event_time--;
144  window.next_trace = event_trace--;
145 
146  // Keep looping while event occurred after start of window and we
147  // haven't hit beginning of array...
148  } while (*event_time > begin_time && event_time != events->times);
149 
150  // Deference event to use as previous
151  window.prev_time = *event_time;
152  window.prev_trace = *event_trace;
153  window.prev_time_valid = event_time != events->times;
154 
155  // Calculate number of events
156  window.num_events = (end_event_time - window.next_time);
157 
158  // Find a vector of dopamine trace markers, with the LSB
159  // entry in the vector corresponding to the oldest trace in the window
160  window.dopamine_trace_markers =
161  events->dopamine_trace_markers >> (count - window.num_events);
162 
163  // Return window
164  return window;
165 }

171 static inline post_event_window_t post_events_next(
172  post_event_window_t window) {
173  // Update previous time and increment next time
174  window.prev_time = *window.next_time++;
175  window.prev_trace = *window.next_trace++;
176 
177  // Time will now be valid for sure!
178  window.prev_time_valid = 1;
179 
180  // Decrement remaining events
181  window.num_events--;
182 
183  // Shift the dopamine trace markers to place the next trace marker at LSB
184  window.dopamine_trace_markers >>= 1;
185  return window;
186 }

202 static inline void post_events_add(
203  uint32_t time, post_event_history_t *events, post_trace_t post_trace,
204  int16_t dopamine_trace, bool dopamine) {
205  if (events->count_minus_one < MAX_POST_SYNAPTIC_EVENTS - 1) {
206  // If there's still space, store time at current end
207  // and increment count minus 1
208  const uint32_t new_index = ++events->count_minus_one;
209  events->times[new_index] = time;
210  events->traces[new_index].post_trace = post_trace;
211  events->traces[new_index].dopamine_trace = dopamine_trace;
212  if (dopamine) {
213  events->dopamine_trace_markers |= (1 << new_index);
214  } else {
215  events->dopamine_trace_markers &= ~(1 << new_index);
216  }
217  } else {
218  // Otherwise Shuffle down elements
219  // **NOTE** 1st element is always an entry at time 0
220  for (uint32_t e = 2; e < MAX_POST_SYNAPTIC_EVENTS; e++) {
221  events->times[e - 1] = events->times[e];
222  events->traces[e - 1] = events->traces[e];
223  }
224  events->dopamine_trace_markers >>= 1;
225 
226  // Stick new time at end
227  events->times[MAX_POST_SYNAPTIC_EVENTS - 1] = time;
228  events->traces[MAX_POST_SYNAPTIC_EVENTS - 1].post_trace = post_trace;
229  events->traces[MAX_POST_SYNAPTIC_EVENTS - 1].dopamine_trace = dopamine_trace;
230  if (dopamine) {
231  events->dopamine_trace_markers |=
232  (1 << (MAX_POST_SYNAPTIC_EVENTS - 1));
233  } else {
234  events->dopamine_trace_markers &=
235  ~(1 << (MAX_POST_SYNAPTIC_EVENTS - 1));
236  }
237  }
238 }
