39 static bool do_sdram_read_and_test(
40  synaptic_row_t row_data, synaptic_row_t row,
41  uint32_t n_bytes_to_transfer) {
42  spin1_memcpy(row_data, row, n_bytes_to_transfer);
43  log_debug("Process synaptic row");
44 
45  // get address of plastic region from row
46  if (synapse_row_plastic_size(row) > 0) {
47  log_debug("Plastic row had entries, so cant be pruned");
48  return true;
49  }
50 
51  // Get address of non-plastic region from row
52  synapse_row_fixed_part_t *fixed_region = synapse_row_fixed_region(row);
53  uint32_t fixed_synapse = synapse_row_num_fixed_synapses(fixed_region);
54  if (fixed_synapse == 0) {
55  log_debug("Plastic and fixed do not have entries, so can be pruned");
56  return false;
57  } else {
58  log_debug("Fixed row has entries, so cant be pruned");
59  return true;
60  }

64 static inline void determine_redundancy(filter_region_t *bitfield_filters) {
65  // Semantic sugar to keep the code a little shorter
66  filter_info_t *filters = bitfield_filters->filters;
67  for (uint32_t i = 0; i < bitfield_filters->n_filters; i++) {
68  filters[i].merged = 0;
69  filters[i].all_ones = 0;
70  int i_atoms = filters[i].n_atoms;
71  int i_words = get_bit_field_size(i_atoms);
72  if (i_atoms == count_bit_field(filters[i].data, i_words)) {
73  filters[i].all_ones = 1;
74  }
75  }
76 }

80 static inline bool generate_bit_field(filter_region_t *bitfield_filters,
81  uint32_t *n_atom_data, void *synaptic_matrix, void *structural_matrix,
82  pre_pop_info_table_t *pre_info, synaptic_row_t row_data) {
83 
84  // Get the location just after the structs for the actual bit fields
85  uint32_t *bit_field_words_location = (uint32_t *)
86  &bitfield_filters->filters[master_pop_table_length];
87  int position = 0;
88 
89  // iterate through the master pop entries
90  log_info("Generating %u bitfields", master_pop_table_length);
91  for (uint32_t i = 0; i < master_pop_table_length; i++) {
92 
93  // determine n_neurons and bit field size
94  uint32_t n_neurons = n_atom_data[i];
95  uint32_t n_words = get_bit_field_size(n_neurons);
96 
97  // Make and clear a bit field
98  bit_field_t bit_field = bit_field_alloc(n_neurons);
99  if (bit_field == NULL) {
100  log_error("Could not allocate dtcm for bit field");
101  return false;
102  }
103  clear_bit_field(bit_field, n_words);
104 
105  master_population_table_entry mp_entry = master_pop_table[i];
106 
107  if (structural_matrix != NULL) {
108 
109  // If this is a structural entry, set all the bits
110  uint32_t dummy1 = 0, dummy2 = 0, dummy3 = 0, dummy4 = 0;
111  if (sp_structs_find_by_spike(pre_info, mp_entry.key,
112  &dummy1, &dummy2, &dummy3, &dummy4)) {
113  for (uint32_t n = 0; n < n_neurons; n++) {
114  bit_field_set(bit_field, n);
115  }
116  }
117  } else {
118 
119  // Go through the addresses of the master pop entry
120  uint32_t pos = mp_entry.start;
121  for (uint32_t j = mp_entry.count; j > 0; j--, pos++) {
122 
123  // Find the base address and row length of the address entry
124  address_list_entry entry = address_list[pos];
125 
126  // Skip invalid addresses
127  if (entry.address == INVALID_ADDRESS) {
128  continue;
129  }
130 
131  // Go through each neuron and check the row
132  for (uint32_t n = 0; n < n_neurons; n++) {
133 
134  // If this neuron is already set, skip it this round
135  if (bit_field_test(bit_field, n)) {
136  continue;
137  }
138 
139  synaptic_row_t row;
140  uint32_t n_bytes_to_transfer;
141  get_row_addr_and_size(entry, (uint32_t) synaptic_matrix,
142  n, &row, &n_bytes_to_transfer);
143 
144  // Check if the row is non-empty and if so set a bit
145  if (do_sdram_read_and_test(row_data, row, n_bytes_to_transfer)) {
146  bit_field_set(bit_field, n);
147  }
148  }
149  }
150  }
151 
152  // Copy details into SDRAM
153  bitfield_filters->filters[i].key = mp_entry.key;
154  bitfield_filters->filters[i].n_atoms = n_neurons;
155  bitfield_filters->filters[i].n_atoms_per_core = mp_entry.n_neurons;
156  bitfield_filters->filters[i].core_shift = mp_entry.mask_shift;
157  spin1_memcpy(&bit_field_words_location[position], bit_field,
158  n_words * sizeof(uint32_t));
159  bitfield_filters->filters[i].data = &bit_field_words_location[position];
160 
161  // Move to the next location in SDRAM for bit fields
162  position += n_words;
163 
164  // free dtcm of bitfield.
165  log_debug("Freeing the bitfield dtcm");
166  sark_free(bit_field);
167  }
168 
169  // write how many entries (thus bitfields) have been generated into sdram
170  bitfield_filters->n_filters = master_pop_table_length;
171  return true;
172 }

175 static bool do_bitfield_generation(
176  uint32_t *n_atom_data_sdram, void *master_pop,
177  void *synaptic_matrix, void *bitfield_filters, void *structural_matrix) {
178 
179  pop_table_config_t *config = (pop_table_config_t *) master_pop;
180  master_pop_table_length = config->table_length;
181 
182  if (master_pop_table_length == 0) {
183  return true;
184  }
185 
186  master_pop_table = &config->data[0];
187  address_list = (address_list_entry *) &config->data[master_pop_table_length];
188 
189  uint32_t n_atom_bytes = master_pop_table_length * sizeof(uint32_t);
190  uint32_t *n_atom_data = spin1_malloc(n_atom_bytes);
191  if (n_atom_data == NULL) {
192  log_error("Couldn't allocate memory for key_to_max_atoms");
193  rt_error(RTE_SWERR);
194  }
195  spin1_memcpy(n_atom_data, n_atom_data_sdram, n_atom_bytes);
196 
197 
198  uint32_t row_max_n_words = 0xFF + N_SYNAPSE_ROW_HEADER_WORDS;
199  synaptic_row_t row_data = spin1_malloc(row_max_n_words * sizeof(uint32_t));
200  if (row_data == NULL) {
201  log_error("Could not allocate dtcm for the row data");
202  return false;
203  }
204 
205  rewiring_data_t rewiring_data;
206  post_to_pre_entry *post_to_pre_table;
207  pre_pop_info_table_t pre_info = {0, NULL};
208  if (structural_matrix != NULL) {
209  sp_structs_read_in_common(
210  structural_matrix, &rewiring_data, &pre_info, &post_to_pre_table);
211  }
212 
213  if (!generate_bit_field(bitfield_filters, n_atom_data, synaptic_matrix,
214  structural_matrix, &pre_info, row_data)) {
215  log_error("Failed to generate bit fields");
216  return false;
217  }
218  determine_redundancy(bitfield_filters);
219  return true;
220 }
