72 // The data structure layout supported by this API is designed for
73 // mixed plastic and fixed synapse rows.
74 //
75 // The data structure is treated as an array of 32-bit words.
76 // Special meanings are ascribed to the 0th and 1st elements
77 // of the array.
78 //
79 // We are expecting the original source address in SDRAM to be
80 // in location 0. The number of array elements in the plastic
81 // region is held in the upper part of row[1]. A tag to indicate
82 // the nature of the synaptic row structure is held in the lower
83 // part of row[1].
84 //
85 // 0: [ N = <plastic elements> | <tag> ]
86 // 1: [ First word of plastic region ]
87 // ...
88 // N: [ Last word of plastic region ]
89 // N+1: [ First word of fixed region ]
90 // ...
91 // M: [ Last word of fixed region ]
92 
93 static inline size_t synapse_row_plastic_size(address_t row) {
94  return (size_t) row[0];
95 }

102 // Returns the address of the non-plastic (or fixed) region
103 static inline address_t synapse_row_fixed_region(address_t row) {

105 }

107 // Within the fixed-region extracted using the above API, fixed[0]
108 // Contains the number of 32-bit fixed synaptic words, fixed[1]
109 // Contains the number of 16-bit plastic synapse control words
110 // (The weights for the plastic synapses are assumed to be stored
111 // In some learning-rule-specific format in the plastic region)
112 // 0: [ F = Num fixed synapses ]
113 // 1: [ P = Size of plastic region in WORDS ]
114 // 2: [ First fixed synaptic word ]
115 // ...
116 // F+1: [ Last fixed synaptic word ]
117 // F+2: [ 1st plastic synapse control word|2nd plastic control word ]
118 // ...
119 // F+1+ceil(P/2): [ Last word of fixed region ]
120 static inline size_t synapse_row_num_fixed_synapses(address_t fixed) {
121  return (size_t) fixed[0];
122 }
