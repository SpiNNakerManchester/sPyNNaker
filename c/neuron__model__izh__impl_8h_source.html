93 static inline void rk2_kernel_midpoint(
94  REAL h, neuron_t *neuron, REAL input_this_timestep) {
95  // to match Mathematica names
96  REAL lastV1 = neuron->V;
97  REAL lastU1 = neuron->U;
98  REAL a = neuron->A;
99  REAL b = neuron->B;
100 
101  REAL pre_alph = REAL_CONST(140.0) + input_this_timestep - lastU1;
102  REAL alpha = pre_alph
103  + (REAL_CONST(5.0) + MAGIC_MULTIPLIER * lastV1) * lastV1;
104  REAL eta = lastV1 + REAL_HALF(h * alpha);
105 
106  // could be represented as a long fract?
107  REAL beta = REAL_HALF(h * (b * lastV1 - lastU1) * a);
108 
109  neuron->V += h * (pre_alph - beta
110  + (REAL_CONST(5.0) + MAGIC_MULTIPLIER * eta) * eta);
111 
112  neuron->U += a * h * (-lastU1 - beta + b * eta);
113 }

130 static state_t neuron_model_state_update(
131  uint16_t num_excitatory_inputs, const input_t *exc_input,
132  uint16_t num_inhibitory_inputs, const input_t *inh_input,
133  input_t external_bias, neuron_t *restrict neuron) {
134  REAL total_exc = 0;
135  REAL total_inh = 0;
136 
137  for (int i =0; i<num_excitatory_inputs; i++) {
138  total_exc += exc_input[i];

140  for (int i =0; i<num_inhibitory_inputs; i++) {
141  total_inh += inh_input[i];
142  }
143 
144  input_t input_this_timestep = total_exc - total_inh

147  // the best AR update so far
148  rk2_kernel_midpoint(neuron->this_h, neuron, input_this_timestep);
149  neuron->this_h = global_params->machine_timestep_ms;
