95 static inline bool initialise_synapse_regions(
96  data_specification_metadata_t *ds_regions,
97  struct synapse_regions regions, weight_t **ring_buffers,
98  uint32_t *row_max_n_words,
99  uint32_t *incoming_spike_buffer_size,
100  bool *clear_input_buffer_of_late_packets,
101  uint32_t *n_recording_regions_used) {
102  // Set up the synapses
103  uint32_t *ring_buffer_to_input_buffer_left_shifts;
104  uint32_t n_neurons;
105  uint32_t n_synapse_types;
106  if (!synapses_initialise(
107  data_specification_get_region(regions.synapse_params, ds_regions),
108  &n_neurons, &n_synapse_types, ring_buffers,
109  &ring_buffer_to_input_buffer_left_shifts,
110  clear_input_buffer_of_late_packets,
111  incoming_spike_buffer_size)) {
112  return false;
113  }
114 
115  // set up direct synapses
116  address_t direct_synapses_address;
117  if (!direct_synapses_initialise(
118  data_specification_get_region(regions.direct_matrix, ds_regions),
119  &direct_synapses_address)) {
120  return false;
121  }
122 
123  // Set up the population table
124  if (!population_table_initialise(
125  data_specification_get_region(regions.pop_table, ds_regions),
126  data_specification_get_region(regions.synaptic_matrix, ds_regions),
127  direct_synapses_address, row_max_n_words)) {
128  return false;
129  }
130  // Set up the synapse dynamics
131  if (!synapse_dynamics_initialise(
132  data_specification_get_region(regions.synapse_dynamics, ds_regions),
133  n_neurons, n_synapse_types, ring_buffer_to_input_buffer_left_shifts)) {
134  return false;
135  }
136 
137  // Set up structural plasticity dynamics
138  if (!synaptogenesis_dynamics_initialise(data_specification_get_region(
139  regions.structural_dynamics, ds_regions), n_recording_regions_used)) {
140  return false;
141  }
142 
143  return true;
144 }
