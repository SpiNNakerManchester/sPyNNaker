85 bool local_only_impl_initialise(void *address){
86  log_info("+++++++++++++++++ CONV init ++++++++++++++++++++");
87  conv_config* sdram_config = address;
88  config = *sdram_config;
89 
90  log_info("post_start = %u, %u, post_end = %u, %u, post_shape = %u, %u",
91  config.post_start.col, config.post_start.row,
92  config.post_end.col, config.post_end.row,
93  config.post_shape.width, config.post_shape.height);
94  log_info("num connectors = %u", config.n_connectors);
95  if (config.n_connectors == 0) {
96  return false;
97  }
98 
99  // Allocate space for connector information and pre-to-post table
100  connectors = spin1_malloc(config.n_connectors * sizeof(connectors[0]));
101  if (connectors == NULL) {
102  log_error("Can't allocate memory for connectors");
103  return false;
104  }
105 
106  // The first connector comes after the configuration in SDRAM
107  connector *conn = (connector *) &sdram_config[1];
108  for (uint32_t i = 0; i < config.n_connectors; i++) {
109  // We need the number of weights to calculate the size
110  uint32_t n_weights = conn->kernel.width * conn->kernel.height;
111  if (n_weights & 0x1) {
112  n_weights += 1;
113  }
114  uint32_t n_bytes = sizeof(*conn) + (n_weights * sizeof(conn->weights[0]));
115 
116  // Copy the data from SDRAM
117  connectors[i] = spin1_malloc(n_bytes);
118  if (connectors[i] == NULL) {
119  log_error("Can't allocate memory for connectors[%u]", i);
120  return false;
121  }
122  spin1_memcpy(connectors[i], conn, n_bytes);
123 
124  log_info("Connector %u: key=0x%08x, mask=0x%08x,"
125  "col_mask=0x%08x, col_shift=%u, row_mask=0x%08x, row_shift=%u",
126  i, connectors[i]->key_info.key, connectors[i]->key_info.mask,
127  connectors[i]->key_info.col_mask, connectors[i]->key_info.col_shift,
128  connectors[i]->key_info.row_mask, connectors[i]->key_info.row_shift);
129  log_info(" pre_start=%u, %u, kernel_shape=%u %u",
130  connectors[i]->pre_start.col, connectors[i]->pre_start.row,
131  connectors[i]->kernel.width, connectors[i]->kernel.height);
132 
133  // Move to the next connector; because it is dynamically sized,
134  // this comes after the last weight in the previous connector
135  conn = (connector *) &conn->weights[n_weights];
136  }
137 
138  return true;
139 }

143 static inline int16_t recip_multiply(int16_t integer, int16_t recip) {
144  int32_t i = integer;
145  int32_t r = recip;
146  return (int16_t) ((i * r) >> RECIP_FRACT_BITS);

153 static inline lc_coord_t map_pre_to_post(connector *connector, lc_coord_t pre,
154  int16_t half_kh, int16_t half_kw) {
155  lc_coord_t post = pre;
156  post.col = recip_multiply(post.col, connector->recip_pool_strides.col);
157  post.row = recip_multiply(post.row, connector->recip_pool_strides.row);
158  post.col = post.col - half_kw + connector->padding.width;
159  post.row = post.row - half_kh + connector->padding.height;
160  post.col = recip_multiply(post.col, connector->recip_strides.col);
161  post.row = recip_multiply(post.row, connector->recip_strides.row);
162  return post;
163 }

169 static inline void do_convolution_operation(
170  uint32_t time, lc_coord_t pre_coord, connector *connector,
171  uint16_t *ring_buffers) {
172  int32_t half_kh = connector->kernel.height / 2;
173  int32_t half_kw = connector->kernel.width / 2;
174  lc_coord_t post_coord = map_pre_to_post(connector, pre_coord, half_kh, half_kw);
175  log_debug("pre row %d, col %d AS post row %d, col %d",
176  pre_coord.row, pre_coord.col, post_coord.row, post_coord.col);
177 
178  int32_t kw = connector->kernel.width;
179  for (int32_t r = -half_kh, kr = 0; r <= half_kh; r++, kr++) {
180  int32_t tmp_row = post_coord.row + r;
181  if ((tmp_row < config.post_start.row) || (tmp_row > config.post_end.row)) {
182  continue;
183  }
184  for (int32_t c = -half_kw, kc = 0; c <= half_kw; c++, kc++) {
185  int32_t tmp_col = post_coord.col + c;
186  if ((tmp_col < config.post_start.col) || (tmp_col > config.post_end.col)) {
187  continue;
188  }
189 
190  // This the neuron id relative to the neurons on this core
191  uint32_t post_index =
192  ((tmp_row - config.post_start.row) * config.post_shape.width)
193  + (tmp_col - config.post_start.col);
194  uint32_t k = (kr * kw) + kc;
195  lc_weight_t weight = connector->weights[k];
196  if (weight == 0) {
197  continue;
198  }
199  uint32_t rb_index = 0;
200  if (weight > 0) {
201  rb_index = synapse_row_get_ring_buffer_index(time + 1,
202  connector->positive_synapse_type, post_index,
203  synapse_type_index_bits, synapse_index_bits,
204  synapse_delay_mask);
205  } else {
206  rb_index = synapse_row_get_ring_buffer_index(time + 1,
207  connector->negative_synapse_type, post_index,
208  synapse_type_index_bits, synapse_index_bits,
209  synapse_delay_mask);
210  weight = -weight;
211  }
212  log_debug("Updating ring_buffers[%u] for post neuron %u = %u, %u, with weight %u",
213  rb_index, post_index, tmp_col, tmp_row, weight);
214 
215  // Add weight to current ring buffer value, avoiding saturation
216  uint32_t accumulation = ring_buffers[rb_index] + weight;
217  uint32_t sat_test = accumulation & 0x10000;
218  if (sat_test) {
219  accumulation = sat_test - 1;
220  }
221  ring_buffers[rb_index] = accumulation;
222  }
223  }
224 }

226 static inline bool key_to_index_lookup(uint32_t spike, connector **conn,
227  uint32_t *core_local_col, uint32_t *core_local_row) {
228  for (uint32_t i = 0; i < config.n_connectors; i++) {
229  connector *c = connectors[i];
230  if ((spike & c->key_info.mask) == c->key_info.key) {
231  uint32_t local_spike = (spike & ~c->key_info.mask) >> c->key_info.n_colour_bits;
232  *conn = c;
233  *core_local_col = (local_spike & c->key_info.col_mask) >> c->key_info.col_shift;
234  *core_local_row = (local_spike & c->key_info.row_mask) >> c->key_info.row_shift;
235  return true;
236  }
237  }
238  return false;
239 }

247 void local_only_impl_process_spike(
248  uint32_t time, uint32_t spike, uint16_t* ring_buffers) {
249  connector *connector;
250  uint32_t core_local_col;
251  uint32_t core_local_row;
252 
253  // Lookup the spike, and if found, get the appropriate parts
254  if (!key_to_index_lookup(
255  spike, &connector, &core_local_col, &core_local_row)) {
256  return;
257  }
258 
259  // compute the population-based coordinates
260  lc_coord_t pre_coord = {
261  core_local_row + connector->pre_start.row,
262  core_local_col + connector->pre_start.col
263  };
264  log_debug("Received spike %u = %u, %u (Global: %u, %u)", spike, core_local_col, core_local_row,
265  pre_coord.col, pre_coord.row);
266 
267  // Compute the convolution
268  do_convolution_operation(time, pre_coord, connector, ring_buffers);
