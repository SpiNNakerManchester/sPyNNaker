87bool local_only_impl_initialise(void *address){
88 log_info("+++++++++++++++++ CONV init ++++++++++++++++++++");
89 conv_config* sdram_config = address;
90 uint32_t n_bytes = sizeof(conv_config) +
91 (sizeof(connector) * sdram_config->n_connectors);
92 config = spin1_malloc(n_bytes);
93 if (config == NULL) {
94 log_error("Can't allocate memory for config!");
95 return false;
96 }
97 spin1_memcpy(config, sdram_config, n_bytes);
98
99 log_info("post_start = %u, %u, post_end = %u, %u, post_shape = %u, %u",
100 config->post_start.col, config->post_start.row,
101 config->post_end.col, config->post_end.row,
102 config->post_shape.width, config->post_shape.height);
103 log_info("num connectors = %u", config->n_connectors);
104
105 if (config->n_connectors == 0) {
106 log_error("No connectors!");
107 return false;
108 }
109
110 // The weights come after the config in SDRAM
111 lc_weight_t *kernel_weights =
112 (lc_weight_t *) &(sdram_config->connectors[config->n_connectors]);
113 uint32_t n_weight_bytes = sizeof(lc_weight_t) * config->n_weights_total;
114 weights = spin1_malloc(n_weight_bytes);
115 if (weights == NULL) {
116 log_error("Can't allocate memory for weights!");
117 return false;
118 }
119 spin1_memcpy(weights, kernel_weights, n_weight_bytes);
120
121 // Print what we have
122 for (uint32_t i = 0; i < config->n_connectors; i++) {
123 log_info("Connector %u: key=0x%08x, mask=0x%08x,"
124 "col_mask=0x%08x, col_shift=%u, row_mask=0x%08x, row_shift=%u",
125 i, config->connectors[i].key_info.key,
126 config->connectors[i].key_info.mask,
127 config->connectors[i].key_info.col_mask,
128 config->connectors[i].key_info.col_shift,
129 config->connectors[i].key_info.row_mask,
130 config->connectors[i].key_info.row_shift);
131 log_info(" pre_start=%u, %u, kernel_shape=%u %u",
132 config->connectors[i].pre_start.col,
133 config->connectors[i].pre_start.row,
134 config->connectors[i].kernel.width,
135 config->connectors[i].kernel.height);
136 }
137
138 return true;
139}

143static inline int16_t recip_multiply(int16_t integer, int16_t recip) {
144 int32_t i = integer;
145 int32_t r = recip;
146 return (int16_t) ((i * r) >> RECIP_FRACT_BITS);
147}

153static inline lc_coord_t map_pre_to_post(connector *connector, lc_coord_t pre,
154 int16_t half_kh, int16_t half_kw) {
155 lc_coord_t post = pre;
156 post.col = recip_multiply(post.col, connector->recip_pool_strides.col);
157 post.row = recip_multiply(post.row, connector->recip_pool_strides.row);
158 post.col = post.col - half_kw + connector->padding.width;
159 post.row = post.row - half_kh + connector->padding.height;
160 post.col = recip_multiply(post.col, connector->recip_strides.col);
161 post.row = recip_multiply(post.row, connector->recip_strides.row);
162 return post;
163}

169static inline void do_convolution_operation(
170 uint32_t time, lc_coord_t pre_coord, connector *connector,
171 uint16_t *ring_buffers) {
172 int32_t half_kh = connector->kernel.height / 2;
173 int32_t half_kw = connector->kernel.width / 2;
174 lc_coord_t post_coord = map_pre_to_post(connector, pre_coord, half_kh, half_kw);
175 log_debug("pre row %d, col %d AS post row %d, col %d",
176 pre_coord.row, pre_coord.col, post_coord.row, post_coord.col);
177 lc_weight_t *connector_weights = &weights[connector->kernel_index];
178
179 int32_t kw = connector->kernel.width;
180 for (int32_t r = -half_kh, kr = 0; r <= half_kh; r++, kr++) {
181 int32_t tmp_row = post_coord.row + r;
182 if ((tmp_row < config->post_start.row) || (tmp_row > config->post_end.row)) {
183 continue;
184 }
185 for (int32_t c = -half_kw, kc = 0; c <= half_kw; c++, kc++) {
186 int32_t tmp_col = post_coord.col + c;
187 if ((tmp_col < config->post_start.col) || (tmp_col > config->post_end.col)) {
188 continue;
189 }
190
191 // This the neuron id relative to the neurons on this core
192 uint32_t post_index =
193 ((tmp_row - config->post_start.row) * config->post_shape.width)
194 + (tmp_col - config->post_start.col);
195 uint32_t k = (kr * kw) + kc;
196 lc_weight_t weight = connector_weights[k];
197 if (weight == 0) {
198 continue;
199 }
200 uint32_t rb_index = 0;
201 if (weight > 0) {
202 rb_index = synapse_row_get_ring_buffer_index(time + connector->delay,
203 connector->positive_synapse_type, post_index,
204 synapse_type_index_bits, synapse_index_bits,
205 synapse_delay_mask);
206 } else {
207 rb_index = synapse_row_get_ring_buffer_index(time + connector->delay,
208 connector->negative_synapse_type, post_index,
209 synapse_type_index_bits, synapse_index_bits,
210 synapse_delay_mask);
211 weight = -weight;
212 }
213 log_debug("Updating ring_buffers[%u] for post neuron %u = %u, %u, with weight %u",
214 rb_index, post_index, tmp_col, tmp_row, weight);
215
216 // Add weight to current ring buffer value, avoiding saturation
217 uint32_t accumulation = ring_buffers[rb_index] + weight;
218 uint32_t sat_test = accumulation & 0x10000;
219 if (sat_test) {
220 accumulation = sat_test - 1;
221 }
222 ring_buffers[rb_index] = accumulation;
223 }
224 }

227static inline bool key_to_index_lookup(uint32_t spike, uint32_t *start_index,
228 uint32_t *end_index) {
229 for (uint32_t i = 0; i < config->n_connectors; i++) {
230 connector *c = &(config->connectors[i]);
231 if ((spike & c->key_info.mask) == c->key_info.key) {
232 *start_index = i;
233 uint32_t e = i + 1;
234 while (e < config->n_connectors) {
235 connector *c_e = &(config->connectors[e]);
236 if ((spike & c_e->key_info.mask) != c_e->key_info.mask) {
237 break;
238 }
239 e = e + 1;
240 }
241 *end_index = e;
242 return true;
243 }
244 }
245 return false;
246}

248static inline void get_row_col(uint32_t spike, uint32_t index,
249 uint32_t *core_local_col, uint32_t *core_local_row) {
250 connector *c = &(config->connectors[index]);
251 uint32_t local_spike = (spike & ~c->key_info.mask) >> c->key_info.n_colour_bits;
252 *core_local_col = (local_spike & c->key_info.col_mask) >> c->key_info.col_shift;
253 *core_local_row = (local_spike & c->key_info.row_mask) >> c->key_info.row_shift;
254}

262void local_only_impl_process_spike(
263 uint32_t time, uint32_t spike, uint16_t* ring_buffers) {
264
265 // Lookup the spike, and if found, get the appropriate parts
266 uint32_t start;
267 uint32_t end;
268 if (!key_to_index_lookup(spike, &start, &end)) {
269 log_warning("Spike %u didn't match any connectors!", spike);
270 return;
271 }
272 log_debug("Received spike %u, using connectors between %u and %u", spike, start, end);
273
274 // compute the population-based coordinates
275 for (uint32_t i = start; i < end; i++) {
276 uint32_t core_local_col;
277 uint32_t core_local_row;
278 connector *connector = &(config->connectors[i]);
279 get_row_col(spike, i, &core_local_col, &core_local_row);
280 lc_coord_t pre_coord = {
281 core_local_row + connector->pre_start.row,
282 core_local_col + connector->pre_start.col
283 };
284 log_debug("Spike %u = %u, %u (Global: %u, %u)", spike, core_local_col, core_local_row,
285 pre_coord.col, pre_coord.row);
286
287 // Compute the convolution
288 do_convolution_operation(time, pre_coord, connector, ring_buffers);
289 }
