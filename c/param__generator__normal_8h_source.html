52 static void *param_generator_normal_initialize(address_t *region) {
53  // Allocate memory for the data
54  struct param_generator_normal *obj =
55  spin1_malloc(sizeof(struct param_generator_normal));
56  struct normal_params *params_sdram = (void *) *region;
57 
58  // Copy the parameters in

60  *region = (void *) params_sdram;

62  log_debug("normal mu = %k, sigma = %k",
63  obj->params.mu, obj->params.sigma);
64 
65  // Initialise the RNG for this generator
66  obj->rng = rng_init(region);

90 static void param_generator_normal_generate(
91  void *generator, uint32_t n_indices, UNUSED uint32_t pre_neuron_index,
92  UNUSED uint16_t *indices, accum *values) {
93  // For each index, generate a normally distributed random value
94  struct param_generator_normal *obj = generator;
95  for (uint32_t i = 0; i < n_indices; i++) {
96  accum value = rng_normal(obj->rng);
97  values[i] = obj->params.mu + (value * obj->params.sigma);
98  }
