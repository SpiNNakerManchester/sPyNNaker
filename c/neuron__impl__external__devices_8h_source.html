105 static inline uint _int_bits(int value) {
106  typedef union _int_bits_union {
107  int int_value;
108  uint uint_value;
109  } _int_bits_union;
110 
111  _int_bits_union converter;
112  converter.int_value = value;
113  return converter.uint_value;
114 }

120 static inline uint _get_payload(enum send_type type, accum value) {
121  switch (type) {
122  case SEND_TYPE_INT:
123  return _int_bits((int) value);
124  case SEND_TYPE_UINT:
125  return (uint) value;
126  case SEND_TYPE_ACCUM:
127  return _int_bits(bitsk(value));
128  case SEND_TYPE_UACCUM:
129  return bitsuk((unsigned accum) value);
130  case SEND_TYPE_FRACT:
131  return _int_bits(bitslr((long fract) value));
132  case SEND_TYPE_UFRACT:
133  return bitsulr((long unsigned fract) value);
134  default:
135  log_error("Unknown enum value %u", value);
136  rt_error(RTE_SWERR);
137  }
138  return 0;
139 }

197 static void neuron_impl_load_neuron_parameters(
198  address_t address, uint32_t next, uint32_t n_neurons,
199  address_t save_initial_state) {
200 
201  // Read the number of steps per timestep
202  n_steps_per_timestep = address[next++];
203  if (n_steps_per_timestep > 1) {
204  log_info("Looping over %u steps each timestep", n_steps_per_timestep);
205  } else if (n_steps_per_timestep == 0) {
206  log_error("bad number of steps per timestep: 0");
207  }
208 
209  neuron_params_t *neuron_params = (neuron_params_t *) &address[next];
210  for (uint32_t i = 0; i < n_neurons; i++) {
211  neuron_model_initialise(&neuron_array[i], &neuron_params[i], n_steps_per_timestep);
212  }
213  next += n_words_needed(n_neurons * sizeof(neuron_params_t));
214 
215  spin1_memcpy(packet_firing_array, &address[next],
216  n_neurons * sizeof(packet_firing_data_t));
217  next += n_words_needed(n_neurons * sizeof(packet_firing_data_t));
218 
219  synapse_types_params_t *synapse_params = (synapse_types_params_t *) &address[next];
220  for (uint32_t i = 0; i < n_neurons; i++) {
221  synapse_types_initialise(&synapse_types_array[i], &synapse_params[i], n_steps_per_timestep);
222  }
223  next += n_words_needed(n_neurons * sizeof(synapse_types_params_t));
224 
225  // If we are to save the initial state, copy the whole of the parameters
226  // to the initial state
227  if (save_initial_state) {
228  spin1_memcpy(save_initial_state, address, next * sizeof(uint32_t));
229  }
230 
231 #if LOG_LEVEL >= LOG_DEBUG
232  for (index_t n = 0; n < n_neurons; n++) {
233  neuron_model_print_parameters(&neuron_array[n]);
234  }
235 #endif // LOG_LEVEL >= LOG_DEBUG
236 }

242 static bool _test_will_fire(packet_firing_data_t *packet_firing) {
243  if (packet_firing->time_until_next_send == 0) {
244  packet_firing->time_until_next_send =
245  packet_firing->timesteps_between_sending;
246  --packet_firing->time_until_next_send;
247  return true;
248  }
249  --packet_firing->time_until_next_send;
250  return false;
251 }

253 SOMETIMES_UNUSED // Marked unused as only used sometimes
258 static void neuron_impl_do_timestep_update(
259  UNUSED uint32_t timer_count, UNUSED uint32_t time, uint32_t n_neurons) {
260 
261  for (uint32_t neuron_index = 0; neuron_index < n_neurons; neuron_index++) {
262  // Get the neuron itself
263  neuron_t *this_neuron = &neuron_array[neuron_index];
264 
265  // Get threshold and additional input parameters for this neuron
266  packet_firing_data_t *the_packet_firing =
267  &packet_firing_array[neuron_index];
268  synapse_types_t *the_synapse_type =
269  &synapse_types_array[neuron_index];
270 
271  // Store whether the neuron has spiked
272  bool will_fire = false;
273 
274  // Loop however many times requested; do this in reverse for efficiency,
275  // and because the index doesn't actually matter
276  for (uint32_t i = n_steps_per_timestep; i > 0; i--) {
277  // Get the voltage
278  state_t soma_voltage = neuron_model_get_membrane_voltage(this_neuron);
279 
280  // Get the exc and inh values from the synapses
281  input_t exc_values[NUM_EXCITATORY_RECEPTORS];
282  input_t *exc_input_values =
283  synapse_types_get_excitatory_input(exc_values, the_synapse_type);
284  input_t inh_values[NUM_INHIBITORY_RECEPTORS];
285  input_t *inh_input_values =
286  synapse_types_get_inhibitory_input(inh_values, the_synapse_type);
287 
288  // Sum g_syn contributions from all receptors for recording
289  REAL total_exc = 0;
290  REAL total_inh = 0;
291 
292  for (int i = 0; i < NUM_EXCITATORY_RECEPTORS; i++) {
293  total_exc += exc_input_values[i];
294  }
295  for (int i = 0; i < NUM_INHIBITORY_RECEPTORS; i++) {
296  total_inh += inh_input_values[i];
297  }
298 
299  // Do recording if on the first step
300  if (i == n_steps_per_timestep) {
301  neuron_recording_record_accum(
302  V_RECORDING_INDEX, neuron_index, soma_voltage);
303  neuron_recording_record_accum(
304  GSYN_EXC_RECORDING_INDEX, neuron_index, total_exc);
305  neuron_recording_record_accum(
306  GSYN_INH_RECORDING_INDEX, neuron_index, total_inh);
307  }
308 
309  // Get any input from an injected current source
310  REAL current_offset = current_source_get_offset(time, neuron_index);
311 
312  // update neuron parameters
313  state_t result = neuron_model_state_update(
314  NUM_EXCITATORY_RECEPTORS, exc_input_values,
315  NUM_INHIBITORY_RECEPTORS, inh_input_values,
316  0, current_offset, this_neuron);
317 
318  // determine if a packet should fly
319  will_fire = _test_will_fire(the_packet_firing);
320 
321  // If spike occurs, communicate to relevant parts of model
322  if (will_fire) {
323  if (the_packet_firing->value_as_payload) {
324  accum value_to_send = result;
325  if (result > the_packet_firing->max_value) {
326  value_to_send = the_packet_firing->max_value;
327  }
328  if (result < the_packet_firing->min_value) {
329  value_to_send = the_packet_firing->min_value;
330  }
331 
332  uint payload = _get_payload(
333  the_packet_firing->type,
334  value_to_send * the_packet_firing->value_as_payload);
335 
336  send_spike_mc_payload(the_packet_firing->key, payload);
337  } else {
338  send_spike_mc(the_packet_firing->key);
339  }
340  }
341 
342  // Shape the existing input according to the included rule
343  synapse_types_shape_input(the_synapse_type);
344  }
345 
346  if (will_fire) {
347  // Record the spike
348  neuron_recording_record_bit(PACKET_RECORDING_BITFIELD, neuron_index);
349  }
350 
351  #if LOG_LEVEL >= LOG_DEBUG
352  neuron_model_print_state_variables(this_neuron);
353  #endif // LOG_LEVEL >= LOG_DEBUG
354  }
355 }
