106 bool connection_generator_fixed_pre_generate(
107  void *generator, uint32_t pre_lo, uint32_t pre_hi,
108  uint32_t post_lo, uint32_t post_hi, UNUSED uint32_t post_index,
109  uint32_t post_slice_start, uint32_t post_slice_count,
110  unsigned long accum weight_scale, accum timestep_per_delay,
111  param_generator_t weight_generator, param_generator_t delay_generator,
112  matrix_generator_t matrix_generator) {
113  // Get the actual ranges to generate within
114  uint32_t post_start = max(post_slice_start, post_lo);
115  uint32_t post_end = min(post_slice_start + post_slice_count - 1, post_hi);
116 
117  struct fixed_pre *obj = generator;
118  // Get how many values can be sampled from
119  uint32_t n_values = pre_hi - pre_lo + 1;
120  // Get the number of connections in each column
121  uint32_t n_conns = obj->params.n_pre;
122 
123  // We have to generate everything for each column, then just take our share,
124  // so we use the population_rng here to ensure all cores do the same thing
125  for (uint32_t post = post_start; post <= post_end; post++) {
126  uint32_t local_post = post - post_slice_start;
127  if (obj->params.with_replacement) {
128  // If with replacement just repeated pick
129  for (uint32_t j = 0; j < n_conns; j++) {
130  accum weight = param_generator_generate(weight_generator);
131  uint16_t delay = rescale_delay(
132  param_generator_generate(delay_generator), timestep_per_delay);
133  uint32_t pre;
134  bool written = false;
135  uint32_t n_retries = 0;
136  do {
137  pre = pre_random_in_range(core_rng, n_values) + pre_lo;
138  if (obj->params.allow_self_connections || pre != post) {
139  written = matrix_generator_write_synapse(
140  matrix_generator, pre, local_post, weight, delay,
141  weight_scale);
142  n_retries++;
143  }
144  } while (!written && n_retries < 10);
145  if (!written) {
146  log_error("Couldn't find a row to write to!");
147  return false;
148  }
149  }
150  } else {
151  // Without replacement uses reservoir sampling to save space
152  uint16_t values[n_conns];
153  uint32_t replace_start = n_conns;
154  for (uint32_t j = 0; j < n_conns; j++) {
155  if (j == post && !obj->params.allow_self_connections) {
156  values[j] = n_conns;
157  replace_start = n_conns + 1;
158  } else {
159  values[j] = j + pre_lo;
160  }
161  }
162  for (uint32_t j = replace_start; j < n_values; j++) {
163  // r = random(0, j) (inclusive); swap j into array if r
164  // is in range
165  if (j != post || obj->params.allow_self_connections) {
166  uint32_t r = pre_random_in_range(core_rng, j + 1);
167  if (r < n_conns) {
168  values[r] = j + pre_lo;
169  }
170  }
171  }
172  for (uint32_t j = 0; j < n_conns; j++) {
173  accum weight = param_generator_generate(weight_generator);
174  uint16_t delay = rescale_delay(
175  param_generator_generate(delay_generator), timestep_per_delay);
176  // Not a lot we can do here!
177  if (!matrix_generator_write_synapse(matrix_generator, values[j],
178  local_post, weight, delay, weight_scale)) {
179  log_warning("Could not write to matrix!");
180  }
181  }
182  }
183  }
184  return true;
185 }
