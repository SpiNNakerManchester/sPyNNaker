109 static uint32_t pre_random_in_range(struct fixed_pre *obj, uint32_t range) {
110  uint32_t u01 = rng_generator(obj->rng) & 0x00007fff;
111  return (u01 * range) >> 15;

133 uint32_t connection_generator_fixed_pre_generate(
134  void *generator, uint32_t pre_slice_start, uint32_t pre_slice_count,
135  uint32_t pre_neuron_index, uint32_t post_slice_start,
136  uint32_t post_slice_count, uint32_t max_row_length, uint16_t *indices) {
137  use(pre_slice_start);
138  use(pre_slice_count);
139 
140  // If there are no connections to be made, return 0
141 
142  // Don't think that this is necessary, unless the user says 0 for some reason?
143  struct fixed_pre *obj = generator;
144  if (max_row_length == 0 || obj->params.n_pre == 0) {
145  return 0;
146  }
147 
148  // Get how many values can be sampled from
149  uint32_t n_values = obj->params.n_pre_neurons;
150 
151  // Get the number of connections in this column
152  uint32_t n_conns = obj->params.n_pre;
153 
154  log_debug("Generating %u from %u possible synapses", n_conns, n_values);
155 
156  // The number of columns is the number of post-slices to do the calculation for
157  uint32_t n_columns = post_slice_count;
158 
159  // If we haven't done so then do the calculations by looping over
160  // the post-slices available here
161  if (pre_neuron_index == 0) {
162  // Ensure the array was freed
163  if (fixed_pre_globals.full_indices != NULL) {
164  log_error("Created out of order!");
165  rt_error(RTE_SWERR);
166  }
167 
168  fixed_pre_globals.n_pre_neurons_done = 0;
169  // Allocate array for each column (i.e. post-slice on this slice)
170  uint16_t (*array)[n_columns][n_conns] =
171  spin1_malloc(n_columns * n_conns * sizeof(uint16_t));
172  fixed_pre_globals.in_sdram = false;
173  if (array == NULL) {
174  log_warning("Could not allocate in DTCM, trying SDRAM");
175  array = sark_xalloc(sv->sdram_heap,
176  n_columns * n_conns * sizeof(uint16_t), 0, ALLOC_LOCK);
177  fixed_pre_globals.in_sdram = true;
178  }
179  if (array == NULL) {
180  log_error("Could not allocate array for indices");
181  rt_error(RTE_SWERR);
182  }
183  fixed_pre_globals.full_indices = array;
184 
185  // Loop over the columns and fill the full_indices array accordingly
186  for (uint32_t n = 0; n < n_columns; n++) {
187  // Sample from the possible connections in this column n_conns times
188  if (obj->params.with_replacement) {
189  // Sample them with replacement
190  if (obj->params.allow_self_connections) {
191  // self connections are allowed so sample
192  for (uint32_t i = 0; i < n_conns; i++) {
193  (*array)[n][i] = pre_random_in_range(obj, n_values);
194  }
195  } else {
196  // self connections are not allowed (on this slice)
197  for (uint32_t i = 0; i < n_conns; i++) {
198  // Set j to the disallowed value, then test against it
199  uint32_t j;
200 
201  do {
202  j = pre_random_in_range(obj, n_values);
203  } while (j == n + post_slice_start);
204 
205  (*array)[n][i] = j;
206  }
207  }
208  } else {
209  // Sample them without replacement using reservoir sampling
210  if (obj->params.allow_self_connections) {
211  // Self-connections are allowed so do this normally
212  for (uint32_t i = 0; i < n_conns; i++) {
213  (*array)[n][i] = i;
214  }
215  // And now replace values if chosen at random to be replaced
216  for (uint32_t i = n_conns; i < n_values; i++) {
217  // j = random(0, i) (inclusive)
218  uint32_t j = pre_random_in_range(obj, i + 1);
219 
220  if (j < n_conns) {
221  (*array)[n][j] = i;
222  }
223  }
224  } else {
225  // Self-connections are not allowed
226  uint32_t replace_start = n_conns;
227  for (uint32_t i = 0; i < n_conns; i++) {
228  if (i == n + post_slice_start) {
229  // set to a value not equal to i for now
230  (*array)[n][i] = n_conns;
231  replace_start = n_conns + 1;
232  } else {
233  (*array)[n][i] = i;
234  }
235  }
236  // And now "replace" values if chosen at random to be replaced
237  for (uint32_t i = replace_start; i < n_values; i++) {
238  if (i != (n + post_slice_start)) {
239  // j = random(0, i) (inclusive)
240  uint32_t j = pre_random_in_range(obj, i + 1);
241 
242  if (j < n_conns) {
243  (*array)[n][j] = i;
244  }
245  }
246  }
247  }
248  }
249  }
250  }
251 
252  // If not in the pre-population view range, then don't generate
253  if ((pre_neuron_index < obj->params.pre_lo) ||
254  (pre_neuron_index > obj->params.pre_hi)) {
255  return 0;
256  }
257 
258  uint16_t (*array)[n_columns][n_conns] = fixed_pre_globals.full_indices;
259 
260  // Loop over the full indices array, only use pre_neuron_index, and
261  // only generate for required columns
262  uint32_t count_indices = 0;
263  for (uint32_t n = 0; n < n_columns; n++) {
264  // Only generate within the specified lo and hi of the population view
265  if ((n + post_slice_start >= obj->params.post_lo) &&
266  (n + post_slice_start <= obj->params.post_hi)) {
267  for (uint32_t i = 0; i < n_conns; i++) {
268  uint32_t j = (*array)[n][i] + obj->params.pre_lo;
269  if (j == pre_neuron_index) {
270  indices[count_indices] = n; // On this slice!
271  count_indices++;
272  }
273  }
274  }
275  }
276 
277  // If all neurons in pre-slice have been done, free memory
278  fixed_pre_globals.n_pre_neurons_done++;
279  if (fixed_pre_globals.n_pre_neurons_done == obj->params.n_pre_neurons) {
280  if (!fixed_pre_globals.in_sdram) {
281  sark_free(fixed_pre_globals.full_indices);
282  } else {
283  sark_xfree(sv->sdram_heap, fixed_pre_globals.full_indices,
284  ALLOC_LOCK);
285  }
286  fixed_pre_globals.full_indices = NULL;
287  }
288 
289  return count_indices;
290 }
