83 static void *connection_generator_fixed_pre_initialise(address_t *region) {
84  // Allocate memory for the parameters
85  struct fixed_pre *obj = spin1_malloc(sizeof(struct fixed_pre));
86 
87  // Copy the parameters in
88  struct fixed_pre_params *params_sdram = (void *) *region;
89  obj->params = *params_sdram++;
90  *region = (void *) params_sdram;
91 
92  // Initialise the RNG
93  obj->rng = rng_init(region);
94  log_debug("Fixed Number Pre Connector parameters: pre_lo = %u, pre_hi = %u, "
95  "post_lo = %u, post_hi = %u, allow self connections = %u, "
96  "with replacement = %u, n_pre = %u, n pre neurons = %u",
97  obj->params.pre_lo, obj->params.pre_hi, obj->params.post_lo, obj->params.post_hi,
98  obj->params.allow_self_connections,
99  obj->params.with_replacement, obj->params.n_pre,
100  obj->params.n_pre_neurons);
101 
102  // Build the array
103 
104  // Get how many values can be sampled from
105  uint32_t n_values = obj->params.n_pre_neurons;
106 
107  // Get the number of connections in each column
108  uint32_t n_conns = obj->params.n_pre;
109 
110  log_debug("Generating %u from %u possible synapses", n_conns, n_values);
111 
112  // The number of columns is the number of post-slices to do the calculation for
113  uint32_t n_columns = obj->params.post_hi - obj->params.post_lo + 1;
114 
115  // Allocate array for each column (i.e. post-slice on this slice)
116  uint16_t (*array)[n_columns][n_conns] =
117  spin1_malloc(n_columns * n_conns * sizeof(uint16_t));
118  fixed_pre_globals.in_sdram = false;
119  if (array == NULL) {
120  log_warning("Could not allocate in DTCM, trying SDRAM");
121  array = sark_xalloc(sv->sdram_heap,
122  n_columns * n_conns * sizeof(uint16_t), 0, ALLOC_LOCK);
123  fixed_pre_globals.in_sdram = true;
124  }
125  if (array == NULL) {
126  log_error("Could not allocate array for indices");
127  rt_error(RTE_SWERR);
128  }
129 
130  // Loop over the columns and fill the full_indices array accordingly
131  for (uint32_t n = 0; n < n_columns; n++) {
132  // Sample from the possible connections in this column n_conns times
133  if (obj->params.with_replacement) {
134  // Sample them with replacement
135  if (obj->params.allow_self_connections) {
136  // self connections are allowed so sample
137  for (uint32_t i = 0; i < n_conns; i++) {
138  (*array)[n][i] = pre_random_in_range(obj, n_values);
139  }
140  } else {
141  // self connections are not allowed (on this slice)
142  for (uint32_t i = 0; i < n_conns; i++) {
143  // Set j to the disallowed value, then test against it
144  uint32_t j;
145 
146  do {
147  j = pre_random_in_range(obj, n_values);
148  } while (j == n); // + post_slice_start);
149 
150  (*array)[n][i] = j;
151  }
152  }
153  } else {
154  // Sample them without replacement using reservoir sampling
155  if (obj->params.allow_self_connections) {
156  // Self-connections are allowed so do this normally
157  for (uint32_t i = 0; i < n_conns; i++) {
158  (*array)[n][i] = i;
159  }
160  // And now replace values if chosen at random to be replaced
161  for (uint32_t i = n_conns; i < n_values; i++) {
162  // j = random(0, i) (inclusive)
163  uint32_t j = pre_random_in_range(obj, i + 1);
164 
165  if (j < n_conns) {
166  (*array)[n][j] = i;
167  }
168  }
169  } else {
170  // Self-connections are not allowed
171  uint32_t replace_start = n_conns;
172  for (uint32_t i = 0; i < n_conns; i++) {
173  if (i == n) { // + post_slice_start) {
174  // set to a value not equal to i for now
175  (*array)[n][i] = n_conns;
176  replace_start = n_conns + 1;
177  } else {
178  (*array)[n][i] = i;
179  }
180  }
181  // And now "replace" values if chosen at random to be replaced
182  for (uint32_t i = replace_start; i < n_values; i++) {
183  if (i != n) { // + post_slice_start)) {
184  // j = random(0, i) (inclusive)
185  uint32_t j = pre_random_in_range(obj, i + 1);
186 
187  if (j < n_conns) {
188  (*array)[n][j] = i;
189  }
190  }
191  }
192  }
193  }
194  }
195 
196  fixed_pre_globals.full_indices = array;
197 
198  return obj;
199 }

230 uint32_t connection_generator_fixed_pre_generate(
231  void *generator, UNUSED uint32_t pre_slice_start,
232  UNUSED uint32_t pre_slice_count,
233  uint32_t pre_neuron_index, uint32_t post_slice_start,
234  uint32_t post_slice_count, uint32_t max_row_length, uint16_t *indices) {
235  // If there are no connections to be made, return 0
236 
237  // Don't think that this is necessary, unless the user says 0 for some reason?
238  struct fixed_pre *obj = generator;
239  if (max_row_length == 0 || obj->params.n_pre == 0) {
240  return 0;
241  }
242 
243  // If not in the pre-population view range, then don't generate
244  if ((pre_neuron_index < obj->params.pre_lo) ||
245  (pre_neuron_index > obj->params.pre_hi)) {
246  return 0;
247  }
248 
249  // Get the number of connections in each column
250  uint32_t n_conns = obj->params.n_pre;
251 
252  // The number of columns is the number of post-slices to do the calculation for
253  uint32_t n_columns = obj->params.post_hi - obj->params.post_lo + 1;
254 
255  fixed_pre_globals.n_pre_neurons_done = 0;
256 
257  uint16_t (*array)[n_columns][n_conns] = fixed_pre_globals.full_indices;
258 
259  // Loop over the full indices array, only use pre_neuron_index, and
260  // only generate for required columns
261  uint32_t count_indices = 0;
262  for (uint32_t n = 0; n < n_columns; n++) {
263  // Only generate within the post-population view
264  if ((n + obj->params.post_lo >= post_slice_start) &&
265  (n + obj->params.post_lo < (post_slice_start + post_slice_count))) {
266  for (uint32_t i = 0; i < n_conns; i++) {
267  uint32_t j = (*array)[n][i] + obj->params.pre_lo;
268  if (j == pre_neuron_index) {
269  // The index is the value locally on the slice
270  indices[count_indices] = n + obj->params.post_lo - post_slice_start;
271  count_indices++;
272  }
273  }
274  }
275  }
276 
277  // If all neurons in pre-slice have been done, free memory
278  fixed_pre_globals.n_pre_neurons_done++;
279  if (fixed_pre_globals.n_pre_neurons_done == obj->params.n_pre_neurons) {
280  if (!fixed_pre_globals.in_sdram) {
281  sark_free(fixed_pre_globals.full_indices);
282  } else {
283  sark_xfree(sv->sdram_heap, fixed_pre_globals.full_indices,
284  ALLOC_LOCK);
285  }
286  fixed_pre_globals.full_indices = NULL;
287  }
288 
289  return count_indices;
290 }
