87 uint32_t connection_generator_fixed_pre_generate(
88  void *data, uint32_t pre_slice_start, uint32_t pre_slice_count,
89  uint32_t pre_neuron_index, uint32_t post_slice_start,
90  uint32_t post_slice_count, uint32_t max_row_length, uint16_t *indices) {
91  use(pre_slice_start);
92  use(pre_slice_count);
93 
94  // If there are no connections to be made, return 0
95 
96  // Don't think that this is necessary, unless the user says 0 for some reason?
97  struct fixed_pre *obj = data;
98  if (max_row_length == 0 || obj->params.n_pre == 0) {
99  return 0;
100  }
101 
102  // Get how many values can be sampled from
103  uint32_t n_values = obj->params.n_pre_neurons;
104 
105  // Get the number of connections in this column
106  uint32_t n_conns = obj->params.n_pre;
107 
108  log_debug("Generating %u from %u possible synapses", n_conns, n_values);
109 
110  // The number of columns is the number of post-slices to do the calculation for
111  uint32_t n_columns = post_slice_count;
112 
113  // If we haven't done so then do the calculations by looping over
114  // the post-slices available here
115  if (pre_neuron_index == 0) {
116  // Ensure the array was freed
117  if (full_indices != NULL) {
118  log_error("Created out of order!");
119  rt_error(RTE_SWERR);
120  }
121 
122  n_pre_neurons_done = 0;
123  // Allocate array for each column (i.e. post-slice on this slice)
124  uint16_t (*array)[n_columns][n_conns] =
125  spin1_malloc(n_columns * n_conns * sizeof(uint16_t));
126  in_sdram = 0;
127  if (array == NULL) {
128  log_warning("Could not allocate in DTCM, trying SDRAM");
129  array = sark_xalloc(sv->sdram_heap,
130  n_columns * n_conns * sizeof(uint16_t), 0, ALLOC_LOCK);
131  in_sdram = 1;
132  }
133  if (array == NULL) {
134  log_error("Could not allocate array for indices");
135  rt_error(RTE_SWERR);
136  }
137  full_indices = array;
138 
139  // Loop over the columns and fill the full_indices array accordingly
140  for (uint32_t n = 0; n < n_columns; n++) {
141  // Sample from the possible connections in this column n_conns times
142  if (obj->params.with_replacement) {
143  // Sample them with replacement
144  if (obj->params.allow_self_connections) {
145  // self connections are allowed so sample
146  for (uint32_t i = 0; i < n_conns; i++) {
147  (*array)[n][i] = pre_random_in_range(obj, n_values);
148  }
149  } else {
150  // self connections are not allowed (on this slice)
151  for (uint32_t i = 0; i < n_conns; i++) {
152  // Set j to the disallowed value, then test against it
153  uint32_t j;
154 
155  do {
156  j = pre_random_in_range(obj, n_values);
157  } while (j == n + post_slice_start);
158 
159  (*array)[n][i] = j;
160  }
161  }
162  } else {
163  // Sample them without replacement using reservoir sampling
164  if (obj->params.allow_self_connections) {
165  // Self-connections are allowed so do this normally
166  for (uint32_t i = 0; i < n_conns; i++) {
167  (*array)[n][i] = i;
168  }
169  // And now replace values if chosen at random to be replaced
170  for (uint32_t i = n_conns; i < n_values; i++) {
171  // j = random(0, i) (inclusive)
172  uint32_t j = pre_random_in_range(obj, i + 1);
173 
174  if (j < n_conns) {
175  (*array)[n][j] = i;
176  }
177  }
178  } else {
179  // Self-connections are not allowed
180  uint32_t replace_start = n_conns;
181  for (uint32_t i = 0; i < n_conns; i++) {
182  if (i == n + post_slice_start) {
183  // set to a value not equal to i for now
184  (*array)[n][i] = n_conns;
185  replace_start = n_conns + 1;
186  } else {
187  (*array)[n][i] = i;
188  }
189  }
190  // And now "replace" values if chosen at random to be replaced
191  for (uint32_t i = replace_start; i < n_values; i++) {
192  if (i != (n + post_slice_start)) {
193  // j = random(0, i) (inclusive)
194  uint32_t j = pre_random_in_range(obj, i + 1);
195 
196  if (j < n_conns) {
197  (*array)[n][j] = i;
198  }
199  }
200  }
201  }
202  }
203  }
204  }
205 
206  // If not in the pre-population view range, then don't generate
207  if ((pre_neuron_index < obj->params.pre_lo) ||
208  (pre_neuron_index > obj->params.pre_hi)) {
209  return 0;
210  }
211 
212  uint16_t (*array)[n_columns][n_conns] = full_indices;
213 
214  // Loop over the full indices array, only use pre_neuron_index, and
215  // only generate for required columns
216  uint32_t count_indices = 0;
217  for (uint32_t n = 0; n < n_columns; n++) {
218  // Only generate within the specified lo and hi of the population view
219  if ((n + post_slice_start >= obj->params.post_lo) &&
220  (n + post_slice_start <= obj->params.post_hi)) {
221  for (uint32_t i = 0; i < n_conns; i++) {
222  uint32_t j = (*array)[n][i] + obj->params.pre_lo;
223  if (j == pre_neuron_index) {
224  indices[count_indices] = n; // On this slice!
225  count_indices++;
226  }
227  }
228  }
229  }
230 
231  // If all neurons in pre-slice have been done, free memory
232  n_pre_neurons_done++;
233  if (n_pre_neurons_done == obj->params.n_pre_neurons) {
234  if (!in_sdram) {
235  sark_free(full_indices);
236  } else {
237  sark_xfree(sv->sdram_heap, full_indices, ALLOC_LOCK);
238  }
239  full_indices = NULL;
240  }
241 
242  return count_indices;
243 }
