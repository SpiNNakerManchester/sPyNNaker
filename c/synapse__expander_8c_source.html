81 static bool read_connection_builder_region(address_t *in_region,
82  address_t synaptic_matrix_region, uint32_t post_slice_start,
83  uint32_t post_slice_count, uint32_t n_synapse_type_bits,
84  uint32_t n_synapse_index_bits, unsigned long accum *weight_scales) {
85  address_t region = *in_region;
86  struct connection_builder_config config;
87  fast_memcpy(&config, region, sizeof(config));
88  region += sizeof(config) / sizeof(uint32_t);
89 
90  // Get the matrix, connector, weight and delay parameter generators
91  matrix_generator_t matrix_generator =
92  matrix_generator_init(config.matrix_type, &region);
93  connection_generator_t connection_generator =
94  connection_generator_init(config.connector_type, &region);
95  param_generator_t weight_generator =
96  param_generator_init(config.weight_type, &region);
97  param_generator_t delay_generator =
98  param_generator_init(config.delay_type, &region);
99 
100  *in_region = region;
101 
102  // If any components couldn't be created return false
103  if (matrix_generator == NULL || connection_generator == NULL
104  || delay_generator == NULL || weight_generator == NULL) {
105  return false;
106  }
107 
108  log_debug("Synaptic matrix offset = %u, delayed offset = %u",
109  config.offset, config.delayed_offset);
110  log_debug("Max row synapses = %u, max delayed row synapses = %u",
111  config.max_row_n_synapses, config.max_delayed_row_n_synapses);
112 
113  // Get the positions to which the data should be written in the matrix
114  address_t synaptic_matrix = NULL;
115  if (config.offset != 0xFFFFFFFF) {
116  synaptic_matrix = &synaptic_matrix_region[config.offset];
117  }
118  address_t delayed_synaptic_matrix = NULL;
119  if (config.delayed_offset != 0xFFFFFFFF) {
120  delayed_synaptic_matrix = &synaptic_matrix_region[config.delayed_offset];
121  }
122  log_debug("Generating matrix at 0x%08x, delayed at 0x%08x",
123  synaptic_matrix, delayed_synaptic_matrix);
124 
125  // Do the generation
126  bool status = matrix_generator_generate(
127  matrix_generator, synaptic_matrix, delayed_synaptic_matrix,
128  config.max_row_n_words, config.max_delayed_row_n_words,
129  config.max_row_n_synapses, config.max_delayed_row_n_synapses,
130  n_synapse_type_bits, n_synapse_index_bits,
131  config.synapse_type, weight_scales,
132  post_slice_start, post_slice_count,
133  config.pre_slice_start, config.pre_slice_count,
134  connection_generator, delay_generator, weight_generator,
135  config.max_stage, config.max_delay_per_stage,
136  config.timestep_per_delay);
137 
138  // Free the neuron four!
139  matrix_generator_free(matrix_generator);
140  connection_generator_free(connection_generator);
141  param_generator_free(weight_generator);
142  param_generator_free(delay_generator);
143 
144  // If failed, log error
145  if (!status) {
146  log_error("\tMatrix generation failed");
147  return false;
148  }
149 
150  // Return success!
151  return true;
152 }

161 static bool run_synapse_expander(data_specification_metadata_t *ds_regions,
162  address_t params_address) {
163  // Read in the global parameters
164  struct expander_config config;
165  fast_memcpy(&config, params_address, sizeof(config));
166  params_address += sizeof(config) / sizeof(uint32_t);
167  log_info("Generating %u edges for %u atoms starting at %u",
168  config.n_in_edges, config.post_slice_count, config.post_slice_start);
169 
170  // Read in the weight scales, one per synapse type
171  unsigned long accum weight_scales[config.n_synapse_types];
172  fast_memcpy(weight_scales, params_address,
173  sizeof(unsigned long accum) * config.n_synapse_types);
174  params_address += 2 * config.n_synapse_types;
175 
176  // Get the synaptic matrix region
177  address_t synaptic_matrix_region = data_specification_get_region(
178  config.synaptic_matrix_region, ds_regions);
179  ds_regions->regions[config.synaptic_matrix_region].checksum = 0;
180  ds_regions->regions[config.synaptic_matrix_region].n_words = 0;
181 
182  // Go through each connector and generate
183  for (uint32_t edge = 0; edge < config.n_in_edges; edge++) {
184  if (!read_connection_builder_region(
185  &params_address, synaptic_matrix_region,
186  config.post_slice_start, config.post_slice_count,
187  config.n_synapse_type_bits,
188  config.n_synapse_index_bits, weight_scales)) {
189  return false;
190  }
191  }
192 
193  return true;
194 }

197 void c_main(void) {
198  sark_cpu_state(CPU_STATE_RUN);
199 
200  log_info("Starting To Build Connectors");
201 
202  // Get pointer to 1st virtual processor info struct in SRAM and get USER1;
203  // This is the ID of the connection builder region from which to read the
204  // rest of the data
205  vcpu_t *virtual_processor_table = (vcpu_t*) SV_VCPU;
206  uint user1 = virtual_processor_table[spin1_get_core_id()].user1;
207 
208  // Get the addresses of the regions
209  data_specification_metadata_t *ds_regions =
210  data_specification_get_data_address();
211  address_t params_address = data_specification_get_region(user1, ds_regions);
212  log_info("\tReading SDRAM at 0x%08x", params_address);
213 
214  // Run the expander
215  if (!run_synapse_expander(ds_regions, params_address)) {
216  log_info("!!! Error reading SDRAM data !!!");
217  rt_error(RTE_ABORT);
218  }
219 
220  log_info("Finished On Machine Connectors!");
221 }
