30 static const uint32_t DMA_WRITE_FLAGS =
31  DMA_WIDTH << 24 | DMA_BURST_SIZE << 21 | DMA_WRITE << 19;

34 static const uint32_t DMA_READ_FLAGS =

39 static inline bool dma_done(void) {
40  return (dma[DMA_STAT] & DMA_CHECK_MASK) == DMA_COMPLETE;
41 }

48 static inline void do_fast_dma_write(void *tcm_address, void *system_address,
49  uint32_t n_bytes) {
50 #if LOG_LEVEL >= LOG_DEBUG
51  // Useful for checking when things are going wrong, but shouldn't be
52  // needed in normal code
53  uint32_t stat = dma[DMA_STAT];
54  if (stat & 0x1FFFFF) {
55  log_error("DMA pending or in progress on write: 0x%08x", stat);
56  rt_error(RTE_SWERR);
57  }
58 #endif
59  uint32_t desc = DMA_WRITE_FLAGS | n_bytes;
60  dma[DMA_ADRS] = (uint32_t) system_address;
61  dma[DMA_ADRT] = (uint32_t) tcm_address;
62  dma[DMA_DESC] = desc;
63 }

88 static inline void wait_for_dma_to_complete(void) {
89 #if LOG_LEVEL >= LOG_DEBUG
90  // Useful for checking when things are going wrong, but shouldn't be
91  // needed in normal code
92  uint32_t n_loops = 0;
93  while (!dma_done() && n_loops < 10000) {
94  n_loops++;
95  }
96  if (!dma_done()) {
97  log_error("Timeout on DMA loop: DMA stat = 0x%08x!", dma[DMA_STAT]);
98  rt_error(RTE_SWERR);
99  }
100 #else
101  // This is the normal loop, done without checking
102  while (!dma_done()) {
103  continue;
104  }
105 #endif
106  dma[DMA_CTRL] = 0x8;
107 }
