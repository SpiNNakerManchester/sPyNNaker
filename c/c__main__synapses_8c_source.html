134 void timer_callback(UNUSED uint unused0, UNUSED uint unused1) {
135  time++;
136  if (simulation_is_finished()) {
137  // Enter pause and resume state to avoid another tick
138  simulation_handle_pause_resume(resume_callback);
139

154 static bool initialise(void) {
155  log_debug("Initialise: started");
156 
157  data_specification_metadata_t *ds_regions;
158  if (!initialise_common_regions(
159  &timer_period, &simulation_ticks, &infinite_run, &time,
160  &recording_flags, store_provenance_data, timer_callback,
161  COMMON_REGIONS, COMMON_PRIORITIES, &ds_regions)) {
162  return false;
163  }
164 
165  // Setup synapses
166  uint32_t incoming_spike_buffer_size;
167  uint32_t row_max_n_words;
168  bool clear_input_buffer_of_late_packets;
169  weight_t *ring_buffers;
170  uint32_t n_rec_regions_used = 0;
171  if (!initialise_synapse_regions(
172  ds_regions, SYNAPSE_REGIONS, &ring_buffers, &row_max_n_words,
173  &incoming_spike_buffer_size,
174  &clear_input_buffer_of_late_packets, &n_rec_regions_used)) {
175  return false;
176  }
177 
178  // Setup for writing synaptic inputs at the end of each run
179  struct sdram_config *sdram_config = data_specification_get_region(
180  SDRAM_PARAMS_REGION, ds_regions);
181  struct key_config *key_config = data_specification_get_region(
182  KEY_REGION, ds_regions);
183 
184  if (!spike_processing_fast_initialise(
185  row_max_n_words, incoming_spike_buffer_size,
186  clear_input_buffer_of_late_packets, n_rec_regions_used, MC,
187  *sdram_config, *key_config, ring_buffers)) {
188  return false;
189  }
190 
191  // Do bitfield configuration last to only use any unused memory
192  if (!population_table_load_bitfields(data_specification_get_region(
193  SYNAPSE_REGIONS.bitfield_filter, ds_regions))) {
194  return false;
195  }
196 
197  // Set timer tick (in microseconds)
198  log_debug("setting timer tick callback for %d microseconds", timer_period);
199  spin1_set_timer_tick(timer_period);
200 
201  recording_reset();
202 
203  log_debug("Initialise: finished");
204  return true;
205 }
