186 static inline uint32_t round_to_next_pot(uint32_t v) {
187  v--;
188  v |= v >> 1;
189  v |= v >> 2;
190  v |= v >> 4;
191  v |= v >> 8;
192  v |= v >> 16;
193  v++;
194  return v;
195 }

200 static bool read_parameters(struct delay_parameters *params) {
201  log_debug("read_parameters: starting");
202 
203  has_key = params->has_key;
204  key = params->key;
205  incoming_key = params->incoming_key;
206  incoming_mask = params->incoming_mask;
207  incoming_neuron_mask = ~incoming_mask;
208  log_debug("\t key = 0x%08x, incoming key = 0x%08x, incoming mask = 0x%08x,"
209  "incoming key mask = 0x%08x",
210  key, incoming_key, incoming_mask, incoming_neuron_mask);
211 
212  num_neurons = params->n_atoms;
213  neuron_bit_field_words = get_bit_field_size(num_neurons);
214 
215  num_delay_stages = params->n_delay_stages;
216  n_delay_in_a_stage = params->n_delay_in_a_stage;
217  max_keys = num_neurons * num_delay_stages;
218 
219  clear_input_buffers_of_late_packets = params->clear_packets;
220 
221  num_delay_slots = num_delay_stages * n_delay_in_a_stage;
222  // We need an extra slot here (to make one clearable after the maximum delay
223  // time), and a power of 2 (to make it easier to loop)
224  uint32_t num_delay_slots_pot = round_to_next_pot(num_delay_slots + 1);
225  num_delay_slots_mask = num_delay_slots_pot - 1;
226 
227  log_info("\t parrot neurons = %u, neuron bit field words = %u,"
228  " num delay stages = %u, num delay slots = %u (pot = %u),"
229  " num delay slots mask = %08x",
230  num_neurons, neuron_bit_field_words,
231  num_delay_stages, num_delay_slots, num_delay_slots_pot,
232  num_delay_slots_mask);
233 
234  // Create array containing a bitfield specifying whether each neuron should
235  // emit spikes after each delay stage
236  neuron_delay_stage_config =
237  spin1_malloc(num_delay_stages * sizeof(bit_field_t));
238  if (neuron_delay_stage_config == NULL) {
239  log_error("failed to allocate memory for array of size %u bytes",
240  num_delay_stages * sizeof(bit_field_t));
241  return false;
242  }
243 
244  // Loop through delay stages
245  for (uint32_t d = 0; d < num_delay_stages; d++) {
246  log_debug("\t delay stage %u", d);
247 
248  // Allocate bit-field
249  neuron_delay_stage_config[d] =
250  spin1_malloc(neuron_bit_field_words * sizeof(uint32_t));
251  if (neuron_delay_stage_config[d] == NULL) {
252  log_error("failed to allocate memory for bitfield of size %u bytes",
253  neuron_bit_field_words * sizeof(uint32_t));
254  return false;
255  }
256 
257  // Copy delay stage configuration bits into delay stage configuration
258  // bit-field
259  spin1_memcpy(neuron_delay_stage_config[d],
260  &params->delay_blocks[d * neuron_bit_field_words],
261  neuron_bit_field_words * sizeof(uint32_t));
262 
263 #if LOG_LEVEL >= LOG_DEBUG
264  io_printf(IO_BUF, "\t\tNeurons set:");
265  for (uint32_t i = 0; i < num_neurons; i++) {
266  if (bit_field_test(neuron_delay_stage_config[d], i)) {
267  io_printf(IO_BUF, " %d", i);
268  }
269  }
270  io_printf(IO_BUF, "\n");
271 #endif
272  }
273 
274  // Allocate array of counters for each delay slot
275  spike_counters = spin1_malloc(num_delay_slots_pot * sizeof(uint8_t*));
276  if (spike_counters == NULL) {
277  log_error("failed to allocate memory for array of size %u bytes",
278  num_delay_slots_pot * sizeof(uint8_t*));
279  return false;
280  }
281 
282  for (uint32_t s = 0; s < num_delay_slots_pot; s++) {
283  // Allocate an array of counters for each neuron and zero
284  spike_counters[s] = spin1_malloc(num_neurons * sizeof(uint8_t));
285  if (spike_counters[s] == NULL) {
286  log_error("failed to allocate memory for bitfield of size %u bytes",
287  num_neurons * sizeof(uint8_t));
288  return false;
289  }
290  zero_spike_counters(spike_counters[s], num_neurons);
291  }
292 
293  log_debug("read_parameters: completed successfully");
294  return true;
295 }

401 static inline void spike_process(void) {
402 
403  // Get current time slot of incoming spike counters
404  uint32_t current_time_slot = time & num_delay_slots_mask;
405  uint8_t *current_time_slot_spike_counters =
406  spike_counters[current_time_slot];
407 
408  log_debug("%d: Current time slot %u", time, current_time_slot);
409 
410  // While there are any incoming spikes
411  spike_t s;
412  uint32_t state = spin1_int_disable();
413  while (in_spikes_get_next_spike(&s)) {
414  spin1_mode_restore(state);
415  n_processed_spikes++;
416 
417  if ((s & incoming_mask) == incoming_key) {
418  // Mask out neuron ID
419  uint32_t neuron_id = key_n(s);
420  if (neuron_id < num_neurons) {
421  // Increment counter
422  if (current_time_slot_spike_counters[neuron_id] ==
423  COUNTER_SATURATION_VALUE) {
424  saturation_count += 1;
425  } else {
426  current_time_slot_spike_counters[neuron_id]++;
427  }
428  log_debug("Incrementing counter %u = %u\n",
429  neuron_id,
430  current_time_slot_spike_counters[neuron_id]);
431  n_spikes_added++;
432  } else {
433  n_packets_dropped_due_to_invalid_neuron_value += 1;
434  log_debug("Invalid neuron ID %u", neuron_id);
435  }
436  } else {
437  n_packets_dropped_due_to_invalid_key += 1;
438  log_debug("Invalid spike key 0x%08x", s);
439  }
440  state = spin1_int_disable();
441  }
442 
443  spike_processing = false;
444  spin1_mode_restore(state);
445 }

451 static void background_callback(uint local_time, UNUSED uint timer_count) {
452  // reset the TDMA for this next cycle.
453  tdma_processing_reset_phase();
454 
455  // Loop through delay stages
456  for (uint32_t d = 0; d < num_delay_stages; d++) {
457  // If any neurons emit spikes after this delay stage
458  bit_field_t delay_stage_config = neuron_delay_stage_config[d];
459  if (nonempty_bit_field(delay_stage_config, neuron_bit_field_words)) {
460  // Get key mask for this delay stage and its time slot
461  uint32_t delay_stage_delay = (d + 1) * n_delay_in_a_stage;
462  if (local_time >= delay_stage_delay) {
463  uint32_t delay_stage_time_slot =
464  (local_time - delay_stage_delay) & num_delay_slots_mask;
465  uint8_t *delay_stage_spike_counters =
466  spike_counters[delay_stage_time_slot];
467 
468  log_debug("%u: Checking time slot %u for delay stage %u",
469  local_time, delay_stage_time_slot, d);
470 
471  // Loop through neurons
472  for (uint32_t n = 0; n < num_neurons; n++) {
473 
474  // If this neuron emits a spike after this stage
475  if (bit_field_test(delay_stage_config, n)) {
476 
477  // Calculate key all spikes coming from this neuron will be
478  // sent with
479  uint32_t neuron_index = ((d * num_neurons) + n);
480  uint32_t spike_key = neuron_index + key;
481 
482  #if LOG_LEVEL >= LOG_DEBUG
483  if (delay_stage_spike_counters[n] > 0) {
484  log_debug("Neuron %u sending %u spikes after delay"
485  "stage %u with key %x",
486  n, delay_stage_spike_counters[n], d,
487  spike_key);
488  }
489  #endif
490 
491  // fire n spikes as payload, 1 as none payload.
492  if (has_key) {
493  if (delay_stage_spike_counters[n] > 1) {
494  log_debug(
495  "%d: sending packet with key %d and payload %d",
496  time, spike_key, delay_stage_spike_counters[n]);
497 
498  tdma_processing_send_packet(
499  spike_key, delay_stage_spike_counters[n],
500  WITH_PAYLOAD, timer_count);
501 
502  // update counter
503  n_spikes_sent += delay_stage_spike_counters[n];
504  } else if (delay_stage_spike_counters[n] == 1) {
505  log_debug("%d: sending spike with key %d", time, spike_key);
506 
507  tdma_processing_send_packet(
508  spike_key, 0, NO_PAYLOAD, timer_count);
509 
510  // update counter
511  n_spikes_sent++;
512  }
513  }
514  }
515  }
516  }
517  }
518  }
519  n_backgrounds_queued--;
520 }

525 static void timer_callback(uint timer_count, UNUSED uint unused1) {
526  uint32_t state = spin1_int_disable();
527  uint32_t n_spikes = in_spikes_size();
528  if (clear_input_buffers_of_late_packets) {
529  in_spikes_clear();
530  }
531  // Record the count whether clearing or not for provenance
532  count_input_buffer_packets_late += n_spikes;
533  time++;
534 
535  // Clear counters
536  if (time > num_delay_slots) {
537  uint32_t clearable_slot = ((time - 1) - num_delay_slots) & num_delay_slots_mask;
538  log_debug("%d: Clearing time slot %d", time, clearable_slot);
539  zero_spike_counters(spike_counters[clearable_slot], num_neurons);
540  }
541 
542  log_debug("Timer tick %u", time);
543 
544  // If a fixed number of simulation ticks are specified and these have passed
545  if (simulation_is_finished()) {
546  // handle the pause and resume functionality
547  simulation_handle_pause_resume(NULL);
548 
549  log_debug("Delay extension finished at time %u, %u received spikes, "
550  "%u processed spikes, %u sent spikes, %u added spikes",
551  time, n_in_spikes, n_processed_spikes, n_spikes_sent,
552  n_spikes_added);
553 
554  log_debug("Delayed %u times", n_delays);
555 
556  // Subtract 1 from the time so this tick gets done again on the next
557  // run
558  time--;
559 
560  simulation_ready_to_read();
561  spin1_mode_restore(state);
562  return;
563  }
564 
565  if (!spin1_schedule_callback(background_callback, time, timer_count, BACKGROUND)) {
566  // We have failed to do this timer tick!
567  n_background_overloads++;
568  } else {
569  n_backgrounds_queued++;
570  if (n_backgrounds_queued > max_backgrounds_queued) {
571  max_backgrounds_queued++;
572  }
573  }
574  spin1_mode_restore(state);
575 }
