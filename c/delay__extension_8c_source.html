162 static inline uint32_t round_to_next_pot(uint32_t v) {
163  v--;
164  v |= v >> 1;
165  v |= v >> 2;
166  v |= v >> 4;
167  v |= v >> 8;
168  v |= v >> 16;
169  v++;
170  return v;
171 }

176 static bool read_parameters(struct delay_parameters *params) {
177  log_debug("read_parameters: starting");
178 
179  has_key = params->has_key;
180  key = params->key;
181  incoming_key = params->incoming_key;
182  incoming_mask = params->incoming_mask;
183  incoming_neuron_mask = ~incoming_mask;
184  log_debug("\t key = 0x%08x, incoming key = 0x%08x, incoming mask = 0x%08x,"
185  "incoming key mask = 0x%08x",
186  key, incoming_key, incoming_mask, incoming_neuron_mask);
187 
188  num_neurons = params->n_atoms;
189  neuron_bit_field_words = get_bit_field_size(num_neurons);
190 
191  num_delay_stages = params->n_delay_stages;
192  n_delay_in_a_stage = params->n_delay_in_a_stage;
193  max_keys = num_neurons * num_delay_stages;
194 
195  uint32_t num_delay_slots = num_delay_stages * n_delay_in_a_stage;
196  uint32_t num_delay_slots_pot = round_to_next_pot(num_delay_slots);
197  num_delay_slots_mask = num_delay_slots_pot - 1;
198 
199  log_debug("\t parrot neurons = %u, neuron bit field words = %u,"
200  " num delay stages = %u, num delay slots = %u (pot = %u),"
201  " num delay slots mask = %08x",
202  num_neurons, neuron_bit_field_words,
203  num_delay_stages, num_delay_slots, num_delay_slots_pot,
204  num_delay_slots_mask);
205 
206  // Create array containing a bitfield specifying whether each neuron should
207  // emit spikes after each delay stage
208  neuron_delay_stage_config =
209  spin1_malloc(num_delay_stages * sizeof(bit_field_t));
210  if (neuron_delay_stage_config == NULL) {
211  log_error("failed to allocate memory for array of size %u bytes",
212  num_delay_stages * sizeof(bit_field_t));
213  return false;
214  }
215 
216  // Loop through delay stages
217  for (uint32_t d = 0; d < num_delay_stages; d++) {
218  log_debug("\t delay stage %u", d);
219 
220  // Allocate bit-field
221  neuron_delay_stage_config[d] =
222  spin1_malloc(neuron_bit_field_words * sizeof(uint32_t));
223  if (neuron_delay_stage_config[d] == NULL) {
224  log_error("failed to allocate memory for bitfield of size %u bytes",
225  neuron_bit_field_words * sizeof(uint32_t));
226  return false;
227  }
228 
229  // Copy delay stage configuration bits into delay stage configuration
230  // bit-field
231  spin1_memcpy(neuron_delay_stage_config[d],
232  &params->delay_blocks[d * neuron_bit_field_words],
233  neuron_bit_field_words * sizeof(uint32_t));
234 
235  for (uint32_t w = 0; w < neuron_bit_field_words; w++) {
236  log_debug("\t\t delay stage config word %u = %08x",
237  w, neuron_delay_stage_config[d][w]);
238  }
239  }
240 
241  // Allocate array of counters for each delay slot
242  spike_counters = spin1_malloc(num_delay_slots_pot * sizeof(uint8_t*));
243  if (spike_counters == NULL) {
244  log_error("failed to allocate memory for array of size %u bytes",
245  num_delay_slots_pot * sizeof(uint8_t*));
246  return false;
247  }
248 
249  for (uint32_t s = 0; s < num_delay_slots_pot; s++) {
250  // Allocate an array of counters for each neuron and zero
251  spike_counters[s] = spin1_malloc(num_neurons * sizeof(uint8_t));
252  if (spike_counters[s] == NULL) {
253  log_error("failed to allocate memory for bitfield of size %u bytes",
254  num_neurons * sizeof(uint8_t));
255  return false;
256  }
257  zero_spike_counters(spike_counters[s], num_neurons);
258  }
259 
260  log_debug("read_parameters: completed successfully");
261  return true;
262 }

356 static inline void spike_process(void) {
357  // turn off interrupts as this function is critical for
358  // keeping time in sync.
359  uint state = spin1_int_disable();
360 
361  // Get current time slot of incoming spike counters
362  uint32_t current_time_slot = time & num_delay_slots_mask;
363  uint8_t *current_time_slot_spike_counters =
364  spike_counters[current_time_slot];
365 
366  log_debug("Current time slot %u", current_time_slot);
367 
368  // While there are any incoming spikes
369  spike_t s;
370  while (in_spikes_get_next_spike(&s)) {
371  n_processed_spikes++;
372 
373  if ((s & incoming_mask) == incoming_key) {
374  // Mask out neuron ID
375  uint32_t neuron_id = key_n(s);
376  if (neuron_id < num_neurons) {
377  // Increment counter
378  if (current_time_slot_spike_counters[neuron_id] ==
379  COUNTER_SATURATION_VALUE) {
380  saturation_count += 1;
381  }else{
382  current_time_slot_spike_counters[neuron_id]++;
383  }
384  log_debug("Incrementing counter %u = %u\n",
385  neuron_id,
386  current_time_slot_spike_counters[neuron_id]);
387  n_spikes_added++;
388  } else {
389  n_packets_dropped_due_to_invalid_neuron_value += 1;
390  log_debug("Invalid neuron ID %u", neuron_id);
391  }
392  } else {
393  n_packets_dropped_due_to_invalid_key += 1;
394  log_debug("Invalid spike key 0x%08x", s);
395  }
396  }
397 
398  // reactivate interrupts as critical section complete
399  spin1_mode_restore(state);
400 }

405 static void timer_callback(uint timer_count, UNUSED uint unused1) {
406  // Process all the spikes from the last timestep
407  spike_process();
408 
409  time++;
410 
411  log_debug("Timer tick %u", time);
412 
413  // If a fixed number of simulation ticks are specified and these have passed
414  if (simulation_is_finished()) {
415  // handle the pause and resume functionality
416  simulation_handle_pause_resume(NULL);
417 
418  log_debug("Delay extension finished at time %u, %u received spikes, "
419  "%u processed spikes, %u sent spikes, %u added spikes",
420  time, n_in_spikes, n_processed_spikes, n_spikes_sent,
421  n_spikes_added);
422 
423  log_debug("Delayed %u times", n_delays);
424 
425  // Subtract 1 from the time so this tick gets done again on the next
426  // run
427  time--;
428 
429  simulation_ready_to_read();
430  return;
431  }
432 
433  // reset the tdma for this next cycle.
434  tdma_processing_reset_phase();
435 
436  // Loop through delay stages
437  for (uint32_t d = 0; d < num_delay_stages; d++) {
438  // If any neurons emit spikes after this delay stage
439  bit_field_t delay_stage_config = neuron_delay_stage_config[d];
440  if (nonempty_bit_field(delay_stage_config, neuron_bit_field_words)) {
441  // Get key mask for this delay stage and it's time slot
442  uint32_t delay_stage_delay = (d + 1) * n_delay_in_a_stage;
443  uint32_t delay_stage_time_slot =
444  (time - delay_stage_delay) & num_delay_slots_mask;
445  uint8_t *delay_stage_spike_counters =
446  spike_counters[delay_stage_time_slot];
447 
448  log_debug("%u: Checking time slot %u for delay stage %u",
449  time, delay_stage_time_slot, d);
450 
451  // Loop through neurons
452  for (uint32_t n = 0; n < num_neurons; n++) {
453 
454  // If this neuron emits a spike after this stage
455  if (bit_field_test(delay_stage_config, n)) {
456 
457  // Calculate key all spikes coming from this neuron will be
458  // sent with
459  uint32_t neuron_index = ((d * num_neurons) + n);
460  uint32_t spike_key = neuron_index + key;
461 
462  if (delay_stage_spike_counters[n] > 0) {
463  log_debug("Neuron %u sending %u spikes after delay"
464  "stage %u with key %x",
465  n, delay_stage_spike_counters[n], d,
466  spike_key);
467  }
468 
469  // fire n spikes as payload, 1 as none payload.
470  if (has_key) {
471  if (delay_stage_spike_counters[n] > 1) {
472  log_debug(
473  "seeing packet with key %d and payload %d",
474  spike_key, delay_stage_spike_counters[n]);
475 
476  tdma_processing_send_packet(
477  spike_key, delay_stage_spike_counters[n],
478  WITH_PAYLOAD, timer_count);
479 
480  // update counter
481  n_spikes_sent += delay_stage_spike_counters[n];
482  } else if (delay_stage_spike_counters[n] == 1) {
483  log_debug("sending spike with key %d", spike_key);
484 
485  tdma_processing_send_packet(
486  spike_key, 0, NO_PAYLOAD, timer_count);
487 
488  // update counter
489  n_spikes_sent++;
490  }
491  }
492  }
493  }
494  }
495  }
496 
497  // Zero all counters in current time slot
498  uint32_t current_time_slot = time & num_delay_slots_mask;
499  zero_spike_counters(spike_counters[current_time_slot], num_neurons);
500 }

503 void c_main(void) {
504  log_info("max dtcm supply %d", sark_heap_max(sark.heap, 0));
505  if (!initialize()) {
506  log_error("Error in initialisation - exiting!");
507  rt_error(RTE_SWERR);
508  }
509 
510  // Start the time at "-1" so that the first tick will be 0
511  time = UINT32_MAX;
512 
513  // Initialise the incoming spike buffer
514  if (!in_spikes_initialize_spike_buffer(IN_BUFFER_SIZE)) {
515  rt_error(RTE_SWERR);
516  }
517 
518  // Set timer tick (in microseconds)
519  log_debug("Timer period %u", timer_period);
520  spin1_set_timer_tick(timer_period);
521 
522  // Register callbacks
523  spin1_callback_on(MC_PACKET_RECEIVED, incoming_spike_callback, MC_PACKET);
524  spin1_callback_on(TIMER_TICK, timer_callback, TIMER);
525 
526  simulation_run();
527 }
