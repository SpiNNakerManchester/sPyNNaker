190 static inline uint32_t round_to_next_pot(uint32_t v) {
191  v--;
192  v |= v >> 1;
193  v |= v >> 2;
194  v |= v >> 4;
195  v |= v >> 8;
196  v |= v >> 16;
197  v++;
198  return v;
199 }

204 static bool read_parameters(struct delay_parameters *params) {
205  log_debug("read_parameters: starting");
206 
207  has_key = params->has_key;
208  key = params->key;
209  incoming_key = params->incoming_key;
210  incoming_mask = params->incoming_mask;
211  incoming_neuron_mask = ~incoming_mask;
212  log_debug("\t key = 0x%08x, incoming key = 0x%08x, incoming mask = 0x%08x,"
213  "incoming key mask = 0x%08x",
214  key, incoming_key, incoming_mask, incoming_neuron_mask);
215 
216  num_neurons = params->n_atoms;
217  neuron_bit_field_words = get_bit_field_size(num_neurons);
218 
219  num_delay_stages = params->n_delay_stages;
220  n_delay_in_a_stage = params->n_delay_in_a_stage;
221  max_keys = num_neurons * num_delay_stages;
222 
223  clear_input_buffers_of_late_packets = params->clear_packets;
224 
225  num_delay_slots = num_delay_stages * n_delay_in_a_stage;
226  // We need an extra slot here (to make one clearable after the maximum delay
227  // time), and a power of 2 (to make it easier to loop)
228  uint32_t num_delay_slots_pot = round_to_next_pot(num_delay_slots + 1);
229  num_delay_slots_mask = num_delay_slots_pot - 1;
230 
231  log_info("\t parrot neurons = %u, neuron bit field words = %u,"
232  " num delay stages = %u, num delay slots = %u (pot = %u),"
233  " num delay slots mask = %08x, n delay in a stage = %u",
234  num_neurons, neuron_bit_field_words,
235  num_delay_stages, num_delay_slots, num_delay_slots_pot,
236  num_delay_slots_mask, n_delay_in_a_stage);
237 
238  // Allocate array of counters for each delay slot
239  spike_counters = spin1_malloc(num_delay_slots_pot * sizeof(uint8_t*));
240  if (spike_counters == NULL) {
241  log_error("failed to allocate memory for array of size %u bytes",
242  num_delay_slots_pot * sizeof(uint8_t*));
243  return false;
244  }
245 
246  for (uint32_t s = 0; s < num_delay_slots_pot; s++) {
247  // Allocate an array of counters for each neuron and zero
248  spike_counters[s] = spin1_malloc(num_neurons * sizeof(uint8_t));
249  if (spike_counters[s] == NULL) {
250  log_error("failed to allocate memory for bitfield of size %u bytes",
251  num_neurons * sizeof(uint8_t));
252  return false;
253  }
254  zero_spike_counters(spike_counters[s], num_neurons);
255  }
256 
257  n_colour_bits = params->n_colour_bits;
258  colour_mask = (1 << n_colour_bits) - 1;
259 
260  log_debug("read_parameters: completed successfully");
261  return true;
262 }

361 static inline void spike_process(void) {
362 
363  // While there are any incoming spikes
364  spike_t s;
365  uint32_t state = spin1_int_disable();
366  while (in_spikes_get_next_spike(&s)) {
367  spin1_mode_restore(state);
368  n_processed_spikes++;
369 
370  if ((s & incoming_mask) == incoming_key) {
371  // Mask out neuron ID
372  uint32_t spike_id = key_n(s);
373  uint32_t spike_colour = spike_id & colour_mask;
374  uint32_t neuron_id = spike_id >> n_colour_bits;
375  if (neuron_id < num_neurons) {
376 
377  // Account for delayed spikes
378  int32_t colour_diff = colour - spike_colour;
379  uint32_t colour_delay = colour_diff & colour_mask;
380 
381  // Get current time slot of incoming spike counters
382  uint32_t time_slot = (time + colour_delay) & num_delay_slots_mask;
383  uint8_t *time_slot_spike_counters = spike_counters[time_slot];
384 
385  // Increment counter
386  if (time_slot_spike_counters[neuron_id] ==
387  COUNTER_SATURATION_VALUE) {
388  saturation_count += 1;
389  } else {
390  time_slot_spike_counters[neuron_id]++;
391  }
392  log_debug("Incrementing counter %u = %u\n",
393  neuron_id,
394  time_slot_spike_counters[neuron_id]);
395  n_spikes_added++;
396  } else {
397  n_packets_dropped_due_to_invalid_neuron_value += 1;
398  log_debug("Invalid neuron ID %u", neuron_id);
399  }
400  } else {
401  n_packets_dropped_due_to_invalid_key += 1;
402  log_debug("Invalid spike key 0x%08x", s);
403  }
404  state = spin1_int_disable();
405  }
406 
407  spike_processing = false;
408  spin1_mode_restore(state);
409 }

423 static void background_callback(uint local_time, UNUSED uint timer_count) {
424  // Loop through delay stages
425  for (uint32_t d = 0; d < num_delay_stages; d++) {
426  uint32_t delay_stage_delay = (d + 1) * n_delay_in_a_stage;
427  if (local_time >= delay_stage_delay) {
428  uint32_t delay_stage_time_slot =
429  (local_time - delay_stage_delay) & num_delay_slots_mask;
430  uint8_t *delay_stage_spike_counters =
431  spike_counters[delay_stage_time_slot];
432 
433  log_debug("%u: Checking time slot %u for delay stage %u (delay %u)",
434  local_time, delay_stage_time_slot, d, delay_stage_delay);
435 
436  // Loop through neurons
437  for (uint32_t n = 0; n < num_neurons; n++) {
438 
439  // If no spikes to send, skip
440  if (delay_stage_spike_counters[n] == 0) {
441  continue;
442  }
443 
444  // Calculate key all spikes coming from this neuron will be
445  // sent with
446  uint32_t neuron_index = ((d * num_neurons) + n);
447  uint32_t spike_key = (key + (neuron_index << n_colour_bits)) | colour;
448 
449  log_debug("Neuron %u sending %u spikes after delay"
450  "stage %u with key %x",
451  n, delay_stage_spike_counters[n], d,
452  spike_key);
453 
454  // fire n spikes as payload, 1 as none payload.
455  if (has_key) {
456  if (delay_stage_spike_counters[n] > 1) {
457  log_debug(
458  "%d: sending packet with key 0x%08x and payload %d",
459  time, spike_key, delay_stage_spike_counters[n]);
460 
461  send_spike_mc_payload(spike_key, delay_stage_spike_counters[n]);
462 
463  // update counter
464  n_spikes_sent += delay_stage_spike_counters[n];
465  } else if (delay_stage_spike_counters[n] == 1) {
466  log_debug("%d: sending spike with key 0x%08x", time, spike_key);
467 
468  send_spike_mc(spike_key);
469 
470  // update counter
471  n_spikes_sent++;
472  }
473  }
474  }
475  }
476  }
477  n_backgrounds_queued--;
478 }

483 static void timer_callback(uint timer_count, UNUSED uint unused1) {
484  uint32_t state = spin1_int_disable();
485  uint32_t n_spikes = in_spikes_size();
486  if (clear_input_buffers_of_late_packets) {
487  in_spikes_clear();
488  }
489  // Record the count whether clearing or not for provenance
490  count_input_buffer_packets_late += n_spikes;
491  time++;
492 
493  // Clear counters
494  if (time > num_delay_slots) {
495  uint32_t clearable_slot = ((time - 1) - num_delay_slots) & num_delay_slots_mask;
496  log_debug("%d: Clearing time slot %d", time, clearable_slot);
497  zero_spike_counters(spike_counters[clearable_slot], num_neurons);
498  }
499 
500  log_debug("Timer tick %u", time);
501 
502  // If a fixed number of simulation ticks are specified and these have passed
503  if (simulation_is_finished()) {
504  // handle the pause and resume functionality
505  simulation_handle_pause_resume(NULL);
506 
507  log_debug("Delay extension finished at time %u, %u received spikes, "
508  "%u processed spikes, %u sent spikes, %u added spikes",
509  time, n_in_spikes, n_processed_spikes, n_spikes_sent,
510  n_spikes_added);
511 
512  log_debug("Delayed %u times", n_delays);
513 
514  // Subtract 1 from the time so this tick gets done again on the next
515  // run
516  time--;
517 
518  simulation_ready_to_read();
519  spin1_mode_restore(state);
520  return;
521  }
522 
523  // Set the colour for the time step
524  colour = time & colour_mask;
525 
526  if (!spin1_schedule_callback(background_callback, time, timer_count, BACKGROUND)) {
527  // We have failed to do this timer tick!
528  n_background_overloads++;
529  } else {
530  n_backgrounds_queued++;
531  if (n_backgrounds_queued > max_backgrounds_queued) {
532  max_backgrounds_queued++;
533  }
534  }
535  spin1_mode_restore(state);
536 }

539 void c_main(void) {
540  log_info("max dtcm supply %d", sark_heap_max(sark.heap, 0));
541  if (!initialize()) {
542  log_error("Error in initialisation - exiting!");
543  rt_error(RTE_SWERR);
544  }
545 
546  // Start the time at "-1" so that the first tick will be 0
547  time = UINT32_MAX;
548 
549  // Initialise the incoming spike buffer
550  if (!in_spikes_initialize_spike_buffer(IN_BUFFER_SIZE)) {
551  rt_error(RTE_SWERR);
552  }
553 
554  // Set timer tick (in microseconds)
555  log_debug("Timer period %u", timer_period);
556  spin1_set_timer_tick(timer_period);
557 
558  // Register callbacks
559  spin1_callback_on(MC_PACKET_RECEIVED, incoming_spike_callback, MC_PACKET);
560  spin1_callback_on(MCPL_PACKET_RECEIVED, incoming_spike_callback, MC_PACKET);
561  spin1_callback_on(TIMER_TICK, timer_callback, TIMER);
562  spin1_callback_on(USER_EVENT, user_callback, USER);
563 
564  simulation_run();
565 }
