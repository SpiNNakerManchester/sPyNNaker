69 static inline void rk2_kernel_midpoint(
70  REAL h, neuron_t *neuron, REAL input_this_timestep) {
71  // to match Mathematica names
72  REAL lastV1 = neuron->V;
73  REAL lastU1 = neuron->U;
74  REAL a = neuron->A;
75  REAL b = neuron->B;
76 
77  REAL pre_alph = REAL_CONST(140.0) + input_this_timestep - lastU1;
78  REAL alpha = pre_alph
79  + (REAL_CONST(5.0) + MAGIC_MULTIPLIER * lastV1) * lastV1;
80  REAL eta = lastV1 + REAL_HALF(h * alpha);
81 
82  // could be represented as a long fract?
83  REAL beta = REAL_HALF(h * (b * lastV1 - lastU1) * a);
84 
85  neuron->V += h * (pre_alph - beta
86  + (REAL_CONST(5.0) + MAGIC_MULTIPLIER * eta) * eta);
87 
88  neuron->U += a * h * (-lastU1 - beta + b * eta);
89 }

96 state_t neuron_model_state_update(
97  uint16_t num_excitatory_inputs, const input_t *exc_input,
98  uint16_t num_inhibitory_inputs, const input_t *inh_input,
99  input_t external_bias, neuron_t *restrict neuron) {
100  REAL total_exc = 0;
101  REAL total_inh = 0;
102 
103  for (int i =0; i<num_excitatory_inputs; i++) {
104  total_exc += exc_input[i];
105  }
106  for (int i =0; i<num_inhibitory_inputs; i++) {
107  total_inh += inh_input[i];
108  }
109 
110  input_t input_this_timestep = total_exc - total_inh
111  + external_bias + neuron->I_offset;
112 
113  // the best AR update so far
114  rk2_kernel_midpoint(neuron->this_h, neuron, input_this_timestep);
115  neuron->this_h = global_params->machine_timestep_ms;
116 
117  return neuron->V;
118 }
