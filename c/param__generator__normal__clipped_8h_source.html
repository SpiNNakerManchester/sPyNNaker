60 static void *param_generator_normal_clipped_initialize(address_t *region) {
61  // Allocate memory for the data
62  struct param_generator_normal_clipped *obj =
63  spin1_malloc(sizeof(struct param_generator_normal_clipped));
64  struct normal_clipped_params *params_sdram = (void *) *region;
65 
66  // Copy the parameters in
67  obj->params = *params_sdram++;
68  *region = (void *) params_sdram;

70  log_debug("normal clipped mu = %k, sigma = %k, low = %k, high = %k",
71  obj->params.mu, obj->params.sigma, obj->params.low,
72  obj->params.high);
73 
74  // Initialise the RNG for this generator

99 static void param_generator_normal_clipped_generate(
100  void *generator, uint32_t n_indices, UNUSED uint32_t pre_neuron_index,
101  UNUSED uint16_t *indices, accum *values) {
102  // For each index, generate a normally distributed random value, redrawing
103  // if outside the given range
104  struct param_generator_normal_clipped *obj = generator;
105  for (uint32_t i = 0; i < n_indices; i++) {
106  uint32_t n_draws = 0;
107  do {
108  accum value = rng_normal(obj->rng);
109  values[i] = obj->params.mu + (value * obj->params.sigma);
110  n_draws++;
111  } while ((values[i] < obj->params.low || values[i] > obj->params.high)
112  && (n_draws < MAX_REDRAWS));
113  if (n_draws == MAX_REDRAWS) {
114  log_error("Maximum number of redraws (%u) exceeded on clipped normal "
115  "distribution with mu=%k, sigma=%k, low=%k, high=%k",
116  n_draws, obj->params.mu, obj->params.sigma, obj->params.low,
117  obj->params.high);
118  rt_error(RTE_SWERR);
119  }
120  }
121 }
