1077 static inline multicast_packet _key_retina(
1078  uint32_t retina_pixels, uint32_t time_stamps, uint32_t uart_id) {
1079  if (retina_pixels == 128 * 128) {
1080  // if fine, create message

1085  .payload_flag = WITH_PAYLOAD
1086  };
1087  }
1088  if (retina_pixels == 64 * 64) {
1089  return (multicast_packet) {
1090  .key = (ACTIVE_RETINA_EVENT_STREAMING_KEYS_CONFIGURATION |
1091  (uart_id << OFFSET_FOR_UART_ID) | instance_key),
1092  .payload = (time_stamps | PAYLOAD_RETINA_64_DOWN_SAMPLING),
1093  .payload_flag = WITH_PAYLOAD
1094  };
1095  }
1096  if (retina_pixels == 32 * 32) {
1097  return (multicast_packet) {
1098  .key = (ACTIVE_RETINA_EVENT_STREAMING_KEYS_CONFIGURATION |
1099  (uart_id << OFFSET_FOR_UART_ID) | instance_key),
1100  .payload = (time_stamps | PAYLOAD_RETINA_32_DOWN_SAMPLING),
1101  .payload_flag = WITH_PAYLOAD
1102  };
1103  }
1104  if (retina_pixels == 16 * 16) {
1105  return (multicast_packet) {
1106  .key = (ACTIVE_RETINA_EVENT_STREAMING_KEYS_CONFIGURATION |
1107  (uart_id << OFFSET_FOR_UART_ID) | instance_key),
1108  .payload = (time_stamps | PAYLOAD_RETINA_16_DOWN_SAMPLING),
1109  .payload_flag = WITH_PAYLOAD
1110  };
1111  }
1112 
1113  log_error("The number of pixels is not supported in this protocol.");
1114  rt_error(RTE_SWERR);
1115  return (multicast_packet) {.key = 0, .payload = 0, .payload_flag = 0};
1116 }

1126 static inline multicast_packet munich_protocol_set_retina_transmission(
1127  bool events_in_key, uint32_t retina_pixels,
1128  bool payload_holds_time_stamps, uint32_t size_of_time_stamp_in_bytes,
1129  uint32_t uart_id) {
1130  // if events in the key.
1131  if (events_in_key) {
1132  if (!payload_holds_time_stamps) {
1133  // not using payloads
1134  return _key_retina(retina_pixels, PAYLOAD_NO_TIMESTAMPS, uart_id);
1135  }
1136  // using payloads
1137  switch (size_of_time_stamp_in_bytes) {
1138  case 0:
1139  return _key_retina(
1140  retina_pixels, PAYLOAD_DELTA_TIMESTAMPS, uart_id);
1141  case 2:
1142  return _key_retina(
1143  retina_pixels, PAYLOAD_TWO_BYTE_TIME_STAMPS, uart_id);
1144  case 3:
1145  return _key_retina(
1146  retina_pixels, PAYLOAD_THREE_BYTE_TIME_STAMPS, uart_id);
1147  case 4:
1148  return _key_retina(
1149  retina_pixels, PAYLOAD_FOUR_BYTE_TIME_STAMPS, uart_id);
1150  default:
1151  log_error("Unknown size of timestamp in bytes: %d\n",
1152  size_of_time_stamp_in_bytes);
1153  rt_error(RTE_SWERR);
1154  return (multicast_packet) {
1155  .key = 0, .payload = 0, .payload_flag = 0
1156  };
1157  }
1158  } else {
1159  // using payloads to hold all events
1160 
1161  // warn users about models
1162  log_warning("The current sPyNNaker models do not support the reception "
1163  "of packets with payloads, therefore you will need to add a "
1164  "adaptor model between the device and sPyNNaker models.");
1165 
1166  // verify that its what the end user wants.
1167  if (payload_holds_time_stamps || (size_of_time_stamp_in_bytes == 0)) {
1168  log_error(
1169  "If you are using payloads to store events, you cannot"
1170  " have time stamps at all.");
1171  rt_error(RTE_SWERR);
1172  }
1173 
1174  // if fine, create message
1175  return (multicast_packet) {
1176  .key = (ACTIVE_RETINA_EVENT_STREAMING_KEYS_CONFIGURATION |
1177  (uart_id << OFFSET_FOR_UART_ID) | instance_key),
1178  .payload = (PAYLOAD_NO_TIMESTAMPS |
1179  PAYLOAD_RETINA_NO_DOWN_SAMPLING_IN_PAYLOAD),
1180  .payload_flag = WITH_PAYLOAD
1181  };
1182  }
