59 static bool current_source_initialise(address_t cs_address, uint32_t n_neurons) {
60  // Avoid the loops if no current sources
61  #if !defined(_CURRENT_SOURCE_DC_H_) && !defined(_CURRENT_SOURCE_AC_H) && \
62  !defined(_CURRENT_SOURCE_STEP_H_) && !defined(_CURRENT_SOURCE_NOISY_H_)
63  io_printf(IO_BUF, "no current sources defined \n");
64  return true;
65  #else
66 
67  n_neurons_on_core = n_neurons;
68 
69  // Read from cs_address; the first value is the number of current sources
70  n_current_sources = cs_address[0];
71 
72  // Don't initialise if no current sources
73  if (n_current_sources != 0) {
74 
75  neuron_current_source = spin1_malloc(n_neurons * sizeof(uint32_t*));
76 
77  // Loop over neurons and read in the current IDs and indices
78  uint32_t next = 1;
79  for (uint32_t n=0; n < n_neurons; n++) {
80  uint32_t n_sources = (uint32_t) cs_address[next];
81  uint32_t struct_size = (1 + (2 * n_sources)) * sizeof(uint32_t);
82  neuron_current_source[n] = spin1_malloc(struct_size);
83  spin1_memcpy(neuron_current_source[n], &cs_address[next], struct_size);
84 
85  next += 1 + (n_sources * 2);
86 
87  }
88 
89  // Read number of each type of current source
90  n_dc_sources = (uint32_t) cs_address[next++];
91  n_ac_sources = (uint32_t) cs_address[next++];
92  n_step_sources = (uint32_t) cs_address[next++];
93  n_noisy_sources = (uint32_t) cs_address[next++];
94 
95  log_debug("Check numbers of sources: AC %u DC %u STEP %u NOISY %u",
96  n_dc_sources, n_ac_sources, n_step_sources, n_noisy_sources);
97 
98  // Now initialise separate sources
99 #ifdef _CURRENT_SOURCE_DC_H_
100  if (!current_source_dc_init(n_dc_sources, &next)) {
101  return false;
102  }
103 #else
104  if (n_dc_sources > 0) {
105  log_error("DC current source is not supported for this build");
106  return false;
107  }
108 #endif
109 
110 #ifdef _CURRENT_SOURCE_AC_H_
111  if (!current_source_ac_init(n_ac_sources, &next)) {
112  return false;
113  }
114 #else
115  if (n_ac_sources > 0) {
116  log_error("AC current source is not supported for this build");
117  return false;
118  }
119 #endif
120 
121 #ifdef _CURRENT_SOURCE_STEP_H_
122  if (!current_source_step_init(cs_address, n_step_sources, &next)) {
123  return false;
124  }
125 #else
126  if (n_step_sources > 0) {
127  log_error("Step current source is not supported for this build");
128  return false;
129  }
130 #endif
131 
132 #ifdef _CURRENT_SOURCE_NOISY_H_
133  if (!current_source_noisy_init(n_noisy_sources, &next)) {
134  return false;
135  }
136 #else
137  if (n_noisy_sources > 0) {
138  log_error("Noisy current source is not supported for this build");
139  return false;
140  }
141 #endif
142 
143  }
144 
145  return true;
146 
147  #endif
148 }

212 static inline REAL current_source_get_offset(uint32_t time, uint32_t neuron_index) {
213  // Avoid the loops if no current sources defined
214  #if !defined(_CURRENT_SOURCE_DC_H_) && !defined(_CURRENT_SOURCE_AC_H) && \
215  !defined(_CURRENT_SOURCE_STEP_H_) && !defined(_CURRENT_SOURCE_NOISY_H_)
216  return ZERO;
217  #else
218 
219  REAL current_offset = ZERO;
220 
221  // Also avoid the loop if no current sources set by user
222  if (n_current_sources != 0) {
223  uint32_t n_current_sources_neuron =
224  neuron_current_source[neuron_index]->n_current_sources;
225  if (n_current_sources_neuron > 0) {
226  for (uint32_t n_cs=0; n_cs < n_current_sources_neuron; n_cs++) {
227  uint32_t cs_id =
228  neuron_current_source[neuron_index]->cs_id_index_list[n_cs].cs_id;
229  uint32_t cs_index =
230  neuron_current_source[neuron_index]->cs_id_index_list[n_cs].cs_index;
231  // Now do the appropriate calculation based on the ID value
232  #ifdef _CURRENT_SOURCE_DC_H_
233  if (cs_id == 1) { // DCSource
234  current_offset += current_source_dc_get_offset(cs_index, time);
235  }
236  #endif
237  #ifdef _CURRENT_SOURCE_AC_H_
238  if (cs_id == 2) { // ACSource
239  current_offset += current_source_ac_get_offset(cs_index, time);
240  }
241  #endif
242  #ifdef _CURRENT_SOURCE_STEP_H_
243  if (cs_id == 3) { // StepCurrentSource
244  current_offset += current_source_step_get_offset(cs_index, time);
245  }
246  #endif
247  #ifdef _CURRENT_SOURCE_NOISY_H_
248  if (cs_id == 4) { // NoisyCurrentSource
249  current_offset += current_source_noisy_get_offset(cs_index, time);
250  }
251  #endif
252  }
253 
254  }
255  }
256 
257  return current_offset;
258 
259  #endif
260 }
