154 void background_callback(uint timer_count, uint local_time) {
155  profiler_write_entry_disable_irq_fiq(PROFILER_ENTER | PROFILER_TIMER);
156 
157  log_debug("Timer tick %u \n", local_time);
158 
159  // Now do neuron time step update
160  neuron_do_timestep_update(local_time, timer_count);
161 
162  profiler_write_entry_disable_irq_fiq(PROFILER_EXIT | PROFILER_TIMER);
163  n_backgrounds_queued--;
164 }

170 void timer_callback(uint timer_count, UNUSED uint unused) {
171  // Disable interrupts to stop MC getting in the way of this bit
172  uint32_t state = spin1_int_disable();
173 
174  // Increment time step
175  time++;
176 
177  // Clear any outstanding spikes
178  local_only_clear_input(time);
179 
180  // Allow things to interrupt again
181  spin1_mode_restore(state);
182 
183  // Process ring buffers for the inputs from last time step
184  process_ring_buffers();
185 
186  /* if a fixed number of simulation ticks that were specified at startup
187  * then do reporting for finishing */
188  if (simulation_is_finished()) {
189 
190  // Enter pause and resume state to avoid another tick
191  simulation_handle_pause_resume(resume_callback);
192 
193  // Pause neuron processing
194  neuron_pause();
195 
196  // Pause common functions
197  common_pause(recording_flags);
198 
199  // Subtract 1 from the time so this tick gets done again on the next
200  // run
201  time--;
202 
203  simulation_ready_to_read();
204  return;
205  }
206 
207  // Push the rest to the background
208  if (!spin1_schedule_callback(background_callback, timer_count, time, BACKGROUND)) {
209  // We have failed to do this timer tick!
210  n_background_overloads++;
211  } else {
212  n_backgrounds_queued++;
213  if (n_backgrounds_queued > max_backgrounds_queued) {
214  max_backgrounds_queued++;
215  }
216  }
217 }

222 static bool initialise(void) {
223  log_debug("Initialise: started");
224 
225  data_specification_metadata_t *ds_regions;
226  if (!initialise_common_regions(
227  &timer_period, &simulation_ticks, &infinite_run, &time,
228  &recording_flags, c_main_store_provenance_data, timer_callback,
229  COMMON_REGIONS, COMMON_PRIORITIES, &ds_regions)) {
230  return false;
231  }
232 
233  // Setup neurons
234  uint32_t n_rec_regions_used;
235  if (!initialise_neuron_regions(
236  ds_regions, NEURON_REGIONS, &n_rec_regions_used)) {
237  return false;
238  }
239 
240  if (!local_only_initialise(
241  data_specification_get_region(LOCAL_ONLY_REGION, ds_regions),
242  data_specification_get_region(LOCAL_ONLY_PARAMS_REGION, ds_regions),
243  n_rec_regions_used, &ring_buffers)) {
244  return false;
245  }
246 
247  // Set timer tick (in microseconds)
248  log_debug("setting timer tick callback for %d microseconds", timer_period);
249  spin1_set_timer_tick(timer_period);
250 
251  log_debug("Initialise: finished");
252  return true;
253 }
