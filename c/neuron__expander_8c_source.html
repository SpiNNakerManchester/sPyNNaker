46 // An array of how much to add to align data to 4-bytes
47 // indexed by [current offset % 4][size to write % 4].
48 // Note sizes are expected to be 1, 2, 4 or 8 (indices 1, 2, 0, 0)
49 // so other values are 0 (but indexing an array is quicker).
50 static const uint32_t ADD[4][3] = {
51  {0, 0, 0}, // Offset 0 - anything goes
52  {3, 0, 1}, // Offset 1 - needs shift for 2, 4 and 8 (indices 2, 0, 0)
53  {2, 0, 0}, // Offset 2 - needs shift for 4 and 8 (indices 0, 0)
54  {1, 0, 1} // Offset 3 - needs shift for 2, 4 and 8 (indices 2, 0, 0)
55 };

57 static inline uint32_t align_offset(uint32_t offset, uint32_t size) {
58  if (size == 0) {
59  log_error("Size of 0!");
60  rt_error(RTE_SWERR);
61  }
62  uint32_t size_mod = size & MOD_4;
63  if (size_mod == 3) {
64  log_error("Size %u unsupported!", size);
65  }
66  return ADD[offset & MOD_4][size & MOD_4];
67 }

170 static bool read_struct_builder_region(void **region,
171  void **neuron_params_region, uint32_t n_neurons) {
172 
173  // Get the config for the repeated struct and move on to the data after
174  neuron_params_struct_t *config = *region;
175  uint8_t *reg = *region;
176  *region = &(reg[config->struct_size_bytes]);
177 
178  // Read items from SDRAM into variables for use later
179  uint32_t n_params = config->n_params;
180  uint32_t bytes_per_repeat = config->bytes_per_repeat;
181  uint32_t n_repeats_total = config->n_repeats_total;
182  if (n_repeats_total == REPEAT_PER_NEURON) {
183  n_repeats_total = n_neurons;
184  }
185  log_debug("Reading %u params, %u bytes per neuron, %u neurons, "
186  "%u bytes to end of struct", n_params, bytes_per_repeat,
187  n_repeats_total, config->struct_size_bytes);
188 
189  // Get the current struct position
190  uint8_t* struct_ptr = *neuron_params_region;
191  *neuron_params_region = &(struct_ptr[bytes_per_repeat * n_repeats_total]);
192 
193  // Keep track of the offset of the param from the start of each struct
194  uint32_t param_offset = 0;
195 
196  // Go through the params in the struct to be expanded
197  neuron_param_t *param = &(config->param[0]);
198  for (uint32_t p = 0; p < n_params; p++) {
199  log_debug(" Param %u, type=%u, n_items=%u", p, param->param_type,
200  param->n_items);
201 
202  // Get the writer for the parameter type
203  type_info *writer = get_type_writer(param->param_type);
204 
205  // Align the offset for the size of parameter to be written
206  param_offset += align_offset(param_offset, writer->size);
207  log_debug(" Writing %u bytes each time to struct offset %u",
208  writer->size, param_offset);
209 
210  // Go through the items and generate
211  uint32_t n_items = param->n_items;
212  uint32_t offset = 0;
213  for (uint32_t i = 0; i < n_items; i++) {
214  neuron_param_item_t item = param->item[i];
215  log_debug(" Item %u, generator=%u, n_repeats=%u",
216  i, item.generator, item.n_repeats);
217 
218  param_generator_t gen = param_generator_init(item.generator, region);
219  if (gen == NULL) {
220  return false;
221  }
222 
223  uint32_t n_repeats = item.n_repeats;
224  if (n_repeats == REPEAT_PER_NEURON) {
225  n_repeats = n_neurons;
226  log_debug(" (Really only repeating %u times!)", n_repeats);
227  }
228 
229  // Generate the requested number of times
230  for (uint32_t r = 0; r < n_repeats; r++) {
231  accum value = param_generator_generate(gen);
232  uint32_t index = offset + param_offset;
233  log_debug(" Writing %k to offset %u", value, index);
234  writer->writer(&(struct_ptr[index]), value);
235  offset += bytes_per_repeat;
236  }
237 
238  // Finish with the generator
239  param_generator_free(gen);
240  }
241 
242  // After writing, add to the offset for the next parameter
243  param_offset += writer->size;
244 
245  // Go to the next param
246  param = (neuron_param_t *) &(param->item[n_items]);
247  }
248 
249  // Return success!
250  return true;
251 }

253 static inline void read_index(uint32_t n_items, recording_index_t *items,
254  uint32_t n_neurons, uint32_t n_neurons_max, uint32_t n_neurons_recording,
255  uint16_t *sdram_out) {
256  // Go through the data
257  uint16_t indices[n_neurons_max];
258  uint32_t neuron_id = 0;
259  uint16_t next_index = 0;
260  for (uint32_t i = 0; i < n_items; i++) {
261  if (neuron_id >= n_neurons) {
262  log_error("The next neuron %u >= the maximum %u", neuron_id, n_neurons);
263  rt_error(RTE_SWERR);
264  }
265  recording_index_t item = items[i];
266  uint32_t n_repeats = item.n_repeats;
267  if (n_repeats == REPEAT_PER_NEURON_RECORDED) {
268  n_repeats = n_neurons;
269  }
270  if (item.is_recording) {
271  for (uint32_t r = 0; r < n_repeats; r++) {
272  if (next_index >= n_neurons_recording) {
273  log_error("The next index %u >= the maximum %u", next_index,
274  n_neurons_recording);
275  rt_error(RTE_SWERR);
276  }
277  indices[neuron_id++] = next_index++;
278  }
279  } else {
280  for (uint32_t r = 0; r < n_repeats; r++) {
281  indices[neuron_id++] = (uint16_t) n_neurons_recording;
282  }
283  }
284  }
285 
286  // Copy to SDRAM
287  uint32_t *index_words = (uint32_t *) &(indices[0]);
288  uint32_t *sdram_out_words = (uint32_t *) sdram_out;
289  for (uint32_t i = 0; i < (n_neurons_max >> 1); i++) {
290  sdram_out_words[i] = index_words[i];
291  }
292 }

331  }

366 static bool run_neuron_expander(data_specification_metadata_t *ds_regions,
367  void *params_address) {
368  // Read in the global parameters
369  expander_config_t *sdram_config = params_address;
370  expander_config_t *config = spin1_malloc(sizeof(expander_config_t));
371  fast_memcpy(config, sdram_config, sizeof(expander_config_t));
372  log_info("Generating %u structs", config->n_structs);
373 
374  // Get the synaptic matrix region
375  void *neuron_params_region = data_specification_get_region(
376  config->neuron_params_region, ds_regions);
377 
378  // Store the RNGs
379  population_rng = &(config->population_rng);
380  core_rng = &(config->core_rng);
381 
382  log_info("Population RNG: %u %u %u %u", population_rng->seed[0],
383  population_rng->seed[1], population_rng->seed[2],
384  population_rng->seed[3]);
385 
386  log_info("Core RNG: %u %u %u %u", core_rng->seed[0],
387  core_rng->seed[1], core_rng->seed[2], core_rng->seed[3]);
388 
389 
390  // Go through each struct and generate
391  void *address = &(sdram_config[1]);
392 
393  // Read the remaining structs
394  uint32_t n_neurons = config->n_neurons;
395  for (uint32_t s = 0; s < config->n_structs; s++) {
396  if (!read_struct_builder_region(&address, &neuron_params_region,
397  n_neurons)) {
398  return false;
399  }
400  }
401 
402  // Read recording data
403  recording_params_t *recording_params = address;
404  recording_params_t *sdram_recording_params = data_specification_get_region(
405  config->neuron_recording_region, ds_regions);
406 
407  // Copy header data
408  uint32_t n_variables = recording_params->n_recordable_variables;
409  uint32_t n_bitfields = recording_params->n_recordable_bit_fields;
410  sdram_recording_params->n_recordable_variables = n_variables;
411  sdram_recording_params->n_recordable_bit_fields = n_bitfields;
412 
413  // Move read and write pointers
414  address = &(recording_params[1]);
415  void *sdram_address = &(sdram_recording_params[1]);
416 
417  // Round up the number of neurons to the next multiple of 2
418  uint32_t n_neurons_max = (n_neurons + CEIL_TO_2) & FLOOR_TO_2;
419 
420  // Do variables
421  for (uint32_t i = 0; i < n_variables; i++) {
422  read_recorded_variable(&address, &sdram_address, n_neurons, n_neurons_max);
423  }
424  // Do bitfields
425  for (uint32_t i = 0; i < n_bitfields; i++) {
426  read_recorded_bitfield(&address, &sdram_address, n_neurons, n_neurons_max);
427  }
428 
429  // Clear checksums to avoid later issues
430  ds_regions->regions[config->neuron_params_region].checksum = 0;
431  ds_regions->regions[config->neuron_params_region].n_words = 0;
432  ds_regions->regions[config->neuron_recording_region].checksum = 0;
433  ds_regions->regions[config->neuron_recording_region].n_words = 0;
434 
435  return true;
436 }

439 void c_main(void) {
440  sark_cpu_state(CPU_STATE_RUN);
441 
442  log_info("Starting To Build Connectors");
443 
444  // Get pointer to 1st virtual processor info struct in SRAM and get USER1;
445  // This is the ID of the connection builder region from which to read the
446  // rest of the data
447  vcpu_t *virtual_processor_table = (vcpu_t*) SV_VCPU;
448  uint user1 = virtual_processor_table[spin1_get_core_id()].user1;
449 
450  // Get the addresses of the regions
451  data_specification_metadata_t *ds_regions =
452  data_specification_get_data_address();
453  void *params_address = data_specification_get_region(user1, ds_regions);
454  log_info("\tReading SDRAM at 0x%08x", params_address);
455 
456  // Run the expander
457  if (!run_neuron_expander(ds_regions, params_address)) {
458  log_info("!!! Error reading SDRAM data !!!");
459  rt_error(RTE_ABORT);
460  }
461 
462  log_info("Finished On Machine Connectors!");
463 }
