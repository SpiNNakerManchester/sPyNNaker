41 // The plastic control words used by Morrison synapses store an axonal delay
42 // in the upper 3 bits.
43 // Assuming a maximum of 16 delay slots, this is all that is required as:
44 //
45 // 1) Dendritic + Axonal <= 15
46 // 2) Dendritic >= Axonal
47 //
48 // Therefore:
49 //
50 // * Maximum value of dendritic delay is 15 (with axonal delay of 0)
51 // - It requires 4 bits
52 // * Maximum value of axonal delay is 7 (with dendritic delay of 8)
53 // - It requires 3 bits
54 //
55 // | Axonal delay | Dendritic delay | Type | Index |
56 // |---------------------------|--------------------|-------------------|--------------------|
57 // | SYNAPSE_AXONAL_DELAY_BITS | SYNAPSE_DELAY_BITS | SYNAPSE_TYPE_BITS | SYNAPSE_INDEX_BITS |
58 // | | | SYNAPSE_TYPE_INDEX_BITS |
59 // |---------------------------|--------------------|----------------------------------------|
60 #ifndef SYNAPSE_AXONAL_DELAY_BITS
61 #define SYNAPSE_AXONAL_DELAY_BITS 3
62 #endif
63 
64 #define SYNAPSE_AXONAL_DELAY_MASK \
65  ((1 << SYNAPSE_AXONAL_DELAY_BITS) - 1)

114 static inline bool synapse_dynamics_stdp_init(
115  address_t *address, stdp_params *params, uint32_t n_synapse_types,
116  uint32_t *ring_buffer_to_input_buffer_left_shifts) {
117 
118  // Load parameters
119  stdp_params *sdram_params = (stdp_params *) *address;
120  spin1_memcpy(params, sdram_params, sizeof(stdp_params));
121 
122  // Load timing dependence data
123  address_t weight_region_address = timing_initialise(
124  (address_t) &sdram_params[1]);
125  if (weight_region_address == NULL) {
126  return false;
127  }
128 
129  // Load weight dependence data
130  address_t weight_result = weight_initialise(
131  weight_region_address, n_synapse_types,
132  ring_buffer_to_input_buffer_left_shifts);
133  if (weight_result == NULL) {
134  return false;
135  }
136 
137  // Update address to after the region just read
138  *address = weight_result;
139  return true;
140 }

155 static inline fixed_stdp_synapse synapse_dynamics_stdp_get_fixed(
156  uint32_t control_word, uint32_t time) {
157  // Extract control-word components
158  // **NOTE** cunningly, control word is just the same as lower
159  // 16-bits of 32-bit fixed synapse so same functions can be used
160  uint32_t delay_dendritic = synapse_row_sparse_delay(control_word,
161  synapse_type_index_bits, synapse_delay_mask);
162  uint32_t delay_axonal = 0; //sparse_axonal_delay(control_word);
163  uint32_t type_index = synapse_row_sparse_type_index(control_word,
164  synapse_type_index_mask);
165  return (fixed_stdp_synapse) {
166  .delay_dendritic = delay_dendritic,
167  .delay_axonal = delay_axonal,
168  .type = synapse_row_sparse_type(
169  control_word, synapse_index_bits, synapse_type_mask),
170  .index = synapse_row_sparse_index(
171  control_word, synapse_index_mask),
172  .type_index = type_index,
173  .ring_buffer_index = synapse_row_get_ring_buffer_index_combined(
174  delay_axonal + delay_dendritic + time, type_index,
175  synapse_type_index_bits, synapse_delay_mask)

179 static inline void synapse_dynamics_stdp_update_ring_buffers(
180  weight_t *ring_buffers, fixed_stdp_synapse s, int32_t weight) {
181  uint32_t accumulation = ring_buffers[s.ring_buffer_index] + weight;
182 
183  uint32_t sat_test = accumulation & 0x10000;
184  if (sat_test) {
185  accumulation = sat_test - 1;
186  plastic_saturation_count++;
187  }
188 
189  ring_buffers[s.ring_buffer_index] = accumulation;
190 }

193 static inline control_t control_conversion(
194  uint32_t id, uint32_t delay, uint32_t type) {
195  control_t new_control =
196  (delay & ((1 << synapse_delay_bits) - 1)) << synapse_type_index_bits;
197  new_control |= (type & ((1 << synapse_type_index_bits) - 1)) << synapse_index_bits;
198  new_control |= id & ((1 << synapse_index_bits) - 1);
199  return new_control;
200 }
