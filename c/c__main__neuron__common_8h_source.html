25 uint64_t udiv64(uint64_t num, uint64_t den) {
26 
27  uint64_t quot = 0;
28  uint64_t qbit = 1;
29 
30  if (den == 0) {
31  /* Intentional divide by zero, without
32  triggering a compiler warning which
33  would abort the build */
34  return 1/((unsigned)den);
35  }
36 
37  /* Left-justify denominator and count shift */
38  while ((int64_t) den >= 0) {
39  den <<= 1;
40  qbit <<= 1;
41  }
42 
43  while (qbit) {
44  if (den <= num) {
45  num -= den;
46  quot += qbit;
47  }
48  den >>= 1;
49  qbit >>= 1;
50  }
51  return quot;
52 }
