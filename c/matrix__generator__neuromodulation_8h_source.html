98 static void matrix_generator_neuromodulation_write_row(
99  void *generator,
100  address_t synaptic_matrix, UNUSED address_t delayed_synaptic_matrix,
101  UNUSED uint32_t n_pre_neurons, uint32_t pre_neuron_index,
102  uint32_t max_row_n_words, UNUSED uint32_t max_delayed_row_n_words,
103  UNUSED uint32_t synapse_type_bits, UNUSED uint32_t synapse_index_bits,
104  uint32_t synapse_type, uint32_t n_synapses, uint16_t *indices,
105  UNUSED uint16_t *delays, uint16_t *weights, UNUSED uint32_t max_stage,
106  UNUSED uint32_t max_delay_per_stage) {
107 
108  matrix_generator_neuromodulation *conf = generator;
109 
110  // The number of words in a row including headers
111  uint32_t n_row_words = max_row_n_words + 3;
112 
113  // The normal row position
114  row_nm_plastic_t *row =
115  (row_nm_plastic_t *) &synaptic_matrix[pre_neuron_index * n_row_words];
116  row->plastic_plastic_size = 1;
117  row->is_neuromodulation = 1;
118  row->is_reward = synapse_type == conf->reward_synapse_type;
119  row->synapse_type = synapse_type;
120 
121  // Set the fixed-fixed size to 0 and point to the fixed-plastic region
122  row_nm_fixed_t *fixed = (row_nm_fixed_t *) &row[1];
123  fixed->fixed_fixed_size = 0;
124  fixed->fixed_plastic_size = n_synapses;
125 
126  // Go through the synapses
127  for (uint32_t synapse = 0; synapse < n_synapses; synapse++) {
128  // Post-neuron index
129  uint32_t post_index = indices[synapse];
130 
131  // Weight
132  uint16_t weight = weights[synapse];
133 
134  fixed->fixed_plastic_data[synapse] = (weight << 16) | post_index;
135  }
136 }
